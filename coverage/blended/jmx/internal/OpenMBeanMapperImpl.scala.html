38   def productToMap(cc: Product): Map[String, Element] = {
39     if(cc.productArity == 0) {
40       // an object?
41       // just store the name
42       Map("object" -> fieldToElement("name", cc.getClass().getName()))
43     }
44     else {
45       val values: Iterator[Any] = cc.productIterator
46       cc.getClass().getDeclaredFields().filter { f =>
47         f.getName != "$outer"
48       }.map { f =>
49         val value = values.next()
50         val element = fieldToElement(f.getName, value)
51         f.getName -> element
52       }.toMap
53     }
54   }
