9 case class Accumulator(
10   count : Long = 0,
11   totalMsec : Long = 0,
12   minMsec : Long = Long.MaxValue,
13   maxMsec : Long = Long.MinValue
14 ) {
15   def record(msec : Long) : Accumulator = {
16     copy(
17       count = count + 1,
18       totalMsec = totalMsec + msec,
19       minMsec = Math.min(minMsec, msec),
20       maxMsec = Math.max(maxMsec, msec)
21     )
22   }
23 
24   val avg : Double = if (count == 0) {
25     0d
26   } else {
27     totalMsec.toDouble / count
28   }
29 }

31 case class Entry(
32   component : String,
33   subComponents : Map[String, String],
34   success : Accumulator = Accumulator(),
35   failed : Accumulator = Accumulator(),
36   lastFailed: Long = -1L,
37   inflight : Long = 0L
38 ) {
39 
40   private def duration(started : ServiceInvocationEvent, completed : ServiceInvocationEvent) : Long =
41     Math.max(0, completed.timestamp - started.timestamp)
42 
43   def update(evt : ServiceInvocationEvent, started : ServiceInvocationStarted) : Entry = evt match  {
44     case _ : ServiceInvocationStarted =>
45       copy(inflight = inflight + 1)
46     case c : ServiceInvocationCompleted =>
47       copy(success = success.record(duration(started, c)), inflight = inflight - 1)
48     case f : ServiceInvocationFailed =>
49       copy(failed = failed.record(duration(started, f)), lastFailed = System.currentTimeMillis(), inflight = inflight - 1)
50   }
51 }

53 case class StatisticsState(
54   mbeanExporter : OpenMBeanExporter,
55   invocations : Map[String, ServiceInvocationStarted] = Map.empty,
56   entries : Map[String, Entry] = Map.empty
57 ) {
58 
59   private val log : Logger = Logger[StatisticsState]
60   private val datakey : ServiceInvocationStarted => String = s => s"${s.component}-${s.subComponents.mkString(",")}"
61 
62   override def toString: String =
63     s"${getClass().getSimpleName()}(${invocations.keys.mkString("(", ",", ")")}, ${entries.keys.mkString("(", ",", ")")})"
64 
65   def update(evt : ServiceInvocationEvent) : StatisticsState = {
66     val (newState, entry) : (StatisticsState, Option[Entry]) = evt match {
67       case started : ServiceInvocationStarted => invocationStarted(started)
68       case completed : ServiceInvocationEvent => complete(completed)
69     }
70 
71     entry.foreach { e =>
72       val toPublish : PublishEntry = PublishEntry.create(e)
73       log.trace(s"Exporting/updating JMX entry [${toPublish.name}] : [$toPublish]")
74       mbeanExporter.export(toPublish, toPublish.name, replaceExisting = true).recover {
75         case NonFatal(e) =>
76           log.warn(e)(s"Could not register mbean with name [${toPublish.name}]")
77       }
78     }
79 
80     newState
81   }
82 
83   private def complete(evt : ServiceInvocationEvent) : (StatisticsState, Option[Entry]) = {
84     log.trace(s"Recording Service invocation event [$evt]")
85     invocations.get(evt.id) match {
86       case None =>
87         log.debug(s"No active service invocation found for [${evt.id}]")
88         (this, None)
89       case Some(started) =>
90         val key : String = datakey(started)
91         entries.get(key) match {
92           case None =>
93             log.warn(s"No statistics entry found for [$key]")
94             (this, None)
95           case Some(e) =>
96             evt match {
97               case _ : ServiceInvocationStarted =>
98                 log.warn(s"Unexpected event for updating completing entry [$key]")
99                 (this, None)
100               case _ =>
101                 val newEntry : Entry = e.update(evt, started)
102                 (copy(
103                   invocations = invocations - evt.id,
104                   entries = entries + (key -> newEntry)
105                 ), Some(newEntry))
106             }
107         }
108     }
109   }
110 
111   private def invocationStarted(evt : ServiceInvocationStarted) : (StatisticsState, Option[Entry]) = {
112     invocations.get(evt.id) match {
113       case None =>
114         log.trace(s"Recording Service invocation start [$evt]")
115         val key : String = datakey(evt)
116         val newEntry : Entry = entries.getOrElse(key, Entry(evt.component, evt.subComponents)).update(evt, evt)
117 
118         val newState : StatisticsState = copy(
119           invocations = invocations + (evt.id -> evt),
120           entries = entries + (key -> newEntry)
121         )
122         (newState, Some(newEntry))
123       case Some(_) =>
124         log.debug(s"Service with id [${evt.id}] has already been started")
125         (this, None)
126     }
127   }
128 }

130 class StatisticsActor(mbeanExporter: OpenMBeanExporter) extends Actor {
131 
132   private[this] val log = Logger[this.type]
133 
134   override def preStart(): Unit = {
135     super.preStart()
136     // register event handler
137     log.debug(s"Subscribing self [${self}] to ServiceInvocationEvents from system.eventStream")
138     context.system.eventStream.subscribe(self, classOf[ServiceInvocationEvent])
139     context.become(working(StatisticsState(mbeanExporter)))
140   }
141 
142   override def receive: Receive = Actor.emptyBehavior
143 
144   private def working(statsData : StatisticsState) : Receive = {
145     case evt : ServiceInvocationEvent =>
146       val newState : StatisticsState = statsData.update(evt)
147       log.debug(s"new statistics state is [$newState]")
148       context.become(working(newState))
149   }
150 }
