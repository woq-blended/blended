22   // Get the content of the envelope as a ByteString which we can write to disk
23   private def extractContent(env : FlowEnvelope) : Try[ByteString] = Try {
24     env.flowMessage match {
25       case tMsg : TextFlowMessage =>
26         val charSet = env.headerWithDefault[String](cfg.charsetHeader, "UTF-8")
27         log.logEnv(env, LogLevel.Debug, s"Using charset [$charSet] to file drop text message [${env.id}]")
28         ByteString(tMsg.getText().getBytes(charSet))
29 
30       case bMsg : BinaryFlowMessage =>
31         bMsg.content
32 
33       case m =>
34         val eTxt = s"Dropping files unsupported for msg [${env.id}] of type [${m.getClass.getName}]"
35         log.logEnv(env, LogLevel.Error, eTxt)
36         throw new Exception(eTxt)
37     }
38   }
39 
40   // Try to get the correlation Id from the message, fall back with the correlation ID from
41   // the FlowEnvelope, finally fall back with the envelope ID
42   private[this] def corrId(env : FlowEnvelope) : String = {
43     env.headerWithDefault[String](
44       "JMSCorrelationID",
45       env.headerWithDefault[String](corrIdHeader(headerConfig.prefix), env.id)
46     )
47   }
48 
49   // extract the drop Command from the envelope
50   private[this] def dropCmd(env : FlowEnvelope)(f : FlowEnvelope => Try[ByteString]) : Try[FileDropCommand] = Try {
51     FileDropCommand(
52       id = env.id,
53       content = f(env).unwrap,
54       directory = env.headerWithDefault[String](cfg.dirHeader, cfg.defaultDir),
55       fileName = env.headerWithDefault[String](cfg.fileHeader, ""),
56       compressed = env.headerWithDefault[Boolean](cfg.compressHeader, false),
57       append = env.headerWithDefault[Boolean](cfg.appendHeader, false),
58       timestamp = env.headerWithDefault[Long](timestampHeader(headerConfig.prefix), System.currentTimeMillis()),
59       properties = Map("JMSCorrelationID" -> corrId(env)) ++ env.flowMessage.header.mapValues(_.value),
60       log = log.underlying
61     )
62   }
63 
64   private[this] def handleError(env: FlowEnvelope, error: Throwable): FileDropResult = {
65     log.logEnv(env, LogLevel.Error, s"Error dropping envelope [${env.id}] to file : [${error.getMessage()}]")
66     val cmd = dropCmd(env)(e => Success(ByteString(""))).get
67     dropActor ! FileDropAbort(env.id, error)
68     FileDropResult(cmd, Some(error))
69   }
70 
71   def dropEnvelope(env : FlowEnvelope) : (FileDropCommand, Future[FileDropResult]) = {
72 
73     val p : Promise[FileDropResult] = Promise()
74 
75     dropCmd(env)(extractContent) match {
76       case Success(cmd) =>
77         implicit val to : Timeout = Timeout(cfg.dropTimeout)
78         implicit val eCtxt : ExecutionContext = system.dispatcher
79 
80         (dropActor ? cmd).mapTo[FileDropResult].onComplete {
81           case Success(r) => r.error match {
82             case None => p.complete(Success(r))
83             case Some(t) => p.complete(Success(handleError(env, t)))
84           }
85           case Failure(t) => p.complete(Success(handleError(env, t)))
86         }
87 
88         (cmd, p.future)
89 
90       case Failure(t) =>
91         val r : FileDropResult = handleError(env, t)
92         p.complete(Success(r))
93         (r.cmd, p.future)
94     }
95   }
