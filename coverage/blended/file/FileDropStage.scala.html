13 /**
14   * The Filedrop Stream consumes messages from a given upstream producing
15   * FlowEnvelope. Each FlowEnvelope will by written to the designated file
16   * drop location using an instance of a file drop actor. The file actor
17   * responds with a FileDropResult, which is passed further downstream.
18   *
19   * Users of the Filedrop Stage must implement the logic of handling FileDropResults
20   * if required.
21   */
22 class FileDropStage(
23   name : String,
24   config : FileDropConfig,
25   headerCfg : FlowHeaderConfig,
26   dropActor: ActorRef,
27   log: FlowEnvelopeLogger
28 )(implicit system: ActorSystem)
29   extends GraphStage[FlowShape[FlowEnvelope, FileDropResult]] {
30 
31   private val in = Inlet[FlowEnvelope](s"FileDropStream($name.in)")
32   private val out = Outlet[FileDropResult](s"FileDropStream($name.out)")
33 
34   override def shape : FlowShape[FlowEnvelope, FileDropResult] = FlowShape(in, out)
35 
36   override def createLogic(inheritedAttributes : Attributes) : GraphStageLogic = {
37     new GraphStageLogic(shape) {
38 
39       private var results : List[FileDropResult] = List.empty
40       private implicit val eCtxt : ExecutionContext = system.dispatcher
41 
42       private val dropper : EnvelopeFileDropper = new EnvelopeFileDropper(
43         cfg = config, headerConfig = headerCfg, dropActor = dropActor, log = log
44       )
45 
46       private val resultCallback : AsyncCallback[FileDropResult] = getAsyncCallback[FileDropResult]{ r =>
47         log.underlying.debug(s"Filedrop result is [$r]")
48         results = r :: results
49 
50         pushResult()
51 
52         // We are ready to receive a new File
53         pull(in)
54       }
55 
56       private def pushResult() : Unit = {
57         if (results.nonEmpty && isAvailable(out)) {
58           push(out, results.last)
59           results = results.take(results.size - 1)
60         }
61       }
62 
63       // We kick off by signalling that we are ready to drop our first file.
64       override def preStart() : Unit = pull(in)
65 
66       // The InHandler accepts messages from upstream and needs to kick off dropping the
67       // message to the file system.
68       setHandler(in, new InHandler {
69         override def onPush() : Unit = {
70           val env = grab(in)
71           log.logEnv(env, LogLevel.Debug, s"Filedropstage [$name] is processing envelope [${env.id}]")
72           val (cmd, result) = dropper.dropEnvelope(env)
73 
74           result.onComplete {
75             case Success(r) => resultCallback.invoke(r)
76             case Failure(t) => resultCallback.invoke(FileDropResult(cmd, Some(t)))
77           }
78         }
79       })
80 
81       // The outhandler grabs the Filedropresult if available and pushes it down stream
82       setHandler(out, new OutHandler {
83         override def onPull() : Unit = pushResult()
84       })
85     }
86   }
87 }
