23 case class FileDropCommand(
24   id : String,
25   content: ByteString,
26   directory: String,
27   fileName: String,
28   compressed: Boolean,
29   append: Boolean,
30   timestamp: Long,
31   properties: Map[String, Any],
32   log : Logger
33 ) {
34 
35   val trimmedFileName : String = fileName.trim()
36 
37   override def equals(obj: Any): Boolean = obj match {
38     case cmd : FileDropCommand =>
39       content.equals(cmd.content) &&
40       directory.equals(cmd.directory) &&
41       fileName.equals(cmd.fileName) &&
42       compressed == cmd.compressed &&
43       append == cmd.append &&
44       timestamp == cmd.timestamp &&
45       properties.equals(cmd.properties)
46     case _ => false
47   }
48 
49   override def hashCode(): Int = toString().hashCode()
50 
51   val timestampAsString : String = FileDropCommand.tsPattern.format(new Date(timestamp))
52 
53   override def toString: String = {
54 
55     s"FileDropCommand[$id](dir = [$directory], fileName = [$fileName], compressed = $compressed, append = $append, " +
56     s"timestamp = [$timestampAsString], content-size = ${content.length}), properties=${properties.mkString("[", ",", "]")}"
57   }
58 
59   // determine the final file name for a file drop
60   val finalFile : File = {
61 
62     if (trimmedFileName.length() < fileName.length()) {
63       log.warn(s"Using trimmed file name [$trimmedFileName] for [${toString()}]")
64     }
65 
66     val file = new File(directory, trimmedFileName)
67 
68     if (!append) {
69       if (file.exists()) {
70         // In case we need to generate a new file name
71         new File(directory, fileName.lastIndexOf('.') match {
72           case -1 => s"dup_${timestampAsString}_${fileName}"
73           case pos => s"${fileName.substring(0, pos)}.dup_${timestampAsString}${fileName.substring(pos)}"
74         })
75       } else {
76         // In case we do not append and we can generate a new file
77         file
78       }
79     } else {
80       // In case we append the content to a file, we keep the same final file name
81       file
82     }
83   }

120   private def checkDirectory(dir: File) : Try[File] = {
121 
122     if (!dir.exists()) {
123       actorLog.debug(s"Creating directory [${dir.getAbsolutePath}]")
124       dir.mkdirs()
125     }
126 
127     if (dir.exists() && dir.isDirectory() && dir.canWrite()) {
128       Success(dir)
129     } else {
130       Failure(new Exception(s"Directory [${dir.getAbsolutePath()}] does not exist or is not writable."))
131     }
132   }

134   // A temp file is only created when we need to append to an existing file.
135   def tmpFile(cmd: FileDropCommand) : Try[Option[File]] = Try {
136 
137     val filter : DirectoryStream.Filter[Path] = new DirectoryStream.Filter[Path] {
138       override def accept(entry: Path): Boolean = {
139         val n : String = entry.toFile().getName()
140         val p : String = cmd.finalFile.getName()
141         n.startsWith(p) && n.endsWith(".tmp")
142       }
143     }
144 
145     def sourceFile(f : File) : Option[File] = {
146       if (f.exists()) {
147         Some(f)
148       } else {
149         val dirStream = Files.newDirectoryStream(new File(cmd.directory).toPath(), filter)
150         val files : List[File] = dirStream.iterator().asScala.toList.map(_.toFile()).sortBy(_.lastModified()).reverse
151         if (files.isEmpty) {
152           None
153         } else {
154           val result : File = files.head
155           cmd.log.warn(s"Using tmp file for append source in [${cmd.id}] : [${result.getAbsolutePath()}]")
156           files.tail.foreach{ f =>
157             cmd.log.debug(s"Removing tmp file [$f]")
158             f.delete()
159           }
160           Some(result)
161         }
162       }
163     }
164 
165     if (cmd.append) {
166       sourceFile(cmd.finalFile) match {
167         case Some(f) =>
168           val tmpName = s"${cmd.fileName}.${cmd.timestampAsString}.tmp"
169           val tmpFile = new File(cmd.directory, tmpName)
170           cmd.log.debug(s"Creating temporary file for [${cmd.id}] [${tmpFile.getAbsolutePath}]")
171           Files.move(f.toPath(), tmpFile.toPath(), REPLACE_EXISTING)
172           Some(tmpFile)
173         case None =>
174           None
175       }
176     } else {
177       None
178     }
179   }

181   // The outfile is the file that will temporarily hold the final content
182   def outFile(cmd: FileDropCommand) : File =
183     new File(cmd.directory, s"${cmd.fileName}.${cmd.timestampAsString}.out")

185   // prepare the output stream, if required for append
186   def prepareOutputStream(cmd: FileDropCommand) : Try[(OutputStream, Option[File], File)] = Try {
187 
188     val tf : Option[File] = tmpFile(cmd).get
189 
190     val of : File = outFile(cmd)
191 
192     val outStream = if (!cmd.append) {
193       new FileOutputStream(of)
194     } else {
195       tf match {
196         case None => new FileOutputStream(of)
197         case Some(f) =>
198           Files.copy(f.toPath, of.toPath)
199           cmd.log.debug(s"Appending to file for [${cmd.id}] [${of.getAbsolutePath()}]")
200           new FileOutputStream(of, true)
201       }
202     }
203 
204     (outStream, tf, of)
205   }

207   // Create the input stream, potentially from a zip stream
208   def inputStream(cmd : FileDropCommand) : Option[InputStream] = {
209     if (cmd.compressed) {
210 
211       try {
212         Some(new GZIPInputStream(new BufferedInputStream(new ByteArrayInputStream(cmd.content.toArray))))
213       } catch {
214         case NonFatal(_) =>
215           val zis = new ZipInputStream(new BufferedInputStream(new ByteArrayInputStream(cmd.content.toArray)))
216           val next = Option(zis.getNextEntry)
217           if (next.isEmpty) None else Some(zis)
218       }
219     } else {
220       Some(new ByteArrayInputStream(cmd.content.toArray))
221     }
222   }

224   private def cleanUp(state : FileDropState) : FileDropState = {
225 
226     val newState : FileDropState = {
227       try {
228         state.is.foreach(_.close())
229         state.os.foreach(_.close())
230 
231         if (state.error.isEmpty) {
232           state.cmd.log.debug(s"Removing tmp file for [${state.cmd.id}] : [${state.tmpFile}]")
233           state.tmpFile.foreach{ tf => Files.delete(tf.toPath()) }
234           state.cmd.log.debug(s"Creating final file for [${state.cmd.id}] : [${state.cmd.finalFile}]")
235           Files.move(state.outFile.toPath(), state.cmd.finalFile.toPath(), REPLACE_EXISTING)
236         }
237 
238         state
239       } catch {
240         case NonFatal(t) =>
241           state.cmd.log.warn(t)(s"Error creating final file for [${state.cmd}]")
242           state.copy(error = Some(t))
243       }
244     }
245 
246     if (newState.error.isDefined) {
247       // in case an error was encountered, we will restore the original file
248       // and forget the append
249       state.tmpFile.foreach{ tf =>
250         try {
251           Files.move(tf.toPath(), state.cmd.finalFile.toPath(), REPLACE_EXISTING)
252         } catch {
253           case NonFatal(t) =>
254             newState.cmd.log.warn(t)(s"Error cleaning up files (move) for [${state.cmd}]")
255         }
256 
257         try {
258           Files.delete(state.outFile.toPath())
259         } catch {
260           case NonFatal(t) =>
261             newState.cmd.log.warn(t)(s"Error cleaning up files (delete) for [${state.cmd}]")
262         }
263       }
264     } else {
265       // In case the command was successful, we will delete the tmpfile
266       // and create the final file
267       newState.cmd.log.info(
268         s"Successfully processed filedrop [${state.cmd}] and created file [${state.cmd.finalFile.getAbsolutePath()}]"
269       )
270     }
271 
272     newState
273   }

275   private[this] def respond(
276     state : FileDropState,
277     pending : Seq[FileDropState]
278   ) : Unit = {
279 
280     val result : FileDropState = cleanUp(state)
281     state.requestor ! FileDropResult.result(result.cmd, result.error)
282 
283     pending match {
284       case Seq() =>
285         actorLog.debug(s"No more pending Filedrops...switching to idle state")
286         context.become(idle(Seq.empty))
287       case s =>
288         val state : FileDropState = s.last
289         actorLog.debug(s"Scheduling pending file drop [${state.cmd.id}]")
290         context.become(dropping(state, s.take(s.size - 1)))
291         self ! FileDropChunk
292     }
293   }

297   // While in dropping state we drop one chunk after the other
298   private def dropping(state : FileDropState, pending : Seq[FileDropState]) : Receive = {
299     case cmd : FileDropCommand =>
300       val newState : FileDropState = createDropState(cmd, sender())
301       cmd.log.debug(s"File drop for [${cmd.id}] is pending, [${pending.size + 1}] pending in total")
302       context.become(dropping(state, pending ++ Seq(newState)))
303       self ! FileDropChunk
304 
305     // Abort the file drop
306     case FileDropAbort(id, t) =>
307       if (state.cmd.id == id) {
308         respond(state.copy(error = Some(t)), pending)
309       } else {
310         pending.find(_.cmd.id == id).foreach{ s =>
311           cleanUp(s)
312           s.requestor ! FileDropResult.result(s.cmd, Some(t))
313           context.become(dropping(state, pending.filter(_.cmd.id != id)))
314         }
315       }
316 
317     // Drop the next chunk to the out directory
318     case FileDropChunk =>
319       (state.is, state.os) match {
320         // Streams are still open, so we can proceed
321         case (Some(in), Some(out)) =>
322           try {
323             // scalastyle:off magic.number
324             val buffer : Array[Byte] = new Array[Byte](4096)
325             // scalastyle:on magic.number
326             val cnt = in.read(buffer)
327             if (cnt >= 0) {
328               out.write(buffer, 0, cnt)
329               self ! FileDropChunk
330             } else {
331               state.cmd.log.debug(s"Successfully wrote message [${state.cmd.id}] to [${state.outFile}].")
332               respond(state.copy(error = None), pending)
333             }
334           } catch {
335             case NonFatal(e) =>
336               state.cmd.log.warn(e)(s"Error while dropping file [${state.cmd}] : [${e.getMessage()}]")
337               respond(state. copy(error = Some(e)), pending)
338           }
339         // The input or output stream is not open
340         case (_,_) =>
341           respond(state.copy(error = Some(new Exception("Illegal stream state in FileDropActor"))), pending)
342       }
343   }

345   private def createDropState(cmd : FileDropCommand, requestor: ActorRef) : FileDropState = {
346     checkDirectory(new File(cmd.directory)) match {
347       case Success(_) => prepareOutputStream(cmd) match {
348         case Success((s, tf, of)) =>
349           FileDropState(requestor = requestor, cmd = cmd, tmpFile = tf,
350             outFile = of, is = inputStream(cmd), os = Some(s), error = None
351           )
352         case Failure(t) =>
353           FileDropState(requestor = requestor, cmd = cmd, tmpFile = None,
354             outFile = outFile(cmd), is = None, os = None, error = Some(t)
355           )
356       }
357 
358       case Failure(t) =>
359         FileDropState(requestor = requestor, cmd = cmd, tmpFile = None,
360           outFile = outFile(cmd), is = None, os = None, error = Some(t)
361         )
362     }
363   }
364 
365   private def idle(pending : Seq[FileDropState]) : Receive = {
366 
367     case FileDropChunk =>
368     case FileDropAbort(_, _) =>
369 
370     case cmd : FileDropCommand =>
371       val state : FileDropState = createDropState(cmd, sender())
372       if(state.error.isEmpty) {
373         context.become(dropping(state, pending))
374         self ! FileDropChunk
375       } else {
376         respond(state, pending)
377       }
378   }
