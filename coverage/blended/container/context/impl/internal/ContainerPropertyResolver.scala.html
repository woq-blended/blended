71   private[this] def lastIndexOfPattern(current : Int, next : Int, s : String, p : Pattern) : Int = {
72 
73     val m : Matcher = p.matcher(s)
74 
75     if (m.find(Math.max(next, 0))) {
76       lastIndexOfPattern(m.start(), m.start() + m.group(0).length, s, p)
77     } else {

82   private[this] def extractVariableElement(
83     line : String,
84     startDelim : Pattern,
85     endDelim : String
86   ) : ExtractedElement = {
87 
88     val idx : Int = lastIndexOfPattern(-1, -1, line, startDelim)
89 
90     idx match {
91       case -1 =>
92         ExtractedElement("", line, "")
93       case start =>
94         line.indexOf(endDelim, start) match {
95           case -1 => throw new PropertyResolverException(s"Error decoding replacement pattern [$line] : missing end delimiter [$endDelim]")
96           case end =>
97 
98             val subline = line.substring(start, end + endDelim.length)
99             val subMatcher : Matcher = startDelim.matcher(subline)
100             subMatcher.find()
101 
102             val patternStart = start + subMatcher.group(0).length
103 
104             ExtractedElement(
105               prefix = line.substring(0, start),
106               line.substring(patternStart, end),
107               postfix = line.substring(end + endDelim.length),
108               modifier = subMatcher.group(2)
109             )
110         }
111     }
112   }

152   private[this] def extractModifier(s : String) : Option[(Modifier, String)] = {
153     val pos = s.indexOf(":")
154     val (modName, params) = if (pos != -1) (s.substring(0, pos), s.substring(pos + 1)) else (s, "")
155     modifiers.get(modName).map { m => (m, params) }

158   private[this] def processRule(rule : String, additionalProps : Map[String, Any]) : Try[String] = {
159 
160     log.trace(s"Processing rule [$rule]")
161 
162     // A rule can have any number of parameterized modifiers, spearated by ","
163     val (ruleName : String, modifier : List[String]) = rule.indexOf("(") match {
164       case -1 => (rule, List.empty)
165       case s => (
166         rule.substring(0, s),
167         rule.substring(s + 1, rule.indexOf(")", s)).split(",").toList
168       )
169     }
170 
171     log.trace(s"rule [$ruleName], [${modifier.mkString(",")}]")
172 
173     // Now we need to find the proper modifiers to apply with their parameters
174     val mods : List[(Modifier, String)] = modifier.map(extractModifier).collect {
175       case Some(m) => m
176     }
177 
178     // First, we resolve the rule from the environment vars or System properties
179     // The resolution is mandatory
180     val applyRules : Try[String] = properties.get(ruleName) match {
181       case Some(s) => Success(s)
182       case None =>
183         Option(
184           additionalProps.getOrElse(
185             rule,
186             System.getenv().getOrDefault(
187               ruleName, System.getProperty(ruleName)
188             )
189           )
190         ) match {
191             case Some(s) => Success(s.toString())
192             case None =>
193               resolver.get(ruleName) match {
194                 case Some(r) => r(ruleName)
195                 case None    => Failure(new PropertyResolverException(s"Unable to resolve property [$rule]"))
196               }
197           }
198     }
199 
200     // Then we apply the collected modifiers left to right to the resolved value
201     val result : Try[String] = mods.foldLeft[Try[String]](applyRules)( (current, value) => current match {
202       case f @ Failure(_) => f
203       case Success(s) => value._1(s, value._2)
204     })
205 
206     result
207   }

209   def resolve(line : String, additionalProps : Map[String, Any] = Map.empty) : Try[AnyRef] = Try {
210     // First we check if we have replacements in "Blended Style"
211 
212     /**
213      * TODO: Map the modifier to installable services
214      * delayed : do not evaluate the inner expression any further
215      * encrypted : resolve an encrypted value
216      */
217     if (lastIndexOfPattern(-1, -1, line, resolveStartDelim) != -1) {
218       val e = extractVariableElement(line, resolveStartDelim, resolveEndDelim)
219       e.modifier match {
220         case "delayed" =>
221           resolve(e.prefix, additionalProps).unwrap + e.pattern + resolve(e.postfix, additionalProps).unwrap
222 
223         case "encrypted" =>
224           val decrypted : String = cryptoSupport.decrypt(e.pattern).get
225           resolve(e.prefix, additionalProps).unwrap.toString() +
226             resolve(decrypted, additionalProps).unwrap.toString() +
227             resolve(e.postfix, additionalProps).unwrap.toString()
228 
229         case _ =>
230           // First we resolve the inner expression to resolve any nested expressions
231           val inner = resolve(e.pattern, additionalProps).unwrap.toString
232           // then we resolve the entire line with the inner expression resolved
233           resolve(e.prefix + processRule(inner, additionalProps).unwrap + e.postfix, additionalProps).unwrap
234 
235       }
236     } else {
237       lastIndexOfPattern(-1, -1, line, evalStartDelim) match {
238         case i if i < 0 =>
239           line
240         case i if i >= 0 =>
241           val e = extractVariableElement(line, evalStartDelim, evalEndDelim)
242           if (e.prefix.isEmpty && e.postfix.isEmpty) {
243             evaluate(e.pattern, additionalProps)
244           } else {
245             resolve(e.prefix + evaluate(e.pattern, additionalProps).toString() + e.postfix, additionalProps).unwrap
246           }
247       }
248     }
249   }

252   private[impl] def evaluate(
253     line : String, additionalProps : Map[String, Any] = Map.empty
254   ) : AnyRef = {
255 
256     val context = new StandardEvaluationContext()
257     classOf[SpelFunctions].getDeclaredMethods().foreach { m =>
258       context.registerFunction(m.getName(), m)
259     }
260 
261     ctCtxt.foreach{ c =>
262       context.setRootObject(c)
263       context.setVariable("ctCtxt", c)
264     }
265 
266     additionalProps.foreach {
267       case (k, v) =>
268         context.setVariable(k, v)
269     }
270 
271     val exp = parseExpression(line).get
272 
273     Option(exp.getValue(context)) match {
274       case None =>
275         log.warn(s"Could not resolve expression [${line}], using empty String")
276         ""
277       case Some(r) =>
278         log.trace(s"Evaluated [$line] to [$r][${r.getClass().getName()}]")
279         r
280     }
281   }
