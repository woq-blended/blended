25   private[internal] def readConfigFile(f : File, fallback : Config) : Config = {
26 
27     if (f.exists() && f.isFile() && f.canRead()) {
28       ConfigFactory.parseFile(f, ConfigParseOptions.defaults().setAllowMissing(false))
29         .withFallback(fallback)
30         .withFallback(sysProps)
31         .withFallback(envProps)
32         .resolve()
33     } else {
34       ConfigFactory.empty()
35     }
36   }
37 
38   private[internal] def fullKeyset(prefix : String, cfg: Config) : List[String] = {
39     val keySet : List[String] = cfg.root().keySet().asScala.toList
40 
41     val toAdd : List[List[String]] = keySet.map{ s =>
42       if (cfg.getIsNull(s)) {
43         List(prefix + s)
44       } else {
45         try {
46           List(prefix + s) ++ fullKeyset(prefix + s + ".", cfg.getConfig(s))
47         } catch {
48           case NonFatal(_) => List(prefix + s)
49         }
50       }
51     }
52 
53     toAdd.flatten
54   }
55 
56   private[internal] def evaluatedConfig(rawCfg : Config, ctContext : ContainerContext) : Try[Config] = Try {
57     val replaced = rawCfg.entrySet().asScala.map{ e =>
58       val v : AnyRef = e.getValue() match {
59         case s if s.valueType() == ConfigValueType.STRING =>
60           val resolved : String = ctContext.resolveString(s.unwrapped().toString()).map(_.toString()).unwrap
61           resolved
62         case o =>
63           o.unwrapped()
64       }
65       e.getKey() -> ConfigValueFactory.fromAnyRef(v)
66     }.toMap.asJava
67 
68     ConfigFactory.parseMap(replaced).resolve()
69   }
70 
71   def safeConfig(cfgDir : String, fileName: String, fallback: Config, ctContext : ContainerContext) : Config = {
72     val file = new File(cfgDir, fileName)
73     log.debug(s"Retrieving config from [${file.getAbsolutePath()}]")
74 
75     evaluatedConfig(readConfigFile(file, fallback), ctContext) match {
76       case Failure(e) =>
77         log.warn(s"Error reading [${file.getAbsolutePath()}] : [${e.getMessage()}], using empty config")
78         ConfigFactory.empty()
79       case Success(cfg) =>
80         cfg
81     }
82   }
