104   private lazy val overlayConfig : Config = {
105     val cfg : Option[File] = brandingProperties.get(RuntimeConfig.Properties.PROFILE_DIR) match {
106       case Some(profileDir) =>
107         brandingProperties.get(RuntimeConfig.Properties.OVERLAYS) match {
108           case Some(overlays) =>
109             val overlayRefs = overlays.split("[,]").toList.map(_.split("[:]", 2)).flatMap {
110               case Array(n, v) => Some(OverlayRef(n, v))
111               case x =>
112                 log.debug("Unsupported overlay: " + x.mkString(":"))
113                 None
114             }.toSet
115             if (overlayRefs.isEmpty) {
116               None
117             } else {
118               val dir = LocalOverlays.materializedDir(overlayRefs, new File(profileDir))
119               val confFile = new File(dir, s"$CONFIG_DIR/application_overlay.conf")
120               if (confFile.exists()) {
121                 log.debug(s"About to read extra application overlay override file: ${confFile}")
122                 Some(confFile)
123               } else {
124                 None
125               }
126             }
127           case _ => None
128         }
129       case _ => None
130     }
131 
132     val result : Config = cfg match {
133       case Some(oc) => ConfigFactory.parseFile(oc, ConfigParseOptions.defaults().setAllowMissing(false))
134       case _        => ConfigFactory.empty()
135     }
136 
137     log.trace(s"After reading overlay config : $result")
138 
139     result
140   }

142   override lazy val containerConfig : Config = {

146     val appCfg : Config =
147       ConfigFactory.parseFile(

149         ConfigParseOptions.defaults().setAllowMissing(false)
150       ).withFallback(sysProps).withFallback(envProps).resolve()
151 
152     // we need to make sure that all keys are available in the resulting config,
153     // even if they point to null values or empty configs
154     val allKeys : List[String] = ConfigLocator.fullKeyset("", appCfg)
155 
156     val nullKeys = allKeys.filter(s => appCfg.getIsNull(s)).map(s => (s -> null)).toMap.asJava
157 
158     val emptyKeys = allKeys.filter { s =>
159       Try { appCfg.getConfig(s) } match {
160         case Success(c) => c.isEmpty()
161         case _ => false
162       }
163     }.map(s => s -> ConfigFactory.empty().root()).toMap.asJava
164 
165     val evaluated = ConfigLocator.evaluatedConfig(appCfg, this).unwrap
166 
167     log.trace(s"After reading application.conf : $evaluated")
168 
169     val resolvedCfg : Config = overlayConfig.withFallback(evaluated)
170       .withFallback(sysProps)
171       .withFallback(envProps)
172       .withFallback(ConfigFactory.parseMap(nullKeys))
173       .withFallback(ConfigFactory.parseMap(emptyKeys))
174       .resolve()
175 
176     log.debug(s"Resolved container config : $resolvedCfg")
177
