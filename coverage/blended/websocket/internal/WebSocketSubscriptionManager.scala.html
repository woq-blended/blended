29 /**
30   * Manage all subscriptions for currently connected web socket clients.
31   */
32 class WebSocketSubscriptionManager extends Actor {
33 
34   private val log : Logger = Logger[WebSocketSubscriptionManager]
35 
36   private val subKey : WebSocketSubscription => String =  { s =>
37     s"${s.context.namespace}:${s.context.name}:${s.cmd.key}:${s.token.user.getName()}"
38   }
39 
40   override def preStart(): Unit = {
41     super.preStart()
42     context.system.eventStream.subscribe(self, classOf[NewSubscription])
43     context.system.eventStream.subscribe(self, classOf[RemoveSubscription])
44     context.become(handling(Map.empty))
45     log.info(s"Started WebSocket subscription manager")
46   }
47 
48   override def receive: Receive = Actor.emptyBehavior
49 
50   private def handling( subscriptions : Map[String, ActorRef]) : Receive = {
51 
52     case NewSubscription(sub) =>
53       val key : String = subKey(sub)
54       if (!subscriptions.contains(key)) {
55         log.info(s"Creating subscription actor for [${key}]")
56         val actor : ActorRef = context.actorOf(WebSocketSubscriptionActor.props(sub))
57         context.watch(actor)
58         context.become(handling(subscriptions + (key -> actor)))
59       } else {
60         log.debug(s"Subscription [$key] is already active.")
61       }
62 
63     case RemoveSubscription(sub) =>
64       val key : String = subKey(sub)
65       subscriptions.get(key).foreach{ a =>
66         log.info(s"Removing subscription [$key]")
67         context.unwatch(a)
68         context.stop(a)
69       }
70       context.become(handling(subscriptions.filterKeys(_ != key)))
71 
72     case Terminated(a) =>
73       subscriptions.find { case (_,sa) => a == sa }.foreach { case (sk, sa) =>
74         log.warn(s"subscription actor for [$sk] terminated, removing subscription.")
75         context.become(handling(subscriptions.filterKeys(_ != sk)))
76       }
77   }

87 class WebSocketSubscriptionActor(subscription : WebSocketSubscription) extends Actor {
88

92 
93   override def preStart(): Unit = {
94     super.preStart()
95     self ! Tick
96   }
97 
98 
99   override def postStop(): Unit = {
100     log.info(s"Subscription for [${subscription.context}][${subscription.cmd}], user [${subscription.token.user}] ended.")
101     super.postStop()
102   }
103 
104   override def receive: Receive = {
105     case Tick =>
106       val user : String = subscription.token.user.getName()
107       log.debug(s"Evaluating subscription command [${subscription.cmd}] for user [$user]")
108       subscription.update.lift(subscription.cmd) match {
109         case Some(o) =>
110           o match {
111             case Success(upd) =>
112               log.debug(s"Got subscription update for user [$user]")
113               emit(
114                 msg = upd, token = subscription.token, context = subscription.context, pickler = subscription.pickler
115               )(context.system)
116             case Failure(e) =>
117               log.error(s"Error getting subscription update : [${e.getMessage()}]")
118           }
119           subscription.interval match {
120             case None => context.stop(self)
121             case Some(t) => context.system.scheduler.scheduleOnce(t, self, Tick)
122           }
123         case None =>
124           log.warn(s"Subscription function is not defined at [${subscription.cmd}] for user [$user], subscription will terminate.")
125           context.stop(self)
126       }
127   }
