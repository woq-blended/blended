27   private[this] lazy val routeImpl : Route = pathSingleSlash {
28     log.debug("Received Web Socket upgrade request")
29     parameter("token") { token =>
30       log.debug(s"Evaluating token [$token]")
31       store.verifyToken(token) match {
32         case Failure(e) =>
33           log.error(s"Could not verify token [$token] : [${e.getMessage}]")
34           complete(StatusCodes.Unauthorized)
35         case Success(verified) =>
36           log.info(s"Starting Web Socket message handler for token [${verified.id}]")
37 
38           store.getToken(verified.id) match {
39             case None =>
40               complete(StatusCodes.BadRequest)
41             case Some(info) =>
42               // A web socket handler is simply a flow Message -> Message
43               // Inbound messages will be treated as commands coming from the
44               // client, outbound message are usually WsEncoded messages that
45               // can carry a status code and a JSON encoded payload
46               handleWebSocketMessages(handlerFlow(info))
47           }
48       }
49     }
50   }
51 
52   private[this] def handlerFlow(info : Token) : Flow[Message, Message, Any] = {
53     Flow[Message]
54       // We will only process TextMessage.Strict variants for now,
55       // so we use a collect here
56       .collect {
57         case TextMessage.Strict(msg) => msg
58       }
59       // Incoming Strings are always treated as commands and processed
60       // by the command handler chain. The result for each and every
61       // command is a WsResult, which is passed back to the client here.
62       // A command *might* send additional messages to the client as a
63       // result of executing the command (i.e. when setting up a listener
64       // on Container events).
65       // The additional messages are not passed through, but are sent directly
66       // to the client as websocket messages
67       .via(cmdHandler.newClient(info))
68       .collect {
69         case enc : WsMessageEncoded =>
70           TextMessage.Strict(enc.json)
71       }
72       .via(reportErrorsFlow)
73   }
74 
75   def reportErrorsFlow[T] : Flow[T, T, Any] =
76     Flow[T]
77       .watchTermination()((_, f) => f.onComplete {
78         case Failure(cause) =>
79           log.error(cause)(s"WS stream failed : [${cause.getMessage()}]")
80         case _ => // ignore regular completion
81       })
82 }
