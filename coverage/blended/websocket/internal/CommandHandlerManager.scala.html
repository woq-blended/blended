15 trait CommandHandlerManager {
16   /**
17     * This method will be called every time a client has successfully authenticated.
18     * The token will contain all the user info including the user id and the permissions
19     * for the user. If required, the token will be passed to the command handlers so that
20     * client specific permissions can be evaluated.
21     * Essentially it will create a flow from Strings to WsUnitMessages. This flow will then
22     * try to decode the incoming command and produce a Websockets message carrying a result.
23     * @param t : The token with the client specific id and permissions.
24     */
25   def newClient(t : Token) : Flow[String, WsMessageEncoded, NotUsed]
26 
27   def addCommandPackage(pkg : WebSocketCommandPackage) : Unit
28   def removeCommandPackage(pkg : WebSocketCommandPackage) : Unit
29 }

31 object CommandHandlerManager {
32 
33   case class AddCommandPackage(handler: WebSocketCommandPackage)
34   case class RemoveCommandPackage(handler: WebSocketCommandPackage)
35   case class NewClient(t : Token, clientActor : ActorRef)
36   case class ClientClosed(t : Token)
37   case class ReceivedMessage(t: Token, s : String)
38   case class WsClientUpdate(
39     msg : WsMessageEncoded,
40     token : Token
41   )
42 
43   /**
44     * Create an empty command handler within an Actor system.
45     */
46   def create(system: ActorSystem): CommandHandlerManager = new CommandHandlerManager {
47     // This creates one actor which will dispatch all incoming client
48     // messages and dispatch them accordingly
49     private val cmdHandler : ActorRef = system.actorOf(Props[CommandHandlerActor])
50 
51     // for each new client we will create a flow which will consume Strings and emit
52     // WSMessageEnvelopes
53     override def newClient(token : Token) : Flow[String, WsMessageEncoded, NotUsed] = {
54       val in = Flow[String]
55         .map(s => ReceivedMessage(token, s))
56         .to(Sink.actorRef[ReceivedMessage](cmdHandler, ClientClosed(token)))
57 
58       // This materializes a new actor for the given client. All messages sent to this actor
59       // will be sent to the client via Web Sockets
60       // The new client will be registered with the DispatcherActor, which will then watch this
61       // actor and dispatch events to the client as long as it is active.
62       val out = Source.actorRef[WsMessageEncoded](1, OverflowStrategy.fail)
63         .mapMaterializedValue { c => cmdHandler ! NewClient(token, c) }
64 
65       Flow.fromSinkAndSourceCoupled(in, out)
66     }
67 
68     override def addCommandPackage(pkg: WebSocketCommandPackage): Unit = cmdHandler ! AddCommandPackage(pkg)
69 
70     override def removeCommandPackage(pkg: WebSocketCommandPackage): Unit = cmdHandler ! RemoveCommandPackage(pkg)
71   }
72 
73   /**
74     * The central Web sockets command handler.
75     *
76     * The central command handler will keep track of all registered specialized
77     * command handlers, so that it can dispatch any inbound WsMessage to the
78     * relevant command handler. It will also keep track of the connected clients,
79     * as the command handlers will eventually send WsMessages to a particular client
80     * via their emit method.
81     */
82   private class CommandHandlerActor extends Actor {
83 
84     private val log : Logger = Logger[CommandHandlerActor]
85 
86     // We start with no handlers and no clients
87     override def preStart(): Unit = {
88       context.become(handling(CommandHandlerState()))
89       context.system.eventStream.subscribe(self, classOf[WsClientUpdate])
90     }
91 
92     override def receive: Receive = Actor.emptyBehavior
93 
94     /**
95       * Handle (de)registration of [[BlendedCommandPackage]]s.
96       * @param state The CommandHandler State managing packages and clients.
97       */
98     private def handlePackages(state : CommandHandlerState) : Receive = {
99       // Manage Ws Command Handler
100       case AddCommandPackage(h) =>
101         context.become(handling(state.addHandler(h)))
102       case RemoveCommandPackage(h) =>
103         context.become(handling(state.removeHandler(h)))
104     }
105 
106     /**
107       * Handle client (dis)connects.
108       * @param state The CommandHandler State managing packages and clients.
109       */
110     private def handleClients(state : CommandHandlerState) : Receive = {
111       // Manage client connects / disconnects
112       case NewClient(info, clientActor) =>
113         context.watch(clientActor)
114         context.become(handling(state.addClient(ClientInfo(info, clientActor))))
115       case ClientClosed(t) =>
116         state.clients.get(t.id).foreach { ci =>
117           ci.clientActor ! actor.Status.Success(Unit)
118         }
119         context.become(handling(state.removeClient(t)))
120       case Terminated(ca) =>
121         state.clients.values.find(_.clientActor == ca).foreach { ci =>
122           context.become(handling(state.removeClient(ci.t)))
123         }
124     }
125 
126     /**
127       * Handle incoming commands.
128       *
129       * This is the main command dispatcher. It will try to decode an incoming
130       * command into a [[WsContext]] and dispatch it to the corresponding
131       * [[BlendedCommandPackage]] for further processing.
132       *
133       * Any implemented commands may use their emit method to dispatch messages
134       * to a particular client.
135       * @param state The CommandHandler State managing packages and clients.
136       */
137     private def handleCommand(state: CommandHandlerState) : Receive = {
138       case rm : ReceivedMessage =>
139         log.debug(s"Handling message [${rm.s}] for client [${rm.t.id}]")
140         Unpickle[WsMessageEncoded].fromString(rm.s) match {
141 
142           case Success(msg) =>
143             state.packageByNS(msg.context.namespace) match {
144               case Some(p) =>
145                 val response : WsContext = p.handleCommand(msg, rm.t)
146                 state.respondToClient(response, rm.t)
147 
148               case None =>
149                 val result = WsContext(
150                   namespace = msg.context.namespace,
151                   name = msg.context.name,
152                   status = StatusCodes.NotFound.intValue,
153                   statusMsg = None
154                 )
155 
156                 log.warn(result.toString())
157                 state.respondToClient(result, rm.t)
158             }
159 
160           case Failure(_) =>
161 
162             val result = WsContext(
163               namespace = "unknown",
164               name = "unknown",
165               status = StatusCodes.BadRequest.intValue,
166               statusMsg = Some(s"Message [${rm.s}] not decoded into valid web socket request")
167             )
168 
169             log.warn(result.toString())
170             state.respondToClient(result, rm.t)
171         }
172     }
173 
174     /**
175       * The actual receive method.
176       *
177       * Here we maintain the command handler state and processes incoming [[WsClientUpdate]]s.
178       * A [[WsClientUpdate]] is normally sent over the Akka event stream by using a [[WebSocketCommandHandler]].
179       * The Client Update will be dispatched to the client associated with the [[Token]] within
180       * the update.
181       *
182       * @param state The CommandHandler State managing packages and clients.
183       */
184     private def handling(state : CommandHandlerState) : Receive =
185       handlePackages(state)
186         .orElse(handleClients(state))
187         .orElse(handleCommand(state))
188         .orElse {
189           // Forward an emitted message from a command handler to the connected client
190           case u : WsClientUpdate =>
191             log.debug(s"Processing WsClientUpdate for [${u.token.id}]")
192             state.clientByToken(u.token).foreach{ ci =>
193               ci.clientActor ! u.msg
194             }
195         }
196   }
