7 /**
8   * WebSocket clients use this message to subscribe to Jmx updates.
9   *
10   * A subscription will always receive the list of registered object
11   * names and changes to that list. If the optional parameter objName
12   * is set, the objName the behavior is as follows:
13   *
14   * If an MBean instance with the given object name is found, the update
15   * will contain updates only for this instance.
16   *
17   * If no MBean instance can be found for the given name, the name will
18   * be used as a pattern to search for MBeans and all direct children
19   * of the given name will be included in the update.
20   *
21   * A direct child in that sense is an MBean that extends the attribute
22   * of the given name by exactly one attribute.
23   *
24   * Subsequent subscribe messages will replace an existing subscription.
25   *
26   * The command handler will emit a subscription update as specified
27   * in the interval. If the interval is <= 0, the command handler
28   * will only emit one update and then remove the subscription.
29   *
30   * @param objName An optional object name to restrict the subscription
31   */
32 case class JmxSubscribe(
33   objName: Option[JmxObjectName],
34   intervalMS: Long,
35 ) extends BlendedJmxMessage {
36   override def key: String = objName.map(_.objectName).getOrElse("None")
37 }
38 
39 /**
40   * WebSocket clients will use this message to cancel their Jmx subscription.
41   */
42 case class JmxUnsubscribe(sub : JmxSubscribe) extends BlendedJmxMessage {
43   override def key: String = sub.key
44 }
