14 /**
15   * A web socket command handler.
16   *
17   * A connected web sockets client will have authenticated itself and will have a [[Token]]
18   * describing the permissions for that particular user. A command handler will receive messages
19   * in the form of WsData objects and will have to decode the command and act accordingly.
20   * It will respond with another WsData object that will carry an encoded response for the
21   * client that has made the request.
22   *
23   * A command handler is built over an arbitrary type T, which is the implementation of the
24   * command payload and response.
25   */
26 trait WebSocketCommandHandler[T] {
27 
28   /**
29      *The package this command belongs to
30     */
31   def cmdPackage : WebSocketCommandPackage
32 
33   /**
34     * The name of the handled command
35     */
36   def name : String
37 
38   /**
39     * A short description of the handled command
40     */
41   def description : String
42 
43   private val log : Logger = Logger(getClass().getName())
44 
45   /**
46     * Execute a command on behalf of a client. All permission information is contained
47     * within the token. The result of the [[handleCommand]] method will simply indicate
48     * whether the command was submitted successfully. (i.e. A command handler for the
49     * command has to be registered). If the command was submitted successfully, eventually
50     * the concrete implementation of the command handler will send one or more messages
51     * to the client who has issued the command.
52     *
53     * For example, retrieving the current version of the WebSocket protocol is a one off
54     * operation and results in a single [[String]]. On the other hand,
55     * subscribing to container events or JMX events normally results in multiple WsMessages
56     * with the event data as payload.
57     *
58     * @param cmd The command to be executed
59     * @param t The token with the security information
60     * @return A [[WsContext]] indicating the success or failure of the command
61     */
62   final def handleCommand(cmd : WsMessageEncoded, t: Token)(implicit up : Unpickler[T]): WsContext = {
63 
64     val json : String = new String(Base64.getDecoder().decode(cmd.content))
65     JsonHelper.decode[T](json) match {
66 
67       case Success(content) =>
68         log.debug(s"Command content is [$content]")
69         if (doHandleCommand.isDefinedAt(content)) {
70           doHandleCommand(content)(t)
71         } else {
72           WsContext(cmdPackage.namespace, name, StatusCodes.NotImplemented.intValue, None)
73         }
74 
75       case Failure(ex) =>
76         WsContext(
77           cmdPackage.namespace, name, StatusCodes.BadRequest.intValue,
78           Some(s"Unable to decode command [${ex.getMessage()}] :")
79         )
80     }
81   }
82 
83   def doHandleCommand : PartialFunction[T,  Token => WsContext]
84 }

109 trait WebSocketCommandPackage {
110 
111   type T
112   /**
113     * The namespace the handle command belongs to
114     */
115   def namespace : String
116 
117   def commands : Seq[WebSocketCommandHandler[T]]
118

120

122 
123   final def handleCommand(
124     cmd : WsMessageEncoded,
125     t : Token
126   ): WsContext = {
127     if (cmd.context.namespace != namespace) {
128       WsContext(
129         namespace = namespace,
130         name = cmd.context.name,
131         StatusCodes.BadRequest.intValue,
132         Some(s"The given namespace [${cmd.context.namespace}] does not match [$namespace]")
133       )
134     } else {
135       commands.find(_.name == cmd.context.name) match {
136         case None =>
137           val r : WsContext = WsContext(
138             cmd.context.namespace, cmd.context.name,
139             StatusCodes.NotFound.intValue,
140             Some(s"The command [${cmd.context.namespace}:${cmd.context.name}] could not be found")
141           )
142           log.warn(r.toString)
143           r
144         case Some(c) => c.handleCommand(cmd, t)(unpickler)
145       }
146     }
147   }
