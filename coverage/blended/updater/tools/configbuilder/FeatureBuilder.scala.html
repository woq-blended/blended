99   def run(cmdline : Cmdline, debug : Boolean) : Unit = {
100     val workDir = new File(cmdline.outputDir.getOrElse("/tmp")).getAbsoluteFile()
101 
102     val file = new File(cmdline.featureFiles)
103     val config = ConfigFactory.parseFile(file, ConfigParseOptions.defaults().setAllowMissing(false)).resolve()
104 
105     val feature = FeatureConfigCompanion.read(config).get
106     Console.err.println(s"Processing feature: ${feature.name} ${feature.version}")
107 
108     val bundles = feature.bundles
109     val mvnUrls = cmdline.mavenUrl // .map { d => new File(d).getAbsoluteFile().toURI().toString() }
110     val mvnGavs = cmdline.mavenArtifacts.map {
111       case (gav, file) => MvnGav.parse(gav) -> file
112     }.collect {
113       case (Success(gav), file) => gav -> file
114     }
115 
116     val bundleFiles = bundles.map { bundle =>
117       bundle -> new File(workDir, RuntimeConfig.resolveFileName(bundle.url).get)
118     }
119 
120     bundleFiles.map {
121       case (bundle, bundleFile) =>
122 
123         if (bundleFile.exists()) {
124           val digest = RuntimeConfigCompanion.digestFile(bundleFile)
125           if (bundle.sha1Sum.isDefined && bundle.sha1Sum != digest) {
126             if (debug) {
127               Console.err.println(s"Bundle file [$bundleFile] has invalid checksum: [$digest], expected: [${bundle.sha1Sum}]")
128             }
129             if (cmdline.discardInvalid) {
130               if (debug) {
131                 Console.err.println(s"Deleting bundle file [$bundleFile]")
132               }
133               bundleFile.delete()
134             }
135           }
136         }
137 
138         if (!bundleFile.exists() && cmdline.downloadMissing) {
139           // lookup in GAV
140           val directUrl = MvnGavSupport.downloadUrls(mvnGavs, bundle.artifact, debug)
141 
142           val urls =
143             if (directUrl.isDefined) directUrl.toSeq
144             else mvnUrls.map(url => RuntimeConfig.resolveBundleUrl(bundle.url, Option(url)).get)
145           urls.find { url =>
146             Console.err.println(s"Downloading ${bundleFile.getName()} from [$url]")
147             RuntimeConfigCompanion.download(url, bundleFile).isSuccess
148           } getOrElse {
149             val msg = s"Could not download [${bundleFile.getName()}] from: [$urls]"
150             Console.err.println(msg)
151             sys.error(msg)
152           }
153         }
154     }
155 
156     val newFeature = if (cmdline.updateChecksums) {
157       val newBundles = bundleFiles.map {
158         case (bundle, bundleFile) =>
159           if (bundleFile.exists()) {
160             val digest = RuntimeConfigCompanion.digestFile(bundleFile)
161             bundle.copy(artifact = bundle.artifact.copy(sha1Sum = digest))
162           } else {
163             val msg = s"Cannot update checksum of missing bundle file: [${bundleFile.getName()}]"
164             Console.err.println(msg)
165             sys.error(msg)
166           }
167       }
168       // config.withValue("bundles", ConfigValueFactory.fromIterable(newBundles.map(c => BundleConfig.toConfig(c).root().unwrapped()).asJava))
169       feature.copy(bundles = newBundles)
170     } else feature
171 
172     if (cmdline.outputFile.isDefined) {
173       ConfigWriter.write(FeatureConfigCompanion.toConfig(newFeature), new File(cmdline.outputFile.get), None)
174     } else {
175       val ps = new PrintStream(Console.out)
176       ConfigWriter.write(FeatureConfigCompanion.toConfig(newFeature), ps, None)
177     }
178   }
