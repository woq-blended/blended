123   def run(
124     args : Array[String],
125     debugLog : Option[String => Unit] = None,
126     infoLog : String => Unit = Console.out.println,
127     errorLog : String => Unit = Console.err.println
128   ) : Unit = {
129 
130     val options : CmdOptions = new CmdOptions()
131 
132     val cp = new CmdlineParser(options)
133     cp.parse(args : _*)
134     if (options.help) {
135       cp.usage()
136       return
137     }
138 
139     val debug : String => Unit = debugLog match {
140       case Some(dl) => dl
141       case None =>
142         if (options.debug) msg => Console.err.println(msg)
143         else msg => {}
144     }
145 
146     debug(s"RuntimeConfigBuilder: ${args.mkString(" ")}")
147 
148     if (options.configFile.isEmpty()) {
149       sys.error("No config file given")
150     }
151 
152     val mvnGavs : Seq[(MvnGav, String)] = options.mavenArtifacts.map {
153       case (gav, file) => MvnGav.parse(gav) -> file
154     }.collect {
155       case (Success(gav), file) => gav -> file
156     }
157 
158     // read feature repo files
159     val features = options.featureRepos.map { fileName =>
160       val featureConfig = ConfigFactory.parseFile(new File(fileName), ConfigParseOptions.defaults().setAllowMissing(false)).resolve()
161       FeatureConfigCompanion.read(featureConfig).get
162     }
163     debug("features: " + features)
164 
165     val configFile = new File(options.configFile).getAbsoluteFile()
166     val outFile = Option(options.outFile.trim())
167       .filter(!_.isEmpty())
168       .orElse(if (options.inPlace) Option(configFile.getPath()) else None)
169       .map(new File(_).getAbsoluteFile())
170 
171     val dir = outFile.flatMap(f => Option(f.getParentFile())).getOrElse(configFile.getParentFile())
172     val config = ConfigFactory
173       .parseFile(configFile, ConfigParseOptions.defaults().setAllowMissing(false))
174       .withFallback(ConfigFactory.parseMap(options.envVars.toMap.asJava))
175       .resolve()
176 
177     val unresolvedRuntimeConfig = RuntimeConfigCompanion.read(config).get
178     //    val unresolvedLocalRuntimeConfig = LocalRuntimeConfig(unresolvedRuntimeConfig, dir)
179 
180     debug("unresolved runtime config: " + unresolvedRuntimeConfig)
181     val resolved = FeatureResolver.resolve(unresolvedRuntimeConfig, features).get
182     debug("runtime config with resolved features: " + resolved)
183 
184     val localRuntimeConfig = LocalRuntimeConfig(resolved, dir)
185 
186     if (options.check) {
187       val issues = localRuntimeConfig.validate(
188         includeResourceArchives = true,
189         explodedResourceArchives = false
190       )
191       if (issues.nonEmpty) {
192         sys.error(issues.mkString("\n"))
193       }
194     }
195 
196     lazy val mvnUrls = resolved.runtimeConfig.properties.get(RuntimeConfig.Properties.MVN_REPO).toSeq ++ options.mavenUrls
197     debug(s"Maven URLs: $mvnUrls")
198 
199     def downloadUrls(b : Artifact) : Seq[String] = {
200       val directUrl = MvnGavSupport.downloadUrls(mvnGavs, b, options.debug)
201       directUrl.map(Seq(_)).getOrElse(mvnUrls.flatMap(baseUrl => RuntimeConfig.resolveBundleUrl(b.url, Option(baseUrl)).toOption).to[Seq])
202     }
203 
204     if (options.downloadMissing) {
205 
206       val files = resolved.allBundles.distinct.map { b =>
207         RuntimeConfigCompanion.bundleLocation(b, dir) -> downloadUrls(b.artifact)
208       } ++ resolved.runtimeConfig.resources.map(r =>
209         RuntimeConfigCompanion.resourceArchiveLocation(r, dir) -> downloadUrls(r))
210 
211       val states = files.par.map {
212         case (file, urls) =>
213           if (!file.exists()) {
214             infoLog(s"Fetching: [$file]")
215             urls.find { url =>
216               debug(s"Downloading [${file.getName()}] from [$url]")
217               RuntimeConfigCompanion.download(url, file).isSuccess
218             }.map { url => file -> Try(file)
219             }.getOrElse {
220               val msg = s"Could not download [${file.getName()}] from: $urls"
221               errorLog(msg)
222               sys.error(msg)
223             }
224           } else file -> Try(file)
225       }.seq
226 
227       val issues = states.collect {
228         case (file, Failure(e)) =>
229           errorLog(s"Could not download: $file (${e.getClass.getSimpleName()}: ${e.getMessage()})")
230           e
231       }
232       if (issues.nonEmpty) {
233         sys.error(issues.mkString("\n"))
234       }
235     }
236 
237     val newRuntimeConfig = if (options.updateChecksums) {
238       var checkedFiles : Map[File, String] = Map()
239 
240       def checkAndUpdate(file : File, r : Artifact) : Artifact = {
241         checkedFiles.get(file).orElse(RuntimeConfigCompanion.digestFile(file)).map { checksum =>
242           checkedFiles += file -> checksum
243           if (r.sha1Sum != Option(checksum)) {
244             infoLog(s"${if (r.sha1Sum.isDefined) "Updating" else "Creating"} checksum for: ${r.fileName.getOrElse(RuntimeConfig.resolveFileName(r.url).get)}")
245             r.copy(sha1Sum = Option(checksum))
246           } else r
247         }.getOrElse(r)
248       }
249 
250       def checkAndUpdateResource(a : Artifact) : Artifact =
251         checkAndUpdate(localRuntimeConfig.resourceArchiveLocation(a), a)
252 
253       def checkAndUpdateBundle(b : BundleConfig) : BundleConfig =
254         b.copy(artifact = checkAndUpdate(localRuntimeConfig.bundleLocation(b), b.artifact))
255 
256       def checkAndUpdateFeatures(f : FeatureConfig) : FeatureConfig =
257         f.copy(bundles = f.bundles.map(checkAndUpdateBundle))
258 
259       ResolvedRuntimeConfig(
260         resolved.runtimeConfig.copy(
261           bundles = resolved.runtimeConfig.bundles.map(checkAndUpdateBundle),
262           resolvedFeatures = resolved.allReferencedFeatures.map(checkAndUpdateFeatures),
263           resources = resolved.runtimeConfig.resources.map(checkAndUpdateResource)
264         )
265       ).runtimeConfig
266     } else {
267       resolved.runtimeConfig
268     }
269 
270     if (options.explodeResources) {
271       newRuntimeConfig.resources.map { r =>
272         val resourceFile = localRuntimeConfig.resourceArchiveLocation(r)
273         if (!resourceFile.exists()) sys.error("Could not unpack missing resource file: " + resourceFile)
274         val blacklist = List("profile.conf", "bundles", "resources", "overlays")
275         Unzipper.unzip(resourceFile, localRuntimeConfig.baseDir, Nil,
276           fileSelector = Some { fileName : String => !blacklist.contains(fileName) },
277           placeholderReplacer = None) match {
278             case Failure(e) => throw new RuntimeException("Could not update resource file: " + resourceFile, e)
279             case _          =>
280           }
281         localRuntimeConfig.createResourceArchiveTouchFile(r, r.sha1Sum) match {
282           case Failure(e) =>
283             throw new RuntimeException("Could not create resource archive touch file for resource file: " + resourceFile, e)
284           case _ =>
285         }
286       }
287     }
288 
289     // read given overlays configs, e.g. A-1 and B-2
290     val overlayConfigs = options.overlayFiles.map { f =>
291       val config = ConfigFactory.parseFile(new File(f))
292       OverlayConfigCompanion.read(config).get
293     }.toSet
294 
295     if (options.writeOverlaysConfig) {
296 
297       // validate configs, e.g. no conflicts
298       val localOverlays = LocalOverlays(overlayConfigs, localRuntimeConfig.baseDir)
299       debug("About to validate overlays: " + localOverlays)
300       val validationResult = localOverlays.validate()
301       debug("overlay validation result: " + validationResult)
302       validationResult match {
303         case Nil => // ok
304         case errors =>
305           sys.error("Inconsistent overlays given:\n- " + errors.mkString("\n- "))
306       }
307 
308       // materialize config generators of overlay configs, e.g. <profileDir>/A-1/B-2/container/application_overlay.conf
309       localOverlays.materialize().get
310 
311       // write resulting overlay config, e.g. <profileDir>/overlays/A-1-B-2.conf (or base.conf)
312       val overlayFile = LocalOverlays.preferredConfigFile(localOverlays.overlayRefs, localOverlays.profileDir)
313       ConfigWriter.write(LocalOverlays.toConfig(localOverlays), overlayFile, None)
314     }
315 
316     if (Option(options.createLaunchConfigFile).isDefined) {
317       val profileBaseDir = options.profileBaseDir
318       val profileLookup = ProfileLookup(
319         profileName = localRuntimeConfig.runtimeConfig.name,
320         profileVersion = localRuntimeConfig.runtimeConfig.version,
321         profileBaseDir = new File("REPLACE_BASE_DIR"),
322         overlays = overlayConfigs.map(_.overlayRef)
323       )
324       val file = new File(options.createLaunchConfigFile)
325       debug("Writing launch config file: " + file)
326       val os = new ByteArrayOutputStream()
327       ConfigWriter.write(ProfileLookup.toConfig(profileLookup), os, None)
328       // FIXME: the following is magic to support env variable in output, but it will work only if th file string has no spaces in it
329       val confOutput = Pattern.
330         compile("[\"]REPLACE_BASE_DIR[\"]").
331         matcher(os.toString()).
332         replaceAll(Matcher.quoteReplacement(profileBaseDir))
333       val ps = new PrintStream(new BufferedOutputStream(new FileOutputStream(file)))
334       try ps.println(confOutput) finally ps.close()
335     }
336 
337     outFile match {
338       case None =>
339         ConfigWriter.write(RuntimeConfigCompanion.toConfig(newRuntimeConfig), Console.out, None)
340       case Some(f) =>
341         debug("Writing config file: " + configFile)
342         ConfigWriter.write(RuntimeConfigCompanion.toConfig(newRuntimeConfig), f, None)
343     }
344 
345   }
