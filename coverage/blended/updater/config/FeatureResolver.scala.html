26           }.toOption.map { fetched =>

28               cache ++= Seq(fetched)

30             }
31           }
32       }
33     }

42   def resolve(feature : FeatureRef, context : ResolveContext) : Try[Seq[FeatureConfig]] = Try {
43     context.fetchFeature(feature) match {
44       case Some(fetchedFeature) => Seq(fetchedFeature) ++ fetchedFeature.features.flatMap(f => resolve(f, context).get)
45       case None                 => sys.error("Could not resolve feature: " + feature)
46     }
47   }
