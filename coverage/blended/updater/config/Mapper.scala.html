6 /**
7  * Mapper functions for bi-directional mapping of domain model case classes to [[java.util.Map]]'s with JVM-only types.

10 
11   def mapArtifact(a : Artifact) : java.util.Map[String, AnyRef] =
12     Map[String, AnyRef](
13       "url" -> a.url,
14       "fileName" -> a.fileName.orNull,
15       "sha1Sum" -> a.sha1Sum.orNull
16     ).asJava
17 
18   def mapBundleConfig(b : BundleConfig) : java.util.Map[String, AnyRef] =
19     Map[String, AnyRef](
20       "artifact" -> mapArtifact(b.artifact),
21       "start" -> java.lang.Boolean.valueOf(b.start),
22       "startLevel" -> b.startLevel.map(i => java.lang.Integer.valueOf(i)).orNull
23     ).asJava
24 
25   def mapFeatureRef(ref : FeatureRef) : java.util.Map[String, AnyRef] =
26     Map[String, AnyRef](
27       "name" -> ref.name,
28       "version" -> ref.version,
29       "url" -> ref.url.orNull
30     ).asJava
31 
32   def mapFeatureConfig(f : FeatureConfig) : java.util.Map[String, AnyRef] =
33     Map[String, AnyRef](
34       "name" -> f.name,
35       "version" -> f.version,
36       "url" -> f.url.orNull,
37       "features" -> f.features.map { f => mapFeatureRef(f) }.asJava,
38       "bundles" -> f.bundles.map { b => mapBundleConfig(b) }.asJava
39     ).asJava
40 
41   def mapRuntimeConfig(runtimeConfig : RuntimeConfig) : java.util.Map[String, AnyRef] =
42     Map[String, AnyRef](
43       "name" -> runtimeConfig.name,
44       "version" -> runtimeConfig.version,
45       "bundles" -> runtimeConfig.bundles.map { b => mapBundleConfig(b) }.asJava,
46       "startLevel" -> java.lang.Integer.valueOf(runtimeConfig.startLevel),
47       "defaultStartLevel" -> java.lang.Integer.valueOf(runtimeConfig.defaultStartLevel),
48       "properties" -> runtimeConfig.properties.asJava,
49       "frameworkProperties" -> runtimeConfig.frameworkProperties.asJava,
50       "systemProperties" -> runtimeConfig.systemProperties.asJava,
51       "features" -> runtimeConfig.features.map(f => mapFeatureRef(f)).asJava,
52       "resources" -> runtimeConfig.resources.map(a => mapArtifact(a)).asJava,
53       "resolvedFeatures" -> runtimeConfig.resolvedFeatures.map(f => mapFeatureConfig(f)).asJava
54     ).asJava

56   def mapRemoteContainerState(s : RemoteContainerState) : java.util.Map[String, AnyRef] =

62   def mapServiceInfo(si : ServiceInfo) : java.util.Map[String, AnyRef] =

66       "timestampMsec" -> java.lang.Long.valueOf(si.timestampMsec),
67       "lifetimeMsec" -> java.lang.Long.valueOf(si.lifetimeMsec),

71   def mapContainerInfo(ci : ContainerInfo) : java.util.Map[String, AnyRef] =
72     Map[String, AnyRef](
73       "containerId" -> ci.containerId,
74       "properties" -> ci.properties.asJava,
75       "serviceInfos" -> ci.serviceInfos.map(si => mapServiceInfo(si)).asJava,

77       "timestampMsec" -> java.lang.Long.valueOf(ci.timestampMsec)

80   def mapUpdateAction(a : UpdateAction) : java.util.Map[String, AnyRef] = a match {
81     case AddRuntimeConfig(id, rc) =>
82       Map(
83         "kind" -> UpdateAction.KindAddRuntimeConfig,
84         "id" -> id,
85         "runtimeConfig" -> mapRuntimeConfig(rc)
86       ).asJava
87     case AddOverlayConfig(id, oc) =>
88       Map(
89         "kind" -> UpdateAction.KindAddOverlayConfig,
90         "id" -> id,
91         "overlay" -> mapOverlayConfig(oc)
92       ).asJava
93     case ActivateProfile(id, profileName, profileVersion, overlays) =>
94       Map(
95         "kind" -> UpdateAction.KindActivateProfile,
96         "id" -> id,
97         "profileName" -> profileName,
98         "profileVersion" -> profileVersion,
99         "overlays" -> overlays.map(o => mapOverlayRef(o)).asJava
100       ).asJava
101     case StageProfile(id, profileName, profileVersion, overlays) =>
102       Map(
103         "kind" -> UpdateAction.KindStageProfile,
104         "id" -> id,
105         "profileName" -> profileName,
106         "profileVersion" -> profileVersion,
107         "overlays" -> overlays.map(o => mapOverlayRef(o)).asJava
108       ).asJava
109   }
110 
111   def mapOverlayConfig(oc : OverlayConfig) : java.util.Map[String, AnyRef] =
112     Map[String, AnyRef](

114       "version" -> oc.version,
115       "generatedConfigs" -> oc.generatedConfigs.map(c => mapGeneratedConfig(c)).asJava,
116       "properties" -> oc.properties.asJava
117     ).asJava

152   def unmapRemoteContainerState(map : AnyRef) : Try[RemoteContainerState] = Try {
153     val m = map.asInstanceOf[java.util.Map[String, AnyRef]].asScala
154     RemoteContainerState(
155       containerInfo = unmapContainerInfo(m("containerInfo")).get,

160   def unmapArtifact(map : AnyRef) : Try[Artifact] = Try {
161     val m = map.asInstanceOf[java.util.Map[String, AnyRef]].asScala
162     Artifact(
163       url = m("url").asInstanceOf[String],
164       fileName = m.get("fileName").flatMap(f => Option(f.asInstanceOf[String])),
165       sha1Sum = m.get("sha1Sum").flatMap(s => Option(s.asInstanceOf[String]))
166     )

169   def unmapUpdateAction(map : AnyRef) : Try[UpdateAction] = Try {
170     val a = map.asInstanceOf[java.util.Map[String, AnyRef]].asScala
171     a("kind") match {
172       case UpdateAction.KindAddRuntimeConfig =>
173         AddRuntimeConfig(
174           id = a("id").asInstanceOf[String],
175           runtimeConfig = unmapRuntimeConfig(a("runtimeConfig")).get
176         )
177       case UpdateAction.KindAddOverlayConfig =>
178         AddOverlayConfig(
179           id = a("id").asInstanceOf[String],
180           overlay = unmapOverlayConfig(a("overlay")).get
181         )
182       case UpdateAction.KindActivateProfile =>
183         ActivateProfile(
184           id = a("id").asInstanceOf[String],
185           profileName = a("profileName").asInstanceOf[String],
186           profileVersion = a("profileVersion").asInstanceOf[String],
187           overlays = a("overlays").asInstanceOf[java.util.Collection[AnyRef]].asScala.map(o => unmapOverlayRef(o).get).toSet
188         )
189       case UpdateAction.KindStageProfile =>
190         StageProfile(
191           id = a("id").asInstanceOf[String],
192           profileName = a("profileName").asInstanceOf[String],
193           profileVersion = a("profileVersion").asInstanceOf[String],
194           overlays = a("overlays").asInstanceOf[java.util.Collection[AnyRef]].asScala.map(o => unmapOverlayRef(o).get).toSet
195         )
196       case kind => sys.error(s"Unsupported update action kind: ${kind}")
197     }
198   }
