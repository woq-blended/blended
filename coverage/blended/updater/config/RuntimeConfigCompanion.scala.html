100     Try {
101       val parentDir = file.getAbsoluteFile().getParentFile() match {
102         case null =>
103           new File(".")
104         case parent =>
105           if (!parent.exists()) {
106             parent.mkdirs()
107           }
108           parent
109       }
110 
111       val tmpFile = File.createTempFile(s".${file.getName()}", "", parentDir)
112       try {
113         val fileStream = new FileOutputStream(tmpFile)
114         val outStream = new BufferedOutputStream(fileStream)
115         try {
116 
117           val connection = new URL(url).openConnection
118           connection.setRequestProperty("User-Agent", "Blended Updater")
119           val inStream = new BufferedInputStream(connection.getInputStream())
120           try {
121             val bufferSize = 1024
122             val buffer = new Array[Byte](bufferSize)
123 
124             while (inStream.read(buffer, 0, bufferSize) match {
125               case count if count < 0 => false
126               case count => {
127                 outStream.write(buffer, 0, count)
128                 true
129               }
130             }) {}
131           } finally {
132             inStream.close()
133           }
134         } finally {
135           outStream.flush()
136           outStream.close()
137           fileStream.flush()
138           fileStream.close()
139         }
140 
141         Files.move(Paths.get(tmpFile.toURI()), Paths.get(file.toURI()),
142           StandardCopyOption.ATOMIC_MOVE);
143 
144         file
145       } catch {
146         case NonFatal(e) =>
147           if (tmpFile.exists()) {
148             tmpFile.delete()
