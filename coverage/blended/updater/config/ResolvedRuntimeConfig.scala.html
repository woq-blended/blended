3 /**
4  * Encapsulated a [RuntimeConfig] guaranteed to contain resolved [FeatureConfig]s for each contained (transitive) [FreatureRef].
5  *
6  * If there are unresolved (transitive) features, this class construction throws with a [java.lang.IllegalArgumentException].
7  *
8  * @see [FeatureResolver] for a way to automatically resolve features, e.g. from remote repositories.
9  *
10  */
11 case class ResolvedRuntimeConfig(runtimeConfig : RuntimeConfig) {
12 
13   {
14     // Check if all feature reference have a according resolved feature
15     def check(features : List[FeatureRef], depChain : List[String]) : Unit = features.foreach { f =>
16       val depName = s"${f.name}-${f.version}"
17       val newDepChain = depName :: depChain
18       require(depChain.find(_ == depName).isEmpty, s"No cycles in feature dependencies allowed, but detected: ${newDepChain.mkString(" required by ")}")
19       val feature = lookupFeature(f)
20       require(feature.isDefined, s"Contains resolved feature: ${newDepChain.mkString(" required by ")}. Known resolved features are: ${runtimeConfig.resolvedFeatures.map(f => s"${f.name}-${f.version}").distinct.mkString(",")}")
21       check(feature.get.features, newDepChain)
22     }
23     check(runtimeConfig.features, s"${runtimeConfig.name}-${runtimeConfig.version}" :: Nil)
24 
25     // force evaluation of framework, which throws if invalid
26     // framework
27 
28     // check, that features do not conflict
29     var seen = Set[(String, String)]()
30     val conflicts = runtimeConfig.features.flatMap { f =>
31       val key = f.name -> f.version
32       if (seen.contains(key)) Some(s"${f.name}-${f.version}")
33       else {
34         seen += key
35         None
36       }
37     }
38     require(conflicts.isEmpty, s"Contains no conflicting resolved features. Multiple features detected: ${conflicts.mkString(", ")}.")
39 
40   }
41 
42   def lookupFeature(featureRef : FeatureRef) : Option[FeatureConfig] = {
43     (runtimeConfig.resolvedFeatures).find(f => f.name == featureRef.name && f.version == featureRef.version)
44   }
45 
46   /**
47    * All referenced features.
48    */
49   def allReferencedFeatures : List[FeatureConfig] = {
50     def find(features : List[FeatureRef]) : List[FeatureConfig] = features.flatMap { f =>
51       val feature = lookupFeature(f).get
52       feature +: find(feature.features)
53     }
54     find(runtimeConfig.features).distinct
55   }
56 
57   /**
58    * All bundles of this runtime config including those trasitively defined in the features.
59    */
60   def allBundles : List[BundleConfig] = (runtimeConfig.bundles ++ allReferencedFeatures.flatMap(_.bundles)).distinct
61 
62   val framework : BundleConfig = {
63     val fs = allBundles.filter(b => b.startLevel == Some(0))
64     require(fs.distinct.size == 1, s"A ResolvedRuntimeConfig needs exactly one bundle with startLevel '0', but this one has (distinct): ${fs.size}${
65       if (fs.isEmpty) "" else fs.mkString("\n  ", "\n  ", "")
66     }")
67     fs.head
68   }
69 }
