54     val profileBaseDir = new File(config.getString("profile.baseDir"))
55     val overlays =
56       if (config.hasPath("profile.overlays"))
57         config.getStringList("profile.overlays").asScala.map { o =>
58           o.split("[:]", 2) match {
59             case Array(n, v) => OverlayRef(n, v)
60             case invalid     => sys.error("Invalid overlay id: " + invalid)
61           }
62         }
63       else Nil
64 
65     ProfileLookup(
66       profileName = profileName,
67       profileVersion = profileVersion,
68       profileBaseDir = profileBaseDir,
69       overlays = overlays.toSet
70     )
71   }
