41   def validate(
42     includeResourceArchives : Boolean,
43     explodedResourceArchives : Boolean
44   ) : Seq[String] = {
45 
46     val artifacts = resolvedRuntimeConfig.allBundles.map(b => bundleLocation(b) -> b.artifact) ++
47       (if (includeResourceArchives) runtimeConfig.resources.map(r => resourceArchiveLocation(r) -> r) else Seq())
48 
49     val artifactIssues = {
50       var checkedFiles : Map[File, String] = Map()
51       artifacts.par.flatMap {
52         case (file, artifact) =>
53           val issue = if (!file.exists()) {
54             Some(s"Missing file: ${file.getName()}")
55           } else {
56             checkedFiles.get(file).orElse(RuntimeConfigCompanion.digestFile(file)) match {
57               case Some(d) =>
58                 checkedFiles += file -> d
59                 if (Option(d) != artifact.sha1Sum) {
60                   Some(s"Invalid checksum of bundle jar: ${file.getName()}")
61                 } else None
62               case None =>
63                 Some(s"Could not evaluate checksum of bundle jar: ${file.getName()}")
64             }
65           }
66           issue.toList
67       }.seq
68     }
69 
70     val resourceIssues = if (explodedResourceArchives) {
71       runtimeConfig.resources.flatMap { artifact =>
72         val touchFile = resourceArchiveTouchFileLocation(artifact)
73         if (touchFile.exists()) {
74           val persistedChecksum = Source.fromFile(touchFile).getLines().mkString("\n")
75           if (artifact.sha1Sum.isDefined && persistedChecksum != artifact.sha1Sum.get) {
76             List(s"Resource ${artifact.fileName.getOrElse(runtimeConfig.resolveFileName(artifact.url).get)} was unpacked from an archive with a different checksum (${persistedChecksum}).")
77           } else Nil
78         } else {
79           List(s"Resource ${artifact.fileName.getOrElse(runtimeConfig.resolveFileName(artifact.url).get)} not unpacked")
80         }
81       }
82     } else Nil
83 
84     artifactIssues ++ resourceIssues
85   }
