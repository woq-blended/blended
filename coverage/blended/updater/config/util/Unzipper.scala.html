35   /**
36    * Extract files from a ZIP archive.
37    * If the list of `selectedFiles` is empty and no `fileSelector` was given, than all files will be extracted.
38    *
39    * @param archive The file denoting a ZIP archive.
40    * @param targetDir The base directory, where the extracted files will be stored.
41    * @param selectedFiles A list of name-file pairs denoting which archive content should be extracted into which file.
42    *   The name of the path inside the archive.
43    *   The file will be the place that file will be extracted to.
44    *   If the file value is `null`, that the file will be extracted into the `targetDir` without any sub directory created.
45    * @param fileSelector A filter used to decide if a file in the archive should be extracted or not.
46    *   `fileSelector` is not able to exclude files already selected with `selectedFiles`.
47    *   If a selector is given (`[scala.Some]`), than only those files will be extracted, for which the selector returns `true`.
48    *
49    * @return A `Seq` of all extracted files.
50    */
51   def unzip(
52     archive : File,
53     targetDir : File,
54     selectedFiles : List[(String, File)],
55     fileSelector : Option[String => Boolean],
56     placeholderReplacer : Option[PlaceholderConfig]
57   ) : Try[Seq[File]] = {
58     if (!archive.exists() || !archive.isFile()) throw new FileNotFoundException(s"Zip file cannot be found: ${archive}")
59     targetDir.mkdirs
60     val is = new FileInputStream(archive)
61     try {
62       unzip(is, targetDir, selectedFiles, fileSelector, placeholderReplacer, Some(archive.getPath()))
63     } finally {
64       is.close()
65     }
66   }

68   /**
69    * Extract files from a ZIP archive.
70    * If the list of `selectedFiles` is empty and no `fileSelector` was given, than all files will be extracted.
71    *
72    * @param inputStream The stream providing a ZIP archive.
73    * @param targetDir The base directory, where the extracted files will be stored.
74    * @param selectedFiles A list of name-file pairs denoting which archive content should be extracted into which file.
75    *   The name of the path inside the archive.
76    *   The file will be the place that file will be extracted to.
77    *   If the file value is `null`, that the file will be extracted into the `targetDir` without any sub directory created.
78    * @param fileSelector A filter used to decide if a file in the archive should be extracted or not.
79    *   `fileSelector` is not able to exclude files already selected with `selectedFiles`.
80    *   If a selector is given (`[scala.Some]`), than only those files will be extracted, for which the selector returns `true`.
81    * @param archive The optional target path of the archive.
82    *
83    * @return A `Seq` of all extracted files.
84    */
85   def unzip(
86     inputStream : InputStream,
87     targetDir : File,
88     selectedFiles : List[(String, File)],
89     fileSelector : Option[String => Boolean],
90     placeholderReplacer : Option[PlaceholderConfig],
91     archive : Option[String]
92   ) : Try[Seq[File]] = Try {
93 
94     log.debug(s"Extracting zip archive ${archive.getOrElse("")} to ${targetDir}")
95 
96     val partial = !selectedFiles.isEmpty || fileSelector.isDefined
97     if (partial) log.debug("Only extracting some content of zip file")
98 
99     val fileWriter : (InputStream, OutputStream) => Unit = placeholderReplacer match {
100       case None => StreamCopy.copy _
101       case Some(PlaceholderConfig(openSeq, closeSeq, escapeChar, props, failOnMissing)) =>
102         val pp = new PlaceholderProcessor(props, openSeq, closeSeq, escapeChar, failOnMissing)
103         (in, out) => pp.process(in, out).get
104     }
105 
106     def findName(name : String) : String = {
107       val index = name.lastIndexOf("/")
108       if (index < 0) name else name.substring(index)
109     }
110 
111     /**
112      * Test if a file is accepted and calculate the output file name
113      */
114     def acceptFile(file : String) : Option[File] = {
115       if (!partial) {
116         // unpack all
117         Some(new File(targetDir, file))
118       } else {
119         // select
120         val selectedFile = selectedFiles.find {
121           case (name, target) => file == name
122         } map {
123           case (name, target) =>
124             Option(target).getOrElse(new File(targetDir, findName(name)))
125         }
126 
127         selectedFile.orElse {
128           fileSelector match {
129             case None => None
130             case Some(s) =>
131               val select = s.apply(file)
132               if (select) Some(new File(targetDir, file)) else None
133           }
134         }
135       }
136     }
137 
138     val filesToExtract = selectedFiles
139     var extractedFilesInv : List[File] = Nil
140 
141     try {
142       val zipIs = new ZipInputStream(inputStream)
143       var zipEntry = zipIs.getNextEntry()
144       val finished = partial && fileSelector.isEmpty && filesToExtract.isEmpty
145       while (zipEntry != null && !finished) {
146         val extractFile : Option[File] = if (partial) {
147           if (zipEntry.isDirectory) {
148             acceptFile(zipEntry.getName).foreach { dir =>
149               log.debug(s"  Creating ${dir.getName()}")
150               dir.mkdirs()
151             }
152             None
153           } else {
154             acceptFile(zipEntry.getName)
155           }
156         } else {
157           if (zipEntry.isDirectory) {
158             log.debug(s"  Creating ${zipEntry.getName()}")
159             new File(targetDir + "/" + zipEntry.getName).mkdirs()
160             None
161           } else {
162             Some(new File(targetDir + "/" + zipEntry.getName))
163           }
164         }
165 
166         if (extractFile.isDefined) {
167           log.debug(s"  Extracting ${zipEntry.getName()}")
168           val targetFile = extractFile.get
169           if (targetFile.exists
170             && !targetFile.getParentFile().isDirectory()) {
171             throw new RuntimeException(
172               "Expected directory is a file. Cannot extract zip content: "
173                 + zipEntry.getName()
174             )
175           }
176           // Ensure, that the directory exists
177           targetFile.getParentFile().mkdirs()
178           val outputStream = new BufferedOutputStream(new FileOutputStream(targetFile))
179           try {
180             fileWriter(zipIs, outputStream)
181           } finally {
182             outputStream.close
183           }
184           extractedFilesInv ::= targetFile
185           if (zipEntry.getTime > 0) {
186             targetFile.setLastModified(zipEntry.getTime)
187           }
188         }
189 
190         zipEntry = zipIs.getNextEntry()
191       }
192 
193     } catch {
194       case NonFatal(e) =>
195         throw new RuntimeException(s"Could not unzip file: ${archive}", e)
196     }
197 
198     if (!filesToExtract.isEmpty) {
199       throw new FileNotFoundException(s"""Could not found file "${filesToExtract.head._1}" in zip archive "${archive}".""")
200     }
201 
202     extractedFilesInv.reverse
203   }
