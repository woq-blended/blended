13 /**
14  * A materialized set of overlays.
15  * The overlays are materialized to the given `profileDir` directory.
16  *
17  * @param overlays   Alls involved overlay config.
18  * @param profileDir The profile directory.
19  */
20 final case class LocalOverlays(overlays : Set[OverlayConfig], profileDir : File) {
21 
22   def overlayRefs : Set[OverlayRef] = overlays.map(_.overlayRef).toSet
23 
24   /**
25    * Validate this set of overlays.
26    * Validation checks for collisions of config names and config settings.
27    *
28    * @return A collection of validation errors, if any.
29    *         If this is empty, the validation was successful.
30    */
31   def validate() : List[String] = {
32     val nameIssues = overlays.groupBy(_.name).collect {
33       case (name, configs) if configs.size > 1 => s"More than one overlay with name '$name' detected"
34     }.toList
35     var seenPropsAndConfigNames = Set[(String, String)]()
36     val propIssues = overlays.toSeq.flatMap { o =>
37       val occ = s"${o.name}-${o.version}"
38       o.properties.keySet.toSeq.flatMap { p =>
39         seenPropsAndConfigNames.find(_._1 == p) match {
40           case Some(first) =>
41             val showOcc = List(first._2, occ).sorted.mkString(", ")
42             List(s"Duplicate property definitions detected. Property: ${p} Occurences: ${showOcc}")
43           case None =>
44             seenPropsAndConfigNames += p -> occ
45             List()
46         }
47       }
48     }
49     val generatorIssues = OverlayConfigCompanion.findCollisions(overlays.toList.flatMap(_.generatedConfigs))
50     nameIssues ++ propIssues ++ generatorIssues
51   }
52 
53   /**
54    * The location of the final applied set of overlays.
55    */
56   def materializedDir : File = LocalOverlays.materializedDir(overlays.map(_.overlayRef), profileDir)
57 
58   /**
59    * Materializes the given local overlays.
60    *
61    * @return The `Success` with the materialized config files or `Failure` in case of any unrecoverable error.
62    */
63   def materialize() : Try[immutable.Seq[File]] = Try {
64     val dir = materializedDir
65     OverlayConfigCompanion.aggregateGeneratedConfigs2(overlays.flatMap(_.generatedConfigs)) match {
66       case Left(issues) =>
67         sys.error("Cannot materialize invalid or inconsistent overlays. Issues: " + issues.mkString(";"))
68       case Right(configByFile) =>
69         configByFile.map {
70           case (fileName, config) =>
71             val file = new File(dir, fileName)
72             file.getParentFile().mkdirs()
73             ConfigWriter.write(config, file, None)
74             file
75         }.toList
76     }
77   }
78 
79   /**
80    * The files that would be generated
81    */
82   def materializedFiles() : Try[immutable.Seq[File]] = Try {
83     val dir = materializedDir
84     OverlayConfigCompanion.aggregateGeneratedConfigs2(overlays.flatMap(_.generatedConfigs)) match {
85       case Left(issues) =>
86         sys.error("Cannot materialize invalid or inconsistent overlays. Issues: " + issues.mkString(";"))
87       case Right(configByFile) =>
88         configByFile.map {
89           case (fileName, config) => new File(dir, fileName)
90         }.toList
91     }
92   }
93 
94   def isMaterialized() : Boolean = {
95     materializedFiles() match {
96       case Success(files) => files.forall { f => f.exists() && f.isFile() }
97       case _              => false
98     }
99   }
100 
101   /**
102    * Return the aggregated properties of the overlay configs.
103    * You need to ensure there are no conflicting properties with `validate`,
104    * otherwise some properties might be override and thus lost.
105    *
106    * @return The properties map.
107    */
108   def properties : Map[String, String] = {
109     overlays.toSeq.sorted.flatMap(o => o.properties).toMap
110   }
111 
112 }

118   def materializedDir(overlays : Iterable[OverlayRef], profileDir : File) : File = {
119     if (overlays.isEmpty) {
120       profileDir
121     } else {
122       val overlayParts = overlays.toList.map(o => s"${o.name}-${o.version}").distinct.sorted
123       new File(profileDir, overlayParts.mkString("/"))
124     }
125   }
126 
127   def preferredConfigFile(overlays : Iterable[OverlayRef], profileDir : File) : File = {
128     val overlayPart =
129       if (overlays.isEmpty) "base"
130       else overlays.toList.map(o => s"${o.name}-${o.version}").distinct.sorted.mkString("_")
131 
132     new File(profileDir, s"overlays/${overlayPart}.conf")
133   }
134 
135   def read(config : Config, profileDir : File) : Try[LocalOverlays] = Try {
136     LocalOverlays(
137       profileDir = profileDir,
138       overlays = config.getObjectList("overlays").asScala.map { c =>
139         OverlayConfigCompanion.read(c.toConfig()).get
140       }.toSet
141     )
142   }
143 
144   def toConfig(localOverlays : LocalOverlays) : Config = {
145     val configs = localOverlays.overlays.toList.sorted.map { o =>
146       val config = OverlayConfigCompanion.toConfig(o)
147       config.root()
148     }
149     val configs2 = ConfigValueFactory.fromIterable(configs.asJava)
150     ConfigFactory.empty().withValue("overlays", configs2)
151   }
152 
153   /**
154    * Find all local overlays installed under the given `profileDir`.
155    */
156   def findLocalOverlays(profileDir : File) : List[LocalOverlays] = {
157     val overlaysDir = new File(profileDir, "overlays")
158     val candidates = Option(overlaysDir.listFiles()).getOrElse(Array()).filter(f => f.isFile() && f.getName().endsWith(".conf"))
159     log.debug(s"About to parse found files and find valid local overlays: ${candidates.mkString(", ")}")
160     val localOverlays = candidates.toList.flatMap { file =>
161       val overlay = Try(ConfigFactory.parseFile(file)).flatMap(c => LocalOverlays.read(c, profileDir))
162       overlay match {
163         case Success(localOverlays) =>
164           log.debug("Found local overlays: " + localOverlays)
165           List(localOverlays)
166         case Failure(e) =>
167           log.error(e)(s"Could not read overlay config file: ${file}")
168           List()
169       }
170     }
171     localOverlays
172   }
