24       filter(b => b.startLevel != Some(0)).

27           location = s"${bundlePrefix}/${bc.jarName.getOrElse(runtimeConfig.resolveFileName(bc.url).get)}",

36     LauncherConfig(
37       frameworkJar = s"${bundlePrefix}/${resolvedRuntimeConfig.framework.jarName.getOrElse(runtimeConfig.resolveFileName(resolvedRuntimeConfig.framework.url).get)}",
38       systemProperties = runtimeConfig.systemProperties,
39       frameworkProperties = runtimeConfig.frameworkProperties,
40       startLevel = runtimeConfig.startLevel,
41       defaultStartLevel = runtimeConfig.defaultStartLevel,
42       bundles = allBundles,
43       branding = runtimeConfig.properties ++ Map(

45         RuntimeConfig.Properties.PROFILE_VERSION -> runtimeConfig.version

50   def launcherConfigToRuntimeConfig(launcherConfig : LauncherConfig, missingPlaceholder : String) : RuntimeConfig = {
51     import RuntimeConfigCompanion._
52 
53     RuntimeConfig(
54       name = launcherConfig.branding.getOrElse(RuntimeConfig.Properties.PROFILE_NAME, missingPlaceholder),
55       version = launcherConfig.branding.getOrElse(RuntimeConfig.Properties.PROFILE_VERSION, missingPlaceholder),
56       startLevel = launcherConfig.startLevel,
57       defaultStartLevel = launcherConfig.defaultStartLevel,
58       frameworkProperties = launcherConfig.frameworkProperties,
59       systemProperties = launcherConfig.systemProperties,
60       bundles = BundleConfig(
61         url = missingPlaceholder,
62         jarName = new File(launcherConfig.frameworkJar).getName(),
63         sha1Sum = digestFile(new File(launcherConfig.frameworkJar)).orNull,
64         start = true,
65         startLevel = 0
66       ) +:
67         launcherConfig.bundles.toList.map { b =>
68           BundleConfig(
69             url = missingPlaceholder,
70             jarName = new File(b.location).getName(),
71             sha1Sum = digestFile(new File(b.location)).orNull,
72             start = b.start,
73             startLevel = b.startLevel
74           )
75         },
76       properties = launcherConfig.branding
77     )
78   }
