28   private def aggregateGeneratedConfigs(generatedConfigs : Iterable[GeneratedConfig]) : Either[Seq[String], Map[String, Map[String, Object]]] = {
29     // seen configurations per target file
30     var fileToConfig : Map[String, Map[String, Object]] = Map()
31     val issues = generatedConfigs.flatMap { gc =>
32       val newConfig = GeneratedConfigCompanion.config(gc).root().unwrapped().asScala.toMap
33       fileToConfig.get(gc.configFile) match {
34         case None =>
35           // no collision
36           fileToConfig += gc.configFile -> newConfig
37           Seq()
38         case Some(existingConfig) =>
39           // check collisions
40           val collisions = existingConfig.keySet.intersect(newConfig.keySet)
41           fileToConfig += gc.configFile -> (existingConfig ++ newConfig)
42           collisions.map(c => s"Double defined config key found: ${c}")
43       }
44     }
45     if (issues.isEmpty) Right(fileToConfig) else Left(issues.toList)
46   }

48   def aggregateGeneratedConfigs2(generatedConfigs : Iterable[GeneratedConfig]) : Either[Seq[String], Map[String, Config]] = {
49     aggregateGeneratedConfigs(generatedConfigs) match {
50       case Left(issues) =>
51         // issue pass through
52         Left(issues)
53       case Right(fileToMap) =>
54         // instead of a Map we aggregate a Config via Config API (to avoid structural changes)
55         Right(
56           generatedConfigs.foldLeft(Map[String, Config]()) { (l, gc) =>
57             val config = GeneratedConfigCompanion.config(gc)
58             l.get(gc.configFile) match {
59               case None =>
60                 l ++ Map(gc.configFile -> config)
61               case Some(existingConfig) =>
62                 val mergedConfig = config.withFallback(existingConfig)
63                 l ++ Map(gc.configFile -> mergedConfig)
64             }
65           }
66         )
67     }

70   def read(config : Config) : Try[OverlayConfig] = Try {
71 
72     OverlayConfig(
73       name = config.getString("name"),
74       version = config.getString("version"),
75       generatedConfigs = if (config.hasPath("configGenerator")) {
76         config.getObjectList("configGenerator").asScala.map { gen =>
77           val genConf = gen.toConfig()
78           val fileName = genConf.getString("file")
79           val genConfig = genConf.getObject("config").toConfig()
80           GeneratedConfigCompanion.create(fileName, genConfig)
81         }.toList
82       } else Nil,
83       properties = ConfigPropertyMapConverter.getKeyAsPropertyMap(config, "properties", Some(() => Map()))
84     )
85   }

87   def toConfig(overlayConfig : OverlayConfig) : Config = {
88     val config = Map(

90       "version" -> overlayConfig.version,
91       "configGenerator" -> overlayConfig.generatedConfigs.map { genConfig =>
92         Map(
93           "file" -> genConfig.configFile,
94           "config" -> GeneratedConfigCompanion.config(genConfig).root().unwrapped()
95         ).asJava
96       }.asJava,
97       "properties" -> ConfigPropertyMapConverter.propertyMapToConfigValue(overlayConfig.properties)
98     ).asJava
99     ConfigFactory.parseMap(config)
100   }
