19   private def fileIssue(file : File, artifact : Artifact) : Option[String] = {
20     if (!file.exists()) {
21       Some(s"File does not exist: $file")
22     } else {
23       artifact.sha1Sum match {
24         case None => None
25         case Some(sha1) => RuntimeConfigCompanion.digestFile(file) match {
26           case Some(`sha1`)   => None
27           case Some(fileSha1) => Some(s"File checksum $fileSha1 does not match $sha1")
28           case None           => Some(s"Could not verify checksum of file $file")
29         }
30       }
31     }
32   }
33 
34   private def artifactUrls(artifact : Artifact) : Try[List[String]] = {
35     val url = artifact.url
36 
37     if (url.startsWith(RuntimeConfig.MvnPrefix)) {
38       log.debug(s"detected Maven url: [$url]")
39       val gav = MvnGav.parse(artifact.url.substring(RuntimeConfig.MvnPrefix.length()))
40       gav.map(gav => mvnRepositories.map(repo => gav.toUrl(repo)))
41     } else {
42       Success(List(url))
43     }
44   }
45 
46   private def downloadArtifact(d : Download) : Reply =
47     artifactUrls(d.artifact) match {
48       case Success(Nil) =>
49         log.error("No urls or no Maven repositories defined")
50         DownloadFailed(d.requestId, s"Could not download file [${d.file}]. Error: No Maven repositories defined.")
51 
52       case Success(url :: _) =>
53         RuntimeConfigCompanion.download(url, d.file) match {
54           case Success(f) =>
55             fileIssue(f, d.artifact) match {
56               case None        => DownloadFinished(d.requestId)
57               case Some(issue) => DownloadFailed(d.requestId, issue)
58             }
59           case Failure(e) =>
60             log.error(e)(s"Could not download file [${d.file}] from [$url]")
61             DownloadFailed(d.requestId, s"Could not download file [${d.file}] from [$url]. Error: ${e.getMessage()}")
62         }
63       case Failure(e) =>
64         log.error(e)(s"Could not download file [${d.file}]")
65         DownloadFailed(d.requestId, s"Could not download file [${d.file}]. Error: ${e.getMessage()}")
66     }
67 
68   def receive : Actor.Receive = LoggingReceive {
69 
70     case d @ Download(reqId, artifact, file) =>
71 
72       fileIssue(file, artifact) match {
73         case None =>
74           sender() ! DownloadFinished(reqId)
75 
76         case Some(_) =>

78       }
79   }
