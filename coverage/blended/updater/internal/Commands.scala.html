35   def showProfiles() : AnyRef = {
36     implicit val timeout = Timeout(5, SECONDS)
37     val activeProfile = env.map(env => ProfileId(env.launchedProfileName, env.launchedProfileVersion, env.overlays.getOrElse(Set.empty).toSet))
38     log.debug(s"acitive profile: ${activeProfile}")
39 
40     val profiles = Await.result(
41       ask(updater, Updater.GetProfiles(UUID.randomUUID().toString())).mapTo[Updater.Result[Set[_]]],
42       timeout.duration
43     ).result
44 
45     s"${profiles.size} profiles:\n${
46       profiles.map {
47         case p : LocalProfile =>
48           val activePart = if (activeProfile.exists(_ == p.profileId)) " (active)" else ""
49           p.profileId + ": " + p.state + activePart
50       }.mkString("\n")
51     }"
52   }

118   def parseOverlays(overlayNameVersion : Seq[String]) : List[OverlayRef] = {
119     if (overlayNameVersion.size % 2 != 0) {
120       sys.error(s"Missing version for overlay ${overlayNameVersion.last}")
121     }
122     overlayNameVersion.sliding(2, 2).map(o => OverlayRef(o(0), o(1))).toList
123   }

125   @varargs
126   def stageProfile(rcName : String, rcVersion : String, overlayNameVersion : String*) : AnyRef = {
127     val overlays = parseOverlays(overlayNameVersion)
128     val overlaysAsString =
129       if (overlays.isEmpty) ""
130       else overlays.toList.sorted.map(o => s"${o.name}-${o.version}").mkString(" with ", " with ", "")
131 
132     val asString = s"${rcName}-${rcVersion}${overlaysAsString}"
133 
134     implicit val timeout = Timeout(1, HOURS)
135     val reqId = UUID.randomUUID().toString()
136     Await.result(
137       ask(updater, Updater.StageProfile(reqId, rcName, rcVersion, overlays.toSet)), timeout.duration
138     ) match {
139         case OperationSucceeded(`reqId`) =>
140           "Staged: " + asString
141         case OperationFailed(`reqId`, reason) =>
142           "Stage failed: " + asString + "\nReason: " + reason
143         case x =>
144           "Stage failed: " + asString + "\nError: " + x
145       }
146   }

148   @varargs
149   def activateProfile(name : String, version : String, overlayNameVersion : String*) : AnyRef = {
150     val overlays = parseOverlays(overlayNameVersion)
151     val overlaysAsString =
152       if (overlays.isEmpty) ""
153       else overlays.toList.sorted.map(o => s"${o.name}-${o.version}").mkString(" with ", " with ", "")
154 
155     val asString = s"${name}-${version}${overlaysAsString}"
156 
157     env match {
158       case Some(UpdateEnv(_, _, Some(lookupFile), _, _, _)) =>
159         implicit val timeout = Timeout(5, MINUTES)
160         val reqId = UUID.randomUUID().toString()
161         Await.result(
162           ask(updater, Updater.ActivateProfile(reqId, name, version, overlays.toSet)), timeout.duration
163         ) match {
164             case OperationSucceeded(`reqId`) =>
165               "Activated: " + asString
166             case OperationFailed(`reqId`, reason) =>
167               "Activation failed: " + asString + "\nReason: " + reason
168             case x =>
169               "Activation failed: " + asString + "\nError: " + x
170           }
171       case _ =>
172         sys.error("No updateable environment detected. No profile lookup file defined.")
173     }
174   }
