75   private def readUpdateEnv() : Option[UpdateEnv] = try {
76     val props = blended.launcher.runtime.Branding.getProperties()
77     log.info("Blended Launcher detected: " + props)
78     val pName = Option(props.getProperty(RuntimeConfig.Properties.PROFILE_NAME))
79     val pVersion = Option(props.getProperty(RuntimeConfig.Properties.PROFILE_VERSION))
80     val pProfileLookupFile = Option(props.getProperty(RuntimeConfig.Properties.PROFILE_LOOKUP_FILE))
81     val pProfilesBaseDir = Option(props.getProperty(RuntimeConfig.Properties.PROFILES_BASE_DIR))
82     val pProfileDir = Option(props.getProperty(RuntimeConfig.Properties.PROFILE_DIR))
83     val overlays = Option(props.getProperty(RuntimeConfig.Properties.OVERLAYS))
84     val overlayRefs = overlays.map { o =>
85       o.split("[,]").toList.map(_.split("[:]", 2)).flatMap {
86         case Array(n, v) => Some(OverlayRef(n, v))
87         case x =>
88           log.debug("Unsupported overlay: " + x.mkString(":"))
89           None
90       }
91     }
92     Some(
93       UpdateEnv(
94         launchedProfileName = pName.get,
95         launchedProfileVersion = pVersion.get,
96         launchProfileLookupFile = pProfileLookupFile.map(f => new File(f)),
97         profilesBaseDir = new File(pProfilesBaseDir.get),
98         launchedProfileDir = pProfileDir.map(f => new File(f)),
99         overlays = overlayRefs
100       )
101     )
102   } catch {
103     case _ : NoClassDefFoundError =>
104       // could not load optional branding class
105       None
106     case _ : NoSuchElementException =>
107       // could not found some required properties
108       None
109   }

111   private def profileActivator(updateEnv : UpdateEnv) = new ProfileActivator {
112     override def apply(
113       newName : String,
114       newVersion : String,
115       newOverlays : Set[OverlayRef]
116     ) : Boolean = {
117       // TODO: Error reporting
118       updateEnv match {
119         case UpdateEnv(_, _, Some(lookupFile), _, _, _) =>
120           val config = ConfigFactory.parseFile(lookupFile).resolve()
121           ProfileLookup.read(config) match {
122             case Success(profileLookup) =>
123               val newConfig = profileLookup.copy(
124                 profileName = newName,
125                 profileVersion = newVersion,
126                 overlays = newOverlays
127               )
128               log.debug(s"About to update profile lookup file: [$lookupFile] with config: [$newConfig]")
129               ConfigWriter.write(ProfileLookup.toConfig(newConfig), lookupFile, None)
130               true
131             case Failure(_) =>
132               false
133           }
134 
135         case _ =>
136           // no lookup file
137           false
138       }
139     }
140   }
