15   def scanForOverlayConfigs(overlayBaseDir : File) : List[OverlayConfig] = {
16     log.debug(s"Scanning for overlays configs in: ${overlayBaseDir}")
17 
18     val confFiles = Option(overlayBaseDir.listFiles).getOrElse(Array()).
19       filter(f => f.isFile() && f.getName().endsWith(".conf"))
20 
21     val configs = confFiles.toList.flatMap { file =>
22       Try {
23         ConfigFactory.parseFile(file).resolve()
24       }.
25         flatMap(OverlayConfigCompanion.read) match {
26           case Success(overlayConfig) =>
27             List(overlayConfig)
28           case Failure(e) =>
29             log.error(e)(s"Could not parse overlay config file: ${file}")
30             List()
31         }
32     }
33 
34     log.debug(s"Found overlay configs: ${configs}")
35     configs
36   }

38   def scanForRuntimeConfigs(installBaseDir : File) : List[LocalRuntimeConfig] = {
39     log.debug(s"Scanning for runtime configs in ${installBaseDir}")
40 
41     val configFiles = Option(installBaseDir.listFiles).getOrElse(Array()).toList.
42       flatMap { nameDir =>
43         Option(nameDir.listFiles).getOrElse(Array()).toList.
44           flatMap { versionDir =>
45             val profileFile = new File(versionDir, "profile.conf")
46             if (profileFile.exists()) Some(profileFile)
47             else None
48           }
49       }
50 
51     log.debug(s"Found potential runtime config files: ${configFiles}")
52 
53     // read configs
54     val runtimeConfigs = configFiles.flatMap { runtimeConfigFile =>
55       Try {
56         val versionDir = runtimeConfigFile.getParentFile()
57         val version = versionDir.getName()
58         val name = versionDir.getParentFile.getName()
59 
60         val config = ConfigFactory.parseFile(runtimeConfigFile, ConfigParseOptions.defaults().setAllowMissing(false)).resolve()
61         val resolved = ResolvedRuntimeConfig(RuntimeConfigCompanion.read(config).get)
62         val local = LocalRuntimeConfig(baseDir = versionDir, resolvedRuntimeConfig = resolved)
63 
64         // consistency checks
65         if (local.runtimeConfig.name == name && local.runtimeConfig.version == version) {
66           List(local)
67         } else {
68           log.warn(s"Profile name and version do not match directory names: ${runtimeConfigFile}")
69           List()
70         }
71       }.getOrElse {
72         log.warn(s"Could not read profile file: ${runtimeConfigFile}")
73         List()
74       }
75     }
76     log.debug(s"Found runtime configs: ${runtimeConfigs.zipWithIndex.map(wi => s"${wi._2}: ${wi._1}").mkString("\n")}")
77 
78     runtimeConfigs
79   }

81   def scanForProfiles(installBaseDir : File, runtimeConfigs : Option[List[LocalRuntimeConfig]] = None) : List[LocalProfile] = {
82     log.debug(s"Scanning for profiles in: ${installBaseDir}")
83 
84     val rcs = runtimeConfigs.getOrElse(scanForRuntimeConfigs(installBaseDir)).toList
85 
86     val runtimeConfigsWithIssues = rcs.flatMap { localConfig =>
87       val issues = localConfig.validate(
88         includeResourceArchives = false,
89         explodedResourceArchives = true
90       ).toList
91       log.debug(s"Runtime config ${localConfig.runtimeConfig.name}-${localConfig.runtimeConfig.version} issues: ${issues}")
92       List(localConfig -> issues)
93 
94     }
95 
96     log.debug(s"Runtime configs (with issues): ${runtimeConfigsWithIssues}")
97 
98     def profileState(issues : List[String]) : LocalProfile.ProfileState = issues match {
99       case Seq()  => LocalProfile.Staged
100       case issues => LocalProfile.Pending(issues)
101     }
102 
103     val fullProfiles = runtimeConfigsWithIssues.flatMap {
104       case (localRuntimeConfig, issues) =>
105         val profileDir = localRuntimeConfig.baseDir
106 
107         // scan for overlays
108         val overlayDir = new File(profileDir, "overlays")
109         val overlayFiles = Option(overlayDir.listFiles()).getOrElse(Array()).filter(f => f.getName().endsWith(".conf")).toList
110         if (overlayFiles.isEmpty) {
111           log.warn(s"Could not found any overlay configs for profile: ${localRuntimeConfig.profileFileLocation}")
112           //          log.info("Migrating legacy profile. Generating base overlay config")
113           //          // We create a transient base overlay
114           // TODO: Remove timely
115           //          List(Profile(localRuntimeConfig, LocalOverlays(Set(), profileDir), profileState(issues)))
116           List()
117         } else overlayFiles.flatMap { file =>
118           Try {
119             ConfigFactory.parseFile(file, ConfigParseOptions.defaults().setAllowMissing(false)).resolve()
120           }.flatMap { c =>
121             LocalOverlays.read(c, profileDir)
122           } match {
123             case Failure(e) =>
124               log.error(e)(s"Could not load overlay config file: ${file}")
125               None
126             case Success(localOverlays) =>
127               val canonicalFile = LocalOverlays.preferredConfigFile(localOverlays.overlays.map(_.overlayRef), profileDir)
128               if (canonicalFile != file) {
129                 log.error(s"Skipping found overlays file because filename does not match the expected file name: ${file}")
130                 List()
131               } else {
132                 val overlayIssues = localOverlays.validate() match {
133                   case Seq() =>
134                     // no conflicts, now check if already materialized
135                     if (localOverlays.isMaterialized()) {
136                       List()
137                     } else {
138                       List("Overlays not materialized")
139                     }
140                   case issues =>
141                     log.error(
142                       s"Skipping found overlays file because it is not valid: ${file}. Issue: ${issues.mkString(" / ")}"
143                     )
144                     issues.toList
145                 }
146                 log.debug(s"Found overlay: ${localOverlays}")
147                 log.debug(s"Found overlay issues: ${issues}")
148                 List(LocalProfile(localRuntimeConfig, localOverlays, profileState(issues ::: overlayIssues)))
149               }
150           }
151         }
152     }
153 
154     fullProfiles
155   }
