56   private[this] def stageInProgress(state : State) : Unit = {
57     val id = state.requestId
58     val config = state.config
59     val progress = state.progress()
60     log.debug(s"Progress: ${progress} for reqestId: ${id}")
61 
62     // TODO: generate config files (overlay)
63 
64     if (state.artifactsToDownload.isEmpty && state.issues.isEmpty && !state.pendingArtifactsToUnpack.isEmpty) {
65       // start unpacking
66       state.pendingArtifactsToUnpack.foreach { a =>
67         unpacker ! Unpacker.Unpack(a.reqId, self, a.file, config.baseDir)
68       }
69       stageInProgress(state.copy(
70         artifactsToUnpack = state.pendingArtifactsToUnpack,
71         pendingArtifactsToUnpack = Nil
72       ))
73     } else if (state.artifactsToDownload.isEmpty && state.artifactsToUnpack.isEmpty) {
74 
75       stagingInProgress = stagingInProgress.filterKeys(id != _)
76       val (profileState, msg) = state.issues match {
77         case Seq()  => LocalProfile.Staged -> OperationSucceeded(id)
78         case issues => LocalProfile.Invalid(issues) -> OperationFailed(id, issues.mkString("; "))
79       }
80 
81       profiles += state.profileId -> LocalProfile(state.config, state.overlays, profileState)
82       state.requestActor ! msg
83 
84     } else {
85       stagingInProgress += id -> state
86     }
87   }

97   def findLocalResource(name : String, sha1Sum : String) : Option[File] =
98     profiles.values.filter { profile =>
99       // only valid profiles
100       profile.state == LocalProfile.Staged
101     }.
102       flatMap { profile =>
103         profile.bundles.filter { b =>
104           // same sha1sum
105           b.sha1Sum == Some(sha1Sum)
106         }.
107           filter { b =>
108             val location = profile.config.bundleLocation(b.artifact)
109             // same file name and file exists
110             location.getName() == name && location.exists()
111           }.
112           map { b =>
113             // extract location
114             profile.config.bundleLocation(b.artifact)
115           }.headOption
116       }.headOption

185   private def publishResultEvent[T](id : String, reply : Try[Any]) = {
186     reply match {
187       case Success(OperationSucceeded(_))      => eventStream.publish(UpdateActionApplied(id))
188       case Success(OperationFailed(_, reason)) => eventStream.publish(UpdateActionApplied(id, Option(reason)))
189       case Failure(e)                          => eventStream.publish(UpdateActionApplied(id, Option(e.getMessage())))
190       case x                                   => log.warn(s"Skip publish event of unsupported reply: [${x}]")
191     }

194   def handleUpdateAction(event : UpdateAction) : Unit = event match {
195     case UAAddRuntimeConfig(id, runtimeConfig) =>
196       log.debug(s"Received add runtime config request (via event stream) for ${runtimeConfig.name}-${runtimeConfig.version} with ID [${id}]")
197 
198       implicit val ec = context.system.dispatcher
199       val timeout = new Timeout(10, TimeUnit.MINUTES)
200 
201       self.ask(AddRuntimeConfig(nextId(), runtimeConfig))(timeout).onComplete { x =>
202         log.debug(s"Finished add runtime config request (via event stream) for ${runtimeConfig.name}-${runtimeConfig.version} with ID [${id}] with result: ${x}")
203         publishResultEvent(id, x)
204       }
205 
206     case UAAddOverlayConfig(id, overlayConfig) =>
207       log.debug(s"Received add overlay config request (via event stream) for ${overlayConfig.name}-${overlayConfig.version} with ID [${id}]")
208 
209       implicit val ec = context.system.dispatcher
210       val timeout = new Timeout(10, TimeUnit.MINUTES)
211 
212       self.ask(AddOverlayConfig(nextId(), overlayConfig))(timeout).onComplete { x =>
213         log.debug(s"Finished add overlay config request (via event stream) for ${overlayConfig.name}-${overlayConfig.version} with ID [${id}] with result: ${x}")
214         publishResultEvent(id, x)
215       }
216 
217     case UAStageProfile(id, name, version, overlayRefs) =>
218       log.debug(s"Received stage profile request (via event stream) for ${name}-${version} and overlays: ${overlayRefs} with ID [${id}]")
219 
220       implicit val ec = context.system.dispatcher
221       val timeout = new Timeout(10, TimeUnit.MINUTES)
222 
223       val request = StageProfile(nextId(), name, version, overlayRefs)
224       self.ask(request)(timeout).onComplete {
225         case x @ Success(OperationFailed(_, reason)) =>
226           log.error(s"Could not stage profile: ${reason}")
227           publishResultEvent(id, x)
228         case x @ Failure(e) =>
229           log.error(e)(s"Could not complete stage profile [${request}]")
230           publishResultEvent(id, x)
231         case x =>
232           log.debug(s"Finished stage profile request (via event stream) for ${name}-${version} and overlays ${overlayRefs}  with ID [${id}] with result: ${x}")
233           publishResultEvent(id, x)
234       }
235 
236     case UAActivateProfile(id, name, version, overlayRefs) =>
237       log.debug(s"Received activate profile request (via event stream) for ${name}-${version} and overlays: ${overlayRefs} with ID [${id}]")
238 
239       implicit val ec = context.system.dispatcher
240       val timeout = new Timeout(10, TimeUnit.MINUTES)
241 
242       self.ask(ActivateProfile(nextId(), name, version, overlayRefs))(timeout).onComplete { x =>
243         log.debug(s"Finished activation profile request (via event stream) for ${name}-${version} and overlays ${overlayRefs} with ID [${id}] with result: ${x}")
244         publishResultEvent(id, x)
245       }
246   }

248   protected def activateProfile(reqActor : ActorRef, msg : ActivateProfile) : Unit = {
249     val ActivateProfile(reqId, name, version, overlays) = msg
250     val profileId = ProfileId(name, version, overlays)
251     log.debug(s"Requested activate profile with id: ${profileId}")
252 
253     profiles.get(profileId) match {
254       case Some(LocalProfile(_, _, LocalProfile.Staged)) =>
255         // write config
256         log.debug(s"About to activate new profile for next startup: ${profileId}")
257         val success = profileActivator(name, version, profileId.overlays)
258         if (success) {
259           reqActor ! OperationSucceeded(reqId)
260           restartFramework()
261         } else {
262           reqActor ! OperationFailed(reqId, "Could not update next startup profile")
263         }
264       case r =>
265         log.debug(s"Could not find staged profile with id ${profileId} but: ${r}")
266         log.trace(s"All known profiles: ${profiles.keySet}")
267         reqActor ! OperationFailed(reqId, "No such staged runtime configuration found")
268     }

272   protected def stageProfile(reqActor : ActorRef, msg : StageProfile) : Unit = {
273     val StageProfile(reqId, name, version, overlayRefs) = msg
274 
275     val profileId = ProfileId(name, version, overlayRefs)
276 
277     // try to find the profile, if not found, we report a failure to the reqActor
278     val foundProfile : Option[LocalProfile] = profiles.get(profileId).orElse {
279       Try {
280         val localRuntimeConfig = runtimeConfigs.find(r => r.runtimeConfig.name == name && r.runtimeConfig.version == version) match {
281           case None     => sys.error(s"No such runtime config found: ${name}-${version}")
282           case Some(rc) => rc
283         }
284 
285         val overlays = overlayRefs.toList.map(ref => ref -> overlayConfigs.find(o => o.overlayRef == ref))
286         overlays.find(v => v._2 == None).map { v =>
287           sys.error(s"No such overlay config found: ${v._1.name}-${v._1.version}")
288         }
289 
290         val localOverlays = LocalOverlays(overlays.map(_._2.get).toSet, localRuntimeConfig.baseDir)
291 
292         LocalProfile(localRuntimeConfig, localOverlays, LocalProfile.Pending(List("Newly staged")))
293 
294       } match {
295         case Failure(e) =>
296           reqActor ! OperationFailed(reqId, e.getMessage())
297           None
298         case Success(p) => Some(p)
299       }
300     }
301 
302     // we found a profile and inspect it
303     foundProfile.foreach {
304       case profile @ LocalProfile(_, _, LocalProfile.Staged) =>
305         // already staged
306         log.debug(s"Profile [${profile.profileId}] already staged")
307         reqActor ! OperationSucceeded(reqId)
308 
309       case profile @ LocalProfile(config, localOverlays, state) =>
310         log.debug(s"About to stage profile [${profile.profileId}] with current state [$state]")
311 
312         val validErrors = localOverlays.validate()
313         if (!validErrors.isEmpty) {
314           reqActor ! OperationFailed(reqId, s"Overlay configuration contains ${validErrors.size} errors: ${validErrors.mkString("; ")}")
315         } else {
316           val overlayFiles = localOverlays.materialize()
317           val overlayFile = LocalOverlays.preferredConfigFile(localOverlays.overlayRefs, localOverlays.profileDir)
318           log.debug(s"About to save applied overlays config to [${overlayFile}]")
319           ConfigWriter.write(LocalOverlays.toConfig(localOverlays), overlayFile, None)
320 
321           if (stagingInProgress.contains(reqId)) {
322             log.error(s"Duplicate id detected. Dropping request [${msg}]")
323           } else {
324             // analyze config
325             overlayFiles match {
326               case Failure(e) =>
327                 reqActor ! OperationFailed(reqId, "Overlay configuration contains errors: " + e.getMessage())
328               case Success(generateFiles) =>
329                 val artifacts = config.resolvedRuntimeConfig.allBundles.map { b =>
330                   ArtifactInProgress(nextId(), b.artifact, config.bundleLocation(b))
331                 } ++
332                   config.runtimeConfig.resources.map { r =>
333                     ArtifactInProgress(nextId(), r, config.resourceArchiveLocation(r))
334                   }
335 
336                 val pendingUnpacks = config.runtimeConfig.resources.map { r =>
337                   ArtifactInProgress(nextId(), r, config.resourceArchiveLocation(r))
338                 }
339 
340                 artifacts.foreach { a =>
341                   artifactDownloader ! ArtifactDownloader.Download(a.reqId, a.artifact, a.file)
342                 }
343 
344                 stageInProgress(State(
345                   requestId = reqId,
346                   requestActor = reqActor,
347                   config = config,
348                   artifactsToDownload = artifacts,
349                   pendingArtifactsToUnpack = pendingUnpacks,
350                   artifactsToUnpack = List.empty,
351                   overlays = localOverlays,
352                   issues = List.empty
353                 ))
354 
355             }
356           }
357         }
358     }

362   def handleProtocol(msg : Protocol) : Unit = msg match {
363 
364     case GetOverlays(reqId) =>
365       sender() ! Result(reqId, overlayConfigs)
366 
367     case GetRuntimeConfigs(reqId) =>
368       sender() ! Result(reqId, runtimeConfigs)
369 
370     case GetProfiles(reqId) =>
371       sender() ! Result(reqId, profiles.values.toSet)
372 
373     case GetProfileIds(reqId) =>
374       sender() ! Result(reqId, profiles.keySet)
375 
376     case AddOverlayConfig(reqId, config) =>
377       overlayConfigs.find(o => o.name == config.name && o.version == config.version) match {
378         case Some(existing) if existing != config =>
379           sender() ! OperationFailed(reqId, s"A different overlay with same name and version (${config.name}-${config.version}) is already registered")
380         case _ =>
381           val confFile = new File(installBaseDir.getParentFile(), s"overlays/${config.name}-${config.version}.conf")
382           confFile.getParentFile().mkdirs()
383           ConfigWriter.write(OverlayConfigCompanion.toConfig(config), confFile, None)
384           overlayConfigs += config
385           sender ! OperationSucceeded(reqId)
386       }
387 
388     case AddRuntimeConfig(reqId, config) =>
389       runtimeConfigs.find(r => r.runtimeConfig.name == config.name && r.runtimeConfig.version == config.version) match {
390         case Some(existing) if existing.runtimeConfig != config =>
391           sender() ! OperationFailed(reqId, s"A different overlay with same name and version (${config.name}-${config.version}) is already registered")
392         case _ =>
393           val confFile = new File(installBaseDir, s"${config.name}/${config.version}/profile.conf")
394           config.resolve() match {
395             case Success(resolved) =>
396               confFile.getParentFile().mkdirs()
397               ConfigWriter.write(RuntimeConfigCompanion.toConfig(config), confFile, None)
398               runtimeConfigs += LocalRuntimeConfig(baseDir = confFile.getParentFile(), resolvedRuntimeConfig = resolved)
399               sender() ! OperationSucceeded(reqId)
400             case Failure(e) =>
401               sender() ! OperationFailed(reqId, s"Given runtime config can't be resolved: ${e.getMessage}")
402           }
403       }
404 
405     case StageNextRuntimeConfig(reqId) =>
406       if (stagingInProgress.isEmpty) {
407         profiles.toIterator.collect {
408           case (id @ ProfileId(name, version, overlays), LocalProfile(_, _, LocalProfile.Pending(_))) =>
409             log.info(s"About to auto-stage profile ${id}")
410             self ! StageProfile(nextId(), name, version, overlays = overlays)
411         }.take(1)
412       }
413 
414     case msg : StageProfile    => stageProfile(sender(), msg)
415 
416     case msg : ActivateProfile => activateProfile(sender(), msg)
417 
418     case GetProgress(reqId) =>
419       val reqActor = sender()
420       stagingInProgress.get(reqId) match {
421         case Some(state) => reqActor ! Progress(reqId, state.progress())
422         case None        => reqActor ! OperationFailed(reqId, "Unknown request ID")
423       }
424 
425   }

440   override def receive : Actor.Receive = LoggingReceive {
441 
442     // from event stream
443     case e : UpdateAction =>
444       log.debug(s"Current profiles [${profiles}]")
445       log.debug(s"Handling UpdateAction message: ${e}")
446       handleUpdateAction(e)
447 
448     // direct protocol
449     case p : Protocol =>
450       log.debug(s"Handling Protocol message: ${p}")
451       handleProtocol(p)
452 
453     case Scan =>
454       log.debug("Handling Scan mesage")
455       overlayConfigs = scanForOverlayConfigs().toSet
456       val rcs = scanForRuntimeConfigs()
457       runtimeConfigs = rcs.toSet
458 
459       val fullProfiles = scanForProfiles(Option(rcs))
460       profiles = fullProfiles.map { profile => profile.profileId -> profile }.toMap
461       log.debug(s"Profiles (after scan): ${profiles}")
462 
463     case PublishProfileInfo =>
464       log.debug("Handling PublishProfileInfo message")
465       val activeProfile = findActiveProfile().map(_.toSingleProfile)
466       val singleProfiles = profiles.values.toList.map(_.toSingleProfile).map { p =>
467         activeProfile match {
468           case Some(a) if p.name == a.name && p.version == a.version && p.overlays.toSet == a.overlays.toSet =>
469             p.copy(overlaySet = p.overlaySet.copy(state = OverlayState.Active))
470           case _ => p
471         }
472 
473       }
474       val toSend = singleProfiles
475       log.debug(s"Publishing profile info to event stream: ${toSend}")
476       eventStream.publish(ProfileInfo(System.currentTimeMillis(), toSend))
477 
478     case PublishServiceInfo =>
479       log.debug("Handling PublishServiceInfo message")
480 
481       val serviceInfo = ServiceInfo(
482         name = context.self.path.toString,
483         serviceType = "Updater",
484         timestampMsec = System.currentTimeMillis(),
485         lifetimeMsec = config.serviceInfoLifetimeMSec,
486         props = Map(
487           "installBaseDir" -> installBaseDir.getAbsolutePath(),
488           "launchedProfileDir" -> launchedProfileDir.map(_.getAbsolutePath()).getOrElse(""),
489           "launchedProfileId" -> launchedProfileId.map(_.toString()).getOrElse("")
490         )
491       )
492       log.debug(s"About to publish service info: ${serviceInfo}")
493       eventStream.publish(serviceInfo)
494 
495     case msg : ArtifactDownloader.Reply => handleArtifactDownloaderReply(msg)
496 
497     case msg : Unpacker.UnpackReply     => handleUnpackerUnpackReply(msg)
498   }

500   protected def handleArtifactDownloaderReply(msg : ArtifactDownloader.Reply) : Unit = {
501     val foundProgress = stagingInProgress.values.flatMap { state =>
502       state.artifactsToDownload.find { bip => bip.reqId == msg.requestId }.map(state -> _).toList
503     }.toList
504     foundProgress match {
505       case Nil =>
506         log.error(s"Unkown download id ${msg.requestId}")
507       case (state, bundleInProgress) :: _ =>
508         val newArtifactsToDownload = state.artifactsToDownload.filter(bundleInProgress != _)
509         msg match {
510           case ArtifactDownloader.DownloadFinished(_) =>
511             stageInProgress(state.copy(
512               artifactsToDownload = newArtifactsToDownload
513             ))
514           case ArtifactDownloader.DownloadFailed(_, error) =>
515             stageInProgress(state.copy(
516               artifactsToDownload = newArtifactsToDownload,
517               issues = s"Download failed for ${bundleInProgress.file} (${error.getClass().getSimpleName()}: ${error})" +: state.issues
518             ))
519         }
520     }
521   }

523   protected def handleUnpackerUnpackReply(msg : Unpacker.UnpackReply) : Unit = {
524     val foundProgress = stagingInProgress.values.flatMap { state =>
525       state.artifactsToUnpack.find { a => a.reqId == msg.reqId }.map(state -> _).toList
526     }.toList
527     foundProgress match {
528       case Nil =>
529         log.error(s"Unkown unpack process ${msg.reqId}")
530       case (state, artifact) :: _ =>
531         val artifactsToUnpack = state.artifactsToUnpack.filter(artifact != _)
532         msg match {
533           case Unpacker.UnpackingFinished(_) =>
534             state.config.createResourceArchiveTouchFile(artifact.artifact, artifact.artifact.sha1Sum) match {
535               case Success(file) =>
536                 stageInProgress(state.copy(
537                   artifactsToUnpack = artifactsToUnpack
538                 ))
539               case Failure(e) =>
540                 stageInProgress(state.copy(
541                   artifactsToUnpack = artifactsToUnpack,
542                   issues = s"Could not create unpacked-marker file for resource file ${artifact.file} (${e.getClass().getSimpleName()}: ${e.getMessage()})" +: state.issues
543                 ))
544             }
545           case Unpacker.UnpackingFailed(_, e) =>
546             stageInProgress(state.copy(
547               artifactsToUnpack = artifactsToUnpack,
548               issues = s"Could not unpack file ${artifact.file} (${e.getClass().getSimpleName()}: ${e.getMessage()})" +: state.issues
549             ))
550         }
551     }
552   }
