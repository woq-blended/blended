11 class FileSystemOverlayConfigPersistor(storageDir : File) extends OverlayConfigPersistor {
12 
13   private[this] val log = Logger[FileSystemOverlayConfigPersistor]
14 
15   private[this] var overlayConfigs : Map[File, OverlayConfig] = Map()
16   private[this] var initalized : Boolean = false
17 
18   def overlayConfigFileName(oc : OverlayConfig) : String = s"${oc.name}-${oc.version}.conf"
19 
20   def initialize() : Unit = {
21     log.debug(s"About to initialize overlay config persistor for storageDir: ${storageDir}")
22     overlayConfigs = if (!storageDir.exists()) {
23       Map()
24     } else {
25       val files = Option(storageDir.listFiles()).getOrElse(Array())
26       val ocs = files.flatMap { file =>
27         val oc = Try {
28           ConfigFactory.parseFile(file).resolve()
29         }.flatMap { rawConfig =>
30           OverlayConfigCompanion.read(rawConfig)
31         }
32         log.debug(s"Found file: ${file} with: {oc}")
33         oc.toOption.map(oc => file -> oc)
34       }
35       ocs.filter { case (file, oc) => file.getName() == overlayConfigFileName(oc) }.toMap
36     }
37     initalized = true
38   }
39 
40   override def persistOverlayConfig(overlayConfig : OverlayConfig) : Unit = {
41     if (!initalized) initialize()
42     val configFile = new File(storageDir, overlayConfigFileName(overlayConfig))
43     if (configFile.exists()) {
44       // collision, what should we do?
45       if (overlayConfigs.get(configFile) == Some(overlayConfig)) {
46         // known and same, so silently ignore
47         log.debug("OverlayConfig already persistent")
48       } else {
49         val msg = "Cannot persist overlay config. Storage location already taken for a different configuration."
50         log.error(s"${msg} Found file ${configFile} with config: ${overlayConfigs.get(configFile)}")
51         sys.error(msg)
52       }
53     }
54     ConfigWriter.write(OverlayConfigCompanion.toConfig(overlayConfig), configFile, None)
55     overlayConfigs += configFile -> overlayConfig
56   }
57 
58   override def findOverlayConfigs() : List[OverlayConfig] = {
59     if (!initalized) initialize()
60     overlayConfigs.values.toList
61   }
