11 class FileSystemRuntimeConfigPersistor(storageDir : File) extends RuntimeConfigPersistor {
12 
13   private[this] val log = Logger[FileSystemRuntimeConfigPersistor]
14 
15   private[this] var runtimeConfigs : Map[File, RuntimeConfig] = Map()
16   private[this] var initialized : Boolean = false
17 
18   def fileName(rc : RuntimeConfig) : String = s"${rc.name}-${rc.version}.conf"
19 
20   def initialize() : Unit = {
21     log.debug(s"About to initialize runtime config persistor for storageDir: ${storageDir}")
22     runtimeConfigs = if (!storageDir.exists()) {
23       Map()
24     } else {
25       val files = Option(storageDir.listFiles()).getOrElse(Array())
26       val rcs = files.flatMap { file =>
27         val rc = Try {
28           ConfigFactory.parseFile(file).resolve()
29         }.flatMap { rawConfig =>
30           RuntimeConfigCompanion.read(rawConfig)
31         }
32         log.debug(s"Found file: ${file} with: ${rc}")
33         rc.toOption.map(rc => file -> rc)
34       }
35       rcs.filter { case (file, rc) => file.getName() == fileName(rc) }.toMap
36     }
37     initialized = true
38   }
39 
40   override def persistRuntimeConfig(runtimeConfig : RuntimeConfig) : Unit = {
41     if (!initialized) initialize()
42     val configFile = new File(storageDir, fileName(runtimeConfig))
43     if (configFile.exists()) {
44       // collision, what should we do?
45       if (runtimeConfigs.get(configFile) == Some(runtimeConfig)) {
46         // known and same, so silently ignore
47         log.debug("RuntimeConfig already persistent")
48       } else {
49         val msg = "Cannot persist runtime config. Storage location already taken for a different configuration."
50         log.error(s"${msg} Found file ${configFile} with config: ${runtimeConfigs.get(configFile)}")
51         sys.error(msg)
52       }
53     }
54     ConfigWriter.write(RuntimeConfigCompanion.toConfig(runtimeConfig), configFile, None)
55     runtimeConfigs += configFile -> runtimeConfig
56   }
57 
58   override def findRuntimeConfigs() : List[RuntimeConfig] = {
59     if (!initialized) initialize()
60     runtimeConfigs.values.toList
61   }
