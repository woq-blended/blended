17   def addAction(containerId : ContainerId, action : UpdateAction) : Unit = {
18     // Logic:
19     // - find previous state
20     // - add new actions
21     // - calc new state
22     // - persist new state
23 
24     log.debug(s"About to add action [${action}] for container [${containerId}]")
25     val state = containerStatePersistor.findContainerState(containerId).getOrElse(ContainerState(containerId = containerId))
26     val actions = state.outstandingActions
27     log.debug(s"Found [${actions.size}] old outstanding actions for container [${containerId}]")
28     val newActions =
29       // we ignore the ID in the existence check
30       if (!actions.exists(_.withId("") == action.withId(""))) {
31         actions ++ List(action)
32       } else {
33         log.debug(s"Ignoring action with ID [${action.id}]. A same action was already scheduled")
34         actions
35       }
36     val newState = state.copy(outstandingActions = newActions)
37     log.debug(s"New container state [${newState}] with [${newState.outstandingActions.size}] outstanding actions")
38     containerStatePersistor.updateContainerState(newState)
39   }

41   def updateContainerState(containerInfo : ContainerInfo) : ContainerState = {
42     // Logic:
43     // - find previous state and extract it's profiles
44     // - analyze outstanding actions (filter those with missing dependencies or those known as applied)
45     // - persist filtered actions as new state
46 
47     log.debug(s"About to analyze update properties from container info for container ID [${containerInfo.containerId}]")
48     log.trace(s"ContainerInfo: [${containerInfo}]")
49 
50     val timeStamp = System.currentTimeMillis()
51     val persistedState = containerStatePersistor.findContainerState(containerInfo.containerId).getOrElse(ContainerState(containerId = containerInfo.containerId))
52 
53     val containerProfiles = containerInfo.profiles
54 
55     val newUpdateActions = persistedState.outstandingActions
56       // remove those marked as applied
57       .filterNot(a => containerInfo.appliedUpdateActionIds.contains(a.id))
58       // filter some inconsistent actions
59       .filter {
60         // TODO: why do we filter here? Performance?
61         // TODO: support for overlays
62         case ActivateProfile(id, n, v, o) =>
63           // exclude already active activate request
64           // FIXME: is this correct, e.g. if a previous action activates another profile?
65           !containerProfiles.exists(p =>
66             p.name == n &&
67               p.version == v &&
68               p.overlaySet.overlays == o &&
69               p.state == OverlayState.Active)
70         case StageProfile(id, n, v, oc) =>
71           // exclude already staged stage request
72           !containerProfiles.exists(p =>
73             p.name == n &&
74               p.version == v &&
75               p.overlaySet.overlays == oc &&
76               Set(OverlayState.Valid, OverlayState.Invalid, OverlayState.Active).exists(_ == p.overlaySet.state))
77         case _ => true
78       }
79     val diff = newUpdateActions.size - persistedState.outstandingActions.size
80     if (diff < 0) {
81       log.debug(s"Removed ${-diff} actions: ${persistedState.outstandingActions.filterNot(a => newUpdateActions.contains(a))}")
82     }
83     if (diff > 0) {
84       log.debug(s"Added ${diff} actions: ${newUpdateActions.filterNot(a => persistedState.outstandingActions.contains(a))}")
85     }
86 
87     val newState = persistedState.copy(
88       profiles = containerProfiles,
89       outstandingActions = newUpdateActions,
90       syncTimeStamp = Some(timeStamp)
91     )
92     containerStatePersistor.updateContainerState(newState)
93     newState
94   }
