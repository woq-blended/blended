22   def renderContainerState(state : ContainerState) : String = {
23     s"""Container ID: ${state.containerId}
24         |  profiles: ${state.profiles.mkString(", ")}
25         |  outstanding actions: ${
26       state.outstandingActions.map {
27         // TODO: overlays
28         case AddRuntimeConfig(id, rc)     => s"add runtime config ${rc.name}-${rc.version}"
29         case AddOverlayConfig(id, oc)     => s"add overlay config ${oc.name}-${oc.version}"
30         case StageProfile(id, n, v, o)    => s"stage ${n}-${v} with ${o.toList.sorted.mkString(" and ")}"
31         case ActivateProfile(id, n, v, o) => s"activate ${n}-${v} with ${o.toList.sorted.mkString(" and ")}"
32       }.mkString(", ")
33     }
34         |  last sync: ${state.syncTimeStamp.map(s => new Date(s)).mkString}""".stripMargin
35   }

37   def remoteShow() : String = {
38     remoteUpdater.getContainerIds().map { id =>
39       s"Update state of container with ID ${id}:\n${remoteUpdater.getContainerState(id).map(renderContainerState)}\n"
40     }.mkString("\n")
41   }

43   def remoteShow(containerId : String) : String = {
44     remoteUpdater.getContainerState(containerId) match {
45       case Some(state) => s"Update state of container with ID ${containerId}:\n${state}\n"
46       case None        => s"Unknown container ID: ${containerId}"
47     }
48   }

50   def profiles() : String = {

52   }

54   def overlays() : String = {

58   def registerProfile(profileFile : String) : Unit = {
59     val file = new File(profileFile)
60     if (!file.exists()) {
61       println(s"File ${file.toURI()} does not exist")
62     } else {
63       println(s"Reading profile from file: ${file.toURI()}")
64       val config = ConfigFactory.parseFile(file).resolve()
65       val runtimeConfig = RuntimeConfigCompanion.read(config)
66       println(s"Profile: ${runtimeConfig}")
67       remoteUpdater.registerRuntimeConfig(runtimeConfig.get)
68     }
69   }
70 
71   def registerOverlay(overlayFile : String) : Unit = {
72     val file = new File(overlayFile)
73     if (!file.exists()) {
74       println(s"File ${file.toURI()} does not exist")
75     } else {
76       println(s"Reading overlay from file: ${file.toURI()}")
77       val config = ConfigFactory.parseFile(file).resolve()
78       val overlayConfig = OverlayConfigCompanion.read(config)
79       println(s"Overlay: ${overlayConfig}")
80       remoteUpdater.registerOverlayConfig(overlayConfig.get)
81     }
82   }
83 
84   def remoteStage(containerId : String, profileName : String, profileVersion : String) : Unit = {
85     remoteUpdater.getRuntimeConfigs().find(rc => rc.name == profileName && rc.version == profileVersion) match {
86       case None => println(s"Profile '${profileName}-${profileVersion}' not found")
87       case Some(rc) =>
88         // FIXME: support for overlay
89         remoteUpdater.addAction(containerId, AddRuntimeConfig(UUID.randomUUID().toString(), rc))
90         remoteUpdater.addAction(containerId, StageProfile(UUID.randomUUID().toString(), profileName, profileVersion, Set.empty))
91         println(s"Scheduled profile staging for container with ID ${containerId}. Config: ${profileName}-${profileVersion}")
92     }
93   }
94 
95   def remoteActivate(containerId : String, profileName : String, profileVersion : String) : Unit = {
96     // FIXME: support for overlays
97     remoteUpdater.addAction(containerId, ActivateProfile(UUID.randomUUID().toString(), profileName, profileVersion, Set.empty))
98     println(s"Scheduled profile activation for container with ID ${containerId}. Profile: ${profileName}-${profileVersion}")
99   }
