18 /**
19  * Logger is a factory for obtaining SLF4J-Loggers
20  */
21 object AkkaLogger {
22   /**
23    * @param logger - which logger
24    * @return a Logger that corresponds for the given logger name
25    */
26   def apply(logger: String): Logger = Logger(logger)
27 
28   /**
29    * @param logClass - the class to log for
30    * @param logSource - the textual representation of the source of this log stream
31    * @return a Logger for the specified parameters
32    */
33   def apply(logClass: Class[_], logSource: String): Logger = logClass match {
34     case c if c == classOf[DummyClassForStringSources] => apply(logSource)
35     case _ => apply(getClass().getName())
36   }
37 }

39 /**
40  * Blended logger.
41  *
42  * The thread in which the logging was performed is captured in
43  * Mapped Diagnostic Context (MDC) with attribute name "sourceThread".
44  */
45 class BlendedLogger extends Actor with BlendedLogging with RequiresMessageQueue[LoggerMessageQueueSemantics] {
46 
47   val mdcThreadAttributeName = "sourceThread"
48   val mdcActorSystemAttributeName = "sourceActorSystem"
49   val mdcAkkaSourceAttributeName = "akkaSource"
50   val mdcAkkaTimestamp = "akkaTimestamp"
51 
52   def receive : Receive = {
53 
54     case event @ Error(cause, logSource, logClass, message) =>
55       cause match {
56         case Error.NoCause | null =>
57           AkkaLogger(logClass, logSource).errorMdc(mdcMap(logSource, event))(Option(message).map(_.toString).orNull)
58         case _ =>
59           AkkaLogger(logClass, logSource).errorMdc(cause)(mdcMap(logSource, event))(Option(message).map(_.toString).getOrElse(cause.getLocalizedMessage))
60       }
61 
62     case event @ Warning(logSource, logClass, message) ⇒
63       event match {
64         case e: LogEventWithCause ⇒
65           AkkaLogger(logClass, logSource).warnMdc(e.cause)(mdcMap(logSource, event))(Option(message).map(_.toString).getOrElse(e.cause.getLocalizedMessage))
66         case _                    ⇒
67           AkkaLogger(logClass, logSource).warnMdc(mdcMap(logSource, event))(Option(message).map(_.toString).orNull)
68       }
69 
70     case event @ Info(logSource, logClass, message) =>
71       AkkaLogger(logClass, logSource).infoMdc(mdcMap(logSource, event))(message.toString)
72 
73     case event @ Debug(logSource, logClass, message) ⇒
74       AkkaLogger(logClass, logSource).debugMdc(mdcMap(logSource, event))(message.toString)
75 
76     case InitializeLogger(_) =>
77       log.info("Blended Akka Logger started")
78       sender() ! LoggerInitialized
79   }
80 
81   @inline
82   final def mdcMap(logSource: String, logEvent: LogEvent) : Map[String, String] = Map(
83     mdcAkkaSourceAttributeName -> logSource,
84     mdcThreadAttributeName -> logEvent.thread.getName(),
85     mdcAkkaTimestamp -> Helpers.currentTimeMillisToUTCString(logEvent.timestamp),
86     mdcActorSystemAttributeName -> context.system.name
87   )
88 }

95 class BlendedLoggingFilter(settings: ActorSystem.Settings, eventStream: EventStream) extends LoggingFilter {
