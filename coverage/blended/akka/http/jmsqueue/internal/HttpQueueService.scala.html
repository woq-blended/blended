25   private[this] def propsToHeaders(m : Message) : List[HttpHeader] = m.getPropertyNames().asScala.map { key =>
26     val realKey = key.toString()

31   private[this] def messageToResponse(result : ReceiveResult) : HttpResponse = {
32 
33     val response = result.msg match {
34       case Failure(_) =>
35         HttpResponse(StatusCodes.InternalServerError)
36       case Success(msg) =>
37         msg match {
38           case None =>
39             HttpResponse(StatusCodes.NoContent)
40           case Some(m) if m.isInstanceOf[TextMessage] =>
41             HttpResponse(
42               status = StatusCodes.OK,
43               headers = propsToHeaders(m),
44               entity = HttpEntity(ContentTypes.`text/plain(UTF-8)`, m.asInstanceOf[TextMessage].getText())
45             )
46           case Some(m) if m.isInstanceOf[BytesMessage] =>
47 
48             val bMsg = m.asInstanceOf[BytesMessage]
49 
50             val body = new Array[Byte](bMsg.getBodyLength().toInt)
51             bMsg.readBytes(body)
52 
53             HttpResponse(
54               status = StatusCodes.OK,
55               headers = propsToHeaders(m),
56               entity = HttpEntity(ContentTypes.`application/octet-stream`, body)
57             )
58           case Some(_) =>
59             log.debug(s"Message received on queue [${result.queue}] for provider [${result.vendor}:${result.provider}] is not a text or binary message. Message discarded ...")
60             HttpResponse(StatusCodes.InternalServerError)
61         }
62     }
63 
64     log.debug(s"Http Response is [$response]")
65     response
66   }

68   private[this] def performReceive(
69     vendor : String,
70     provider : String,
71     queue : String
72   ) : ConnectionFactory => ReceiveResult = { cf =>
73 
74     var conn : Option[Connection] = None
75     var sess : Option[Session] = None
76     var consumer : Option[MessageConsumer] = None
77 
78     try {
79       conn = Some(cf.createConnection())
80       conn.get.start()
81       sess = Some(conn.get.createSession(false, Session.CLIENT_ACKNOWLEDGE))
82       consumer = Some(sess.get.createConsumer(sess.get.createQueue(queue)))
83 
84       Option(consumer.get.receive(100)) match {
85         case None =>
86           log.debug(s"No message available on queue [$queue] for [$vendor:$provider]")
87           ReceiveResult(vendor, provider, queue, Success(None))
88         case Some(m) =>
89           log.debug(s"Received message from queue [$queue] for [$vendor:$provider] : [$m]")
90           m.acknowledge()
91           ReceiveResult(vendor, provider, queue, Success(Some(m)))
92       }
93 
94     } catch {
95       case NonFatal(e) =>
96         log.debug(s"Error receiving message from queue [$queue] for [$vendor:$provider]. Cause: ${e.getMessage()}")
97         ReceiveResult(vendor, provider, queue, Failure(e))
98     } finally {
99       Future {
100         blocking {
101           try {
102             consumer.foreach(_.close())
103           } catch {
104             case NonFatal(t) => log.debug(s"Error closing consumer [${t.getMessage()}]")
105           }
106 
107           try {
108             sess.foreach(_.close())
109           } catch {
110             case NonFatal(t) => log.debug(s"Error closing session [${t.getMessage()}]")
111           }
112 
113           try {
114             conn.foreach(_.close())
115           } catch {
116             case NonFatal(t) => log.debug(s"Error closing connection [${t.getMessage()}]")
117           }
118         }
119       }
120     }
121   }

123   private[this] def receive(vendor : String, provider : String, queue : String) : HttpResponse = {
124 
125     withConnectionFactory[HttpResponse](vendor, provider) {
126       case None =>
127         val msg = s"No connection factory found for [$vendor:$provider]"
128         log.warn(msg)
129         messageToResponse(ReceiveResult(vendor, provider, queue, Failure(new Exception(msg))))
130       case Some(cf) =>
131         messageToResponse(performReceive(vendor, provider, queue)(cf))
132     }
133   }

135   val httpRoute : Route = {
136     path(Segments) { path =>
137       get {
138         entity(as[HttpRequest]) { request =>
139           log.debug(s"Http operation request received at [$path] : $request")
140 
141           path match {
142             case qPath :: queue :: Nil =>
143               qConfig.httpQueues.find { case ((v, p), c) => c.path == qPath } match {
144                 case None =>
145                   log.warn(s"No queues configured for path [$qPath].")
146                   complete(StatusCodes.Unauthorized)
147                 case Some(((v, p), c)) =>
148                   if (c.queueNames.contains(queue)) {
149                     complete(receive(v, p, queue))
150                   } else {
151                     log.warn(s"Queue [$queue] is not configured for [$v:$p] at path [$qPath]")
152                     complete(StatusCodes.Unauthorized)
153                   }
154               }
155             case _ =>
156               log.warn(s"Request [${path.mkString("/")}] does not match the form 'path/queue'")
157               complete(StatusCodes.BadRequest)
158           }
159         }
160       }
161     }
