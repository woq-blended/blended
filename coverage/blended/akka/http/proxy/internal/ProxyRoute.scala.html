36   def handle(requestPath : String) : Route = {
37 
38     implicit val _actorSystem = actorSystem
39     implicit val materializer = ActorMaterializer()
40     import scala.concurrent.ExecutionContext.Implicits.global
41 
42     val timeoutResponse = HttpResponse(
43       StatusCodes.GatewayTimeout,
44       entity = s"The proxy request did not responded after ${proxyConfig.timeout} seconds"
45     )
46 
47     withRequestTimeout(proxyConfig.timeout.seconds, request => timeoutResponse) { ctx : RequestContext =>
48 
49       val simpleClient : HttpRequest => Future[HttpResponse] = sslContext match {
50         case Some(sslCtx) =>
51           // Use the explicit SSL Context for connections
52           Http().singleRequest(_ : HttpRequest, connectionContext = ConnectionContext.https(sslContext = sslCtx))
53         case None =>
54           // Use default SSL Context for connections
55           Http().singleRequest(_ : HttpRequest)
56       }
57 
58       val authHeader : Seq[HttpHeader] =
59         (proxyConfig.user, proxyConfig.password) match {
60           case (Some(u), Some(p)) => Seq(
61             Authorization(BasicHttpCredentials(u, p))
62           )
63           case (_, _) => Seq.empty
64         }
65 
66       val request = ctx.request
67 
68       val uri = Uri(
69         if (requestPath.isEmpty)  {
70           proxyConfig.uri
71         } else {
72           s"${proxyConfig.uri}/${requestPath}"
73         }
74       // Keep the query part of the original request
75       ).copy(rawQueryString = request.uri.rawQueryString)
76 
77       val headers = request.headers.filter{
78         header =>
79           // keep headers, but not the host header
80           if (header.is(Host.lowercaseName)) {
81             false
82           // remove the authorization header if we have found user / pwd in the config
83           } else if (header.is(Authorization.lowercaseName)){
84             authHeader.isEmpty
85           } else {
86             true
87           }
88       } ++ authHeader
89 
90       log.info(s"Received HttpRequest [${request}] at endpoint [${proxyConfig.path}] and path [${requestPath}] with query [${request.uri.queryString()}]")
91 
92       log.info(s"About to request [$uri] with method [${request.method}] with entity [${request.entity}] and headers [${headers}]")
93 
94       // the final request to the target host
95       val proxyReq = HttpRequest(method = request.method, uri = uri, entity = request.entity).withHeaders(headers)
96       log.debug(s"Final http request [${proxyReq}]")
97 
98       def handleResponse(request : HttpRequest, redirectCount : Int, config : ProxyTarget) : Future[HttpResponse] = simpleClient(request).flatMap { response =>
99         response.status match {
100 
101           case e : StatusCodes.ServerError =>
102             //            response.discardEntityBytes()
103             // consume entity (for log)
104             val re = response.entity
105             log.warn(s"503 Bad Gateway. The upstream (proxied) server returned with error [${e}] and response entity [${re}]")
106             Future.successful(HttpResponse(StatusCodes.BadGateway))
107 
108           case r : StatusCodes.Redirection if redirectCount > 0 && Seq(301, 302, 307, 308).contains(r.intValue) =>
109 
110             val newUri = response.header[Location].get.uri
111             val newRedirectCount = redirectCount - 1
112 
113             val newRequest : HttpRequest = config.redirectHeaderPolicy match {
114               case RedirectHeaderPolicy.Client_Only =>
115                 request
116               case RedirectHeaderPolicy.Redirect_Merge =>
117                 log.debug("Merging headers from redirect address into request header.")
118 
119                 response.headers.foldLeft(request) { (r, h) =>
120                   if (!r.getHeader(h.name()).isPresent) r.withHeaders(h) else r
121                 }
122               case RedirectHeaderPolicy.Redirect_Replace =>
123                 log.debug("Replacing request headers with headers from redirect address")
124                 val oldHeaders = request.headers
125                 oldHeaders.foreach { h => request.removeHeader(h.name()) }
126                 request.withHeaders(response.headers)
127             }
128 
129             log.info(s"${r}. Retry request with new URI [${newUri}], headers [${newRequest.headers}] and redirectCount [${newRedirectCount}]")
130 
131             // Always consume the response entity streams
132             response.discardEntityBytes()
133 
134             // create new request and retry
135             handleResponse(newRequest.withUri(newUri), newRedirectCount, config)
136 
137           case s =>
138             log.debug(s"${s}. Received upstream response with status [${response.status}] and headers [${response.headers}]")
139             Future.successful(response)
140         }
141       }
142 
143       ctx.complete(handleResponse(proxyReq, proxyConfig.redirectCount, proxyConfig))
144     }
