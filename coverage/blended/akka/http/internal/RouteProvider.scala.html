31   private[this] def updateRoutes() : Unit = {

33     currentRoute = contexts.foldLeft(initialRoute) { (route, context) =>
34       route ~ pathPrefix(PathMatchers.separateOnSlashes(context.prefix)) {

36       }

40   def dynamicAdapt(capsuleContext : CapsuleContext, bundleContext : BundleContext) : Unit = {
41 
42     def addContext(httpContext : HttpContext) : Unit = {
43       log.info(s"Adding http context: ${httpContext}")
44       // we currently allow only one route for each prefix
45       contexts = contexts.filter(c => c.prefix != httpContext.prefix)
46       contexts :+= httpContext
47       updateRoutes()
48     }
49 
50     def modifyContext(httpContext : HttpContext) : Unit = addContext _
51 
52     def removeContext(httpContext : HttpContext) : Unit = {
53       // we currently allow only one route for each prefix
54       contexts = contexts.filter(c => c.prefix != httpContext.prefix)
55       updateRoutes()
56     }

58     class WatchCapsule(
59       override protected val capsuleContext : CapsuleContext,
60       override protected val bundleContext : BundleContext

62       extends Capsule
63       with ServiceWatching

65 
66       override def start() : Unit = {
67         // We listen for supported services and add them to the route
68         // wait for context registrations, and add them to the main route
69         log.info("Listening for HttpContext registrations...")
70         watchServices[HttpContext] {
71           case ServiceWatcherEvent.AddingService(httpContext, watchContext)   => addContext(httpContext)
72           case ServiceWatcherEvent.ModifiedService(httpContext, watchContext) => modifyContext(httpContext)
73           case ServiceWatcherEvent.RemovedService(httpContext, watchContext)  => removeContext(httpContext)
74         }
75 
76         def toContext[S <: AnyRef](route : Route, ref : ServiceReference[S]) : Option[HttpContext] = {
77           Option(ref.getProperty("context")) match {
78             case Some(prefix : String) if !prefix.trim().isEmpty() =>
79               val ctx = SimpleHttpContext(prefix, route)
80               log.debug(s"Wrapped Route with prefix-property to HttpContext: ${ctx}")
81               Some(ctx)
82             case _ =>
83               log.warn(s"Missing or unsupported property 'prefix' defined for service reference: ${ref}. Skipping registration.")
84               None
85 
86           }
87         }
88 
89         log.info("Listening for Route registrations...")
90         watchServices[Route] {
91           case ServiceWatcherEvent.AddingService(route, watchContext)   => toContext(route, watchContext.ref).foreach(addContext)
92           case ServiceWatcherEvent.ModifiedService(route, watchContext) => toContext(route, watchContext.ref).foreach(modifyContext)
93           case ServiceWatcherEvent.RemovedService(route, watchContext)  => toContext(route, watchContext.ref).foreach(removeContext)
94         }
95 
96       }
97       override def stop() : Unit = {
98         // We unregister all routes
99         contexts = Seq()
100         updateRoutes()
101       }
102     }
103

105     capsuleContext.addCapsule(capsule)
106   }
