132   private def filterHeaders(headers : Seq[HttpHeader]) : collection.immutable.Seq[HttpHeader] = {

137   private def performRequest(opKey : String, request : HttpRequest, actor : ActorRef): Future[HttpResponse] = {
138     operations.get(opKey) match {
139       case None =>
140         log.warn(s"Http operation [$opKey] is not configured.")
141 
142         Future(HttpResponse(
143           status = StatusCodes.NotFound,
144           entity = HttpEntity.Strict(request.entity.contentType, ByteString.empty),
145           headers = filterHeaders(request.headers)
146         ))
147 
148       case Some(opCfg) =>
149         val cType : ContentType = request.entity.contentType
150 
151         val validContentTypes = opCfg.contentTypes match {
152           case None    => defaultContentTypes
153           case Some(l) => l
154         }
155 
156         validContentTypes.filter(_ == cType.mediaType.value) match {
157           case Nil =>
158             log.warn(s"Content-Type [${cType.value}] not supported.")
159 
160             Future(HttpResponse(
161               status = StatusCodes.InternalServerError,
162               entity = HttpEntity(cType, ByteString.empty),
163               headers = filterHeaders(request.headers)
164             ))
165 
166           case _ :: _ =>
167             requestReply(opKey, opCfg, cType, request, actor)
168         }
169     }
170   }

172   private def requestReply(operation : String, opCfg : JmsOperationConfig, cType : ContentType, request : HttpRequest, actor : ActorRef) : Future[HttpResponse] = {
173 
174     val data : Future[Seq[ByteString]] = request.entity.getDataBytes().runWith(Sink.seq[ByteString], materializer)
175 
176     data.map { result =>
177       val content: Array[Byte] = result.flatten.toArray
178 
179       val envId : String = UUID.randomUUID().toString()
180 
181       val header : Seq[(String, Any)] = Seq(
182         destHeader(headerCfg.prefix) -> opCfg.destination,
183         replyToHeader(headerCfg.prefix) -> s"${responseDestination.name}",
184         corrIdHeader(headerCfg.prefix) -> s"${osgiCfg.ctContext.uuid}##$envId",
185         "Content-Type" -> cType.mediaType.value
186       ) ++ opCfg.header.map{ case (k,v) => k -> v }
187 
188       val env: FlowEnvelope = FlowEnvelope(FlowMessage(content)(FlowMessage.props(header:_*).get), envId)
189 
190       log.debug(s"Received request [${env.id}] of length [${content.length}], encoding [${opCfg.encoding}], content type [${cType.mediaType}]")
191       log.debug(s"Request envelope is [$env]")
192 
193       val response : Future[HttpResponse] = addRequest(env.id, request)
194       system.scheduler.scheduleOnce(opCfg.timeout.millis){
195         timeoutRequest(env.id, opCfg.timeout.millis)
196       }
197       actor ! env
198 
199       checkComplete(env.id, opCfg)
200 
201       response
202     }.flatten
203   }

249   private def handleResponse(env : FlowEnvelope) : Unit = synchronized {
250 
251     def createEntity(req : HttpRequest, msg : FlowMessage) : ResponseEntity = msg match {
252       case tMsg: TextFlowMessage => HttpEntity(req.entity.contentType, tMsg.getText().getBytes())
253       case bMsg: BinaryFlowMessage => HttpEntity(req.entity.contentType, bMsg.getBytes())
254       case _ => HttpEntity(req.entity.contentType, ByteString.empty)
255     }
256 
257     pendingRequests.get(env.id) match {
258       case Some((req, p)) =>
259         log.info(s"Received response for HTTP request [$env]")
260         env.exception match {
261           case Some(t) =>
262             log.warn(t)(t.getMessage())
263             p.complete(Failure(t))
264           case None =>
265             p.complete(Success(HttpResponse(
266               StatusCodes.OK,
267               entity = createEntity(req, env.flowMessage),
268               headers = filterHeaders(req.headers)
269             )))
270         }
271       case None =>
272         log.warn(s"No pending request for received response [$env]")
273     }
274   }
