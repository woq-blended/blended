25   override def receive : Receive = open

27   private[akka] def logPending() : Unit = {
28     log.trace(s"Semaphore has [${pending.size}] actors waiting")
29   }
30 
31   private[akka] def acquire(a : Acquire) : Unit = {
32     pending = pending.filterNot(_.actor == a.actor)
33     a.actor ! Acquired
34     logPending()
35     context.become(locked(a))
36   }
37 
38   private[akka] def release(current : Acquire, a : ActorRef) : Unit = {
39     if (current.actor == a) {
40       log.trace(s"Releasing actor [$a]")
41       context.unwatch(a)
42       pending match {
43         case Nil => context.become(open)
44         case l   => acquire(l.last)
45       }
46     } else {
47       pending = pending.filter(_.actor != a)
48       logPending()
49     }
50   }
51 
52   private[akka] def open : Receive = {
53     case a : Acquire =>
54       context.watch(a.actor)
55       acquire(a)
56     case r : Release => // do nothing
57 
58     case m =>
59       log.warn(s"Unexpected message of type [${m.getClass().getName()}] in semaphore [$id]")
60   }
61 
62   private[akka] def locked(lockedBy : Acquire) : Receive = {
63     case a : Acquire =>
64       context.watch(a.actor)
65       if (lockedBy.actor == a.actor) {
66         a.actor ! Acquired
67       } else {
68         pending = (a :: pending).distinct
69         logPending()
70         a.actor ! Waiting
71       }
72 
73     case r : Release   => release(lockedBy, r.actor)
74     case Terminated(a) => release(lockedBy, a)
75   }
