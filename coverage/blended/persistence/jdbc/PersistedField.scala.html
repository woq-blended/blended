46     def extractValue(key : String, value : Any, parent : Option[PersistedField] = None) : Seq[PersistedField] = {
47       val baseFieldId = parent.map(_.fieldId)
48       value match {
49         case null =>
50           Seq(PersistedField(fieldId = nextId(), name = key, baseFieldId = baseFieldId, typeName = TypeName.Null))
51         case value : String =>
52           if (value.size <= maxStringLength) {
53             Seq(PersistedField(fieldId = nextId(), name = key, baseFieldId = baseFieldId, valueString = Some(value), typeName = TypeName.String))
54           } else {
55             val newBase = PersistedField(fieldId = nextId(), name = key, baseFieldId = baseFieldId, typeName = TypeName.LongString)
56             Seq(newBase) ++ value.toList.grouped(maxStringLength).map(_.mkString).toList.zipWithIndex.flatMap {
57               case (v, i) =>
58                 extractValue(i.toString(), v, Some(newBase))
59             }
60           }
61         case value : Boolean =>
62           Seq(PersistedField(fieldId = nextId(), name = key, baseFieldId = baseFieldId, valueLong = Some(if (value) 1 else 0), typeName = TypeName.Boolean))
63         case value : Long =>
64           Seq(PersistedField(fieldId = nextId(), name = key, baseFieldId = baseFieldId, valueLong = Some(value), typeName = TypeName.Long))
65         case value : Int =>
66           Seq(PersistedField(fieldId = nextId(), name = key, baseFieldId = baseFieldId, valueLong = Some(value), typeName = TypeName.Int))
67         case value : Short =>
68           Seq(PersistedField(fieldId = nextId(), name = key, baseFieldId = baseFieldId, valueLong = Some(value), typeName = TypeName.Short))
69         case value : Byte =>
70           Seq(PersistedField(fieldId = nextId(), name = key, baseFieldId = baseFieldId, valueLong = Some(value), typeName = TypeName.Byte))
71         case value : Double =>
72           Seq(PersistedField(fieldId = nextId(), name = key, baseFieldId = baseFieldId, valueDouble = Some(value), typeName = TypeName.Double))
73         case value : Float =>
74           Seq(PersistedField(fieldId = nextId(), name = key, baseFieldId = baseFieldId, valueDouble = Some(value), typeName = TypeName.Float))
75         case value : ju.Map[_, _] =>
76           val newBase = if (key == "" && parent.isEmpty) {
77             // Root map
78             None
79           } else {
80             Some(PersistedField(fieldId = nextId(), name = key, baseFieldId = baseFieldId, typeName = TypeName.Object))
81           }
82           newBase.toSeq ++ value.entrySet().asScala.toList.flatMap { entry =>
83             extractValue(entry.getKey().asInstanceOf[String], entry.getValue(), parent = newBase)
84           }
85         case value : ju.Collection[_] =>
86           val newBase = PersistedField(fieldId = nextId(), name = key, baseFieldId = baseFieldId, typeName = TypeName.Array)
87           Seq(newBase) ++ value.asScala.zipWithIndex.flatMap {
88             case (v, i) =>
89               extractValue(i.toString(), v, Some(newBase))
90           }
91         case x => throw new IllegalArgumentException("Unsupported type: " + Option(x).map(_.getClass()).orNull)
92       }
93     }
94 
95     extractValue(key = "", value = data)
96   }

98   def toJuMap(persistedFields : Seq[PersistedField]) : ju.Map[String, _ <: AnyRef] = {

100     def fieldExtract(field : PersistedField, others : Seq[PersistedField]) : AnyRef = {
101       field.typeName match {
102         case TypeName.Null    => null
103         case TypeName.Boolean => jl.Boolean.valueOf(field.valueLong.map(_ != 0).get)
104         case TypeName.Byte    => jl.Byte.valueOf(field.valueLong.map(_.toByte).get)
105         case TypeName.Short   => jl.Short.valueOf(field.valueLong.map(_.toShort).get)
106         case TypeName.Int     => jl.Integer.valueOf(field.valueLong.map(_.toInt).get)
107         case TypeName.Long    => jl.Long.valueOf(field.valueLong.get)
108         case TypeName.String  => field.valueString.get
109         case TypeName.Double  => jl.Double.valueOf(field.valueDouble.get)
110         case TypeName.Float   => jl.Float.valueOf(field.valueDouble.map(_.toFloat).get)
111         case TypeName.Array =>
112           val (col, colOther) = others.partition(_.baseFieldId == Some(field.fieldId))
113           val collection = new ju.LinkedList[AnyRef]()
114           col.foreach { colField =>
115             collection.add(fieldExtract(colField, colOther))
116           }
117           collection
118         case TypeName.Object => internMap(others, Some(field.fieldId))
119         case TypeName.LongString =>
120           val (col, colOther) = others.partition(_.baseFieldId == Some(field.fieldId))
121           col.sortBy(_.fieldId).map(slice => fieldExtract(slice, colOther)).mkString
122       }
123     }
124 
125     def internMap(fields : Seq[PersistedField], parentId : Option[Long]) : ju.Map[String, _ <: AnyRef] = {
126       val (root, other) = fields.partition(_.baseFieldId == parentId)
127       val map = new ju.LinkedHashMap[String, AnyRef]()
128       root.foreach { field =>
129         val value = fieldExtract(field, other)
130         map.put(field.name, value)
131       }
132       map
133     }
134 
135     internMap(persistedFields, None)
136   }
