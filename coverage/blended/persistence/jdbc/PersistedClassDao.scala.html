45   def persist(persistedClass : PersistedClass) : PersistedClass = {
46     log.trace(s"About to persist class [${persistedClass.name}:${persistedClass.id}] with ${persistedClass.fields.size} fields")
47 
48     // persist the holder class
49     val persistedClassId = {
50       val pcCols = Seq(PC.Name)
51       val sql = s"insert into ${PC.Table} (${pcCols.mkString(",")}) values (${pcCols.map(":" + _).mkString(",")})"
52 
53       val paramSource = new MapSqlParameterSource()
54       paramSource.addValue(PC.Name, persistedClass.name)
55       val keyHolder = new GeneratedKeyHolder()
56 
57       jdbcTemplate.update(sql, paramSource, keyHolder)
58       keyHolder.getKeys().get(PC.Id).asInstanceOf[java.lang.Long]
59     }
60 
61     // persist the fields
62     {
63       val pfCols = Seq(
64         PF.HolderId, PF.FieldId, PF.BaseFieldId,
65         PF.Name, PF.ValueLong, PF.ValueDouble, PF.ValueString,
66         PF.TypeName
67       )
68       val sql = s"insert into ${PF.Table} (${pfCols.mkString(",")}) values (${pfCols.map(":" + _).mkString(",")})"
69 
70       // TODO: add batch processing
71       // we work in-memory, so we expect no performance gain from batch operation for now
72       persistedClass.fields.foreach { field =>
73         val paramSource = new MapSqlParameterSource()
74         paramSource.addValue(PF.HolderId, persistedClassId)
75         paramSource.addValue(PF.FieldId, field.fieldId)
76         paramSource.addValue(PF.BaseFieldId, field.baseFieldId.orNull)
77         paramSource.addValue(PF.Name, field.name)
78         paramSource.addValue(PF.ValueLong, field.valueLong.orNull)
79         paramSource.addValue(PF.ValueDouble, field.valueDouble.orNull)
80         paramSource.addValue(PF.ValueString, field.valueString.orNull)
81         paramSource.addValue(PF.TypeName, field.typeName.name)
82         jdbcTemplate.update(sql, paramSource)
83       }
84     }
85 
86     persistedClass.copy(id = Some(persistedClassId))
87   }

119   def fieldRowMapper(prefix : String = "") : RowMapper[(Long, PersistedField)] = { (rs, _) =>
120 
121     val holderId = rs.getLong(prefix + PF.HolderId)
122 
123     val fieldId = rs.getLong(prefix + PF.FieldId)
124     val baseFieldId = Option(rs.getLong(prefix + PF.BaseFieldId)).filter(_ != 0)
125     val name = rs.getString(prefix + PF.Name)
126     val valueLong = Option(rs.getLong(prefix + PF.ValueLong))
127     val valueDouble = Option(rs.getDouble(prefix + PF.ValueDouble))
128     val valueString = Option(rs.getString(prefix + PF.ValueString))
129     val typeNameString = rs.getString(prefix + PF.TypeName)
130     val typeName = TypeName.fromString(typeNameString).getOrElse {
131       throw new IllegalStateException(s"Unsupported type name [$typeNameString] found in database column ${PF.TypeName}." +
132         s" Supported types: [${TypeName.values.mkString(", ")}]")
133     }
134 
135     holderId -> PersistedField(fieldId, baseFieldId, name, valueLong, valueDouble, valueString, typeName)
136   }

154   def inferPersistedClassesFromFields(pClass : String, fields : Seq[(Long, PersistedField)]) : sci.Seq[PersistedClass] = {
155     val byId = fields.foldLeft(Map[Long, List[PersistedField]]()) { (map, rs) =>
156       val id = rs._1
157       val field = rs._2
158       val tail = map.getOrElse(id, Nil)
159       map + (id -> (field :: tail))
160     }
161     byId.toList.map { case (id, f) => PersistedClass(id = Some(id), name = pClass, fields = f.reverse) }
162   }

166   def createByExampleQuery(pClass : String, selectCols : Seq[String], fields : Seq[PersistedField]) : (String, MapSqlParameterSource) = {
167     val mainField = "field"
168     val cls = "cls"
169 
170     // 3 mutable parts, we build up a query, after all
171     var queryFields : List[String] = mainField :: Nil
172     var queryCriterias : List[String] = Nil
173     val queryParams = new MapSqlParameterSource()
174 
175     queryCriterias ::= s"$mainField.${PF.HolderId} = $cls.${PC.Id} and $cls.${PC.Name} = :clsName"
176     queryParams.addValue("clsName", pClass)
177 
178     fields.foreach { field =>
179       val fName = _fName(field.fieldId)
180       queryFields ::= fName
181 
182       // match holder-id
183       queryCriterias ::= s"$fName.${PF.HolderId} = $mainField.${PF.HolderId}"
184 
185       // match name
186       queryCriterias ::= s"$fName.${PF.Name} = :${fName}Name"
187       queryParams.addValue(fName + "Name", field.name)
188 
189       // match type name
190       queryCriterias ::= s"$fName.${PF.TypeName} = :${fName}TypeName"
191       queryParams.addValue(fName + "TypeName", field.typeName.name)
192 
193       // We only match value fields if they belong to the field type
194       field.typeName match {
195         case TypeName.String =>
196           // match string value
197           field.valueString match {
198             case Some(v) =>
199               queryCriterias ::= s"$fName.${PF.ValueString} = :${fName}ValueString"
200               queryParams.addValue(fName + "ValueString", v)
201 
202             case None =>
203               queryCriterias ::= s"$fName.${PF.ValueString} is null"
204           }
205 
206         case TypeName.Boolean | TypeName.Long | TypeName.Int | TypeName.Byte =>
207           // match long value
208           field.valueLong match {
209             case Some(v) =>
210               queryCriterias ::= s"$fName.${PF.ValueLong} = :${fName}ValueLong"
211               queryParams.addValue(fName + "ValueLong", v.longValue())
212 
213             case None =>
214               queryCriterias ::= s"$fName.${PF.ValueLong} is null"
215           }
216 
217         case TypeName.Double | TypeName.Float =>
218           // match double value
219           field.valueDouble match {
220             case Some(v) =>
221               queryCriterias ::= s"$fName.${PF.ValueDouble} = :${fName}ValueDouble"
222               queryParams.addValue(fName + "ValueDouble", v.doubleValue())
223 
224             case None =>
225               queryCriterias ::= s"$fName.${PF.ValueDouble} is null"
226           }
227 
228         case _ =>
229         // Array and Object types do not define values,
230         // but have childs with hold the parent id
231       }
232 
233       field.baseFieldId.foreach { baseFieldId =>
234         queryCriterias ::= s"$fName.${PF.BaseFieldId} = ${_fName(baseFieldId)}.${PF.FieldId}"
235       }
236     }
237 
238     // Example
239     // Field 1: PeristedField(fieldId = 1, name = k1, valueLong = None, valueDouble = None, valueString = "v1", typeName = "String"
240     // Field 2: PeristedField(fieldId = 2, name = k2, valueLong = "2", valueDouble = None, valueString = None, typeName = "Long"
241     // select field.id, field.typeName, ...
242     // from PersistedField field, PersistedField field1, PersistedField field2, PersistedClass pclass
243     // where field.holderId = pclass.id and pclass.name = :pclass
244     //   and field.holderId = field1.holderId and field1.typeName = "String" and field1.name = "k1" and field1.valueString = "v1"
245     //   and field.holderId = field2.holderId and field2.typeName = "Long" and field2.name = "k2" and field2.valueDouble = 2
246     val sql = s"select ${selectCols.map(mainField + "." + _).mkString(", ")} " +
247       s"\nfrom ${PC.Table} $cls, ${queryFields.reverse.map(PF.Table + " " + _).mkString(", ")} " +
248       s"\nwhere ${queryCriterias.reverse.mkString(" and ")}"
249 
250     sql -> queryParams
251   }

253   def findByFields(pClass : String, fields : Seq[PersistedField]) : sci.Seq[PersistedClass] = {
254     val pfCols = Seq(
255       PF.HolderId, PF.FieldId, PF.BaseFieldId,
256       PF.Name, PF.ValueLong, PF.ValueDouble, PF.ValueString,
257       PF.TypeName

262     log.trace(s"Generated query: [$sql]")
263     log.trace(s"parameter map: [${queryParams.getValues}]")
264

266     inferPersistedClassesFromFields(pClass, allFields.asScala)

269   def deleteByFields(pClass : String, fields : Seq[PersistedField]) : Long = {
270     val cols = Seq(PF.HolderId)
271     val (sql, queryParams) = createByExampleQuery(pClass, cols, fields)
272     val classIds = jdbcTemplate.queryForList(sql, queryParams, classOf[java.lang.Long]).asScala.toList.distinct
273     log.trace(s"Found ${classIds.size} class entries to be deleted. Ids: [$classIds]")
274 
275     if (classIds.nonEmpty) {
276       val sql = s"delete from ${PC.Table} where ${PC.Id} in (:deleteIds)"
277       val paramMap = new MapSqlParameterSource()
278       paramMap.addValue("deleteIds", classIds.asJava)
279       log.trace(s"delete query: [$sql], params: [${paramMap.getValues()}]")
280       jdbcTemplate.update(sql, paramMap)
281     } else {
282       0L
283     }
284   }
