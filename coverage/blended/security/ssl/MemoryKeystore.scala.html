12 case class MemoryKeystore(certificates : Map[String, CertificateHolder]) {
13 
14   private[this] val log : Logger = Logger[MemoryKeystore]
15   private[this] val millisPerDay : Long = 1.day.toMillis
16 
17   /**
18    * Return those aliases of certificates which were updated or newly added.
19    */
20   def changedAliases : List[String] = certificates.collect {
21     case (key, CertificateHolder(_, _, _, change)) if change.changed => key
22   }.toList
23 
24   /**
25    * The in memory keystore is consistent if and only if all certificates have a private key defined
26    * or none of it does have a private key defined.
27    */
28   def consistent : Boolean = {
29     certificates.values.forall(_.privateKey.isDefined) || certificates.values.forall(_.privateKey.isEmpty)
30   }
31 
32   /**
33    * Updates/replaces the certificate (chain) for the given alias and updates
34    * the `change` flag of the associated [[CertificateHolder]].
35    */
36   def update(alias : String, cert : CertificateHolder) : Try[MemoryKeystore] = Try {
37 
38     val newState = certificates.get(alias) match {
39       case None => CertificateChange.Added
40       case Some(oldCert) =>
41         oldCert match {
42           case CertificateHolder(_, _, _, CertificateChange.Unchanged) =>
43             if (oldCert.dump == cert.dump) CertificateChange.Unchanged
44             else CertificateChange.Updated
45           case CertificateHolder(_, _, _, change) => change
46         }
47     }
48     log.info(s"Updating memory keystore [alias=$alias] and change state [${newState}]")
49     val result : MemoryKeystore = MemoryKeystore(certificates + (alias -> cert.copy(change = newState)))
50 
51     if (result.consistent) {
52       log.info(s"Updated keystore aliases : [${result.certificates.keys}]")
53       result
54     } else {
55       throw new InconsistentKeystoreException("Keystore must be consistent after update")
56     }
57   }
58 
59   def findByPrincipal(principal : X500Principal) : Option[CertificateHolder] = {
60     certificates.values.find { ch =>
61       ch.subjectPrincipal.isDefined &&
62         ch.subjectPrincipal.forall(_.equals(principal))
63     }
64   }

66   def nextCertificateTimeout() : Try[Date] = Try {
67     certificates match {
68       case e if e.isEmpty => new Date()
69       case m              => m.values.map(_.chain.head.getNotAfter).min
70     }
71   }

73   private[this] def refreshCertificate(
74     certCfg : CertificateConfig,
75     providerMap : Map[String, CertificateProvider],
76     oldCert : Option[CertificateHolder]
77   ) : Try[MemoryKeystore] = Try {
78     providerMap.get(certCfg.provider) match {
79       case None =>
80         log.warn(s"Certificate provider [${certCfg.provider}] not found, not updating certificate [${certCfg.alias}]")
81         this
82       case Some(p) =>
83         val newCert = p.refreshCertificate(oldCert, certCfg.cnProvider).get
84         log.info(s"Obtained certificate for alias [${certCfg.alias}] : [$newCert]")
85         update(certCfg.alias, newCert).get
86     }
87   }
88 
89   private[ssl] def changed(
90     certConfigs : List[CertificateConfig],
91     providerMap : Map[String, CertificateProvider]
92   ) : Try[MemoryKeystore] = Try {
93 
94     certConfigs match {
95       // No further certificate configs to check, returning result
96       case Nil => this
97 
98       // Otherwise we check the certificate specifies by the cert config at the head of
99       // the list and update the result
100       case head :: tail =>
101         certificates.get(head.alias) match {
102           // The keystore already has a certificate with the requested alias
103           case Some(serverCertificate) =>
104             val certInfo = X509CertificateInfo(serverCertificate.chain.head)
105             val remaining = certInfo.notAfter.getTime() - System.currentTimeMillis()
106 
107             if (remaining <= head.minValidDays * millisPerDay) {
108               log.info(s"Certificate [${head.alias}] is about to expire in ${remaining.toDouble / millisPerDay} days...refreshing certificate")
109 
110               refreshCertificate(head, providerMap, Some(serverCertificate)) match {
111                 case Success(newMs) =>
112                   newMs.changed(tail, providerMap).get
113 
114                 case Failure(t) =>
115                   log.info(t)(s"Could not refresh certificate [${head.alias}], reusing the existing one.")
116                   changed(tail, providerMap).get
117               }
118             } else {
119               log.info(s"Server certificate [${head.alias}] is still valid.")
120               changed(tail, providerMap).get
121             }
122 
123           // The keystore does not yet have a certificate for that alias
124           case None =>
125             log.info(s"Certificate with alias [${head.alias}] does not yet exist.")
126             refreshCertificate(head, providerMap, None) match {
127               case Success(newMs) =>
128                 newMs.changed(tail, providerMap).get
129 
130               case Failure(t) =>
131                 log.error(t)(s"Could not initially create certificate [${head.alias}] : [${t.getMessage()}].")
132                 // changed(tail, providerMap).get
133                 throw new InitialCertificateProvisionException(s"Could not initially create certificate [${head.alias}] : [${t.getMessage()}].")
134             }
135         }
136     }
137   }
