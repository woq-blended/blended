57    * using the given [[PasswordHasher]] to hash the passwords (`keyPass` and `storePass`).
58    */
59   def fromConfig(cfg : Config, hasher : PasswordHasher, ctCtxt : ContainerContext) : CertificateManagerConfig = {
60 
61     val clientOnly: Boolean = cfg.getBoolean("clientOnly", false)
62 
63     val maintainTruststore: Boolean = cfg.getBoolean("maintainTruststore", true)
64 
65     val keystoreCfg: Option[KeystoreConfig] = if (clientOnly) {
66       None
67     } else {
68       Some(KeystoreConfig.fromConfig(cfg, hasher, ctCtxt))
69     }
70 
71     val providers: List[String] = cfg.getStringList("providerList", List.empty)
72 
73     val certConfigs = cfg.getConfigMap("certificates", Map.empty).map {
74       case (k, v) =>
75         CertificateConfig.fromConfig(k, v, ctCtxt)
76     }.toList
77 
78     val refresherConfig = cfg.getConfigOption("refresher").map(c => RefresherConfig.fromConfig(c).get)
79 
80     val skipInitialCheck = cfg.getBoolean("skipInitialCheck", false)
81 
82     val cyphers = cfg.getStringList("validCypherSuites", List.empty)
83 
84     CertificateManagerConfig(
85       clientOnly = clientOnly,
86       maintainTruststore = maintainTruststore,
87       keystoreCfg = keystoreCfg,
88       certConfigs = certConfigs,
89       refresherConfig = refresherConfig,
90       skipInitialCheck = skipInitialCheck,
91       providerList = providers,
92       validCypherSuites = cyphers
93     )
