15   def nextRefreshScheduleTime(validEnd : Date, refresherConfig : RefresherConfig, now : Option[Date] = None) : Date = {
16     // calc next schedule date
17     val cal = new GregorianCalendar()
18     cal.clear()
19     cal.setTime(validEnd)
20     cal.add(Calendar.DAY_OF_MONTH, -1 * refresherConfig.minValidDays)
21 
22     val curNow = now.map(_.getTime()).getOrElse(System.currentTimeMillis())
23 
24     // ensure the threshold is not in the past
25     val threshold = math.max(cal.getTimeInMillis(), curNow)
26     cal.setTimeInMillis(threshold)
27 
28     // apply configured check time
29     cal.set(Calendar.HOUR_OF_DAY, refresherConfig.hourOfDay)
30     cal.set(Calendar.MINUTE, refresherConfig.minuteOfDay)
31     cal.set(Calendar.SECOND, 0)
32     cal.set(Calendar.MILLISECOND, 0)
33 
34     if (cal.getTimeInMillis() < threshold) {
35       cal.add(Calendar.DAY_OF_MONTH, 1)
36     }
37 
38     cal.getTime()
39   }
40 }

42 class CertificateRefresher(
43   override val bundleContext : BundleContext,
44   certMgr : CertificateManagerImpl,
45   cfg : RefresherConfig,
46   scope : CapsuleScope
47 ) extends Capsule with ServiceConsuming {
48 
49   private[this] val log = Logger[CertificateRefresher]
50 
51   private[this] val timerName = "refresh-certificate"
52   private[this] val timer = new Timer(timerName, true)
53 
54   override def start() : Unit = scheduleRefresh(cfg)
55 
56   override def stop() : Unit = {
57     log.debug(s"Cancelling timer [$timerName]")
58     timer.cancel()
59   }
60 
61   /**
62    * Schedule the next refresh attempt for the given certificate.
63    * If the Domino capsule stops, this timer will also be cancelled.
64    */
65   def scheduleRefresh(refresherConfig : RefresherConfig) : Unit = {
66 
67     certMgr.nextCertificateTimeout().get.foreach { timeOut =>
68       val nextScheduleTime = CertificateRefresher.nextRefreshScheduleTime(timeOut, refresherConfig)
69       val task = new RefreshTask(certMgr, refresherConfig)
70       log.debug(s"Scheduling new timer task with timer [$timerName] to start at [$nextScheduleTime]")
71       timer.schedule(task, nextScheduleTime)
72     }
73   }
74 
75   /**
76    * This task tries to refresh a certificate.
77    * If positive, depending on config, the new certificate is re-published as SslContextProvider or the whole OSGi container will be restarted.
78    */
79   class RefreshTask(certMgr : CertificateManagerImpl, refresherConfig : RefresherConfig) extends TimerTask {
80     override def run() : Unit = {
81       log.debug(s"About to start refresh timer task. Trying to update certificate(s)")
82       // request new cert
83       certMgr.checkCertificates() match {
84         case Failure(e) =>
85           log.debug(e)("Automatic certifcate refresh failed. Continuing with old SslContextProvider")
86           scheduleRefresh(refresherConfig)
87 
88         case Success(None) =>
89         // do nothing
90 
91         case Success(Some(newKs)) =>
92           if (newKs.changedAliases.isEmpty) {
93             // no cert update
94             log.debug("Automatic certificate refresh could not obtain an updated certificate. Continuing with old SslContextProviver")
95             scheduleRefresh(refresherConfig)
96           } else {
97             // cert update
98             log.info(s"Automatic certificate refresh returned new certificate(s)")
99             refresherConfig.onRefreshAction match {
100               case RefresherConfig.Refresh =>
101 
102                 log.info("About to remove old SslContextProvider from registry")
103                 scope.stop()
104 
105                 log.info("Registering new SslContextProvider for new KeyStore")
106                 certMgr.registerSslContextProvider()
107                 scheduleRefresh(refresherConfig)
108 
109               case RefresherConfig.Restart =>
110                 withService[FrameworkService, Unit] {
111                   case Some(frameworkService) =>
112                     // we want to restart the container, so no de- and re-registration of the ssl context
113                     log.warn("Requesting framework restart")
114                     frameworkService.restartContainer(
115                       "The certificate required for the SSL context was refreshed. An restart is required to cleanly use the new certificate.",
116                       saveLogs = true
117                     )
118 
119                   case None =>
120                     log.error("Could not acquire a FrameworkService to restart the OSGi container. Skipping certificate refresh.")
121                     scheduleRefresh(refresherConfig)
122                 }
123             }
124           }
125       }
126     }
127   }
