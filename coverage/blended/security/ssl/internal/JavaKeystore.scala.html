30   def saveKeyStore(ms : MemoryKeystore) : Try[MemoryKeystore] = Try {
31 
32     val ks : KeyStore = loadKeyStoreFromFile().get
33 
34     ms.certificates.filter(_._2.change.changed).foreach {
35       case (alias, cert) =>
36         keypass match {
37           case None =>
38             ks.setCertificateEntry(alias, cert.chain.last)
39           case Some(pwd) =>
40             cert.privateKey match {
41               case None    => throw new Exception(s"Certificate for [${cert.subjectPrincipal}] is missing the private key")
42               case Some(k) => ks.setKeyEntry(alias, k, pwd, cert.chain.toArray)
43             }
44         }
45     }
46 
47     saveKeyStoreToFile(ks).get
48     MemoryKeystore(ms.certificates.mapValues(_.copy(change = CertificateChange.Unchanged)))
49   }

51   private[ssl] def loadKeyStoreFromFile() : Try[KeyStore] = Try {
52 
53     log.info(s"Initializing key store of type [$storetype] from file [${keystore.getAbsolutePath()}] ...")
54 
55     val ks = KeyStore.getInstance(storetype)
56 
57     if (keystore.exists()) {

59       try {
60         ks.load(fis, storepass)

63       }
64     } else {

66       // scalastyle:off null

74   private[ssl] def saveKeyStoreToFile(ks : KeyStore) : Try[KeyStore] = Try {

88   // Extract a single certificate from the underlying keystore
89   // If a keypass is set, we will also extract the private key of the certificate
90   private[ssl] def extractCertificate(ks : KeyStore, alias : String) : Try[CertificateHolder] = Try {
91 
92     val chain : List[X509Certificate] = Option(ks.getCertificateChain(alias)) match {
93       case None => Option(ks.getCertificate(alias)) match {
94         case None =>
95           throw new Exception(s"Certificate for alias [$alias] not found.")
96         case Some(c) => List(c.asInstanceOf[X509Certificate])
97       }
98       case Some(c) => c.toList.map(_.asInstanceOf[X509Certificate])
99     }
100 
101     val pubKey : PublicKey = chain.head.getPublicKey()
102     val privKey : Option[PrivateKey] = keypass.map { pwd =>
103       ks.getKey(alias, pwd).asInstanceOf[PrivateKey]
104     }
105 
106     CertificateHolder.create(publicKey = pubKey, privateKey = privKey, chain = chain).get
107   }

109   private[ssl] def memoryKeystore(ks : KeyStore) : Try[MemoryKeystore] = Try {
110 
111     val certs : Map[String, CertificateHolder] = ks.aliases().asScala.map { alias =>
112       (alias, extractCertificate(ks, alias).get)
113     }.toMap
114 
115     val result = MemoryKeystore(certs)
116 
117     if (result.consistent) {
118       result
119     } else {
120       throw new InconsistentKeystoreException(s"KeyStore [${keystore.getAbsolutePath()}] is inconsistent.")
121     }
122   }
