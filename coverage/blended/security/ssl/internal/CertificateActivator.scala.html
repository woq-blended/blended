17   private[this] def setupSelfSignedProvider(cfg : Config, ctCtxt : ContainerContext) : Unit = {
18     // Should we provide a CertifacteProvider with a self-signed certificate?
19     cfg.getConfigOption("selfsigned") match {
20       case Some(selfCfg) =>
21         val selfSignedProvider = new SelfSignedCertificateProvider(SelfSignedConfig.fromConfig(selfCfg, ctCtxt))
22         selfSignedProvider.providesService[CertificateProvider](Map(
23           "provider" -> "default"
24         ))
25       case None =>
26         log.warn("No config entry 'selfsigned' found. Skipping provision of SelfSignedCertificateProvider")
27     }
28   }

30   private[this] def setupCertificateManager(
31     mgrConfig : CertificateManagerConfig
32   ) : Unit = {
33 
34     def waitForProvider(providerNames : List[String], provider : Map[String, CertificateProvider]) : Unit = {
35       providerNames match {
36         case Nil =>
37           val mgr = new CertificateManagerImpl(bundleContext, capsuleContext, mgrConfig, provider)
38           mgr.providesService[CertificateManager]
39           addCapsule(mgr)
40         case head :: tail =>
41           log.info(s"Waiting for certificate provider [$head]")
42           whenAdvancedServicePresent[CertificateProvider](s"(provider=$head)") { p =>
43             log.info(s"Certificate provider [$head] available.")
44             waitForProvider(tail, provider + (head -> p))
45           }
46       }
47     }
48 
49     val distinctProviderNames : List[String] =
50       (mgrConfig.providerList ::: mgrConfig.certConfigs.map(_.provider)).distinct
51 
52     waitForProvider(distinctProviderNames, Map.empty)
53   }
54

67 
68         whenServicePresent[MBeanServer] { server =>

70           server.registerMBean(info, objName)
71

74           }
75         }
