26 trait CertificateChecker {

28   /**
29    * Check a single certificate and determine the result messages for the validity check of that

49 class RemainingValidityChecker(minValidDays : Int) extends CertificateChecker {
50 
51   private val millisPerDay : Long = 1.day.toMillis
52 
53   override def checkCertificate(alias : String, cert : CertificateHolder) : Option[CertificateCheckResult] = {
54 
55     val certInfo : X509CertificateInfo = X509CertificateInfo(cert.chain.head)
56     val remaining : Long = certInfo.notAfter.getTime() - System.currentTimeMillis()
57 
58     Some(
59       if (remaining <= minValidDays * millisPerDay) {
60         val msg = s"Certificate for [${cert.chain.head.getSubjectX500Principal()}] is about to expire in ${remaining.toDouble / millisPerDay} days"
61         CertificateCheckResult(alias, cert, Seq((ResultLevel.WARN, msg)))
62       } else {
63         val msg = s"Certificate for [${cert.chain.head.getSubjectX500Principal()}] is still valid"
64         CertificateCheckResult(alias, cert, Seq((ResultLevel.INFO, msg)))
65       }
66     )
67   }
