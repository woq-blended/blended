39   private[internal] def registerSslContextProvider() : CapsuleScope = capsuleContext.executeWithinNewCapsuleScope {
40 
41     val sslCtxtProvider = new SslContextProvider()
42 
43     javaKeystore.map(_.loadKeyStoreFromFile().get).foreach { ks =>
44       log.debug("Registering SslContextProvider type=server")
45       val serverCtxt : SSLContext = sslCtxtProvider.serverContext(ks, cfg.keystoreCfg.get.keyPass.toCharArray())
46       SSLContext.setDefault(serverCtxt)
47 
48       serverCtxt.providesService[SSLContext](Map("type" -> "server"))
49       log.info(s"Server SSLContext : ${new SslContextInfo(serverCtxt, cfg.validCypherSuites).toString()}")
50     }
51 
52     sslCtxtProvider.clientContext.providesService[SSLContext](Map("type" -> "client"))
53   }

106   /**
107    * @return When successful, an updated server keystore
108    */
109   override def checkCertificates() : Try[Option[MemoryKeystore]] = Try {
110 
111     // for all configured providers update the trusted certificates
112     if (cfg.maintainTruststore) {
113       providerMap.foreach {
114         case (_, provider) =>
115           provider.rootCertificates() match {
116             case Success(certs) => certs.foreach { ms =>
117               new TrustStoreRefresher(ms).refreshTruststore() match {
118                 case Success(_) =>
119                 case Failure(t) =>
120                   log.warn(s"Unable to update trust store : [${t.getMessage()}]")
121               }
122             }
123             case Failure(t) =>
124               log.warn(s"Unable to update trust store : [${t.getMessage()}]")
125           }
126       }
127     }
128 
129     // first refresh the server certificates if required
130     log.debug("Loading keystore...")
131     val ks: Option[MemoryKeystore] = loadKeyStore().get
132     log.debug(s"Loaded keystore [$ks]")
133 
134     ks.map { ms =>
135       log.debug(s"Refreshing certificates for keystore [$ms]")
136       val changedKs = ms.refreshCertificates(cfg.certConfigs, providerMap).get
137       log.debug(s"Changed certificate aliases [${changedKs.certificates.collect{
138         case c if c._2.change.changed => c._1 -> c._2.change
139       }}]")
140 
141       log.debug(s"Saving keystore...")
142       val jks = javaKeystore.get
143       jks.saveKeyStore(changedKs) match {
144         case Failure(t) =>
145           log.warn(t)(s"Failed to save keystore to file [${jks.keystore.getAbsolutePath()}] : [${t.getMessage()}]")
146           throw t
147         case Success(_) => changedKs
148       }
149     }
150   }
