12 /**
13  * A simple case class to hold a certificate chain in memory. A certificate chain
14  * in this class is always sorted, so that the ultimate root certificate is at the
15  * end of the chain and the certificate is always at the head of the chain.
16  *
17  * The keys always belong to the certificate at the head of the chain. The private
18  * key is optional - In that case the certificate can not be used as a server side
19  * certificate.
20  *
21  * The constructor is private, so that instances of the class can only be created
22  * with the factory method(s) in the companion object. These methods will create the
23  * sorted chain verify the signatures of each certificate within the chain.
24  */
25 case class CertificateHolder private (
26   publicKey : PublicKey,
27   privateKey : Option[PrivateKey],
28   chain : List[X509Certificate],
29   change : CertificateChange = CertificateChange.Unchanged
30 ) {
31 
32   val subjectPrincipal : Option[X500Principal] = chain.headOption.map(_.getIssuerX500Principal())
33 
34   // retrieve the keyPair of the certificate
35   val keyPair : Option[KeyPair] = privateKey.map(pk => new KeyPair(publicKey, pk))
36 
37   // A simple String representation of the certificate chain
38   override def toString : String = chain.map { c =>
39     X509CertificateInfo(c).toString
40   }.mkString("\n", "\n", "")
41 
42   // A complete dump of the certificate chain as a String
43   def dump : String = {
44     chain.map { c => c.toString() }.mkString("\n" + "*" * 30, "\n\n--- Signed by ---\n\n", "*" * 30)
45   }
46 
47   lazy val keypairValid : Option[Boolean] = privateKey.map { pk =>
48     val enc = Cipher.getInstance("RSA")
49 
50     enc.init(Cipher.ENCRYPT_MODE, pk)
51     val s1 = new Date().toString()
52     val encrypred = enc.doFinal(s1.getBytes())
53 
54     val dec = Cipher.getInstance("RSA")
55     dec.init(Cipher.DECRYPT_MODE, publicKey)
56     val s2 = new String(dec.doFinal(encrypred))
57 
58     s1.equals(s2)
59   }
60 }

76   def create(chain : List[Certificate]) : Try[CertificateHolder] = Try {
77 
78     val x509Chain : List[X509Certificate] = chain.map(_.asInstanceOf[X509Certificate])
79     x509Chain.find { c => c.getSubjectDN().equals(c.getIssuerDN()) } match {
80       case None =>
81         throw new MissingRootCertificateException
82       case Some(root) =>
83         root.verify(root.getPublicKey())
84         val sortedChain : List[X509Certificate] = sort(x509Chain.filter(c => !c.equals(root)))(root :: Nil).get
85         create(
86           publicKey = sortedChain.head.getPublicKey(),
87           privateKey = None,
88           chain = sortedChain
89         ).get
90     }
91   }

93   def create(publicKey : PublicKey, privateKey : Option[PrivateKey], chain : List[Certificate]) : Try[CertificateHolder] = Try {
94 
95     val sortedChain : List[X509Certificate] = {
96 
97       chain.map(_.asInstanceOf[X509Certificate]) match {
98         // chain must not be empty
99         case Nil =>
100           throw new EmptyCertificateChainException
101 
102         case certs => certs.find { c => c.getSubjectDN().equals(c.getIssuerDN()) } match {
103           // chain must have a root certificate
104           case None =>
105             throw new MissingRootCertificateException
106 
107           case Some(root) =>
108             // The root must have signed itself correctly
109             root.verify(root.getPublicKey())
110             // We kick off the sort with the root certificate as a starting point
111             sort(certs.filter(c => !c.equals(root)))(root :: Nil).get
112         }
113       }
114     }
115 
116     CertificateHolder(publicKey, privateKey, sortedChain)
117   }
118 
119   // A test that yields true if and only if the certificate is not self signed AND was signed by
120   // the given principal
121   private def signedBy(issuer : Principal) : X509Certificate => Boolean = c =>
122     !c.getIssuerDN().equals(c.getSubjectDN()) && c.getIssuerDN().equals(issuer)
123 
124   // Helper function to sort the certificates of a given chain so that any certificate in the chain is
125   // signed by it's successor. This implies that the root certificate is always the last element in the list
126   private def sort(remaining : List[X509Certificate])(sorted : List[X509Certificate]) : Try[List[X509Certificate]] = Try {
127     remaining match {
128       // We have visited and sorted all certificates
129       case Nil => sorted
130       // for the head of the already sorted certificates we look for the one that has been signed
131       // by it and prepend it to the list of sorted certificates
132       case rest =>
133         rest.find(signedBy(sorted.head.getSubjectDN())) match {
134           case None =>
135             throw new CertificateChainException(s"No signed certificate found for certificate [${X509CertificateInfo(sorted.head)}]")
136 
137           case Some(next) =>
138             if (sorted.contains(next)) {
139               throw new CertificateChainException("Certificate chain must not contain circular references")
140             } else {
141               // the certificate we have found must be signed by the first element in the sorted list
142               // correctly
143               next.verify(sorted.head.getPublicKey())
144               sort(remaining.filter(c => !c.equals(next)))(next :: sorted).get
145             }
146         }
147     }
148   }
