23 class MemoryTokenStore extends Actor {

25   import TokenStoreMessages._
26 
27   private[this] val tokens : mutable.Map[String, Token] = mutable.Map.empty
28 
29   override def receive : Receive = {
30     case GetToken(id) =>
31       sender() ! tokens.get(id)
32 
33     case RemoveToken(id) =>
34       sender() ! tokens.remove(id)
35 
36     case StoreToken(t : Token) =>
37       val r = sender()
38       if (tokens.get(t.id).isDefined) {
39         r ! Failure(new Exception(s"token with id [${t.id}] already exists."))
40       } else {
41         tokens += (t.id -> t)
42         r ! Success(t)
43       }
44 
45     case ListTokens      => sender() ! tokens.values.toSeq
46 
47     case RemoveAllTokens => tokens.clear()
48   }
