29   private[this] def auth(creds : BasicHttpCredentials) : Option[Subject] = {
30 
31     val cbHandler = new CallbackHandler {
32       override def handle(callbacks : Array[Callback]) : Unit = {
33         callbacks.foreach { cb : Callback =>
34           cb match {
35             case nameCallback : NameCallback    => nameCallback.setName(creds.username)
36             case pwdCallback : PasswordCallback => pwdCallback.setPassword(creds.password.toCharArray())
37             case other                          => throw new UnsupportedCallbackException(other, "The submitted callback is not supported")
38           }
39         }
40       }
41     }
42 
43     val lc : LoginContext = new LoginContext("blended", cbHandler)
44 
45     try {
46       lc.login()
47       Some(lc.getSubject())
48     } catch {
49       case t : Throwable =>
50         log.error(t)(s"Login failed for [${creds.username}]")
51         None
52     }
53   }

55   private[this] def myUserPassAuthenticator(credentials : Option[HttpCredentials]) : Future[AuthenticationResult[Subject]] =
56     Future {
57       credentials match {
58         case Some(creds : BasicHttpCredentials) => auth(creds) match {
59           case Some(s) => AuthenticationResult.success(s)
60           case None    => AuthenticationResult.failWithChallenge(challenge)
61         }
62         case _ => AuthenticationResult.failWithChallenge(challenge)
