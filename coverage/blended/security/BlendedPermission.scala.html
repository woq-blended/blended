38   /**
39    * Merge this permission with another permission.
40    *
41    * If one of the permissions to be merged has an empty permission class, the result wil always
42    * have an empty permission class as well, in other words the result will not grant access to
43    * any object.
44    *
45    * Also if the permission classes of the two permissions are different, the result will have
46    * an empty permission class.
47    *
48    * @param other The permission to merged.
49    * @return The resulting permission.
50    */
51   def merge(other : BlendedPermission) : BlendedPermission = {
52     (permissionClass, other.permissionClass) match {
53       case (None, _) => BlendedPermission(None)
54       case (_, None) => BlendedPermission(None)
55       case (Some(c1), Some(c2)) => if (c1 != c2) {
56         BlendedPermission(None)
57       } else {
58 
59         // if for any given property one of the permissions does not have a restriction defined
60         // the result will also not restrict on that property.
61 
62         val props = properties
63           // we need the properties defined in both permissions
64           .filterKeys(k => other.properties.get(k).isDefined)
65           // then we combine the properties of both permissions
66           .map { case (k, v) => (k, (v ++ other.properties.getOrElse(k, sci.Seq.empty)).distinct) }
67 
68         BlendedPermission(
69           permissionClass = permissionClass,
70           properties = props
71         )
72       }
73     }
74   }

90   def merge(permissions : Seq[BlendedPermission]) : BlendedPermission = permissions match {
91     case Seq()          => BlendedPermission(None)
92     case h +: Seq()     => h
93     case h +: s +: rest => merge(h.merge(s) +: rest)
