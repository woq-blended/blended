32   // obtain the initial LDAP Context
33   private[this] lazy val dirContext : DirContext = {
34 
35     try {
36       val env : mutable.Map[String, String] = mutable.Map(
37         Context.INITIAL_CONTEXT_FACTORY -> classOf[LdapCtxFactory].getName(),
38         Context.PROVIDER_URL -> ldapCfg.url,
39         Context.SECURITY_AUTHENTICATION -> "simple"
40       )
41 
42       env ++= ldapCfg.systemUser.map(u => Context.SECURITY_PRINCIPAL -> u)
43       env ++= ldapCfg.systemPassword.map(u => Context.SECURITY_CREDENTIALS -> u)
44 
45       new InitialDirContext(new util.Hashtable[String, Object](env.asJava))
46     } catch {
47       case NonFatal(t) =>
48         log.error(t)(t.getMessage())
49         throw new LoginException(t.getMessage())
50     }
51   }

53   @throws[LoginException]
54   override def doLogin() : Boolean = {
55 
56     // First of all we will obtain the directory context
57

76   @throws[LoginException]
77   private[this] def validateUser() : String = {
78 
79     try {
80       val (user, pwd) = extractCredentials()
81 
82       val constraint = new SearchControls()
83       constraint.setSearchScope(SearchControls.SUBTREE_SCOPE)
84 
85       val userSearchFormat = new MessageFormat(ldapCfg.userSearch)
86       val filter = userSearchFormat.format(Array(doRFC2254Encoding(user)))
87 
88       val r = dirContext.search(ldapCfg.userBase, filter, constraint)
89 
90       LdapSearchResult(r).result match {
91         case Nil => throw new LoginException(s"User [$user] not found in LDAP.")
92         case head :: tail =>
93           if (tail.nonEmpty) {
94             log.warn(s"Search for user [$user] returned [${1 + tail.length}] records, using first record only.")
95           }
96 
97           val name = head.getNameInNamespace()
98 
99           dirContext.addToEnvironment(Context.SECURITY_PRINCIPAL, name)
100           dirContext.addToEnvironment(Context.SECURITY_CREDENTIALS, pwd)
101 
102           // scalastyle:off null
103           dirContext.getAttributes("", null)
104           // scalastyle:on null
105           log.info(s"User [$user] authenticated with LDAP name [$name]")
106           name
107       }
108     } catch {
109       case t : Throwable =>
110         log.error(t)(t.getMessage())
111         throw new LoginException(t.getMessage())
112     } finally {
113       ldapCfg.systemUser match {
114         case None    => dirContext.removeFromEnvironment(Context.SECURITY_PRINCIPAL)
115         case Some(u) => dirContext.addToEnvironment(Context.SECURITY_PRINCIPAL, u)
116       }
117       ldapCfg.systemPassword match {
118         case None    => dirContext.removeFromEnvironment(Context.SECURITY_CREDENTIALS)
119         case Some(p) => dirContext.addToEnvironment(Context.SECURITY_CREDENTIALS, p)
120       }
121     }
122   }

124   @throws[LoginException]
125   override def getGroups(member : String) : List[String] = {
126     val constraint = new SearchControls()
127     constraint.setSearchScope(SearchControls.SUBTREE_SCOPE)
128 
129     val groupSearchFormat = new MessageFormat(ldapCfg.groupSearch)
130     val filter = groupSearchFormat.format(Array(doRFC2254Encoding(member)))
131 
132     val r = dirContext.search(ldapCfg.groupBase, filter, constraint)
133 
134     LdapSearchResult(r).result.map { sr =>
135       sr.getAttributes().get(ldapCfg.groupAttribute).get().toString()
136     }
137   }

140   // convenience method to encode a filter string
141   private[this] def doRFC2254Encoding(inputString : String) : String = inputString match {
142     case s if s.isEmpty()        => ""
143     case s if s.startsWith("\\") => "\\5c" + doRFC2254Encoding(s.substring(1))
144     case s if s.startsWith("*")  => "\\2a" + doRFC2254Encoding(s.substring(1))
145     case s if s.startsWith("(")  => "\\28" + doRFC2254Encoding(s.substring(1))
146     case s if s.startsWith(")")  => "\\29" + doRFC2254Encoding(s.substring(1))
147     case s if s.startsWith("\u0000") => "\\00" + doRFC2254Encoding(s.substring(1))
148     case s                       => s.substring(0, 1) + doRFC2254Encoding(s.substring(1))
149   }
