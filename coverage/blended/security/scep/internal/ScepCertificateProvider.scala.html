60   override def refreshCertificate(existing : Option[CertificateHolder], cnProvider : CommonNameProvider) : Try[CertificateHolder] = {
61     log.info(s"Trying to refresh the server certificate via SCEP from [${cfg.url}]")
62     existing match {
63       case None =>
64         log.info("Obtaining initial server certificate from SCEP server.")
65         enroll(None, cnProvider)
66       case Some(c) =>
67         log.info("Refreshing certificate previously obtained from SCEP server.")
68         enroll(Some(c), cnProvider)

94   private[this] def enroll(
95     inCert : Option[CertificateHolder],
96     cnProvider : CommonNameProvider
97   ) : Try[CertificateHolder] = Try {
98 
99     val reqCert : CertificateHolder = inCert match {
100       case None =>
101         log.info(s"Requesting initial certificate from SCEP server at [${cfg.url}].")
102         val selfSigned = selfSignedCertificate(cnProvider).get
103         selfSigned
104       case Some(c) =>
105         log.info(s"Refreshing certificate from SCEP server at [${cfg.url}].")
106         c
107     }
108 
109     val privKey = reqCert.privateKey.get
110 
111     val csrBuilder = new JcaPKCS10CertificationRequestBuilder(new X500Principal(cnProvider.commonName().get), reqCert.publicKey)
112     csrBuilder.addAttribute(PKCSObjectIdentifiers.pkcs_9_at_challengePassword, new DERPrintableString(cfg.scepChallenge))
113 
114     if (cnProvider.alternativeNames().get.nonEmpty) {
115       val altNames : Array[GeneralName] = cnProvider.alternativeNames().get.map { n =>
116         log.info(s"Adding alternative dns name [$n] to SCEP certificate request.")
117         new GeneralName(GeneralName.dNSName, n)
118       }.toArray
119 
120       val names = new GeneralNames(altNames)
121       val extGen = new ExtensionsGenerator()
122       val sanExt = extGen.addExtension(Extension.subjectAlternativeName, false, names)
123 
124       csrBuilder.addAttribute(PKCSObjectIdentifiers.pkcs_9_at_extensionRequest, extGen.generate())
125     }
126 
127     val csrSignerBuilder = new JcaContentSignerBuilder(cfg.csrSignAlgorithm)
128     val csrSigner = csrSignerBuilder.build(privKey)
129     val csr = csrBuilder.build(csrSigner)
130 
131     log.debug(s"csr: ${dumpCsr(csr)}")
132 
133     val response = scepClient.enrol(reqCert.chain.head, privKey, csr)
134 
135     // TODO: Active wait is baaaad
136     while (response.isPending()) {
137       log.info(s"Waiting for PKI response from [${cfg.url}]")
138       Thread.sleep(1000)
139     }
140 
141     if (response.isFailure()) {
142       val info = response.getFailInfo()
143       log.error(s"Certificate provisioning failed: [$info]")
144       sys.error(info.toString)
145     } else {
146       val store = response.getCertStore()
147       val certs : List[Certificate] = store.getCertificates(null).asScala.toList
148 
149       log.info(s"Retrieved [${certs.length}] certificates from [${cfg.url}].")
150 
151       CertificateHolder.create(
152         publicKey = reqCert.publicKey,
153         privateKey = Some(privKey),
154         chain = certs
155       ).get
156     }
157   }
