45   /**
46    * Run the scep client app.
47    *
48    * @param args
49    * @throws ExitAppException The signal the exit code of the application
50    */
51   def run(args : Array[String]) : Unit = {
52     val cmdline = new Cmdline()
53     val cp = new CmdlineParser(cmdline)
54     cp.setProgramName("java -jar scep-client.jar")
55     cp.setAboutLine("Standalone SCEP client, which can create and update Java key stores from a remote SCEP server.")
56     try {
57       cp.parse(args : _*)
58     } catch {
59       case e : CmdlineParserException => throw new ExitAppException(ExitCode.InvalidCmdline, Option(e.getLocalizedMessage()), e)
60     }
61 
62     if (cmdline.help || args.isEmpty) {
63       cp.usage()
64       throw new ExitAppException(ExitCode.Ok)
65     }
66 
67     val salt = cmdline.salt.getOrElse("scep-client")
68 
69     cmdline.password.foreach { pass =>
70       println(new PasswordHasher(salt).password(pass))
71     }
72 
73     cmdline.infoFile.foreach { infoFile =>
74       readInfoFile(infoFile)
75     }
76 
77     cmdline.csrFile.foreach { csrFile =>
78       readCsrFile(csrFile)
79     }
80 
81     if (cmdline.refreshCerts) {
82       refreshCert(
83         salt,
84         timeout = Duration(cmdline.timeout, TimeUnit.SECONDS),
85         baseDir = new File(cmdline.baseDir.getOrElse(".")).getAbsoluteFile(),
86         cmdline.expectRefresh
87       )
88     }
89

92   def readInfoFile(infoFileName : String) : Unit = {
93     val file = new File(infoFileName).getAbsoluteFile()
94     if (!file.exists() || !file.isFile()) {
95       throw new RuntimeException(s"File does not exists: ${file}")
96     }
97     log.debug(s"About to parse file ${file}")
98     val reader = new FileReader(file)
99     val parser = new PEMParser(reader)
100     val readObject = parser.readObject()
101     log.debug(s"Parsed object [${readObject}]")
102     readObject match {
103       case encKeyPair : PEMEncryptedKeyPair =>
104         log.debug(s"Got an encrypted key pair - need a password to decrypt")
105       case keyPair : PEMKeyPair =>
106         log.debug(s"Got an (unencrypted) key pair - no password needed")
107       case cert : X509CertificateHolder =>
108         log.debug(s"Got an certificate holder. subject [${cert.getSubject()}], version [${cert.getVersionNumber()}], extensions [${cert.getExtensions()}]")
109       case csr : PKCS10CertificationRequest =>
110         log.debug(s"Got a CSR. subject [${csr.getSubject()}], " +
111           s"attributes [${csr.getAttributes().map(a => s"Attribute(type [${a.getAttrType()}], values [${a.getAttrValues}])").mkString(", ")}]")
112     }
113   }

115   def readCsrFile(csrFile : String) : Unit = {
116     val file = new File(csrFile).getAbsoluteFile()
117     if (!file.exists() || !file.isFile()) {
118       throw new RuntimeException(s"File does not exist: ${file}")
119     }
120     log.debug(s"About to parse file ${file}")
121     val reader = new FileReader(file)
122     val parser = new PEMParser(reader)
123     val readObject = parser.readObject()
124     log.debug(s"Parsed object [${readObject}]")
125     readObject match {
126 
127       case csr : PKCS10CertificationRequest =>
128         val subjectPublicKeyInfo = csr.getSubjectPublicKeyInfo()
129         val details = Seq(
130           "subject" -> csr.getSubject(),
131           "attributes" -> csr.getAttributes().toList,
132           "signature" -> csr.getSignature().toList,
133           "signatureAlgorithm" -> csr.getSignatureAlgorithm(),
134           "subjectPublicKeyInfo" -> subjectPublicKeyInfo,
135           "subjectPublicKeyInfo.algorithm" -> subjectPublicKeyInfo.getAlgorithm(),
136           "subjectPublicKeyInfo.algorithm.algorithm" -> subjectPublicKeyInfo.getAlgorithm().getAlgorithm(),
137           "subjectPublicKeyInfo.algorithm.parameters" -> subjectPublicKeyInfo.getAlgorithm().getParameters(),
138           "subjectPublicKeyInfo.publicKeyData" -> subjectPublicKeyInfo.getPublicKeyData()
139         )
140 
141         log.debug(s"Got a CSR. Info:\n  ${details.map(t => s"${t._1}: ${t._2}").mkString("\n  ")}")
142 
143       case other =>
144         val msg = s"File [${csrFile}] has no supported CSR file format"
145         log.error(msg)
146         throw new ExitAppException(ExitCode.Error, Some(msg))
147     }
148   }

150   def refreshCert(salt : String, timeout : Duration, baseDir : File, expectCertChange: Boolean) : Unit = {
151     implicit val executionContext = scala.concurrent.ExecutionContext.global
152     val refresher = new CertRefresher(salt, baseDir)
153 
154     val checking = refresher.checkCert()
155 
156     try {
157       Await.ready(checking, atMost = timeout).value.get match {
158         case Success(r) =>
159           log.debug(s"Successfully gathered certificates")
160           refresher.stop()
161 
162           if (expectCertChange && r.changedAliases.isEmpty) {
163             throw new ExitAppException(
164               exitCode = ExitCode.NoCertsRefreshed,
165               errMsg = Some(s"Successfully gathered unchanged certificates.")
166             )
167           }
168 
169           throw new ExitAppException(
170             exitCode = ExitCode.Ok,
171             errMsg = Some(s"Successfully refreshed certificates")
172           )
173 
174         case Failure(e) =>
175           refresher.stop()
176           throw new ExitAppException(
177             exitCode = ExitCode.Error,
178             errMsg = Some(s"Error: Could not refresh certificates.\nReason: ${e.getMessage()}\nSee log file for details."),
179             cause = e
180           )
181       }
182     } catch {
183       case e : TimeoutException =>
184         refresher.stop()
185         throw new ExitAppException(
186           exitCode = ExitCode.Timeout,
187           errMsg = Some(s"Error: Could not refresh certificates.\nReason: Timeout after [${timeout}]\nSee log file for details."),
188           cause = e
189         )
190     }
191   }
