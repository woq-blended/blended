68 
69     val bis = new BufferedInputStream(new ByteArrayInputStream(content))
70     val bos = new ByteArrayOutputStream()

73       log.debug("Trying to use GZIP compression")
74       new GZIPInputStream(new BufferedInputStream(new ByteArrayInputStream(content)))
75     } catch {
76       case NonFatal(e) =>

92     }

108   def getFiles(dirName : String, filter : FileFilter, recursive : Boolean) : List[File] = {
109 
110     val f : File = new File(dirName)
111 
112     if (f.exists) {
113       if (f.isFile()) {
114         List(f.getAbsoluteFile())
115       } else {
116         f.listFiles(filter).flatMap {
117           x =>
118             if (x.isFile) {
119               List(x.getAbsoluteFile())
120             } else {
121               if (recursive) {
122                 getFiles(x.getAbsolutePath, filter, recursive)
123               } else {
124                 List.empty
125               }
126             }
127         }.toList
128       }
129     } else {
130       List.empty
131     }
132   }
