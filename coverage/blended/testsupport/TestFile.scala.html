17     var failure = false

138   def deleteRecursive(files : File*) : Unit = files.foreach { file =>
139     if (file.isDirectory()) {
140       file.listFiles() match {
141         case null  =>
142         case files => deleteRecursive(files : _*)

148   def listFilesRecursive(file : File) : Seq[String] = {
149     def list(file : File, prefix : String) : List[String] = {
150       file.listFiles() match {
151         case null => Nil
152         case files =>
153           files.toList.flatMap { f =>
154             val newPrefix = if (prefix.isEmpty()) f.getName() else s"${prefix}/${f.getName()}"
155             newPrefix :: list(f, newPrefix)
156           }
157       }
158     }

163   def expectFiles(dir : File, exact : Boolean, files : String*) : Unit = {
164     val existing = listFilesRecursive(dir).toSet
165     val expected = files.toSet
166     val missing = expected.diff(existing)
167     if (!missing.isEmpty) {
168       throw new AssertionError(s"${missing.size} missing files in directory [${dir}]: [${missing.mkString(",")}]")
169     }
170     if (exact) {
171       val tooMany = existing.diff(expected)
172       if (!tooMany.isEmpty) {
173         throw new AssertionError(s"${tooMany.size} unexpected files in directory [${dir}]: [${tooMany.mkString(",")}]")
174       }
175     }
