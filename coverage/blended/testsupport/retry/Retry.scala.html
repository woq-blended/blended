11   /**
12    * Executed and in case of an failure retries an operation `op`. As long as there are retries left, the next retry starts after `delay` times.
13    *
14    * @param delay The time between a failure and the next retry.
15    * @param retries The max count of retries, before giving up.
16    * @param onRetry Action to be run before a retry.
17    * @param op The operation to be executed and, iff failed, retried.
18    * @param ec ExecutionContext to run the inner futures with.
19    * @param s The Scheduler used to schedule the next retry.
20    *
21    * @return The Future containing the result of `op` or the last failure.
22    */
23   def retry[T](
24     delay : FiniteDuration,
25     retries : Int,
26     onRetry : (Int, Throwable) => Unit = (n, e) => Logger[Retry.type].debug(e)(s"Retrying after failed execution (${n} retries left)")
27   )(
28     op : => T
29   )(implicit ec : ExecutionContext, s : Scheduler) : Future[T] =
30     Future { op } recoverWith {
31       case e : Throwable if retries > 0 => akka.pattern.after(delay, s)({
32         onRetry(retries - 1, e)
33         retry(delay, retries - 1, onRetry)(op)(ec, s)
34       })
35     }

37   /**
38    * Executed and in case of an failure retries an operation `op`. As long as there are retries left, the next retry starts after `delay` times.
39    *
40    * @param delay The time between a failure and the next retry.
41    * @param retries The max count of retries, before giving up.
42    * @param onRetry Action to be run before a retry.
43    * @param finalDelay The overall delay forwarded to [[Await]], before failing the whole retry block.
44    * @param op The operation to be executed and, iff failed, retried.
45    * @param ex ExecutionContext to run the inner futures with.
46    * @param s The Scheduler used to schedule the next retry.
47    *
48    * @return The result of `op` or throws an exception.
49    */
50   def unsafeRetry[T](
51     delay : FiniteDuration,
52     retries : Int,
53     onRetry : (Int, Throwable) => Unit = (n, e) => Logger[Retry.type].debug(e)(s"Retrying after failed execution (${n} retries left)"),
54     finalDelay : Option[FiniteDuration] = None
55   )(
56     op : => T
57   )(implicit ec : ExecutionContext, s : Scheduler) : T = {
58     val res = retry(delay, retries, onRetry)(op)(ec, s)
59     Await.result(res, finalDelay.getOrElse(delay * retries + 2.seconds))
60   }
