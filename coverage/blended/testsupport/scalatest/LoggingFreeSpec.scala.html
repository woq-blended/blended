8 /**
9  * Same as [[org.scalatest.FreeSpecLike]] but log the start and the end of each test case to SLF4j in Debug level.
10  *
11  * If you want also see the last thrown exception of a test case (this is most probably a failed assertion),
12  * you can use [[LoggingFreeSpec.logException]].
13  *
14  * Example:
15  * {{{
16  *   "A failing test (that also appears in log file)" in logException {
17  *     assert(1 == 0)
18  *   }
19  * }}}
20  *
21  * @see LoggingFreeSpec
22  */
23 trait LoggingFreeSpecLike extends FreeSpecLike {
24 
25   abstract protected override def runTest(testName : String, args : Args) : Status = {
26     val log : Logger = Logger[this.type]
27     log.info(s"Starting test case: ${testName}")
28     var reported = false
29     try {
30       val status = super.runTest(testName, args)
31       log.info(s"${if (status.succeeds()) "Finished" else "Failed"} test case: ${testName}\n" + ("-" * 80))
32       reported = true
33       status
34     } finally {
35       if (!reported) log.info(s"Finished test case: ${testName}\n" + ("-" * 80))
36     }
37   }
38 
39   def logException[T](f : => T) : T = try {
40     f
41   } catch {
42     case NonFatal(e) =>
43       Logger[this.type].error(e)("Exception caught")
44       throw e
45   }
46 
47 }
