28 case class BridgeStreamConfig(

30   inbound : Boolean,
31   // the JMS connection factory to be used to consume messages

33   // Jms Destination to consume messages from

82   private class BridgeDestinationResolver(settings : JmsProducerSettings) extends MessageDestinationResolver(settings) {
83     override def destination(env : FlowEnvelope) : Try[JmsDestination] = {
84 
85       val transId : String = env.header[String](headerConfig.headerTransId).getOrElse("UNNKNOWN")
86 
87       val dest : Try[JmsDestination] = (env.header[String](headerConfig.headerBridgeVendor), env.header[String](headerConfig.headerBridgeProvider)) match {
88         case (Some(v), Some(p)) =>
89           // if the target connection factory is the connection factory targeted by the message, we will just
90           // resolve the destination, otherwise we will make another hop via an outbound bridge queue
91           if (v == bridgeCfg.toCf.vendor && p == bridgeCfg.toCf.provider) {
92             val result = super.destination(env)
93             bridgeLogger.debug(s"Routing message to [$v:$p][$result]")
94             result
95           } else {
96             val result : Try[JmsDestination] = Try {
97               val intern : BridgeProviderConfig = bridgeCfg.registry.internalProvider.get
98               if (intern.vendor == v && intern.provider == p) {
99                 intern.outbound
100               } else {
101                 JmsDestination.create(s"${intern.outbound.asString}.$v.$p").get
102               }
103             }
104             bridgeLogger.debug(s"Rerouting outbound message [$transId] via [$result]")
105             result
106           }
107         case _ =>
108           val result = super.destination(env)
109           bridgeLogger.debug(s"Routing message to [${settings.connectionFactory.vendor}:${settings.connectionFactory.provider}][$result]")
110           result
111       }
112 
113       dest match {
114         case Success(d) =>
115           Success(d)
116         case Failure(t) =>
117           log.debug(s"Failed to resolve target destination for [$transId] from [${destHeader(headerConfig.prefix)}] ")
118           bridgeCfg.outboundAlternates.find{ s => env.header[String](s).isDefined } match {
119             case None =>
120               log.warn(s"Failed to resolve alternative destination for [$transId] from [${bridgeCfg.outboundAlternates.mkString(",")}]")
121               throw t
122             case Some(d) =>
123               val dest : String = env.header[String](d).get
124               log.warn(s"Resolved destination for [$transId] to [$dest] from property[$d]")
125               JmsDestination.create(dest)
126           }
127       }
128     }
129   }

156   protected val (isInbound, internalCf) : (Boolean, Try[IdAwareConnectionFactory]) = {
157 
158     if ((bridgeCfg.fromCf.vendor, bridgeCfg.fromCf.provider) == internalId) {
159       (false, Success(bridgeCfg.fromCf))
160     } else if ((bridgeCfg.toCf.vendor, bridgeCfg.toCf.provider) == internalId) {

167   // The jmsSource provides the inbound stream of FlowEnvelopes that need to be passed onwards
168   // to the target JMS destination
169   protected def jmsSource : Source[FlowEnvelope, NotUsed] = {
170 
171     // configure the consumer
172     val srcSettings = JmsConsumerSettings(
173       log = envLogger,
174       connectionFactory = bridgeCfg.fromCf,
175       headerCfg = bridgeCfg.headerCfg,
176       logLevel = _ => if (isInbound) LogLevel.Info else LogLevel.Debug
177     )
178       .withAcknowledgeMode(AcknowledgeMode.ClientAcknowledge)
179       .withDestination(Some(bridgeCfg.fromDest))
180       .withSessionCount(bridgeCfg.listener)
181       .withSelector(bridgeCfg.selector)
182       .withSubScriberName(bridgeCfg.subscriberName)
183 
184     val stats : Graph[FlowShape[FlowEnvelope, FlowEnvelope], NotUsed] = FlowProcessor.startStats(
185       name = streamId + "-stats",
186       log = envLogger,
187       component = jmxComponent,
188       subComp = Map(
189         "direction" -> (if (isInbound) "inbound" else "outbound"),
190         "jmsvendor" -> bridgeCfg.fromCf.vendor,
191         "provider" -> bridgeCfg.fromCf.provider,
192         "srcdest" -> bridgeCfg.fromDest.asString
193       ) ++ bridgeCfg.toDest.map(d => "to" -> d.asString).toMap,
194       headerCfg = bridgeCfg.headerCfg
195     )
196 
197     val src : Source[FlowEnvelope, NotUsed] = {
198       val result : Source[FlowEnvelope, NotUsed] = Source.fromGraph(new JmsConsumerStage(
199         name = streamId + "-source",
200         consumerSettings = srcSettings
201       )).via(stats)
202 
203       // set the transaction from a the system property blended.streams.transactionShard
204       // Maintaining the transaction shard in the message will ensure that all transaction events
205       // with the same event id will end up in the same transaction destination.
206       transShard match {
207         case None => result
208         case Some(shard) => result.via(Flow.fromFunction[FlowEnvelope, FlowEnvelope]{ env =>
209           env.withHeader(bridgeCfg.headerCfg.headerTransShard, shard, false).get
210         })
211       }
212     }
213 
214 
215     // If we need to transform additional headers on the inbound leg
216     if (bridgeCfg.inbound && bridgeCfg.header.nonEmpty) {
217 
218       bridgeLogger.debug(s"Creating Stream with header configs [${bridgeCfg.header}]")
219 
220       val header: Graph[FlowShape[FlowEnvelope, FlowEnvelope], NotUsed] = HeaderTransformProcessor(
221         name = streamId + "-header",
222         log = envLogger,
223         rules = bridgeCfg.header,
224         ctCtxt = bridgeCfg.ctCtxt
225       ).flow(envLogger)
226 
227       src.via(header)
228     } else {
229       bridgeLogger.debug(s"Creating Stream without additional header configs")
230       src
231     }
232   }

245   // The producer to send the current envelope to the retry queue in case of an error
246   // We only forward the envelope to the retry Queue if the retry destination is set
247   // AND the bridge direction is outbound
248   protected def jmsRetry : Flow[FlowEnvelope, FlowEnvelope, NotUsed] = {
249 
250     val skipRetry : Flow[FlowEnvelope, FlowEnvelope, NotUsed] =
251       Flow.fromGraph(FlowProcessor.log(LogLevel.Debug, envLogger, "Skipping retry"))
252 
253     retryDest match {
254       case None =>
255         bridgeLogger.debug(s"No retry destination set, retry mechanism will be disabled")
256         skipRetry
257 
258       case Some(d) => if (isInbound) {
259         bridgeLogger.debug(s"Retry mechanism will be disabled for inbound bridge direction")
260         skipRetry
261       } else {
262         logEnvelope(s"Forwarding to retry [$d]")
263           .via(
264             Flow.fromFunction[FlowEnvelope, FlowEnvelope] { env =>
265               env.withHeader(bridgeCfg.headerCfg.headerRetryDestination, JmsDestination.asString(bridgeCfg.fromDest)).get
266             }
267           )
268           .via(
269             jmsProducer(
270               name = streamId + "-retry",
271               settings = toSettings(
272                 bridgeCfg.fromCf,
273                 Some(d)
274               ).copy(
275                 clearPreviousException = true,
276                 // We want to log the first send to the retry queue in info level and all others to debug,
277                 // so that we can decide whether we want to see only the first or all in splunk
278                 logLevel = env => if (retryCount(env) == 0) LogLevel.Info else LogLevel.Debug
279               ),
280               autoAck = false
281             )
282           )
283       }
284     }
285   }

287   // Decide whether a tracking event should be generated for this bridge step
288   protected[bridge] val trackFilter : Graph[FanOutShape2[FlowEnvelope, FlowEnvelope, FlowEnvelope], NotUsed] = FlowProcessor.partition[FlowEnvelope] { env =>
289 
290     val doTrack : Boolean = bridgeCfg.trackTransaction match {
291       case TrackTransaction.Off => false
292       case TrackTransaction.On  => true
293       case TrackTransaction.FromMessage =>
294         bridgeLogger.trace(s"Getting tracking mode from message property [${bridgeCfg.headerCfg.headerTrack}]")
295         val msgTrack = env.header[Boolean](bridgeCfg.headerCfg.headerTrack)
296         msgTrack.getOrElse(false)
297     }
298 
299     bridgeLogger.debug(s"Tracking for envelope [${env.id}] is [$doTrack]")
300 
301     doTrack
302   }

313   // flow to generate a transaction event from the current envelope and send it to the
314   // JMS transaction endpoint
315   protected def transactionFlow : Flow[FlowEnvelope, FlowEnvelope, NotUsed] = {
316 
317     val g : Graph[FlowShape[FlowEnvelope, FlowEnvelope], NotUsed] = GraphDSL.create() { implicit b =>
318 
319       import GraphDSL.Implicits._
320 
321       val doLog = b.add(logEnvelope("Before Tracking"))
322 
323       // First we decide wether we need to track the transaction
324       val doTrack = b.add(trackFilter)
325       doLog.out ~> doTrack.in
326 
327       // The actual send of the transaction
328       val send = sendTransaction
329 
330       val merge = b.add(Merge[FlowEnvelope](2))
331 
332       val sendError = b.add(FlowProcessor.partition[FlowEnvelope](_.exception.isEmpty))
333       val mergeError = b.add(Merge[FlowEnvelope](2))
334       val retry = b.add(jmsRetry)
335 
336       doTrack.out0 ~> send ~> sendError.in
337 
338       sendError.out0 ~> mergeError.in(0)
339       sendError.out1 ~> retry ~> mergeError.in(1)
340 
341       mergeError.out ~> merge.in(0)
342       doTrack.out1 ~> merge.in(1)
343 
344       FlowShape(doLog.in, merge.out)
345     }
346 
347     Flow.fromGraph(g)
348   }

353   val stream : Source[FlowEnvelope, NotUsed] = {
354 
355     val g : Graph[FlowShape[FlowEnvelope, FlowEnvelope], NotUsed] = GraphDSL.create() { implicit b =>
356       import GraphDSL.Implicits._
357 
358       // We will forward the message to the target destination
359       val forward = b.add(jmsSend)
360       val complete = b.add(FlowProcessor.completeStats(streamId + "completeStats", envLogger, bridgeCfg.headerCfg))
361 
362       // Then we have a path where the send is successful and one where the send has failed
363       val sendError = b.add(FlowProcessor.partition[FlowEnvelope] {
364         _.exception.isEmpty
365       })
366       forward ~> complete ~> sendError.in
367 
368       val merge = b.add(Merge[FlowEnvelope](2))
369 
370       // In case the send was success full we will track the transaction if required and then
371       val transaction = b.add(transactionFlow)
372 
373       sendError.out0 ~> transaction ~> merge.in(0)
374 
375       // In case of an exception we will
376       // - pass the message to the retry queue if the direction is outbound && the retry is enabled
377       //   if the send to the retry fails, the envelope will simply be rejected
378       // - simply deny the message if the direction is inbound or the retry is disabled
379 
380       val retry = b.add(jmsRetry)
381       sendError.out1 ~> retry ~> merge.in(1)
382 
383       // Finally, we acknowledge or reject the envelope, depending whether we have encountered an exception
384       val ack = b.add(new AckProcessor(streamId + "-ack").flow)
385 
386       merge.out ~> ack.in
387 
388       FlowShape(forward.in, ack.out)
389     }
390 
391     val clearRetrying : Graph[FlowShape[FlowEnvelope, FlowEnvelope], NotUsed] = FlowProcessor.fromFunction("clearRetrying", envLogger){ env => Try {
392       env.removeHeader(bridgeCfg.headerCfg.headerRetrying)
393     }}
394 
395     jmsSource
396       .via(g)
397       .via(clearRetrying)
398   }
