99   private[this] def createInboundStream(in : InboundConfig, cf : IdAwareConnectionFactory, internal : Boolean) : Unit = {
100 
101     val toDest = if (internal) {
102       JmsDestination.create(ctrlCfg.registry.internalProvider.get.inbound.asString).get
103     } else {
104       JmsDestination.create(
105         ctrlCfg.registry.internalProvider.get.inbound.asString + "." + cf.vendor + "." + cf.provider
106       ).get
107     }
108 
109     val inCfg = BridgeStreamConfig(
110       inbound = true,
111       fromCf = cf,
112       fromDest = in.from,
113       toCf = ctrlCfg.internalCf,
114       toDest = Some(toDest),
115       outboundAlternates = Seq.empty,
116       listener = in.listener,
117       selector = in.selector,
118       registry = ctrlCfg.registry,
119       headerCfg = ctrlCfg.headerCfg,
120       trackTransaction = if (ctrlCfg.trackInbound) {
121         TrackTransaction.On
122       } else {
123         TrackTransaction.Off
124       },
125       subscriberName = in.subscriberName,
126       header = in.header,
127       ctCtxt = Some(ctrlCfg.ctCtxt),
128       sessionRecreateTimeout = in.sessionRecreateTimeout
129     )
130 
131     val builder = ctrlCfg.streamBuilderFactory(system)(materializer)(inCfg, ctrlCfg.streamsCfg)
132     val actor = context.actorOf(StreamController.props[FlowEnvelope, NotUsed](
133       streamName = builder.streamId,
134       src = builder.stream,
135       streamCfg = ctrlCfg.streamsCfg
136     )(onMaterialize = _ => ()))
137 
138     streams += (builder.streamId -> actor)
139   }
140 
141   private[this] def createOutboundStream(cf : IdAwareConnectionFactory, internal : Boolean, alternates : Seq[String]) : Unit = {
142 
143     val fromDest = if (internal) {
144       JmsDestination.create(ctrlCfg.registry.internalProvider.get.outbound.asString).get
145     } else {
146       JmsDestination.create(
147         ctrlCfg.registry.internalProvider.get.outbound.asString + "." + cf.vendor + "." + cf.provider
148       ).get
149     }
150 
151     // TODO: Make listener count configurable
152     val outCfg = BridgeStreamConfig(
153       inbound = false,
154       headerCfg = ctrlCfg.headerCfg,
155       fromCf = ctrlCfg.internalCf,
156       fromDest = fromDest,
157       toCf = cf,
158       toDest = None,
159       outboundAlternates = alternates,
160       listener = 3,
161       selector = None,
162       registry = ctrlCfg.registry,
163       trackTransaction = TrackTransaction.FromMessage,
164       subscriberName = None,
165       header = List.empty,
166       sessionRecreateTimeout = 1.second
167     )
168 
169     val builder = ctrlCfg.streamBuilderFactory(system)(materializer)(outCfg, ctrlCfg.streamsCfg)
170     val actor = context.actorOf(StreamController.props[FlowEnvelope, NotUsed](
171       streamName = builder.streamId,
172       src = builder.stream,
173       streamCfg = ctrlCfg.streamsCfg
174     )(onMaterialize = _ => ()))
175 
176     streams += (builder.streamId -> actor)
177   }

179   override def receive : Receive = {
180     case AddConnectionFactory(cf) =>
181 
182       ctrlCfg.registry.internalProvider match {
183         case Success(p) =>
184           val internal = p.vendor == cf.vendor && p.provider == cf.provider
185           log.info(s"Adding connection factory [${cf.id}], internal [$internal]")
186 
187           // Create inbound streams for all matching inbound configs
188           val inbound : List[InboundConfig] = ctrlCfg.inbound.filter { in =>
189             ProviderFilter(in.vendor, in.provider).matches(cf)
190           }
191 
192           log.debug(s"Creating Streams for inbound destinations : [${inbound.mkString(",")}]")
193           inbound.foreach { in => createInboundStream(in, cf, internal) }
194 
195           createOutboundStream(cf, internal, ctrlCfg.outboundAlternates)
196         case Failure(_) =>
197           log.warn("No internal JMS provider found in config")
198       }
199 
200     case RemoveConnectionFactory(cf) =>
201       log.info(s"Removing connection factory [${cf.vendor}:${cf.provider}]")
202 
203       streams.filter { case (key, _) => key.startsWith(cf.id) }.foreach {
204         case (id, stream) =>
205           log.info(s"Stopping stream [$id]")
206           stream ! StreamController.Stop
207           streams -= id
208       }
209   }
