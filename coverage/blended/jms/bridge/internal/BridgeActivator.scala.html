24   private[this] def getProperty(context : ServiceWatcherContext[IdAwareConnectionFactory], name : String) : Option[String] = {
25     Option(context.ref.getProperty(name)) match {
26       case None    => None
27       case Some(o) => Some(o.toString())
28     }
29   }
30   private[this] def identifyCf(context : ServiceWatcherContext[IdAwareConnectionFactory]) : Try[(String, String)] = Try {
31     (getProperty(context, "vendor"), getProperty(context, "provider")) match {
32       case (Some(v), Some(p)) => (v, p)
33       case (v, p) =>
34         val msg = s"Detected connection Factory [$v, $p] is missing either the vendor or provider property."
35         throw new Exception(msg)
36     }
37   }

40   // the factory with stream builders throwing particular exceptions

42     bridgeCfg : BridgeStreamConfig, streamsCfg : BlendedStreamsConfig
43   ) : BridgeStreamBuilder = new BridgeStreamBuilder(bridgeCfg, streamsCfg)(system, materializer)

115                 case Success(_) =>
116                   bridge ! BridgeController.AddConnectionFactory(cf)
117                 case Failure(t) =>
118                   log.warn(t.getMessage)
119               }
120 
121               case ModifiedService(cf, context) => identifyCf(context) match {
122                 case Success(_) =>
123                   bridge ! BridgeController.RemoveConnectionFactory(cf)

125                 case Failure(t) =>
126                   log.warn(t.getMessage)
127               }
128 
129               case RemovedService(cf, context) => identifyCf(context) match {
130                 case Success(_) =>
131                   bridge ! BridgeController.RemoveConnectionFactory(cf)
132                 case Failure(t) =>
133                   log.warn(t.getMessage)
134               }
135             }
136 
137             onStop {
138               log.info("Stopping JMS bridge supervising actor.")
139               osgiCfg.system.stop(bridge)
140             }
141           } catch {
142             case NonFatal(t) =>
143               log.error(t)("Error starting JMS bridge")
144           }
145         }
146       }
147     }
