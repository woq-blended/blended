14 class ProviderFilter(vendor : String, provider : Option[String]) {
15 
16   private[this] val log = Logger[ProviderFilter]
17 
18   def matches[T <: ProviderAware](p : T) : Boolean = {
19 
20     val result = (vendor, provider) match {
21       // we need the vendor to match
22       case (p.vendor, s) => s match {
23         // The provider to search for is not specified => match
24         case None             => true
25         // The provider to search for matches exactly => match
26         case Some(p.provider) => true
27         // if the provider to search for is a regex => matches if the regex matches
28         case Some(pattern)    => p.provider.matches(pattern)
29       }
30 
31       // If the vendor does not match, nothing matches
32       case (_, _) => false
33     }
34 
35     log.debug(s"Checking Provider [${p.vendor},${p.provider}] with [$vendor, $provider], matched [$result]")
36     result
37   }
38 
39   def listProviderFilter[T <: ProviderAware](l : List[T], vendor : String, provider : Option[String])(implicit classTag : ClassTag[T]) : List[T] =
40     l.filter(matches)
41 
42   def singleProviderFilter[T <: ProviderAware](l : List[T], vendor : String, provider : String)(implicit evidence : ClassTag[T]) : Option[T] = {
43 
44     l.filter(matches) match {
45       case Nil =>
46         l.filter { p => p.vendor == vendor && p.provider.isEmpty } match {
47           case Nil =>
48             log.warn(s"No config entry found for [$vendor, $provider]")
49             None
50           case head :: tail =>
51             if (tail.nonEmpty) log.warn(s"Config entry for [$vendor] is not unique.")
52             Some(head)
53         }
54       case head :: tail =>
55         if (tail.nonEmpty) log.warn(s"Config entry for [$vendor, $provider] is not unique.")
56         Some(head)
57     }
58   }
59 }
