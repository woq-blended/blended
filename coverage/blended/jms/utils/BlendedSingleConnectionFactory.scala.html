47

49         val jmxServer = ManagementFactory.getPlatformMBeanServer
50         val jmxBean = new ConnectionMonitor(vendor, provider, clientId)

52         val objName = new ObjectName(s"blended:type=ConnectionMonitor,vendor=$vendor,provider=$provider")
53 
54         if (jmxServer.isRegistered(objName)) {
55           try {
56             jmxServer.unregisterMBean(objName)
57           } catch {
58             case NonFatal(t) =>
59               log.warn(s"Failed to deregister MBean [${objName.toString()}]:[${t.getMessage()}]")
60           }
61         }
62 
63         try {
64           jmxServer.registerMBean(jmxBean, objName)
65         } catch {
66           case NonFatal(t) =>
67             log.warn(s"Could not register MBean [${objName.toString}]:[${t.getMessage()}]")
68         }
69 
70         Some(jmxBean)
71       } else {
72         None
73       }
74 
75       // Simply create the state monitor for that particular connection
76       system.actorOf(ConnectionStateMonitor.props(vendor, provider, bundleContext, mbean), monitorName)
77       log.info(s"Connection State Monitor [$stateMgrName] created.")
78       Some(system.actorOf(ConnectionStateManager.props(config, holder), stateMgrName))
79     } else {
80       log.info(s"Connection State Monitor [$stateMgrName] is disabled by config setting.")
81       None
82     }

86   @throws[JMSException]
87   override def createConnection() : Connection = {
88 
89     if (cfEnabled) {
90       try {
91         holder.getConnection() match {
92           case Some(c) => c
93           case None    => throw new Exception(s"Error connecting to [$id].")
94         }
95       } catch {
96         case NonFatal(e) =>
97           val msg = s"Error getting Connection Factory [${e.getMessage()}]"
98           log.error(msg)

100           e match {
101             case ex : Exception => jmsEx.setLinkedException(ex)
102             case _ =>
103           }
104           throw jmsEx
105       }
106     } else {
107       throw new JMSException(s"Connection for provider [$id] is disabled.")
108     }
109   }
