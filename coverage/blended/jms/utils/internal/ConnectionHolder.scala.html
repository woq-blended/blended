33   @throws[JMSException]
34   def connect() : Connection = {
35     conn match {
36       case Some(c) => c
37       case None =>
38 
39         if (!connecting.getAndSet(true)) {
40           try {
41             log.info(s"Creating underlying connection for provider [$vendor:$provider] as user [${config.defaultUser}] with client id [${config.clientId}]")
42 
43             val cf : ConnectionFactory = getConnectionFactory()
44 
45             val c = config.defaultUser match {
46               case None       => cf.createConnection()
47               case Some(user) => cf.createConnection(user, config.defaultPassword.orNull)
48             }
49 
50             try {
51               c.setClientID(config.clientId)
52 
53               c.setExceptionListener(new ExceptionListener {
54                 override def onException(e : JMSException) : Unit = {
55                   log.warn(s"Exception encountered in connection for provider [$vendor:$provider] : ${e.getMessage()}")
56                   system.eventStream.publish(Reconnect(vendor, provider, Some(e)))
57                 }
58               })
59             } catch {
60               case NonFatal(e) =>
61                 log.error(s"Error setting client Id [${config.clientId}]...Closing Connection...")
62                 c.close()
63                 throw e
64             }
65 
66             c.start()
67 
68             log.info(s"Successfully connected to [$vendor:$provider] with clientId [${config.clientId}]")
69             val wrappedConnection = new BlendedJMSConnection(c)
70             conn = Some(wrappedConnection)
71 
72             wrappedConnection
73           } catch {
74             case NonFatal(t) =>
75               val msg : String = s"Error creating connection [$vendor:$provider] : [${t.getMessage()}]"
76               log.warn(msg)
77               val e : JMSException = new JMSException(msg)
78               e.setLinkedException(new Exception(t))
79               throw e
80           } finally {
81             connecting.set(false)
82           }
83 
84         } else {
85           throw new JMSException(s"Connection Factory for provider [$vendor:$provider] is still connecting.")
86         }
87     }
88   }
89 
90   def close() : Try[Unit] = {
91 
92     conn match {
93       case None => Success(())
94       case Some(c) =>
95         log.info(s"Closing underlying connection for provider [$provider]")
96         try {

99         } catch {
100           case NonFatal(t) =>

102         } finally {

119       config.properties ++ config.ctxtClassName.map(c => Context.INITIAL_CONTEXT_FACTORY -> c).toMap
120 
121     cfgMap.foreach {
122       case (k, v) =>
123         envMap.put(k, v)
124     }
125 
126     log.info(s"Initial context properties [${cfgMap.mkString(", ")}]")
127 
128     envMap
129   }
130 
131   override def getConnectionFactory() : ConnectionFactory = {
132 
133     val oldLoader = Thread.currentThread().getContextClassLoader()
134 
135     var context : Option[Context] = None
136 
137     try {
138 
139       val (name, contextFactoryClass) = (config.jndiName, config.ctxtClassName) match {
140         case (Some(n), Some(c)) => (n, c)
141         case (_, _) =>
142           throw new JMSException(s"Context Factory class and JNDI name have to be defined for JNDI lookup [$vendor:$provider].")
143       }
144 
145       config.jmsClassloader.foreach(Thread.currentThread().setContextClassLoader)
146 
147       context = Some(new InitialContext(initialContextEnv))
148       log.info(s"Looking up JNDI name [$name]")
149       context.get.lookup(name).asInstanceOf[ConnectionFactory]
150     } catch {
151       case NonFatal(t) =>
152         val sw = new StringWriter()
153         t.printStackTrace(new PrintWriter(sw))
154         log.warn(s"Could not lookup ConnectionFactory : [${t.getMessage()}]")
155         log.error(sw.toString)
156         val ex : JMSException = new JMSException("Could not lookup ConnectionFactory")
157         throw ex
158     } finally {
159       context.foreach { c =>
160         log.info(s"Closing Initial Context Factory [${config.ctxtClassName}] : [${config.jndiName}]")
161         c.close()
162       }
163       Thread.currentThread().setContextClassLoader(oldLoader)
164     }
165   }

193       }
194

196     } catch {
197       case NonFatal(t) =>
198         log.warn(s"Could not create ConnectionFactory : [${t.getMessage()}]")
199         val ex : JMSException = new JMSException("Could not create ConnectionFactory")
200         throw ex
201     } finally {
202       Thread.currentThread().setContextClassLoader(oldLoader)
203     }
