69   // ---- State: Disconnected
70   def disconnected()(state: ConnectionState) : Receive = {
71 
72     // Upon a CheckConnection message we will kick off initiating and monitoring the connection
73     case cc : CheckConnection =>
74       log.debug(s"Trying to initialize connection")
75       initConnection(state, cc.now)
76 
77     case _ : KeepAliveEvent => // do nothing

80   // ---- State: Connected
81   def connected()(state : ConnectionState) : Receive = {
82 
83     // we simply eat up the CloseTimeOut messages that might still be going for previous
84     // connect attempts
85     case ConnectTimeout(_) => // do nothing, this will just get rid of irrelevant warnings in the log
86 
87     case cc : CheckConnection => // do nothing
88 
89     case d @ Disconnect(_) => disconnect(state)
90 
91     case KeepAliveMissed(v,p,n) =>
92       if(config.vendor == v && config.provider == p) {
93         log.debug(s"Updating missed KeepAlives to [$n]")
94         switchState(connected(), state.copy(missedKeepAlives = n))
95       }
96 
97     case MaxKeepAliveExceeded(v, p) =>
98       if (config.vendor == v && config.provider == p) {
99         log.warn(s"Maximum number of missed keep alives has been reached : [${config.maxKeepAliveMissed}]")
100         reconnect(state)
101       }
102 
103     case _ : KeepAliveEvent => // do nothing
104   }

106   // ---- State: Connecting
107   def connecting()(state : ConnectionState) : Receive = {
108 
109     case _ : CheckConnection =>
110 
111     case ConnectResult(t, Left(e)) =>
112       if (t == state.lastConnectAttempt.getOrElse(new Date())) {
113         state.controller.foreach(context.system.stop)
114         switchState(disconnected(), stopController(state).copy(status = Disconnected))
115         if (!checkRestartForFailedReconnect(state, Some(e))) {
116           checkConnection(config.minReconnect)
117         }
118       }
119 
120     // We successfully connected, record the connection and timestamps
121     case ConnectResult(t, Right(c)) =>
122       if (t == state.lastConnectAttempt.getOrElse(0L)) {
123         conn = Some(new BlendedJMSConnection(c))
124         // checkConnection(schedule)
125         switchState(connected(), publishEvents(state, s"Successfully connected to JMS provider").copy(
126           status = Connected,
127           firstReconnectAttempt = None,
128           lastConnect = Some(new Date()),
129           missedKeepAlives = 0
130         ))
131       }
132 
133     case ConnectTimeout(t) =>
134       log.debug(s"Encountered Connection timeout (connectAttempt:$t), firstConnectAttempt: ${state.firstReconnectAttempt}, " +
135         s"lastConnectAttempt: ${state.lastConnectAttempt}")
136       if (!checkRestartForFailedReconnect(state, None)) {
137         switchState(connecting(), connect(state))
138       }
139 
140     case _ : KeepAliveEvent => // do nothing
141   }

143   // State: Closing
144   def closing()(state : ConnectionState) : Receive = {
145 
146     case _ : CheckConnection => // do nothing
147 
148     // All good, happily disconnected
149     case ConnectionClosed =>
150       conn = None
151       checkConnection(config.minReconnect, true)
152       state.controller.foreach(context.system.stop)
153       switchState(
154         disconnected(),
155         stopController(publishEvents(state, s"JMS connection successfully closed."))
156           .copy(
157             status = Disconnected, lastDisconnect = Some(new Date())
158           )
159       )
160 
161     // Once we encounter a timeout for a connection close we initiate a Container Restart via the monitor
162     case CloseTimeout =>
163       val msg : String = s"Unable to close JMS connection for provider in [${config.minReconnect}]s]. Restarting container ..."
164       switchState(restarting(), state.copy(status = RestartContainer(new Exception(msg))))
165 
166     case _ : KeepAliveEvent => // do nothing
167   }

172   def jmxOperations(state : ConnectionState) : Receive = {
173     case cmd : ConnectionCommand =>
174       if (cmd.vendor == config.vendor && cmd.provider == config.provider) {
175         if (cmd.disconnectPending) {
176           disconnect(state)
177         } else {
178           if (cmd.connectPending) {
179             self ! CheckConnection(cmd.reconnectNow)
180           }
181         }
182       }

185   def handleReconnectRequest(state : ConnectionState) : Receive = {
186     case Reconnect(v, p, None) => if (v == config.vendor && p == config.provider && state.status == Connected) {
187       log.info(s"Initiating JMS reconnect")
188       reconnect(state)
189     }
190     case Reconnect(v, p, Some(r)) => if (v == config.vendor && p == config.provider && state.status == Connected) {
191       log.info(s"Initiating JMS reconnect after connection exception [${r.getMessage()}]")
192       reconnect(state)
193     }
194   }

203   // A convenience method to let us know which state we are switching to
204   private[this] def switchState(rec : StateReceive, newState : ConnectionState) : Unit = {
205 
206     val nextState : ConnectionState =
207       publishEvents(newState, s"Connection State Manager switching to state [${newState.status}]")
208 
209     currentReceive = rec
210     currentState = nextState
211     context.system.eventStream.publish(ConnectionStateChanged(nextState))
212     context.become(
213       rec(nextState)
214         .orElse(jmxOperations(nextState))
215         .orElse(handleReconnectRequest(nextState))
216         .orElse(controllerStopped(nextState))
217         .orElse(handleQueryState(nextState))
218         .orElse(unhandled)
219     )
220   }

228     case Terminated(a) => s.controller match {

235   // We simply stay in the same state and maintain the list of events
236   def publishEvents(s : ConnectionState, msg : String*) : ConnectionState = {
237 
238     msg.foreach(m => log.info(m))
239     val tsMsg = msg.map { m => df.format(new Date()) + " " + m }
240 
241     val newEvents = if (tsMsg.size >= s.maxEvents) {
242       tsMsg.reverse.take(s.maxEvents)
243     } else {
244       tsMsg.reverse ++ s.events.take(s.maxEvents - tsMsg.size)
245     }
246 
247     val newState : ConnectionState = s.copy(events = newEvents.toList)
248     newState
249   }

251   // To initialise the connection we check whether we have been connected at some point in the history of
252   // this container has been started. If so, we will let not attempt to reconnect before the time specified
253   // in the config has passed.
254   // If not, we will try to connect immediately.
255 
256   private[this] def initConnection(s : ConnectionState, now : Boolean) : Unit = {
257 
258     val remaining : Double = s.lastDisconnect match {
259       case None    => 0
260       case Some(l) => config.minReconnect.toMillis - (System.currentTimeMillis() - l.getTime())
261     }
262 
263     // if we were ever disconnected from the JMS provider since the container start we will check
264     // whether the reconnect interval has passed, otherwise we will connect immediately
265     if (!now && s.lastDisconnect.isDefined && remaining > 0) {
266       switchState(
267         currentReceive,
268         publishEvents(s, s"Container is waiting to reconnect, remaining wait time [${remaining / 1000.0}]s")
269       )
270       checkConnection((remaining + 1).seconds)
271     } else {
272       switchState(connecting(), connect(s))
273     }
274   }

276   // A simple convenience method to schedule the next connection check to ourselves
277   private[this] def checkConnection(delay : FiniteDuration, force : Boolean = false) : Unit = {
278     log.info(s"Scheduling JMS connection check in [$delay]")
279     context.system.scheduler.scheduleOnce(delay, self, CheckConnection(false))

282   private[this] def connect(state : ConnectionState) : ConnectionState = {
283 
284     var events : List[String] = List(s"Creating connection to JMS")
285 
286     val lastConnectAttempt = new Date()
287 
288     context.system.scheduler.scheduleOnce(config.connectTimeout, self, ConnectTimeout(lastConnectAttempt))
289 
290     // This only happens if we have configured a maximum reconnect timeout in the config and we ever
291     // had a connection since this container was last restarted and we haven't started the timer yet
292     val newState : ConnectionState = (if (config.maxReconnectTimeout.isDefined && state.firstReconnectAttempt.isEmpty) {
293       events = (s"Starting max connect timeout monitor with [${config.maxReconnectTimeout}]s") :: events
294       restartController(state).copy(firstReconnectAttempt = Some(lastConnectAttempt))
295     } else {
296       restartController(state)
297     })
298 
299     newState.controller.foreach(_ ! Connect(lastConnectAttempt, config.clientId))
300 
301     // push the events into the newState in reverse order and set
302     // the new state name
303     publishEvents(newState, events.reverse.toArray : _*).copy(
304       status = Connecting,
305       lastConnectAttempt = Some(lastConnectAttempt)
306     )
307   }

309   private[this] def restartController(s : ConnectionState) : ConnectionState = {
310     val newController : ActorRef = context.actorOf(JmsConnectionController.props(holder, ConnectionCloseActor.props(holder)))
311     // We start watching the controller, so we can react in case it dies
312     context.watch(newController)
313 
314     stopController(s).copy(controller = Some(newController))
315   }

317   private[this] def stopController(s: ConnectionState) : ConnectionState = {
318     s.controller.foreach{ a =>
319       log.debug(s"Stopping JMS Connection controller")
320       context.unwatch(a)
321       context.stop(a)
322     }
323     s.copy(controller = None)

326   private[this] def checkRestartForFailedReconnect(s: ConnectionState, e: Option[Throwable]): Boolean = {
327 
328     var result = false
329 
330     e.foreach{ t =>
331       log.error(t)(s"Error connecting to JMS provider.")
332     }
333 
334     if (config.maxReconnectTimeout.isDefined && s.firstReconnectAttempt.isDefined) {
335       s.firstReconnectAttempt.foreach { t =>
336 
337         val restart : Boolean = config.maxReconnectTimeout.exists { to =>
338           (System.currentTimeMillis() - t.getTime()).millis > to
339         }
340 
341         if (restart) {
342           val msg : String = s"Unable to reconnect to JMS provider in [${config.maxReconnectTimeout}]s. Restarting container ..."
343           log.warn(msg)
344           context.system.eventStream.publish(ConnectionStateChanged(s.copy(status = RestartContainer(new Exception(msg)))))
345           result = true
346         }
347       }
348     }
349 
350     result
351   }

353   // Once we decided to close the connection we cancel our PingTimer
354   // and start to cleanup. This goes into it's own state, so that we can
355   // catch connections that cannot be closed within a reasonable time.
356   // Experience has shown that for a close timeout it is best to restart
357   // the container.
358   private[this] def disconnect(s : ConnectionState) : Unit = {
359     // Notify the connection controller of the disconnect
360     s.controller.foreach(_ ! Disconnect(config.minReconnect))
361     switchState(closing(), s.copy(status = Closing))
362   }
