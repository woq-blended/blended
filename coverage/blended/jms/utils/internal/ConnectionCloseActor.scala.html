15 /**
16  * This Actor will execute and monitor a single close operation on a given JMS connection
17  * and then stop itself.
18  * @param holder
19  */
20 class ConnectionCloseActor(holder : ConnectionHolder, retryInterval : FiniteDuration) extends Actor with ActorLogging {
21 
22   private[this] implicit val eCtxt : ExecutionContext = context.system.dispatcher
23 
24   case object Tick
25 
26   private[this] def doClose() : Unit = {
27     Future {
28       holder.close() match {
29         case Success(_) =>
30           log.debug(s"Connection [${holder.vendor}:${holder.provider}] successfully closed.")
31           self ! ConnectionClosed
32         case Failure(t) =>
33           log.warning(s"Error closing connection [${holder.vendor}:${holder.provider}] : [${t.getMessage()}] -- Assuming the connection is dead")
34           self ! ConnectionClosed
35       }
36     }
37   }
38 
39   override def receive : Receive = {
40 
41     case Disconnect(t) =>
42       // Just schedule a timeout message in case the Future takes too long
43       context.become(waiting(sender(), context.system.scheduler.scheduleOnce(t, self, CloseTimeout)))
44       doClose()
45   }
46 
47   def waiting(caller : ActorRef, timer : Cancellable) : Receive = {
48 
49     case Tick =>
50       doClose()
51 
52     case CloseTimeout =>
53       timer.cancel()
54       caller ! CloseTimeout
55       context.stop(self)
56 
57     case ConnectionClosed =>
58       timer.cancel()
59       caller ! ConnectionClosed
60       context.stop(self)
61   }
