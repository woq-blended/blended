13   def create(destName : String) : Try[JmsDestination] = Try {
14

18       case 0 => throw new IllegalArgumentException("No destination name given")
19       case 1 => JmsQueue(name(0))
20       case 2 => name(0) match {
21         case QUEUETAG => JmsQueue(name(1))
22         case TOPICTAG => JmsTopic(name(1))
23         case _        => throw new JMSException(s"String representation of JmsDestination must start with either [$QUEUETAG:] or [$TOPICTAG:]")
24       }
25       case 3 =>
26         if (name(0) == TOPICTAG) {
27           JmsDurableTopic(name(2), name(1))
28         } else {
29           throw new IllegalArgumentException("Only names for durable Topics have 3 components")
30         }
31 
32       case _ => throw new IllegalArgumentException(s"Illegal format for destination name [$name]")
33     }
34   }

36   def create(jmsDest : Destination) : Try[JmsDestination] = Try {
37     jmsDest match {
38       case t : Topic => JmsTopic(t.getTopicName())
39       case q : Queue => JmsQueue(q.getQueueName())
40       case _         => throw new IllegalArgumentException(s"Unknown destination type [${jmsDest.getClass().getName()}]")
41     }
42   }
