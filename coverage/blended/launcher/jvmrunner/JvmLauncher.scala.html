49   def run(args : Array[String]) : Int = {
50     val config = checkConfig(parse(args)).get
51     log.debug("JvmLauncherConfig = " + config)
52     config.action match {
53 
54       // Try to start the inner JVM
55       case Some("start") =>
56         log.debug("Request: start process")
57         runningProcess match {
58           case Some(_) =>
59             log.debug("Process already running")
60             sys.error("Already started")
61           case None =>
62             var retVal = 1
63             do {
64               // If the container JVM terminated with exit code 2, we will restart
65               // the container JVM.
66               if (retVal == 2) {
67                 config.restartDelaySec match {
68                   // In some cases we need a cool down period before we restart the container
69                   // JVM. In that case we will wait for the specified number of seconds.
70                   case Some(delay) =>
71                     log.debug("Waiting " + delay + " seconds before restarting the container.")
72                     try {
73                       Thread.sleep(delay * 1000)
74                     } catch {
75                       case e : InterruptedException =>
76                         log.debug("Delay interrupted!")
77                     }
78                   case _ =>
79                 }
80                 log.debug("About to restart the container process.")
81               } else {
82                 log.debug("About to start process")
83               }
84 
85               // Starting the container requires 2 steps:
86               // First, we will start the container with the parameter --write-system-properties.
87               // This will cause that the container runtime evaluates the current profile and it's
88               // overlays. In this mode the container will dump the determined System Properties
89               // into a file and terminate.
90               //
91               // We will then start the container with the calculated system properties.
92               //
93               // This gives us the opportunity to handle properties configuring the JVM itself with
94               // the blended overlay mechanism.
95 
96               log.info("-" * 80)
97               log.info("Starting container in write properties mode")
98               log.info("-" * 80)
99               val sysProps : Map[String, String] = {
100                 val sysPropsFile = File.createTempFile("jvmlauncher", ".properties")
101                 val p = startJava(
102                   classpath = config.classpath,
103                   jvmOpts = config.jvmOpts.toArray,
104                   arguments = config.otherArgs.toArray ++ Array("--write-system-properties", sysPropsFile.getAbsolutePath()),
105                   interactive = false,
106                   errorsIntoOutput = false,
107                   shutdownTimeout = config.shutdownTimeout,
108                   directory = new File(".").getAbsoluteFile()
109                 )
110                 val retVal = p.waitFor
111                 if (retVal != 0) {
112                   log.error(s"The launcher-process to retrieve the JVM properties exited with an unexpected return code: ${retVal}. We try to read the properties file anyway!")
113                 }
114                 val props = new Properties()
115                 Try {
116                   ARM.using(new FileInputStream(sysPropsFile)) { inStream =>
117                     props.load(inStream)
118                   }
119                 }.recover {
120                   case e : Throwable => log.error(e)("Could not read properties file")
121                 }
122                 sysPropsFile.delete()
123                 props.asScala.toList.toMap
124               }
125 
126               // Now we can extract the JVM memory settings if given
127               val xmsOpt = sysProps.collect { case (OverlayConfigCompanion.Properties.JVM_USE_MEM, x) => s"-Xms${x}" }
128               val xmxOpt = sysProps.collect { case (OverlayConfigCompanion.Properties.JVM_MAX_MEM, x) => s"-Xmx${x}" }
129               val ssOpt = sysProps.collect { case (OverlayConfigCompanion.Properties.JVM_STACK_SIZE, x) => s"-Xss${x}" }
130 
131               log.info("-" * 80)
132               log.info("Starting blended container instance")
133               log.info("-" * 80)
134               val p = startJava(
135                 classpath = config.classpath,
136                 jvmOpts = (config.jvmOpts ++ xmsOpt ++ xmxOpt ++ ssOpt ++ sysProps.map { case (k, v) => s"-D${k}=${v}" }).toArray,
137                 arguments = config.otherArgs.toArray,
138                 interactive = config.interactive,
139                 errorsIntoOutput = false,
140                 shutdownTimeout = config.shutdownTimeout,
141                 directory = new File(".").getAbsoluteFile()
142               )
143               log.debug("Process started: " + p)
144               runningProcess = Option(p)
145               retVal = p.waitFor
146               log.info("-" * 80)
147               log.info(s"Blended container instance terminated with exit code [$retVal]")
148               log.info("-" * 80)
149               runningProcess = None
150             } while (retVal == 2)
151             retVal
152         }
153 
154       // try to stop the inner JVM
155       case Some("stop") =>
156         log.debug("Request: stop process")
157         runningProcess match {
158           case None =>
159             log.debug("No process running")
160             sys.error("Not started")
161           case Some(p) =>
162             p.stop()
163         }
164 
165       // All other commands are considered to be errors
166       case a @ _ =>
167         sys.error(s"Not a valid action : [$a]")
168     }
169   }

171   case class JvmLauncherConfig(
172     classpath : Seq[File] = Seq(),
173     otherArgs : Seq[String] = Seq(),
174     action : Option[String] = None,

193   @tailrec
194   final def parse(args : Seq[String], initialConfig : JvmLauncherConfig = JvmLauncherConfig()) : JvmLauncherConfig = {
195 
196     args match {
197       case Seq() =>
198         initialConfig
199       case Seq("--", rest @ _*) =>
200         initialConfig.copy(otherArgs = rest)
201       case Seq("start", rest @ _*) if initialConfig.action.isEmpty =>
202         parse(rest, initialConfig.copy(action = Option("start")))
203       case Seq("stop", rest @ _*) if initialConfig.action.isEmpty =>
204         parse(rest, initialConfig.copy(action = Option("stop")))
205       case Seq(cp, rest @ _*) if initialConfig.classpath.isEmpty && cp.startsWith("-cp=") =>
206         // Also support ":" on non-windows platform
207         val cps = cp.substring("-cp=".length).split("[;]").toSeq.map(_.trim()).filter(!_.isEmpty).map(new File(_))
208         parse(rest, initialConfig.copy(classpath = cps))
209       case Seq(delay, rest @ _*) if initialConfig.restartDelaySec.isEmpty && delay.startsWith("-restartDelay=") =>
210         val delaySec = delay.substring("-restartDelay=".length).toInt
211         parse(rest, initialConfig.copy(restartDelaySec = Option(delaySec)))
212       case Seq(jvmOpt, rest @ _*) if jvmOpt.startsWith("-jvmOpt=") =>
213         val opt = jvmOpt.substring("-jvmOpt=".length).trim()
214         parse(rest, initialConfig.copy(jvmOpts = initialConfig.jvmOpts ++ Seq(opt).filter(!_.isEmpty)))
215       case Seq(interactive, rest @ _*) if interactive.startsWith("-interactive") =>
216         val iAct = interactive.substring("-interactive=".length).toBoolean
217         parse(rest, initialConfig.copy(interactive = iAct))
218       case Seq(maxShutdown, rest @ _*) if maxShutdown.startsWith("-maxShutdown") =>
219         val seconds : Int = Integer.parseInt(maxShutdown.substring("-maxShutdown=".length()))
220         parse(rest, initialConfig.copy(shutdownTimeout = seconds.seconds))
221       case _ =>
222         sys.error("Cannot parse arguments: " + args)
223     }
224   }

237   private def startJava(
238     classpath : Seq[File],
239     jvmOpts : Array[String],
240     arguments : Array[String],
241     interactive : Boolean,
242     errorsIntoOutput : Boolean,
243     directory : File,
244     shutdownTimeout : FiniteDuration
245   ) : RunningProcess = {
246 
247     log.debug("About to run Java process")
248 
249     // lookup java by JAVA_HOME env variable
250     val java = Option(System.getenv("JAVA_HOME")) match {
251       case Some(javaHome) => s"$javaHome/bin/java"
252       case None           => "java"
253     }
254 
255     log.debug("Using java executable: " + java)
256 
257     val cpArgs = Option(classpath) match {
258       case None | Some(Seq()) => Array[String]()
259       case Some(cp)           => Array("-cp", pathAsArg(classpath))
260     }
261     log.debug("Using classpath args: " + cpArgs.mkString(" "))
262 
263     log.debug(s"Using JVM options ${jvmOpts.mkString("[\n", "\n", "\n]")}")
264     val command = Array(java) ++ cpArgs ++ jvmOpts ++ arguments
265 
266     log.debug(s"Using JVM arguments [${arguments.mkString("\n")}]")
267 
268     val pb = new ProcessBuilder(command : _*)
269     log.debug("Run command: " + command.mkString(" "))
270     pb.environment().putAll(sys.env.asJava)
271     pb.directory(directory)
272     // if (!env.isEmpty) env.foreach { case (k, v) => pb.environment().put(k, v) }
273     val p = pb.start
274 
275     new RunningProcess(p, errorsIntoOutput, interactive, shutdownTimeout)
276   }
