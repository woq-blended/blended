84   private[this] def containerId(f : File, createContainerID : Boolean, onlyIfMissing : Boolean) : Try[String] = {
85 
86     val idFile = new File(containerConfigDirectory, containerIdFile)
87 
88     if (idFile.exists() && idFile.isDirectory) {
89       val msg = s"The file [${idFile.getAbsoluteFile}] exists and is a directory"
90       log.error(msg)
91       Console.err.println(msg)
92       sys.error(msg)
93     }
94 
95     val generateId = createContainerID && (!onlyIfMissing || !idFile.exists())
96 
97     if (generateId && idFile.exists() && !idFile.canWrite()) {
98       reportError(s"Container Id File [${idFile.getAbsolutePath}] is not writable")
99     }
100 
101     if (generateId && idFile.exists()) idFile.delete()
102 
103     if (generateId) {
104       log.info("Creating new container id")
105       val uuid : CharSequence = UUID.randomUUID().toString.toCharArray
106       Files.write(idFile.toPath, Seq(uuid).asJava)
107     }
108 
109     Try {
110       val lines = Files.readAllLines(Paths.get(idFile.getAbsolutePath))
111       if (!lines.isEmpty) lines.get(0) else sys.error("Empty container ID file")
112     }
113   }

115   private[this] def createAndPrepareLaunch(configs : Configs, createContainerId : Boolean, onlyIfMissing : Boolean) : Launcher = {
116 
117     val launcher = new Launcher(configs.launcherConfig)
118 
119     val errors = configs.profileConfig match {
120       case Some(localConfig) =>
121         // Expose the List of mandatory container properties as a System Property
122         // This will be evaluated by the Container Identifier Service
123         val propNames = localConfig.resolvedRuntimeConfig.runtimeConfig.properties.getOrElse(RuntimeConfig.Properties.PROFILE_PROPERTY_KEYS, "")
124         System.setProperty(RuntimeConfig.Properties.PROFILE_PROPERTY_KEYS, propNames)
125 
126         localConfig.validate(
127           includeResourceArchives = false,
128           explodedResourceArchives = true
129         )
130 
131       case None =>
132         // if no RuntimeConfig, just check existence of bundles
133         launcher.validate()
134     }
135 
136     if (errors.nonEmpty) sys.error("Could not start the OSGi Framework. Details:\n" + errors.mkString("\n"))
137

139       case Failure(e) =>

142           case Some(_) =>
143             // Profile mode, this is an error
144             log.error(e)(msg)
145             Console.err.println(msg)
146             sys.error(msg)
147           case None =>
148             // simple config mode, this is not an error
149             log.warn(e)(msg)
150         }
151       case Success(id) => log.info(s"ContainerId is [$id] ")
152     }

157   /**
158    * Use this method instead of `main` if you do not want to exit the VM
159    * and instead get an [LauncherException] in case of a error.
160    *
161    * @throws LauncherException
162    */
163   def run(args : Array[String]) : Unit = {
164     val cmdline = parseArgs(args).get
165     val handleFrameworkRestart = cmdline.handleFrameworkRestart
166     var firstStart = true
167     var retVal : Int = 0
168 
169     do {
170       val configs = try {
171         readConfigs(cmdline)
172       } catch {
173         case e : Throwable =>
174           log.error(e)("Could not read configs")
175           throw e
176       }
177       log.debug(s"Configs: $configs")
178 
179       cmdline.writeSystemProperties match {
180         case Some(propFile) =>
181           log.info("Running with --write-system-properties. About to generate properties file and exit")
182           val fileProps = new Properties()
183           configs.launcherConfig.systemProperties.foreach { case (k, v) => fileProps.setProperty(k, v) }
184           try {
185             ARM.using(new FileOutputStream(propFile)) { stream =>
186               fileProps.store(stream, "Generated by Launcher")
187               log.info(s"Wrote system properties file: $propFile")
188             }
189             retVal = 0
190           } catch {
191             case e : Throwable =>
192               log.error(e)(s"Could not write system properties file: $propFile")
193               retVal = 1
194           }
195         case None =>
196           val createContainerId = firstStart && (cmdline.resetContainerId || cmdline.initContainerId)
197           val launcher = createAndPrepareLaunch(configs, createContainerId, cmdline.initContainerId)
198           retVal = launcher.run(cmdline)
199           firstStart = false
200       }
201     } while (handleFrameworkRestart && retVal == 2)
202 
203     if (retVal != 0) throw new LauncherException("", errorCode = retVal)
204   }

211   def readConfigs(cmdline : Cmdline) : Configs = {
212     cmdline.configFile match {
213       case Some(configFile) =>
214         log.info(s"About to read configFile: [$configFile]")
215         val config = ConfigFactory.parseFile(new File(configFile), ConfigParseOptions.defaults().setAllowMissing(false)).resolve()
216         Configs(LauncherConfig.read(config))
217       case None =>
218         val profileLookup : Option[ProfileLookup] = cmdline.profileLookup.map { pl =>
219           log.info(s"About to read profile lookup file: [$pl]")
220           val c = ConfigFactory.parseFile(new File(pl), ConfigParseOptions.defaults().setAllowMissing(false)).resolve()
221           ProfileLookup.read(c).map { pl =>
222             pl.copy(profileBaseDir = pl.profileBaseDir.getAbsoluteFile())
223           }.get
224         }
225 
226         val profile : String = profileLookup match {
227           case Some(pl) =>
228             pl.materializedDir.getPath()
229           case None =>
230             cmdline.profileDir match {
231               case Some(profile) => profile
232               case None =>
233                 sys.error("Either a config file or a profile dir or file or a profile lookup path must be given")
234             }
235         }
236 
237         val (profileDir, profileFile) = if (new File(profile).isDirectory()) {
238           profile -> new File(profile, "profile.conf")
239         } else {
240           Option(new File(profile).getParent()).getOrElse(".") -> new File(profile)
241         }
242 
243         log.info(s"Using profile directory : [$profileDir]")
244         log.info(s"Using profile file      : [${profileFile.getAbsolutePath}]")
245 
246         val config = ConfigFactory.parseFile(profileFile, ConfigParseOptions.defaults().setAllowMissing(false)).resolve()
247 
248         val runtimeConfig = ResolvedRuntimeConfig(RuntimeConfigCompanion.read(config).get)
249         val launchConfig = ConfigConverter.runtimeConfigToLauncherConfig(runtimeConfig, profileDir)
250 
251         var brandingProps = Map(
252           RuntimeConfig.Properties.PROFILE_DIR -> profileDir
253         )
254         var overlayProps = Map[String, String]()
255 
256         profileLookup.foreach { pl =>
257           brandingProps ++= Map(
258             RuntimeConfig.Properties.PROFILE_LOOKUP_FILE -> new File(cmdline.profileLookup.get).getAbsolutePath(),
259             RuntimeConfig.Properties.PROFILES_BASE_DIR -> pl.profileBaseDir.getAbsolutePath(),
260             RuntimeConfig.Properties.OVERLAYS -> pl.overlays.map(or => s"${or.name}:${or.version}").mkString(",")
261           )
262 
263           val knownOverlays = LocalOverlays.findLocalOverlays(new File(profileDir).getAbsoluteFile())
264           knownOverlays.find(ko => ko.overlayRefs.toSet == pl.overlays.toSet) match {
265             case None =>
266               if (pl.overlays.nonEmpty) {
267                 sys.error("Cannot find specified overlay set: " + pl.overlays.toSeq.sorted.mkString(", "))
268               } else {
269                 log.error("Cannot find the empty overlay set (aka 'base.conf'). To be compatible with older version, we continue here as no real information is missing")
270               }
271             case Some(localOverlays) =>
272               val newOverlayProps = localOverlays.properties
273               log.debug("Found overlay provided properties: " + newOverlayProps)
274               overlayProps ++= newOverlayProps
275           }
276         }
277 
278         Configs(
279           launcherConfig = launchConfig.copy(
280             branding = launchConfig.branding ++ brandingProps,
281             systemProperties =
282               SystemPropertyResolver.resolve((launchConfig.systemProperties ++ overlayProps) + ("blended.container.home" -> profileDir))
283           ),
284           profileConfig = Some(LocalRuntimeConfig(runtimeConfig, new File(profileDir)))
285         )
286     }

291   class RunningFramework(val framework : Framework) {
292 
293     def awaitFrameworkStop(framwork : Framework) : Int = {
294       val event = framework.waitForStop(0)
295       event.getType match {
296         case FrameworkEvent.ERROR =>
297           log.info(event.getThrowable())("Framework has encountered an error: ")
298           1
299         case FrameworkEvent.STOPPED =>
300           log.info("Framework has been stopped by bundle " + event.getBundle)
301           0
302         case FrameworkEvent.STOPPED_UPDATE =>
303           log.info("Framework has been updated by " + event.getBundle + " and need a restart")
304           2
305         case _ =>
306           log.info("Framework stopped. Reason: " + event.getType + " from bundle " + event.getBundle)
307           0
308       }
309     }
310 
311     val shutdownHook : Thread = new Thread("framework-shutdown-hook") {
312       override def run() : Unit = {
313         log.info("Caught kill signal: stopping framework")
314         framework.stop()
315         awaitFrameworkStop(framework)
316         BrandingProperties.setLastBrandingProperties(new Properties())
317       }
318     }
319 
320     Runtime.getRuntime.addShutdownHook(shutdownHook)
321 
322     def waitForStop() : Int = {
323       try {
324         awaitFrameworkStop(framework)
325       } catch {
326         case NonFatal(x) =>
327           log.error(x)(s"Framework was interrupted. Cause: [${x.getMessage()}]")
328           1
329       } finally {
330         BrandingProperties.setLastBrandingProperties(new Properties())
331         Try {
332           Runtime.getRuntime.removeShutdownHook(shutdownHook)
333         }
334       }
335     }
336   }

369   /**
370    * Run an (embedded) OSGiFramework based of this Launcher's configuration.
371    */
372   def start(cmdLine : Cmdline) : Try[Framework] = Try {
373     log.info(s"Starting OSGi framework based on config: ${config}")
374 
375     // Try to locate and load the OSGi framework factory
376     val frameworkURL = new File(config.frameworkJar).getAbsoluteFile.toURI().normalize().toURL()
377     log.info("Framework Bundle from: " + frameworkURL)
378     if (!new File(frameworkURL.getFile()).exists) throw new RuntimeException("Framework Bundle does not exist")
379     val cl = new URLClassLoader(Array(frameworkURL), getClass.getClassLoader)
380     log.debug("About to load FrameworkFactory")
381     val frameworkFactory = ServiceLoader.load(classOf[FrameworkFactory], cl).iterator().next()
382     log.debug("Loaded framework factory: " + frameworkFactory)
383 
384     // Statically export branding properties
385     val brandingProps = {
386       val brandingProps = new Properties()
387       config.branding.foreach { case (k, v) => brandingProps.setProperty(k, v) }
388       BrandingProperties.setLastBrandingProperties(brandingProps)
389       log.debug("Exposing branding via class " + classOf[BrandingProperties].getName() + ": " + brandingProps)
390       brandingProps
391     }
392 
393     // Set system properties found in config
394     config.systemProperties foreach { p =>
395       log.info(s"Setting System property [${p._1}] to [${p._2}]")
396       System.setProperty(p._1, p._2)
397     }
398 
399     log.info("About to create framework instance...")
400     val framework = frameworkFactory.newFramework(config.frameworkProperties.asJava)
401     log.debug("Framework created: " + framework)
402 
403     log.debug("About to adapt framework to FrameworkStartLevel")
404     val frameworkStartLevel = framework.adapt(classOf[FrameworkStartLevel])
405     frameworkStartLevel.setInitialBundleStartLevel(config.defaultStartLevel)
406 
407     log.debug("About to start framework")
408     framework.start()
409     log.info(s"Framework started. State: ${framework.getState}")
410 
411     {
412       val props = new Hashtable[String, AnyRef]()
413       props.put("blended.launcher", "true")
414       framework.getBundleContext.registerService(classOf[Properties], brandingProps, props)
415     }
416 
417     log.info(s"Installing [${config.bundles.size}] bundles")
418     val context = framework.getBundleContext()
419     val osgiBundles = config.bundles.map { b =>
420       log.info(s"Installing Bundle: ${b}")
421       // TODO: What happens here, if the JAR is not a bundle?
422       val osgiBundle = context.installBundle(new File(b.location).getAbsoluteFile.toURI().normalize().toString())
423       log.info("Bundle installed: " + b)
424       val bundleStartLevel = osgiBundle.adapt(classOf[BundleStartLevel])
425       log.debug(s"Setting start level for bundle ${osgiBundle.getSymbolicName()} to ${b.startLevel}")
426       bundleStartLevel.setStartLevel(b.startLevel)
427       InstalledBundle(b, osgiBundle)
428     }
429     log.info(s"${osgiBundles.size} bundles installed")
430 
431     def isFragment(b : InstalledBundle) = b.bundle.getHeaders.get(Constants.FRAGMENT_HOST) != null
432 
433     // Iterate over start levels and activate bundles in the correct order
434     1.to(config.startLevel).map { startLevel =>
435       log.info(s"------ Entering start level [$startLevel] ------")
436       frameworkStartLevel.setStartLevel(startLevel, new FrameworkListener() {
437         override def frameworkEvent(event : FrameworkEvent) : Unit = {
438           log.debug(s"Active start level [${startLevel}] reached")
439         }
440       })
441 
442       // A bundle needs to start if it is configured with start=true within the profile or the
443       // bundle symbolic name is set in the Systemproperty blended.laucher.startbundles
444 
445       val extraStartBundles : List[String] = Option(System.getProperty(Launcher.extraStartBundle)) match {
446         case None    => List.empty
447         case Some(s) => s.split(",").toList
448       }
449 
450       val bundlesToStart = osgiBundles.filter { b =>
451         val startConfigured : Boolean =
452           b.jarBundle.startLevel == startLevel && b.jarBundle.start && !isFragment(b)
453 
454         startConfigured || extraStartBundles.contains(b.bundle.getSymbolicName())
455       }
456 
457       log.info(s"Starting [${bundlesToStart.size}] bundles")
458 
459       val startedBundles = bundlesToStart.map { bundle =>
460         val result = Try {
461           bundle.bundle.start()
462         }
463         log.info(s"State of [${bundle.bundle.getSymbolicName}] : [${bundle.bundle.getState}]")
464 
465         result match {
466           case Success(_) =>
467           case Failure(t) =>
468             val sw = new StringWriter()
469             t.printStackTrace(new PrintWriter(sw))
470             log.error("\n" + sw.getBuffer().toString() + "\n")
471         }
472         bundle -> result
473       }
474       log.info(s"[${startedBundles.filter(_._2.isSuccess).size}] bundles started")
475 
476       val failedBundles = startedBundles.filter(_._2.isFailure)
477 
478       if (!failedBundles.isEmpty) {
479         log.warn(s"Could not start some bundles:\n${
480           failedBundles.map(failed => s"\n - ${failed._1}\n ---> ${failed._2}")
481         }")
482 
483         if (cmdLine.strict) {
484           // in strict mode, bundles that failed to start fail the whole container
485           log.warn("Shutting down container due to bundle start failures.")
486           framework.stop()
487         }
488       }
489     }
490 
491     val bundlesInInstalledState = osgiBundles.filter(b => b.bundle.getState() == Bundle.INSTALLED && !isFragment(b))
492 
493     // now we try to also resolve the remaining bundles
494     if (bundlesInInstalledState.nonEmpty) {
495       log.debug(s"The following bundles are in installed state: ${bundlesInInstalledState.map(b => s"${b.bundle.getSymbolicName}-${b.bundle.getVersion}")}")
496       log.info("Resolving installed bundles")
497       val frameworkWiring = framework.adapt(classOf[FrameworkWiring])
498       frameworkWiring.resolveBundles(null /* all bundles */ )
499       val secondAttemptInstalled = osgiBundles.filter(b => b.bundle.getState() == Bundle.INSTALLED && !isFragment(b))
500 
501       val msg : String = s"The following bundles could not be resolved : ${
502         secondAttemptInstalled.map(
503           b => s"${b.bundle.getSymbolicName}-${b.bundle.getVersion}"
504         ).mkString("\n", "\n", "")
505       }"
506 
507       System.err.println(msg)
508       log.warn(msg)
509 
510       if (secondAttemptInstalled.nonEmpty && cmdLine.strict) {
511         // in strict mode, nor resolved bundles fail the whole container
512         log.error("Shutting down container due to unresolved bundles.")
513         framework.stop()
514       }
515     }
516 
517     log.info("Laucher finished starting of framework and bundles. Awaiting framework termination now.")
518     // Framework and bundles started
519 
520     framework
521   }
