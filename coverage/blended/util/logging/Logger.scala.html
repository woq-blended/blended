86  * Factory apply-methods try to find a SLF4J-Logger on the classpath
87  * and fall-back to use the `java.util.logging` API if no SLF4J logger could be loaded.
88  */
89 object Logger {
90 
91   private[this] val mdcPropsMut : mutable.Map[String, String] = mutable.Map.empty[String, String]
92   private[this] var mdcMapIntern : Map[String, String] = Map.empty
93 
94   /**
95    * Create a Logger instance by deriving the logger name from the fully qualified class name.
96    */
97   def apply[T : ClassTag] : Logger = {
98     val name = scala.reflect.classTag[T].runtimeClass.getName
99     apply(name)
100   }
101 
102   /**
103    * Create a Logger instance with the given name.
104    */
105   def apply(name : String) : Logger = {
106     try {
107       // we expect class loading errors if no slf4j is present
108       new LoggerSlf4j(slf4j.LoggerFactory.getLogger(name))
109     } catch {
110       case _ : NoClassDefFoundError | _ : ClassNotFoundException =>
111         try {
112           // fall back to jul
113           new LoggerJul(jul.Logger.getLogger(name))
114         } catch {
115           case _ : NoClassDefFoundError | _ : ClassNotFoundException =>
116             new LoggerNoOp(name)
117         }
118     }
119   }
120 
121   def mdcProps : Map[String, String] = mdcMapIntern
122 
123   def setProps(props: Map[String, String]) : Unit = synchronized {
124     props.foreach{ case (k,v) => mdcPropsMut.put(k,v) }
125     mdcMapIntern = mdcPropsMut.toMap
126   }
127 
128   def clearMdc() : Unit = synchronized {
129     mdcPropsMut.clear()
130   }
