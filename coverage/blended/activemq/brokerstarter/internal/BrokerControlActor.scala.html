83 class BrokerControlActor(brokerCfg : BrokerConfig, cfg : OSGIActorConfig, sslCtxt : Option[SSLContext])
84   extends Actor {
85 
86   private[this] val log = Logger[BrokerControlActor]
87   private[this] var broker : Option[BrokerService] = None
88   private[this] var svcReg : Option[ServiceRegistration[_]] = None
89   private[this] val uuid = UUID.randomUUID().toString()
90 
91   override def toString : String = s"BrokerControlActor($brokerCfg)"

93   private[this] def startBroker() : Unit = {
94 
95     val oldLoader = Thread.currentThread().getContextClassLoader()
96 
97     val cfgDir = cfg.ctContext.profileConfigDirectory
98     val uri = s"file://$cfgDir/${brokerCfg.file}"
99 
100     try {
101       log.info(s"Starting ActiveMQ broker [${brokerCfg.brokerName}] with config file [$uri] ")
102 
103       Thread.currentThread().setContextClassLoader(classOf[DefaultBrokerFactory].getClassLoader())
104 
105       BrokerFactory.setStartDefault(false)
106       val brokerFactory = new XBeanBrokerFactory()
107 
108       val b : BrokerService = brokerFactory.createBroker(new URI(uri))
109       broker = Some(b)
110 
111       if (brokerCfg.withAuthentication) {
112         val plugins : List[BrokerPlugin] = new JaasAuthenticationPlugin(brokerCfg) :: Option(b.getPlugins()).map(_.toList).getOrElse(List.empty)
113         b.setPlugins(plugins.toArray)
114       } else {
115         log.info(s"The broker [${brokerCfg.brokerName}] will start without authentication")
116       }
117 
118       sslCtxt.foreach { ctxt =>
119         val amqSslContext = new org.apache.activemq.broker.SslContext()
120         amqSslContext.setSSLContext(ctxt)
121         b.setSslContext(amqSslContext)
122       }
123 
124       val actor = context.self
125 
126       // TODO: set Datadirectories from Code ??
127       b.setBrokerName(brokerCfg.brokerName)
128       b.setStartAsync(false)
129       b.setSchedulerSupport(true)
130       b.start()
131       b.waitUntilStarted()
132       log.info(s"ActiveMQ broker [${brokerCfg.brokerName}] started successfully.")
133       actor ! BrokerControlActor.BrokerStarted(uuid)
134 
135     } catch {
136       case NonFatal(t) =>
137         log.warn(t)(s"Error starting ActiveMQ broker [${brokerCfg.brokerName}]")
138         throw t
139     } finally {
140       Thread.currentThread().setContextClassLoader(oldLoader)
141     }
142   }
143 
144   private[this] def registerService(brokerCfg : BrokerConfig) : Unit = {
145     if (svcReg.isEmpty) {
146       new Object with ServiceProviding {
147 
148         override protected def capsuleContext : CapsuleContext = new SimpleDynamicCapsuleContext()
149 
150         override protected def bundleContext : BundleContext = cfg.bundleContext
151 
152         val url = s"vm://${brokerCfg.brokerName}?create=false"
153 
154         val jmsCfg : ConnectionConfig = brokerCfg.copy(properties = brokerCfg.properties + ("brokerURL" -> url))
155 
156         val cf = new BlendedSingleConnectionFactory(
157           config = jmsCfg,
158           bundleContext = Some(cfg.bundleContext)
159         )(system = cfg.system)
160 
161         svcReg = Some(cf.providesService[ConnectionFactory, IdAwareConnectionFactory](Map(
162           "vendor" -> brokerCfg.vendor,
163           "provider" -> brokerCfg.provider,
164           "brokerName" -> brokerCfg.brokerName
165         )))
166       }
167     }
168   }
169 
170   private[this] def stopBroker() : Unit = {
171 
172     broker.foreach { b =>
173       log.info(s"Stopping ActiveMQ Broker [${brokerCfg.brokerName}]")
174       try {
175         b.stop()
176         b.waitUntilStopped()
177       } catch {
178         case NonFatal(t) =>
179           log.error(t)(s"Error stopping ActiveMQ broker [${brokerCfg.brokerName}]")
180       } finally {
181         try {
182           log.info(s"Removing OSGi service for Activemq Broker [${brokerCfg.brokerName}]")
183           svcReg.foreach(_.unregister())
184         } catch {
185           case _ : IllegalStateException => // was already unregistered
186         }
187       }
188     }
189 
190     broker = None
191     svcReg = None
192   }
193 
194   override def preRestart(reason : Throwable, message : Option[Any]) : Unit = {
195     log.error(reason)(s"Error starting Active MQ broker [${brokerCfg.brokerName}]")
196     super.preRestart(reason, message)
197   }
198 
199   override def postStop() : Unit = broker.foreach { b =>
200     b.stop()
201     b.waitUntilStopped()
202   }

206   override def receive : Receive = {
207     case BrokerControlActor.StartBroker =>
208       log.trace(s"Received StartBroker Command for [$brokerCfg] [$jvmId][$uuid-${BrokerControlActor.debugCnt.incrementAndGet()}]")
209       if (broker.isEmpty) { startBroker() }
210     case started : BrokerControlActor.BrokerStarted =>
211       log.trace(s"Received BrokerStarted Event for [$brokerCfg] [$jvmId][$uuid-${BrokerControlActor.debugCnt.incrementAndGet()}]")
212       if (started.uuid == uuid) {
213         broker.foreach { _ => registerService(brokerCfg) }
214       }
215     case BrokerControlActor.StopBroker =>
216       log.trace(s"Received StopBroker Command for [$brokerCfg] [$jvmId][$uuid-${BrokerControlActor.debugCnt.incrementAndGet()}]")
217       stopBroker()
218   }
