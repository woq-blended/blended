27   @throws[Exception]
28   override def addConnection(context: ConnectionContext, info: ConnectionInfo): Unit = {
29 
30     // First we check if we need to create a new Security context or can reuse the existing one
31     Option(context.getSecurityContext()) match {
32       case None =>
33         log.debug(s"Authenticating user [${info.getUserName()}] for broker [$getBrokerName()]")
34         val secCtxt : SecurityContext = authenticate(info.getUserName(), info.getPassword(), null)
35         context.setSecurityContext(secCtxt)
36         securityContexts.add(secCtxt)
37       case Some(_) =>
38     }
39 
40     // At this point we either have thrown an exception or we do have a security context set
41     Try {
42       super.addConnection(context, info)
43     } match {
44       case Success(_) => // do nothing
45       case Failure(e) =>
46         log.warn(s"Failed to add connection for user [${info.getUserName()}] in broker [$getBrokerName()] : [${e.getMessage()}]")
47         securityContexts.remove(context.getSecurityContext())
48         context.setSecurityContext(null)
49         throw e
50     }
51   }

53   @throws[SecurityException]
54   override def authenticate(username: String, password: String, peerCertificates: Array[X509Certificate]): SecurityContext = {
55 
56     (Option(username), Option(password)) match {
57       case (Some(u), _) =>
58         log.debug(s"Trying to authenticate [$username] for broker [${getBrokerName()}]")
59         try {
60           val lc = new LoginContext("Test", new PasswordCallbackHandler(u, Option(password).map(_.toCharArray()).getOrElse(Array.empty)))
61           lc.login()
62           val subj : Subject = lc.getSubject()
63           val g : Set[Principal] = subj.getPrincipals().asScala.filter(_.isInstanceOf[GroupPrincipal]).toSet
64           new SecurityContext(u) {
65             override def getPrincipals: util.Set[Principal] = g.asJava
66           }
67         } catch {
68           case NonFatal(t) =>
69             log.warn(s"Error logging in [$u] to [${brokerCfg.brokerName}] : [${t.getMessage()}]")
70             throw new SecurityException(t)
71         }
72       case (None, _) =>
73         brokerCfg.anonymousUser match {
74           case Some(u) =>
75             log.debug(s"Authenticating anonymous user name [$u] to broker [${getBrokerName()}] with groups [${brokerCfg.anonymousGroups}]")
76             val groups : Set[Principal] = brokerCfg.anonymousGroups.map(g => new GroupPrincipal(g)).toSet
77             new SecurityContext(u) {
78               override def getPrincipals: util.Set[Principal] = groups.asJava
79             }
80           case None =>
81             throw new SecurityException(s"Anonymous access to broker [${getBrokerName()}] is disabled")
82         }
83     }
84   }
