19 class RoundtripConnectionVerifier(
20   probeMsg : () => FlowEnvelope,
21   verify : FlowEnvelope => Boolean,
22   requestDest : JmsDestination,
23   responseDest : JmsDestination,
24   headerConfig : FlowHeaderConfig,
25   retryInterval : FiniteDuration = 1.second,

31   private val log : Logger = Logger[RoundtripConnectionVerifier]

35   override def verifyConnection(cf : IdAwareConnectionFactory)(implicit eCtxt : ExecutionContext) : Future[Boolean] = {
36     after[Unit](10.millis, system.scheduler) {
37       Future {
38         probe(cf)
39       }
40     }
41 
42     verified.future
43   }

45   private def probe(cf : IdAwareConnectionFactory) : Unit = {
46 
47     implicit val materializer : Materializer = ActorMaterializer()
48     implicit val eCtxt : ExecutionContext = system.dispatcher
49 
50     val id : String = UUID.randomUUID().toString()
51 
52     val probeEnv : FlowEnvelope = probeMsg()
53       .withHeader(corrIdHeader(headerConfig.prefix), id, true).get
54       .withHeader(replyToHeader(headerConfig.prefix), responseDest.asString).get
55 
56     val pSettings : JmsProducerSettings = JmsProducerSettings(
57       log = envLogger,
58       headerCfg = headerConfig,
59       connectionFactory = cf,
60       jmsDestination = Some(requestDest),
61       timeToLive = Some(receiveTimeout * 2),
62       destinationResolver = s => new MessageDestinationResolver(s),
63       logLevel = _ => LogLevel.Debug
64     )
65 
66     log.info(s"Running verification probe for connection [${cf.vendor}:${cf.provider}]")
67 
68     sendMessages(pSettings, envLogger, probeEnv) match {
69       case Success(s) =>
70         log.info(s"Request message sent successfully to [${requestDest.asString}] : [$probeEnv]")
71         s.shutdown()
72 
73         val collector : Collector[FlowEnvelope] = receiveMessages(
74           headerCfg = headerConfig,
75           cf = cf,
76           dest = responseDest,
77           log = envLogger,
78           listener = 1,
79           selector = Some(s"JMSCorrelationID='$id'"),
80           timeout = Some(receiveTimeout)
81         )
82 
83         collector.result.onComplete {
84           case Success(l) => l match {
85             case Nil =>
86               log.debug(s"No response received to verify connection [${cf.vendor}:${cf.provider}]")
87               scheduleRetry(cf)
88             case h :: _ =>
89               val result : Boolean = verify(h)
90               log.info(s"Verification result for client connection [${cf.vendor}:${cf.provider}] is [$result]")
91               verified.complete(Success(result))
92           }
93 
94           case Failure(t) =>
95             log.debug(s"Failed to receive verification response to verify connection [${cf.vendor}:${cf.provider}] : [${t.getMessage()}]")
96             scheduleRetry(cf)
97         }
98 
99       case Failure(t) =>
100         log.debug(s"Failed to send verification request to verify connection [${cf.vendor}:${cf.provider}] : [${t.getMessage()}]")
101         scheduleRetry(cf)
102     }
103   }
