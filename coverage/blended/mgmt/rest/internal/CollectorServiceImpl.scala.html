61   override def processContainerInfo(info : ContainerInfo) : ContainerRegistryResponseOK = {
62     log.debug(s"Processing container info: ${info}")
63 
64     eventStream.foreach {
65       _.publish(UpdateContainerInfo(info))
66     }
67 
68     // next call has side-effect
69     val updated = updater.updateContainerState(info)
70     // we now have updated outstanding actions
71     val actions = updated.outstandingActions
72 
73     val state = RemoteContainerState(info, actions)
74     remoteContainerStatePersistor.updateState(state)
75 
76     // ...and send these outstanding actions to the remote container
77     ContainerRegistryResponseOK(info.containerId, actions)
78   }
