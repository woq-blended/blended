159   def rolloutProfileRoute : Route = {
160     path("rollout" / "profile") {
161       post {
162         requirePermission("rollout") {
163           entity(as[RolloutProfile]) { rolloutProfile =>
164             // check existence of profile
165             getRuntimeConfigs().find(rc => rc.name == rolloutProfile.profileName && rc.version == rolloutProfile.profileVersion) match {
166               case None =>
167                 reject(ValidationRejection(s"Unknown profile ${rolloutProfile.profileName} ${rolloutProfile.profileVersion}"))
168               case Some(rc) =>
169                 // check existence of overlays
170                 findMissingOverlayRef(rolloutProfile.overlays) match {
171                   case Some(r) =>
172                     log.error(s"The rollout request contains an overlay reference [${r}] which is not known. " +
173                       s"You have to register an overlay before using it in a rollout. " +
174                       s"Known overlay configs are: [${getOverlayConfigs().map(c => c.overlayRef)}]")
175                     reject(ValidationRejection(s"Unknown overlay ${r.name} ${r.version}"))
176                   case None =>
177                     // all ok, complete
178                     complete {
179                       log.debug("looks good, rollout can continue")
180                       rolloutProfile.containerIds.foreach { containerId =>
181 
182                         // Make sure, we have the required runtime config
183                         addUpdateAction(
184                           containerId = containerId,
185                           updateAction = AddRuntimeConfig(
186                             UUID.randomUUID().toString(),
187                             runtimeConfig = rc
188                           )
189                         )
190 
191                         // Also register all required overlay configs
192                         val ocs = getOverlayConfigs()
193                         rolloutProfile.overlays.map { o =>
194                           val oc = ocs.find(oc => oc.overlayRef == o).get
195                           addUpdateAction(
196                             containerId = containerId,
197                             updateAction = AddOverlayConfig(
198                               UUID.randomUUID().toString(),
199                               overlay = oc
200                             )
201                           )
202                         }
203 
204                         // finally stage the new runtime config
205                         addUpdateAction(
206                           containerId = containerId,
207                           updateAction = StageProfile(
208                             UUID.randomUUID().toString(),
209                             profileName = rolloutProfile.profileName,
210                             profileVersion = rolloutProfile.profileVersion,
211                             overlays = rolloutProfile.overlays
212                           )
213                         )
214 
215                       }
216 
217                       s"Recorded ${rolloutProfile.containerIds.size} rollout actions"
218                     }
219                 }
220             }
221           }
222         }
223       }
224     }

230       withSizeLimit(1024 * 1024 * 100) {

263   /**
264    * Process the input stream as it it were a ZIP file stram containing the deploymentpack for a profile.
265    * If the deployment was successful, this method returns the profile name and version as tuple, else the exception is returned.
266    *
267    * @return Tuple of profile name and version.
268    */
269   def processDeploymentPack(repoId : String, zipFile : File) : Try[(String, String)] = {
270     log.debug(s"About to process deploymentpack as inputstream for repoId: ${repoId}")
271 
272     // create temp file to find a free name, than delete and create dir with that name
273     val tempDir = File.createTempFile("upload", "")
274     tempDir.delete()
275     tempDir.mkdirs()
276 
277     val unzipped = Unzipper.unzip(
278       archive = zipFile,
279       targetDir = tempDir,
280       selectedFiles = List(),
281       fileSelector = None,
282       placeholderReplacer = None
283     )
284 
285     val result = unzipped.flatMap { files =>
286       Try {
287         log.debug(s"Extraced files: ${files}")
288         val profileConfFile = files.find(f => f.getName() == "profile.conf").get
289         val config = ConfigFactory.parseFile(profileConfFile)
290         val local = RuntimeConfigCompanion.
291           read(config).
292           flatMap(_.resolve()).
293           flatMap(c => Try { LocalRuntimeConfig(c, tempDir) }).
294           get
295 
296         val issues =
297           local.validate(includeResourceArchives = true, explodedResourceArchives = false) ++
298             local.resolvedRuntimeConfig.allBundles.filter(b => !b.url.startsWith("mvn:")).map(u => s"Unsupported bundle URL: ${u}") ++
299             local.runtimeConfig.resources.filter(b => !b.url.startsWith("mvn:")).map(u => s"Unsupported resource URL: ${u}")
300 
301         if (!issues.isEmpty) sys.error(issues.mkString("; "))
302 
303         // everything is ok
304         log.debug(s"Uploaded profile.conf: ${local}")
305         registerRuntimeConfig(local.runtimeConfig)
306 
307         // we know the urls all start with "mvn:"
308         // now install bundles
309         local.resolvedRuntimeConfig.allBundles.map { b =>
310           val file = local.bundleLocation(b)
311           val path = MvnGav.parse(b.url.substring("mvn:".size)).get.toUrl("")
312           installBundle(repoId, path, file, b.artifact.sha1Sum)
313         }
314         // install resources
315         local.runtimeConfig.resources.map { r =>
316           val file = local.resourceArchiveLocation(r)
317           val path = MvnGav.parse(r.url.substring("mvn:".size)).get.toUrl("")
318           installBundle(repoId, path, file, r.sha1Sum)
319         }
320         local.runtimeConfig.name -> local.runtimeConfig.version
321       }
322     }
323     deleteRecursive(tempDir)
324     result
325   }

327   def deleteRecursive(files : File*) : Unit = files.foreach { file =>
328     if (file.isDirectory()) {
329       file.listFiles() match {

331         case files => deleteRecursive(files : _*)
332       }
333     }
