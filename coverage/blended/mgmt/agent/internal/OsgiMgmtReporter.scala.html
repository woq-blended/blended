11 /**
12  * Actor, that collects various container information and send's it to a remote management container.
13  *
14  * Sources of information:
15  *
16  * * [[ServiceInfo]] from the Akka event stream
17  * * `([[Long]], List[[[Profile]]])` from the Akka event stream
18  *
19  * Send to remote container:
20  *
21  * * [[ContainerInfo]] send via HTTP POST request
22  *
23  * Configuration:
24  *
25  * This actor reads a configuration class [[MgmtReporterConfig]] from the [[OSGIActorConfig]].
26  * Only if all necessary configuration are set (currently `initialUpdateDelayMsec` and `updateIntervalMsec`),
27  * the reporter sends information to the management container.
28  * The target URL of the management container is configured with the `registryUrl` config entry.
29  *
30  */
31 class OsgiMgmtReporter(cfg : OSGIActorConfig) extends OSGIActor(cfg) with MgmtReporter {
32 
33   import MgmtReporter._
34 
35   private[this] val log = Logger[OsgiMgmtReporter]
36 
37   val config : Try[MgmtReporterConfig] = MgmtReporterConfig.fromConfig(cfg.config) match {
38     case f @ Failure(e) =>
39       log.warn(e)("Incomplete management reporter config. Disabled connection to management server.")
40       f
41     case x =>
42       log.info(s"Management reporter config: $x")
43       x
44   }
45 
46   override protected val ctContext : ContainerContext = cfg.ctContext
