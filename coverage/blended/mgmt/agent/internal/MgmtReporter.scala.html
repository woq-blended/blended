20 /**
21  * Actor, that collects various container information and send's it to a remote management container.
22  *
23  * Sources of information:
24  *
25  * * [[ServiceInfo]] from the Akka event stream
26  * * `([[Long]], List[[[Profile]]])` from the Akka event stream
27  *
28  * Send to remote container:
29  *
30  * * [[ContainerInfo]] send via HTTP POST request
31  *
32  * Configuration:
33  *
34  * This actor reads a configuration class [[MgmtReporterConfig]] from the [[OSGIActorConfig]].
35  * Only if all necessary configuration are set (currently `initialUpdateDelayMsec` and `updateIntervalMsec`),
36  * the reporter sends information to the management container.
37  * The target URL of the management container is configured with the `registryUrl` config entry.
38  *
39  */
40 trait MgmtReporter extends Actor with PrickleSupport {

62   override def preStart() : Unit = {
63     super.preStart()
64 
65     config foreach { config =>
66       if (config.initialUpdateDelayMsec < 0 || config.updateIntervalMsec <= 0) {
67         log.warn("Inappropriate timing configuration detected. Disabling automatic container status reporting")
68       } else {
69         log.info(s"Activating automatic container status reporting with update interval [${config.updateIntervalMsec}]")
70         context.system.scheduler.scheduleOnce(config.initialUpdateDelayMsec.millis, self, Tick)
71       }
72     }
73 
74     context.system.eventStream.subscribe(context.self, classOf[ServiceInfo])
75     context.system.eventStream.subscribe(context.self, classOf[ProfileInfo])
76     context.system.eventStream.subscribe(context.self, classOf[UpdateActionApplied])
77 
78     context.become(reporting(MgmtReporterState(
79       serviceInfos = Map.empty,
80       lastProfileInfo = ProfileInfo(0L, Nil),
81       appliedUpdateActionIds = List.empty
82     )))
83   }
84 
85   override def postStop() : Unit = {
86     context.system.eventStream.unsubscribe(context.self)
87     super.postStop()
88   }

90   private def handleTick(state : MgmtReporterState) : Receive = {
91     case Tick =>
92       config.foreach { cfg =>
93 
94         val info = ContainerInfo(
95           containerId = ctContext.uuid,
96           properties = ctContext.properties,
97           serviceInfos = state.serviceInfos.values.toList,
98           profiles = state.lastProfileInfo.profiles,
99           timestampMsec = System.currentTimeMillis(),
100           appliedUpdateActionIds = state.appliedUpdateActionIds
101         )
102         log.debug(s"Performing report [$info].")
103 
104         val entity = Marshal(info).to[MessageEntity]
105 
106         val request = entity.map { entity =>
107           HttpRequest(
108             uri = cfg.registryUrl,
109             method = HttpMethods.POST,
110             entity = entity
111           )
112         }
113 
114         // TODO think about ssl
115         val responseFuture = request.flatMap { request =>
116           Http(context.system).singleRequest(request)
117         }.map(r => r -> state.appliedUpdateActionIds)
118 
119         import akka.pattern.pipe
120         responseFuture.pipeTo(self)
121 
122         context.system.scheduler.scheduleOnce(cfg.updateIntervalMsec.millis, self, Tick)
123       }
124   }
125 
126   private def handleEvents(state : MgmtReporterState) : Receive = {
127     // from event stream
128     case serviceInfo @ ServiceInfo(name, _, _, _, _) =>
129       log.debug(s"Update service info for: $name")
130       context.become(reporting(state.copy(state.serviceInfos ++ Map(name -> serviceInfo))))
131 
132     // from event stream
133     case pi @ ProfileInfo(timestamp, _) =>
134       if (timestamp > state.lastProfileInfo.timeStamp) {
135         log.debug("Update profile info to: " + pi)
136         context.become(reporting(state.copy(lastProfileInfo = pi)))
137       } else {
138         log.debug(s"Ignoring profile info with timestamp [${timestamp.underlying()}] which is older than "
139           + s"[${state.lastProfileInfo.timeStamp.underlying()}]: $pi")
140       }
141 
142     case UpdateActionApplied(id, _) =>
143       context.become(reporting(state.copy(appliedUpdateActionIds = id :: state.appliedUpdateActionIds)))
144   }
145 
146   private def handleHttpRequests(state : MgmtReporterState) : Receive = {
147     case (response @ HttpResponse(status, _, entity, _), appliedIds : List[String]) =>
148       status match {
149         case StatusCodes.OK =>
150           // As the server accepted also the list of applied update action IDs
151           // we remove those from the list
152           context.become(reporting(state.copy(
153             appliedUpdateActionIds = state.appliedUpdateActionIds.filterNot(appliedIds.contains)
154           )))
155 
156           import akka.pattern.pipe
157 
158           // OK; unmarshal and process
159           Unmarshal(entity).to[ContainerRegistryResponseOK].pipeTo(self)
160 
161         case _ =>
162           log.warn(s"Non-OK response ${config.map(c => c.registryUrl).getOrElse("")} from node: $response")
163           response.discardEntityBytes()
164       }
165 
166     case ContainerRegistryResponseOK(id, actions) =>
167       log.debug(s"Reported [$id] to management node")
168       if (actions.nonEmpty) {
169         log.info(s"Received ${actions.size} update actions from management node: $actions")
170         actions.foreach { action : UpdateAction =>
171           log.debug(s"Publishing event to event stream: $action")
172           context.system.eventStream.publish(action)
173         }
174       }
175   }
176 
177   def receive : Receive = Actor.emptyBehavior
178 
179   private def reporting(state : MgmtReporterState) : Receive = LoggingReceive {
180     handleTick(state)
181       .orElse(handleHttpRequests(state))
182       .orElse(handleEvents(state))
183   }
184 }
