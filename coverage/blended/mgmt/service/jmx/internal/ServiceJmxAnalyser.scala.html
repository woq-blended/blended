38   def serviceInfo(instance : ObjectInstance, svcConfig : SingleServiceConfig, template : ServiceTypeTemplate) : ServiceInfo = {
39 
40     def transformOne(name : List[String], value : Object) : List[(String, String)] = value match {
41       case cd if cd.isInstanceOf[CompositeDataSupport] =>
42         val data = cd.asInstanceOf[CompositeDataSupport]
43         data.getCompositeType().keySet().asScala.map { k =>
44           data.get(k) match {
45             case innerCd if innerCd.isInstanceOf[CompositeData] => transformOne(name ::: List(k), innerCd)
46             case v => List((name.mkString(".") + "." + k, v.toString()))
47           }
48         }.toList.flatten
49       case a => List((name.mkString("."), a.toString()))
50     }
51 
52     def transformAll(name : List[String], attrs : List[Attribute]) : List[(String, String)] = attrs match {
53       case head :: tail =>
54         val newName = name ::: List(head.getName())
55         transformOne(newName, head.getValue) ::: transformAll(name, tail)
56       case Nil => List.empty
57     }
