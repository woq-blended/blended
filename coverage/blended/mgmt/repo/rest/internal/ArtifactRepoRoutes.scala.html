18     artifactRepos.find(a => a.repoId == repo) match {
19       case None =>
20         log.debug(s"No repository with id: [$repo]")

22           StatusCodes.NotFound -> s"No repository with id: [$repo]"
23         }

31             log.debug(s"Could not find file at path: [$path]")
32             repository.listFiles(path) match {
33               case Iterator.empty => complete(StatusCodes.NotFound)
34               case files =>
35                 //                respondWithMediaType(MediaTypes.`text/html`) &

37                   val indexBase = repository.repoId + "/" + path

39                   val goUp = if (path.isEmpty()) "" else """<br/><a href="..">..</a>"""
40 
41                   val renderedFiles = files.map { f =>
42                     s"""<br/><a href="$f">$f</a>"""
43                   }.mkString
44 
45                   val html = s"""|<html>
46                     |<head><title>Index of $indexBase</title></head>
47                     |<body>
48                     |<h1>Index of $indexBase</h1>
49                     |<hr>
50                     |$goUp
51                     |$renderedFiles
52                     |<hr>
53                     |</body>
