27   def withCheckedFilePath[T](path : String)(f : File => T) : Try[T] = Try {
28     val base = baseDir.toURI().normalize()
29     val toCheck = new File(baseDir, path).toURI().normalize()
30     if (!toCheck.getPath().startsWith(base.getPath())) sys.error("invalid path given")
31     val pathToCheck = toCheck.getPath().substring(base.getPath().length())
32     if (pathToCheck.startsWith("..") || pathToCheck.startsWith("/..")) sys.error("invalid path given")
33     new File(toCheck)
34   }.map(f)

37     withCheckedFilePath(path) { file =>

42   /**
43    * Try to find a file under the given path.
44    * @return The checksum of the found file or [[None]] if the file was not found.
45    */
46   def findFileSha1Checksum(path : String) : Option[String] = {
47     findFile(path) flatMap { file =>
48       val sha1Stream = new DigestInputStream(new BufferedInputStream(new FileInputStream(file)), MessageDigest.getInstance("SHA"))
49       try {
50         while (sha1Stream.read != -1) {}
51         Option(bytesToString(sha1Stream.getMessageDigest.digest))
52       } catch {
53         case NonFatal(_) => None
54       } finally {
55         sha1Stream.close()
56       }
57     }
58   }

65   def listFiles(path : String) : Iterator[String] = {
66     withCheckedFilePath(path) { file =>
67       if (!file.exists()) Iterator.empty else {
68         def getFiles(dir : Path) : Iterator[Path] = {
69           val files = Files.newDirectoryStream(dir).iterator().asScala
70           files.flatMap { f =>
71             val isDir = Files.isDirectory(f)
72             val file = Iterator(f).filter(f => !isDir)
73             val childs = if (isDir) getFiles(f) else Iterator.empty
74             file ++ childs
75           }
76         }
77 
78         val files = getFiles(file.toPath())
79         val baseFile = file.toURI()
80         files.map(f => baseFile.relativize(f.toUri().normalize()).getPath())
81       }
82     }.getOrElse(Iterator.empty)
83   }

85   override def uploadFile(path : String, fileContent : InputStream, sha1Sum : Option[String]) : Try[Unit] =
86     withCheckedFilePath(path) { file =>
87       if (file.exists()) {
88         // file already exists
89         if (file.isFile()) {
90           // collision
91           // no checksum or checksum differs means we cannot ensure the artifact is identical, so we abort
92           sha1Sum match {
93             case None =>
94               throw new ArtifactCollisionException(s"There is already an artifact installed under path: $path")
95             case sum =>
96               val existingSum = findFileSha1Checksum(path)
97               if (sum != existingSum) {
98                 log.info(s"Artifact with different checksum (existing: $existingSum, new: $sum) already present: $path")
99                 throw new ArtifactCollisionException(s"There is already an artifact with a different checksum installed under path: $path")
100               } else {
101                 // else nothing to do
102                 log.info(s"Artifact with same checksum already present: $path")
103               }
104           }
105         } else {
106           log.error(s"Artifact path [$path] is a directory. Cannot upload")
107           // e.g. an existing directory
108           throw new IllegalArgumentException(s"The given path [$path] cannot be used as artifact path")
109         }
110       } else {
111         // file does not exists, installing now
112         Option(file.getParentFile()).map(_.mkdirs())
113         val fos = new FileOutputStream(file)
114         try {
115           log.debug(s"About to save file: $file")
116           StreamCopy.copy(fileContent, fos)
117         } finally {
118           fos.close()
119         }
120       }
121     }
