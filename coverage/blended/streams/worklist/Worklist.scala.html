61   def flow(name: String, log : FlowEnvelopeLogger): Flow[WorklistEvent, WorklistEvent, NotUsed] =
62     Flow.fromGraph(new WorklistGraphStage(name, log))
63 
64   private class WorklistGraphStage(name: String, log : FlowEnvelopeLogger) extends GraphStage[FlowShape[WorklistEvent, WorklistEvent]] {
65 
66     private val in = Inlet[WorklistEvent](s"$name.in")
67     private val out = Outlet[WorklistEvent](s"$name.out")
68 
69     override def shape : FlowShape[WorklistEvent, WorklistEvent] = FlowShape(in, out)
70 
71     private object CurrentItemState {
72       def apply(item: WorklistItem): CurrentItemState = CurrentItemState(item, WorklistStateStarted, System.currentTimeMillis())
73     }
74 
75     private case class CurrentItemState(
76       item: WorklistItem,
77       state: WorklistState,
78       created: Long
79     )
80 
81     private object CurrentWorklistState {
82       def apply(
83         started: WorklistStarted
84       ): CurrentWorklistState = new CurrentWorklistState(
85         started.worklist.id,
86         mutable.Map(started.worklist.items.map(i => i.id -> CurrentItemState(i)) : _*),
87         timeout = started.timeout
88       )
89     }
90 
91     private case class CurrentWorklistState(
92       id : String,
93       items : mutable.Map[String, CurrentItemState],
94       timeout : FiniteDuration
95     )
96 
97     private object Tick
98 
99     /* -----------------------------------------------------------------------------------------*/
100     override def createLogic(inheritedAttributes : Attributes) : GraphStageLogic = new TimerGraphStageLogic(shape) {
101 
102       private val activeWorklists : mutable.Map[String, CurrentWorklistState] = mutable.Map.empty
103       private val outEvents : mutable.Queue[WorklistEvent] = mutable.Queue.empty
104 
105       override def preStart() : Unit = {
106         schedulePeriodically(Tick, 100.millis)
107         pull(in)
108       }
109 
110       override def postStop() : Unit = {
111         cancelTimer(Tick)
112       }
113 
114       override protected def onTimer(timerKey : Any) : Unit = {
115 
116         timerKey match {
117           case Tick =>
118             activeWorklists.values.foreach { awl =>
119               val now = System.currentTimeMillis()
120               // We only examine items in started state
121               val timedOut = awl.items.values
122                 .filter(i => i.state == WorklistStateStarted)
123                 // and of those we want only the timed out ones
124                 .filter(i => now - i.created > awl.timeout.toMillis)
125 
126               // Then we update the timed out items in the worklist
127               timedOut.foreach(i => awl.items.put(i.item.id, i.copy(state = WorklistStateTimeout)))
128 
129               // finally we check the awl
130               checkWorklist(awl)
131             }
132           case _ =>
133         }
134       }
135 
136       private def pushEvent(event : WorklistEvent) : Unit = {
137         if (outEvents.isEmpty && isAvailable(out)) {
138           push(out, event)
139         } else {
140           outEvents.enqueue(event)
141         }
142       }
143 
144       private def startWorklist(event : WorklistStarted) : Unit = {
145         activeWorklists.get(event.worklist.id) match {
146           case None =>
147             log.underlying.debug(s"Starting Worklist [${event.worklist.id}]")
148             activeWorklists += (event.worklist.id -> CurrentWorklistState(event))
149             pushEvent(event)
150           case Some(awl) =>
151             log.underlying.warn(s"Received start event for worklist [${awl.id}] that is already in use, ignoring StartWorklist event")
152         }
153       }
154 
155       def sendEvent(wl : CurrentWorklistState, state : WorklistState) : Unit = {
156         // Todo : Set fail reason
157         val event = WorklistTerminated(
158           worklist = Worklist(id = wl.id, wl.items.values.map(_.item).toSeq),
159           state = state,
160           reason = None
161         )
162         pushEvent(event)
163         activeWorklists -= wl.id
164         log.underlying.debug(s"Sent WorklistEvent [$event], [${activeWorklists.size}] remaining worklists ")
165       }
166 
167       private def checkWorklist(wl : CurrentWorklistState) : Unit = {
168 
169         val existsByState : WorklistState => Boolean = state => wl.items.exists(i => i._2.state == state)
170 
171         // First we check for timeouts
172         existsByState(WorklistStateTimeout) match {
173           case true =>
174             sendEvent(wl, WorklistStateTimeout)
175           case false =>
176             // We have failed if at least one item has failed
177             existsByState(WorklistStateFailed) match {
178               case true =>
179                 sendEvent(wl, WorklistStateFailed)
180               case false =>
181                 // We have handled all errors here
182                 existsByState(WorklistStateStarted) match {
183                   // we still have unprocessed items in this worklist
184                   case true =>
185                   // Everything that is not started, must have completed, so we are done
186                   case false =>
187                     sendEvent(wl, WorklistStateCompleted)
188                 }
189             }
190         }
191       }
192 
193       private def processSteps(event : WorklistStepCompleted) : Unit = {
194         activeWorklists.get(event.worklist.id) match {
195           case None =>
196             log.underlying.trace(s"Worklist [${event.worklist.id}] is not active, ignoring event [$event]")
197           case Some(awl) =>
198             // Event should be anything but started
199             if (event.state == WorklistStateStarted) {
200               log.underlying.trace(s"Unexpected state [${event.state}] in process complete event [$event]")
201             } else {
202 
203               val filter : String => Boolean = id => awl.items.isDefinedAt(id)
204 
205               val missingEventIds = event.worklist.items.map(_.id).filterNot(filter)
206               if (missingEventIds.nonEmpty) {
207                 log.underlying.warn(s"Event references untracked item ids [${missingEventIds.mkString(",")}] for [$event]")
208               }
209 
210               val newEvents : Map[String, CurrentItemState] = event.worklist.items.filter(i => filter(i.id)).map { i =>
211                 val s = awl.items.get(i.id).get
212                 val newItem = s.copy(item = i, state = event.state)
213                 (newItem.item.id -> newItem)
214               }.toMap
215 
216               val newAwl = awl.copy(
217                 items = awl.items.filter { case (k, _) => !newEvents.contains(k) } ++ newEvents
218               )
219 
220               activeWorklists.put(event.worklist.id, newAwl)
221               checkWorklist(newAwl)
222             }
223         }
224       }
225 
226       private def terminateWorklist(event : WorklistTerminated) : Unit = {
227         activeWorklists.get(event.worklist.id) match {
228           case None      =>
229           case Some(awl) => sendEvent(awl, event.state)
230         }
231       }
232 
233       setHandler(
234         in, new InHandler {
235         override def onPush() : Unit = {
236           val event = grab(in)
237 
238           event match {
239             case s : WorklistStarted       => startWorklist(s)
240             case s : WorklistStepCompleted => processSteps(s)
241             case s : WorklistTerminated    => terminateWorklist(s)
242           }
243 
244           pull(in)
245         }
246       }
247       )
248 
249       setHandler(
250         out, new OutHandler {
251         // If we have any pending events, push then down stream
252         override def onPull() : Unit = if (outEvents.nonEmpty) {
253           push(out, outEvents.dequeue())
254         }
255       }
256       )
257 
258     }
259   }
