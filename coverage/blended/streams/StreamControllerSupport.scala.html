22   val nextInterval : FiniteDuration => BlendedStreamsConfig => FiniteDuration = { interval => streamCfg =>
23 
24     val noise = {
25       val d = rnd.nextDouble().abs
26       (d - d.floor) / (1 / (streamCfg.random * 2)) - streamCfg.random
27     }
28 
29     var newIntervalMillis : Double =
30       if (streamCfg.exponential) {
31         interval.toMillis * 2
32       } else {
33         interval.toMillis + streamCfg.minDelay.toMillis
34       }
35 
36     newIntervalMillis = scala.math.min(
37       streamCfg.maxDelay.toMillis,
38       newIntervalMillis + newIntervalMillis * noise
39     )
40 
41     newIntervalMillis.toLong.millis
42   }

44   private def restart(streamCfg : BlendedStreamsConfig, interval : FiniteDuration, startedAt : Option[Long]) : Unit = {
45     val nextStart : FiniteDuration = startedAt match {
46       case None => nextInterval(interval)(streamCfg)
47       case Some(s) =>
48         if (System.currentTimeMillis() - s < streamCfg.maxDelay.toMillis) {
49           nextInterval(interval)(streamCfg)
50         } else {
51           streamCfg.minDelay
52         }
53     }
54 
55     log.info(s"Scheduling restart of Stream [${name}] in [$nextStart]")
56 
57     context.system.scheduler.scheduleOnce(nextStart, self, StreamController.Start)
58 
59     context.become(starting(streamCfg, nextStart))
60   }

62   def starting(streamCfg : BlendedStreamsConfig, interval : FiniteDuration) : Receive = {
63     case StreamController.Stop =>
64       context.stop(self)
65 
66     case StreamController.Start =>
67       log.info(s"Initializing StreamController [${name}] with config [$streamCfg]")
68 
69       startStream() match {
70         case Success( (mat, killswitch, done) ) =>
71 
72           // We schedule a reset, which will cause the backoff timer to reset
73           val timer : Cancellable = context.system.scheduler.scheduleOnce(streamCfg.resetAfter, self, StreamController.Reset)
74 
75           materialized(mat)
76 
77           done.onComplete {
78             case Success(_) =>
79               self ! StreamController.StreamTerminated(None)
80             case Failure(t) =>
81               self ! StreamController.StreamTerminated(Some(t))
82           }
83 
84           context.become(running(
85             streamCfg = streamCfg,
86             killSwitch = killswitch,
87             interval = interval,
88             startedAt = System.currentTimeMillis(),
89             resetTimer = Some(timer)
90           ))
91 
92         case Failure(t) =>
93           log.warn(s"Stream [$name] terminated with exception [${t.getMessage()}]")
94           restart(streamCfg, interval, None)
95       }
96 
97   }

99   def running(
100     streamCfg : BlendedStreamsConfig,
101     killSwitch : KillSwitch,
102     interval : FiniteDuration,
103     startedAt : Long,
104     resetTimer : Option[Cancellable]
105   ) : Receive = {
106     case StreamController.Stop =>
107       killSwitch.shutdown()
108       context.become(stopping)
109 
110     case StreamController.Reset =>
111       log.debug(s"Resetting timer for stream controller [$name]")
112       context.become(running(
113         streamCfg = streamCfg,
114         killSwitch = killSwitch,
115         interval = streamCfg.minDelay,
116         startedAt = System.currentTimeMillis(),
117         resetTimer = None
118       ))
119 
120     case StreamController.Abort(t) =>
121       killSwitch.abort(t)
122       context.become(stopping)
123 
124     case StreamController.StreamTerminated(t) =>
125       log.info(s"Stream [${name}] terminated ...")
126       if (t.isDefined || (!streamCfg.onFailureOnly)) {
127         t.foreach { e =>
128           log.error(e.getMessage)
129         }
130 
131         resetTimer.foreach(_.cancel())
132         restart(streamCfg, interval, Some(startedAt))
133       } else {
134         context.stop(self)
135       }
136   }

139     case StreamController.StreamTerminated(_) => context.stop(self)
