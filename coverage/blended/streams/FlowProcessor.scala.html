19   def transform[T](name: String, log: FlowEnvelopeLogger)(f : FlowEnvelope => Try[T])(implicit clazz : ClassTag[T])
20     : Graph[FlowShape[FlowEnvelope, Either[FlowEnvelope, T]], NotUsed] = {
21 
22     Flow.fromFunction[FlowEnvelope, Either[FlowEnvelope, T]] { env =>
23       env.exception match {
24         case None =>
25           log.logEnv(env, LogLevel.Debug, s"Starting function [${env.id}]:[$name]")
26           val start = System.currentTimeMillis()
27           f(env) match {
28             case Success(s) =>
29               log.logEnv(env, LogLevel.Debug, s"Function [${env.id}]:[$name] completed in [${System.currentTimeMillis() - start}]ms")
30               Right(s)
31             case Failure(t) =>
32               log.logEnv(env.withException(t), LogLevel.Warn, s"Failed to create [${clazz.runtimeClass.getName()}] in [${env.id}]:[$name]")
33               Left(env.withException(t))
34           }
35         case Some(_) =>
36           log.logEnv(env, LogLevel.Debug, s"Not executing function [${env.id}]:[$name] as envelope has exception [${env.exception.map(_.getMessage()).getOrElse("")}].")
37           Left(env)
38       }
39     }.named(name)
40   }

42   def fromFunction(name: String, log: FlowEnvelopeLogger)(f: IntegrationStep) : Graph[FlowShape[FlowEnvelope, FlowEnvelope], NotUsed] = {
43 
44     Flow.fromFunction[FlowEnvelope, FlowEnvelope] { env : FlowEnvelope =>
45 
46       env.exception match {
47         case None =>
48           log.logEnv(env, LogLevel.Debug, s"Starting Integration step [${env.id}]:[$name]")
49           val start = System.currentTimeMillis()
50 
51           val result = f(env) match {
52             case Success(s) =>
53               log.logEnv(env, LogLevel.Debug, s"Integration step [${env.id}]:[$name] completed in [${System.currentTimeMillis() - start}]ms")
54               s
55 
56             case Failure(t) =>
57               log.logEnv(env.withException(t), LogLevel.Debug, s"Exception in FlowProcessor [${env.id}]:[$name] for message [${env.flowMessage}] : [${t.getClass().getSimpleName()} - ${t.getMessage()}]", false)
58               env.withException(t)
59           }
60 
61           result
62         case Some(_) =>
63           log.logEnv(env, LogLevel.Debug, s"Skipping integration step [${env.id}]:[$name] due to exception caught in flow.")
64           env
65       }
66     }.named(name)

74   def splitEither[L, R]() : Graph[FanOutShape2[Either[L, R], L, R], NotUsed] = {
75     GraphDSL.create() { implicit b =>
76       import GraphDSL.Implicits._
77 
78       val branches = b.add(Broadcast[Either[L, R]](2))
79       val isLeft = b.add(Flow[Either[L, R]].filter(_.isLeft).map(_.left.get))
80       val isRight = b.add(Flow[Either[L, R]].filter(_.isRight).map(_.right.get))
81 
82       branches ~> isLeft
83       branches ~> isRight
84 
85       new FanOutShape2(branches.in, isLeft.out, isRight.out)
86     }
87   }
