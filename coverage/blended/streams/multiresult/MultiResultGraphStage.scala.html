7 // A stage that may generate multiple responses from one incoming message
8 class MultiResultGraphStage[T, U](name : String)(f : T => List[U]) extends GraphStage[FlowShape[T,U]] {
9   private val in : Inlet[T] = Inlet[T](s"MultiResult.$name.in")
10   private val out : Outlet[U] = Outlet[U](s"MultiResult.$name.out")
11 
12   private val log : Logger = Logger(classOf[MultiResultGraphStage[_,_]].getName())
13 
14   override val shape : FlowShape[T, U] = FlowShape.of(in, out)
15 
16   override def createLogic(inheritedAttributes: Attributes): GraphStageLogic = new GraphStageLogic(shape) {
17 
18     private var pendingValues: List[U] = Nil
19 
20     // as long as values are still available, we will push them one by one
21     // if no more messages are available to push downstream, we will pull
22     // the inlet and eventually a new inbound message will be pulled from
23     // upstream, yielding in a new list of values to be pushed.
24     private def push(): Unit = {
25       pendingValues match {
26         case Nil =>
27           if (!hasBeenPulled(in)) {
28             log.trace("Pulling new input")
29             pull(in)
30           }
31         case h :: t =>
32           if (isAvailable(out)) {
33             push(out, h)
34             log.trace(s"Pushing value [$h], rest is [$t]")
35             pendingValues = t
36           }
37       }
38     }
39 
40     setHandlers(
41       in, out,
42       new InHandler with OutHandler {
43         override def onPush(): Unit = {
44           // We calculate the list of responses
45           pendingValues = f(grab(in))
46           // and start to push one by one
47           push()
48         }
49 
50         override def onPull(): Unit = push()
51       })
52 
53     // We start by pulling the inlet
54     override def preStart(): Unit = {
55       pull(in)
56     }
57   }
58 }
