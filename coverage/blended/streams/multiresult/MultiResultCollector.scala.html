58 class MultiResultCollector(
59   replicator : FlowEnvelope => List[FlowEnvelope],
60   processSingle : Flow[FlowEnvelope, FlowEnvelope, NotUsed],
61   respondTo : ActorRef,
62   timeout : Option[FiniteDuration],
63   log : FlowEnvelopeLogger
64 ) extends Actor {
65 
66   private val bufferSize : Int = 10
67   private case class MultiResultTimeout(timeout : FiniteDuration)
68 
69   private implicit val system : ActorSystem = context.system
70   private implicit val eCtxt : ExecutionContext = context.dispatcher
71   private implicit val materializer : Materializer = ActorMaterializer()
72 
73 
74   override def receive : Receive = {
75     case env : FlowEnvelope =>
76       log.logEnv(env, LogLevel.Debug, s"Processing envelope in MultiResultCollector : [${env.id}]")
77 
78       val copies : List[FlowEnvelope] = replicator(env)
79 
80       copies.map(_.exception).find(_.isDefined) match {
81         case Some(Some(t)) =>
82           log.logEnv(env, LogLevel.Warn, s"Failed to create copies from envelope [${env.id}] : [${t.getMessage()}]")
83           respond(env.withException(t), None)
84 
85         case _ =>
86 
87           // Here we will collect the individual results
88           val collector : Collector[FlowEnvelope] = Collector[FlowEnvelope](
89             name = s"MultiResultProcessor-${UUID.randomUUID().toString()}",
90             completeOn = Some{ l : Seq[FlowEnvelope] =>
91               l.size == copies.size || l.exists(_.exception.isDefined)
92             }
93           )
94 
95           // We need to schedule a Timeout message, which will be triggered if the subflows could not
96           // be executed within the given time frame
97           val timer : Option[Cancellable] = timeout.map{ t =>
98             context.system.scheduler.scheduleOnce(t, self, MultiResultTimeout(t))
99           }
100 
101           // Once the collector has completed we will pass the result to ourselves
102           collector.result.onComplete { r => self ! r }
103 
104           // We need to set up the sub flow that accepts messages from an actor and passes
105           // each message through the given sub flow while the final result is collected
106           // by the collector
107           val collActor : ActorRef = Source.actorRef[FlowEnvelope](bufferSize, OverflowStrategy.fail)
108             .viaMat(processSingle)(Keep.left)
109             .toMat(Sink.actorRef[FlowEnvelope](collector.actor, Done))(Keep.left)
110             .run()
111 
112           context.become(collecting(env, timer))
113 
114           copies.foreach{env => collActor ! env }
115       }
116   }
117 
118   private def collecting(env : FlowEnvelope, timer : Option[Cancellable]) : Receive = {
119     case r : Try[List[FlowEnvelope]] => r match {
120       case Success(l) => l.map(_.exception).find(_.isDefined).flatten match {
121         case None =>
122           log.logEnv(env, LogLevel.Info, s"Successfully executed sub flows for [${env.id}]")
123           respond(env, timer)
124         case Some(t) =>
125           log.logEnv(env, LogLevel.Warn, s"Subflow for [${env.id}] threw exception [${t.getMessage()}]")
126           respond(env.withException(t), timer)
127       }
128       case Failure(t) =>
129         log.logEnv(env, LogLevel.Warn, s"Failed to process subflows of [${env.id}] : ${t.getMessage()}")
130         respond(env.withException(t), timer)
131     }
132 
133     case MultiResultTimeout(t) =>
134       val e : Throwable = new MultiResultTimeoutException(env.id, t)
135       respond(env.withException(e), None)
136   }
137 
138   private def respond(env : FlowEnvelope, timer : Option[Cancellable]): Unit = {
139     log.logEnv(env, LogLevel.Debug, s"Multiresult processor result is [$env]", false)
140     timer.foreach(_.cancel())
141     respondTo ! env
142     context.stop(self)
143   }
144 }
