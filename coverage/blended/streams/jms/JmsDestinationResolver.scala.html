19   def createJmsMessage(session : Session, env : FlowEnvelope) : Try[Message] = Try {
20 
21     val flowMsg = env.flowMessage
22 
23     val msg = flowMsg match {
24       case t :
25         // scalastyle:off null
26         TextFlowMessage => session.createTextMessage(Option(t.body()).map(_.toString).orNull)
27       // scalastyle:on null
28       case b :
29         BinaryFlowMessage =>
30         val r = session.createBytesMessage()
31         r.writeBytes(b.getBytes().toArray)
32 
33         r
34       case _ => session.createMessage()
35     }
36 
37     flowMsg.header.filter {
38       case (k, _) => !k.startsWith("JMS")
39     }.foreach {
40       case (k, v) =>
41         val propName = settings.keyFormatStrategy.handleSendKey(k)
42         v match {
43           // scalastyle:off null
44           case _ : UnitMsgProperty => msg.setObjectProperty(propName, null)
45           case o                   => msg.setObjectProperty(propName, o.value)
46           // scalastyle:on null
47         }
48     }
49 
50     // Always set the env id as id for the message
51     msg.setStringProperty(settings.headerCfg.headerTransId, env.id)
52 
53     replyTo(session, env).get.foreach(msg.setJMSReplyTo)
54     // Always try to get the CorrelationId from the flow Message
55     correlationId(env).foreach(msg.setJMSCorrelationID)
56 
57     msg
58   }

76     env.header[String](replyToHeader(headerConfig.prefix)) match {
77       case None    => None
78       case Some(d) => Some(JmsDestination.create(d).map(dest => dest.create(session)).get)
79     }
80   }

82   // Get the destination from the message
83   def destination(env : FlowEnvelope) : Try[JmsDestination] = {
84 
85     val flowMsg : FlowMessage = env.flowMessage
86 
87     Try {
88 
89       log.trace(s"Trying to resolve destination for [${env.id}] from header [${destHeader(headerConfig.prefix)}]")
90 
91       val d = flowMsg.header[String](s"${destHeader(headerConfig.prefix)}") match {
92         case Some(s) =>
93           JmsDestination.create(s).get
94 
95         case None =>
96           log.trace(s"Trying to resolve destination for [${env.id}] from settings.")
97           settings.jmsDestination match {
98             case Some(dest) => dest
99             case None =>
100               throw new JMSException(s"Could not resolve JMS destination for [${env.id}]")
101           }
102       }
103 
104       log.debug(s"Resolved destination for [${env.id}] to [${d.asString}]")
105       d
106     }
107   }
108 
109   val priority : FlowMessage => Int = { flowMsg =>
110     flowMsg.header[Int](priorityHeader(headerConfig.prefix)) match {
111       case Some(p) => p
112       case None    => settings.priority
113     }
114   }

116   val timeToLive : FlowMessage => Option[FiniteDuration] = { flowMsg =>
