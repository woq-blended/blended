13 /**
14   * Manage JMS sessions on behalf of a JMS streams source or sink.
15   *
16   * @param conn The JMS connection that will be used to create the sessions.
17   * @param maxSessions The maximum number of sessions this manager will create.
18   */
19 class JmsSessionManager(
20   name : String,
21   conn : Connection,
22   maxSessions : Int
23 ) {
24 
25   private val log : Logger = Logger[JmsSessionManager]
26 
27   /**
28     * A callback that will be called after a new session has been created.
29     */
30   def onSessionOpen : JmsSession => Try[Unit] = { _ => Success(()) }
31 
32   /**
33     * A callback that will be called immediately before a session will be closed.
34     */
35   def beforeSessionClose : JmsSession => Try[Unit] = { _ => Success(()) }
36 
37   /**
38     * A callback that will be called after a session has been closed.
39     */
40   def afterSessionClose : JmsSession => Try[Unit] = { _ => Success(()) }
41 
42   def onError : Throwable => Unit = _ => ()
43 
44   // We maintain a map of currently open sessions
45   private val sessions : mutable.Map[String, JmsSession] = mutable.Map.empty
46 
47   def isOpen(id : String) : Boolean = sessions.contains(id)
48 
49   def getSession(id : String): Try[Option[JmsSession]] = {
50 
51     sessions.get(id) match {
52       case Some(s) =>
53         log.trace(s"Reusing existing session for session [$id] in [$name]")
54         Success(Some(s))
55       case None =>
56         if (sessions.size < maxSessions) {
57           try {
58             log.debug(s"Creating session [$id] in [$name]")
59             val s : JmsSession = JmsSession(
60               conn.createSession(false, Session.CLIENT_ACKNOWLEDGE), id
61             )
62             sessions.put(s.sessionId, s)
63             onSessionOpen(s)
64             Success(Some(s))
65           } catch {
66             case NonFatal(t) =>
67               log.error(s"Failed to create session in [$name] : [${t.getMessage()}]")
68               Failure(t)
69           }
70         } else {
71           log.debug(s"No free session slot available in [$name] for [$id] : [$maxSessions].")
72           Success(None)
73         }
74     }
75   }
76 
77   def closeSession(id : String) : Try[Unit] = Try {
78     sessions.remove(id).map { sess =>
79       log.debug(s"Closing session [${sess.sessionId}]")
80       beforeSessionClose(sess)
81       sess.closeSession() match {
82         case Success(_) => afterSessionClose(sess)
83         case Failure(t) => onError(t)
84       }
85     }.getOrElse( Success() )
86   }
87 
88   def closeSessionAsync(id : String)(system : ActorSystem) : Future[Unit] =
89     sessions.remove(id).map { _.closeSessionAsync()(system) }.getOrElse( Future {}(system.dispatcher) )
90 
91   def closeAll() : Try[Unit] = {
92     log.trace(s"Closing [${sessions.size}] sessions for [$name]")
93     sessions.values.map{ sess =>
94       closeSession(sess.sessionId)
95     }.find(_.isFailure).getOrElse(Success(()))
96   }
97 }
