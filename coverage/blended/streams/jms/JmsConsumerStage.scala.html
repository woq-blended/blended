19 /**
20  * The JmsSource realizes an inbound Stream of FlowMessages consumed from
21  * a given JMS destination. The messages can be consumed using an arbitrary sink
22  * of FlowMessages. The sink must acknowledge or deny the message eventually. If
23  * the message is not acknowledged with a given time frame, the message will be
24  * denied and eventually redelivered.
25  *
26  * A JMS session will only consume one message at a time, the next message will
27  * be consumed only after the previous message of that session has been denied
28  * or acknowledged. The messages that are waiting for acknowledgement are maintained
29  * within a map of (session-id / inflight messages).
30  *
31  * Any exception while consuming a message within a session will cause that session
32  * to close, so that any inflight messages for that session will be redelivered. A
33  * new session will be created automatically for any sessions that have been closed
34  * with such an exception.
35  *
36  * One of the use cases is a retry pattern. In that pattern the message must remain
37  * in the underlying destination for a minimum amount of time (i.e. 5 seconds).
38  * The message receive loop will check the JMSTimestamp against the System.currentTimeMillis
39  * to check whether a message can be passed downstream already.
40  */

54   private class JmsAckContext(
55     inflightId : String,
56     env : FlowEnvelope,
57     val jmsMessage : Message,
58     val jmsMessageAck : Message => Unit,
59     val session : JmsSession,

63     override def deny(): Unit = {
64       sessionClose(session)
65       consumerSettings.log.logEnv(
66         env, LogLevel.Debug, s"Message [${envelope.id}] has been denied. Closing receiving session."
67       )

70     override def acknowledge(): Unit = {
71       jmsMessageAck(jmsMessage)

73         env, LogLevel.Debug, s"Acknowledged envelope [${envelope.id}] for session [${session.sessionId}]"

90     private val srcDest : JmsDestination = consumerSettings.jmsDestination match {
91       case Some(d) => d
92       case None => throw new IllegalArgumentException(s"Destination must be set for consumer in [$id]")
93     }

95     private val vendor : String = consumerSettings.connectionFactory.vendor
96     private val provider : String = consumerSettings.connectionFactory.provider

98     private[this] val consumer : mutable.Map[String, MessageConsumer] = mutable.Map.empty
99 
100     private[this] def addConsumer(s : String, c : MessageConsumer) : Unit = {
101       consumer.put(s, c)
102       consumerSettings.log.underlying.debug(s"Jms Consumer count of [$name] is [${consumer.size}]")
103       nextPollRelative = None
104       pollImmediately.invoke()
105     }
106 
107     private[this] def removeConsumer(s : String) : Unit = {
108       consumer.remove(s)
109       consumerSettings.log.underlying.debug(s"Consumer count of [$name] is [${consumer.size}]")
110     }

112     private val closeSession : AsyncCallback[JmsSession] = getAsyncCallback(s => connector.closeSession(s.sessionId))
113     private val ackMessage : AsyncCallback[Message] = getAsyncCallback[Message](m => m.acknowledge())
114 
115     private lazy val connector : JmsConnector = new JmsConnector(id, consumerSettings)(session => Try {
116       // After session opened
117       consumerSettings.log.underlying.debug(
118         s"Creating message consumer for session [${session.sessionId}], " +
119           s"destination [$srcDest] and selector [${consumerSettings.selector}]"
120       )
121       session.createConsumer(srcDest, consumerSettings.selector) match {
122 
123         case Success(c) =>
124           addConsumer(session.sessionId, c)
125 
126         case Failure(e) =>
127           consumerSettings.log.underlying.debug(s"Failed to create consumer for session [${session.sessionId}] : [${e.getMessage()}]")
128           closeSession.invoke(session)
129       }
130     })( s => Try {
131       // before session close
132       consumer.get(s.sessionId).foreach{ c =>
133         consumerSettings.log.underlying.debug(s"Closing message consumer for [${s.sessionId}]")
134         c.close()
135         removeConsumer(s.sessionId)
136       }
137     })(
138       //after Session close
139       _ => Success(())
140     )(
141       // error handler
142       handleError.invoke
143     )
144 
145     override protected def freeInflightSlot(): Option[String] =
146       determineNextSlot(inflightSlots.filter(id => connector.isOpen(id))) match {
147         case Some(s) => Some(s)
148         case None => determineNextSlot(inflightSlots)
149       }

151     /** The id's of the available inflight slots */
152     override protected val inflightSlots : List[String] =
153       1.to(consumerSettings.sessionCount).map(i => s"$id-$i").toList
154 
155     private var nextPollRelative : Option[FiniteDuration] = None
156     override protected def nextPoll(): Option[FiniteDuration] = {
157       nextPollRelative match {
158         case None =>
159           Some(consumerSettings.pollInterval)
160         case Some(npr) =>
161           nextPollRelative = None
162           consumerSettings.log.underlying.debug(s"Overriding next poll interval with [$npr]")
163           Some(npr)
164       }
165     }
166 
167     private def receive(session : JmsSession) : Try[Option[Message]] = Try {
168 
169       val msg : Option[Message] = consumer.get(session.sessionId).flatMap { c =>
170         if (consumerSettings.receiveTimeout.toMillis <= 0) {
171           Option(c.receiveNoWait())
172         } else {
173           Option(c.receive(consumerSettings.receiveTimeout.toMillis))
174         }
175       }
176 
177       val result : Option[Message] = msg match {
178         case None => None
179 
180         case Some(m) =>
181           minMessageDelay match {
182             case Some(d) =>
183               val age : Long = System.currentTimeMillis() - m.getJMSTimestamp()
184               if (age <= d.toMillis) {
185                 closeSession.invoke(session)
186                 nextPollRelative = Some( (d.toMillis - age).millis )
187                 consumerSettings.log.underlying.debug(s"Message has not reached the minimum message delay yet ...rescheduling in [$nextPollRelative]")
188                 None
189               } else {
190                 nextPollRelative = None
191                 Some(m)
192               }
193             case None =>
194               msg
195           }
196       }
197 
198       result
199     }

201     private def createEnvelope(message : Message, ackHandler : AcknowledgeHandler) : Try[FlowEnvelope] = Try {
202 
203       val flowMessage : FlowMessage = JmsFlowSupport.jms2flowMessage(headerConfig)(consumerSettings)(message).unwrap
204 
205       val envelopeId : String = flowMessage.header[String](headerConfig.headerTransId) match {
206         case None =>
207           val newId = UUID.randomUUID().toString()
208           consumerSettings.log.underlying.trace(s"Created new envelope id [$newId]")
209           newId
210         case Some(s) =>
211           consumerSettings.log.underlying.trace(s"Reusing transaction id [$s] as envelope id")
212           s
213       }
214 
215       FlowEnvelope(flowMessage, envelopeId)
216         .withHeader(headerConfig.headerTransId, envelopeId).unwrap
217         .withRequiresAcknowledge(true)
218         .withAckHandler(Some(ackHandler))
219     }
220 
221     override protected def doPerformPoll(id: String, ackHandler: AcknowledgeHandler): Try[Option[JmsAckContext]] = Try {
222       connector.getSession(id) match {
223         case Some(s) =>
224 
225           receive(s).unwrap
226             .map{ m =>
227               val e : FlowEnvelope = createEnvelope(m, ackHandler).unwrap
228 
229               val now : Long = System.currentTimeMillis()
230 
231               val msgAge : Long = now - e.header[Long](timestampHeader(headerConfig.prefix)).getOrElse(now)
232 
233               consumerSettings.log.logEnv(e, consumerSettings.logLevel(e),
234                 s"Message received [${e.id}][${consumerSettings.jmsDestination.map(_.asString)}] after " +
235                   s"[$msgAge]ms in [${s.sessionId}] : ${e.flowMessage}"
236               )
237 
238               // We signal that we have received a message for the underlying connection factory
239               actorSystem.eventStream.publish(MessageReceived(vendor, provider, e.id))
240 
241               new JmsAckContext(
242                 inflightId = id,
243                 env = e,
244                 jmsMessage = m,
245                 jmsMessageAck = ackMessage.invoke,
246                 session = s,
247                 sessionClose = closeSession.invoke
248               )
249             }
250 
251         case None =>
252           None
253       }
254     }
255 
256     override def preStart(): Unit = {
257       super.preStart()
258 
259       stateListener = Some(ConnectionStateListener.create(
260         vendor = consumerSettings.connectionFactory.vendor,
261         provider = consumerSettings.connectionFactory.provider
262       ){ event => event.state.status match {
263         case Disconnected =>
264           val msg : String = s"Underlying JMS connection closed for [$id]"
265           consumerSettings.log.underlying.debug(msg)
266           val t : Throwable = new Exception(msg)
267           handleError.invoke(t)
268         case _ =>
269       }})
270     }
271 
272     override def postStop(): Unit = {
273       log.underlying.debug(s"Stopping JmsConsumerStage [$id].")
274       stateListener.foreach(actorSystem.stop)
275       connector.closeAll()
276       super.postStop()
277     }
278   }
