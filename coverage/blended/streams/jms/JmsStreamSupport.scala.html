19 trait JmsStreamSupport {
20   /**
21    * Process a sequence of messages with a given flow. If any of the processed
22    * messages cause an exception in the provided flow, the Stream will terminated
23    * with this exception been thrown.
24    *
25    * The resulting stream will expose a killswitch, so that it stays
26    * open and the test code needs to tear it down eventually.
27    */
28 
29   def processMessages(
30     processFlow : Flow[FlowEnvelope, FlowEnvelope, _],
31     msgs : FlowEnvelope*
32   )(implicit system : ActorSystem) : Try[KillSwitch] = Try {
33 
34     implicit val materializer : Materializer = ActorMaterializer()
35     implicit val eCtxt : ExecutionContext = system.dispatcher
36 
37     val hasException : AtomicBoolean = new AtomicBoolean(false)
38     val sendCount : AtomicInteger = new AtomicInteger(0)
39 
40     val (((actor : ActorRef, killswitch : KillSwitch), done : Future[Done]), errEnv : Future[Option[FlowEnvelope]]) =
41       Source.actorRef[FlowEnvelope](msgs.size, OverflowStrategy.fail)
42         .viaMat(processFlow)(Keep.left)
43         .viaMat(KillSwitches.single)(Keep.both)
44         .watchTermination()(Keep.both)
45         .via(Flow.fromFunction[FlowEnvelope, FlowEnvelope] { env =>
46           if (env.exception.isDefined) {
47             env.exception.foreach { t =>
48               hasException.set(true)
49               throw t
50             }
51           } else {
52             sendCount.incrementAndGet()
53           }
54           env
55         })
56         .filter(_.exception.isDefined)
57         .toMat(Sink.headOption[FlowEnvelope])(Keep.both)
58         .run()
59 
60     // Send all the messages
61     msgs.foreach(m => actor ! m)
62 
63     // We will wait until all messages have passed through the process flow and check if
64     // any have thrown an exception causing the stream to fail
65     do {
66       Thread.sleep(10)
67 
68       if (hasException.get()) {
69         Await.result(errEnv, 1.second).flatMap(_.exception).foreach(t => throw t)
70       }
71 
72       // if the stream has is finished before sending off all the messages, something went wrong.
73       if (done.isCompleted) {
74         throw new Exception("Failed to send messages to stream")
75       }
76     } while (!hasException.get && sendCount.get < msgs.size)
77 
78     killswitch
79   }

99   def receiveMessages(
100     headerCfg : FlowHeaderConfig,
101     cf : IdAwareConnectionFactory,
102     dest : JmsDestination,
103     log : FlowEnvelopeLogger,
104     listener : Integer = 2,
105     minMessageDelay : Option[FiniteDuration] = None,
106     selector : Option[String] = None,
107     completeOn : Option[Seq[FlowEnvelope] => Boolean] = None,
108     timeout : Option[FiniteDuration]
109   )(implicit system : ActorSystem, materializer : Materializer) : Collector[FlowEnvelope] = {
110 
111     val listenerCount : Int = if (dest.isInstanceOf[JmsQueue]) {
112       listener
113     } else {
114       1
115     }
116 
117     val collected : FlowEnvelope => Unit = { env =>
118       log.logEnv(env, LogLevel.Debug, s"Acknowledging envelope [${env.id}]")
119       env.acknowledge()
120     }
121 
122     val source : Source[FlowEnvelope, NotUsed] = jmsConsumer(
123       name = dest.asString,
124       settings =
125         JmsConsumerSettings(log = log, headerCfg = headerCfg, connectionFactory = cf)
126           .withAcknowledgeMode(AcknowledgeMode.ClientAcknowledge)
127           .withSessionCount(listenerCount)
128           .withDestination(Some(dest))
129           .withSelector(selector),
130       minMessageDelay = minMessageDelay
131     )
132 
133     StreamFactories.runSourceWithTimeLimit(
134       name = dest.asString,
135       source = source,
136       timeout = timeout,
137       onCollected = Some(collected),
138       completeOn = completeOn
139     )
140   }

142   def jmsProducer(
143     name : String,
144     settings : JmsProducerSettings,
145     autoAck : Boolean
146   )(implicit system : ActorSystem) : Flow[FlowEnvelope, FlowEnvelope, NotUsed] = {
147 
148     implicit val materializer : Materializer = ActorMaterializer()
149 
150     val f = Flow.fromGraph(new JmsProducerStage(name, settings)).named(name)
151 
152     if (autoAck) {
153       f.via(new AckProcessor(s"ack-$name").flow.named(s"ack-$name"))
154     } else {
155       f
156     }
157   }
