22   /**
23     * Initialize the connection to be used for this stage. If the connection can't be established,
24     * the stage will fail with an exception. For recovery, the stage should be wrapped within a
25     * [[blended.streams.StreamControllerSupport]].
26     */
27   private val connection : Option[Connection] = Try {
28     jmsSettings.log.underlying.debug(s"Trying to create JMS connection for stream [$id]")
29     jmsSettings.connectionFactory.createConnection()
30   } match {
31     case Success(c) =>
32       Some(c)
33     case Failure(e) =>
34       handleError(e)
35       None
36   }
37 
38   private val sessionMgr : Try[JmsSessionManager] = Try {
39     connection match {
40       case Some(c) =>
41         new JmsSessionManager(
42           name = id,
43           conn = c,
44           maxSessions = jmsSettings.sessionCount
45         ) {
46           override def onSessionOpen: JmsSession => Try[Unit] = onSessionOpened
47           override def beforeSessionClose: JmsSession => Try[Unit] = beforeSessionCloseCallback
48           override def afterSessionClose: JmsSession => Try[Unit] = afterSessionCloseCallback
49           override def onError: Throwable => Unit = handleError
50         }
51 
52       case None =>
53         throw new IllegalStateException(s"No connection available ... no session manager created")
54     }
55   }
56 
57   private def withSessionMgr[T](f : JmsSessionManager => T) : Try[T] = Try {
58     sessionMgr match {
59       case Success(mgr) => f(mgr)
60       case Failure(t) => throw t
61     }
62   }
63 
64   def getSession(id : String) : Option[JmsSession] = withSessionMgr[Option[JmsSession]]{ mgr =>

80   }
