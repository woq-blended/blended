25   private val inlet : Inlet[FlowEnvelope] = Inlet[FlowEnvelope](s"JmsProducer($name.in)")
26   private val outlet : Outlet[FlowEnvelope] = Outlet[FlowEnvelope](s"JmsProducer($name.out)")

31 
32     // Just to identify the Source stage in log statements
33     private val id : String = name
34     private val rnd : Random = new Random()
35 
36     private val handleError : AsyncCallback[Throwable] = getAsyncCallback[Throwable]{ t =>
37       failStage(t)
38     }
39 
40     private var stateListener : Option[ActorRef] = None
41     private var producer : Map[String, MessageProducer] = Map.empty
42 
43     private[this] def addProducer(s : String, p : MessageProducer) : Unit = {
44       producer = producer + (s -> p)
45       producerSettings.log.underlying.debug(s"Producer count of [$id] is [${producer.size}]")
46     }
47 
48     private[this] def removeProducer(s : String) : Unit = {
49       if (producer.contains(s)) {
50         producer = producer.filterKeys(_ != s)
51         producerSettings.log.underlying.debug(s"Producer count of [$id] is [${producer.size}]")
52       }
53     }
54 
55     private val connector : JmsConnector = new JmsConnector(id, producerSettings)(s => Try {
56       // scalastyle:off null
57       val p : MessageProducer = s.session.createProducer(null)
58       // scalastyle:on null
59       addProducer(s.sessionId, p)
60     })( s => Try {
61       producer.get(s.sessionId).foreach{ p =>
62         producerSettings.log.underlying.debug(s"Closing message producer for [${s.sessionId}]")
63         p.close()
64         removeProducer(s.sessionId)
65       }
66     }
67     )(_ =>  Success(()))(
68       handleError.invoke
69     )
70 
71     private val closeSession : AsyncCallback[JmsSession] = getAsyncCallback(s => connector.closeSession(s.sessionId))
72 
73     private def pushMessage(env : FlowEnvelope) : Unit = {
74       if (producer.nonEmpty) {
75         val idx: Int = rnd.nextInt(producer.size)
76         val (key, jmsProd): (String, MessageProducer) = producer.toIndexedSeq(idx)
77 
78         connector.getSession(key) match {
79           case Some(s) =>
80             push(outlet, sendEnvelope(env)(s)(jmsProd))
81 
82           case None =>
83             producerSettings.log.underlying.debug(s"No producer session available in [$id]")
84             scheduleOnce(Push(env), 10.millis)
85         }
86       } else {
87         // Kick off to initialize the producers
88         0.to(producerSettings.sessionCount).foreach{ i =>
89           connector.getSession(name + "-" + i)
90         }
91 
92         producerSettings.log.underlying.debug(s"No producer available")
93         scheduleOnce(Push(env), 10.millis)
94       }
95     }
96 
97     private val sendEnvelope : FlowEnvelope => JmsSession => MessageProducer => FlowEnvelope = env => jmsSession => producer => {
98 
99       try {
100         val sendParams = JmsFlowSupport.envelope2jms(producerSettings, jmsSession.session, env).unwrap
101 
102         val sendTtl : Long = sendParams.ttl match {
103           case Some(l) =>
104             if (l.toMillis < 0L) {
105               producerSettings.log.logEnv(env, LogLevel.Warn,
106                 s"The message [${env.id}] has expired and wont be sent to the JMS destination."
107               )
108             }
109             l.toMillis
110           case None => 0L
111         }
112 
113         if (sendTtl >= 0L) {
114           val dest : Destination = sendParams.destination.create(jmsSession.session)
115           producer.send(
116             dest, sendParams.message, sendParams.deliveryMode.mode, sendParams.priority, sendTtl
117           )
118 
119           val logDest = s"${producerSettings.connectionFactory.vendor}:${producerSettings.connectionFactory.provider}:$dest"
120           producerSettings.log.logEnv(env, producerSettings.logLevel(env),
121             s"Successfully sent message [${env.id}] to [$logDest] with headers [${env.flowMessage.header.mkString(",")}] " +
122               s"with parameters [${sendParams.deliveryMode}, ${sendParams.priority}, ${sendParams.ttl}]", false
123           )
124         }
125 
126         if (producerSettings.clearPreviousException) {
127           env.clearException()
128         } else {
129           env
130         }
131       } catch {
132         case NonFatal(t) =>
133           producerSettings.log.logEnv(env, LogLevel.Debug,
134             s"Error sending message [${env.id}] to [${producerSettings.jmsDestination}] in [${jmsSession.sessionId}]"
135           )
136           closeSession.invoke(jmsSession)
137           env.withException(t)
138       }
139     }
140 
141     // First simply pass the pull upstream if any
142     setHandler(
143       outlet,
144       new OutHandler {
145         override def onPull() : Unit = {
146           pull(inlet)
147         }
148       }
149     )
150 
151     // We can only start pushing message after at least one session is available
152     setHandler(
153       inlet,
154       new InHandler {
155         override def onPush() : Unit = {
156           val env = grab(inlet)
157           pushMessage(env)
158         }
159       }
160     )
161 
162     override protected def onTimer(timerKey: Any): Unit = timerKey match {
163       case Push(env) => pushMessage(env)
164     }
165 
166     override def preStart(): Unit = {
167       super.preStart()
168 
169       stateListener = Some(ConnectionStateListener.create(
170         vendor = producerSettings.connectionFactory.vendor,
171         provider = producerSettings.connectionFactory.provider
172       ){ event => event.state.status match {
173         case Disconnected =>
174           val msg : String = s"Underlying JMS connection closed for [$id]"
175           producerSettings.log.underlying.warn(msg)
176           val t : Throwable = new Exception(msg)
177           handleError.invoke(t)
178         case _ =>
179       }})
180     }
181 
182     override def postStop(): Unit = {
183       producerSettings.log.underlying.debug(s"Closing JMS Producer stage [$id]")
184       stateListener.foreach(system.stop)
185       connector.closeAll()
186       super.postStop()
187     }
188   }
