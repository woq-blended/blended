39   // make sure all headers are set without encountering an exception
40   val header : FlowProcessor.IntegrationStep = env => Try {
41 
42     // Decide if we are already within the retryLoop
43     val retrying : Boolean = env.header[String](headerCfg.headerRetrying).nonEmpty
44     val maxRetries : Long = env.header[Long](headerCfg.headerMaxRetries).getOrElse(retryCfg.maxRetries)
45 
46     // In case we are not yet in the retry loop we will reset the retry counter in any case
47     val retryCount : Long = if (retrying) {
48       env.header[Long](headerCfg.headerRetryCount).getOrElse(0L) + 1
49     } else {
50       1L
51     }
52 
53     val retryTimeout : Long = env.header[Long](headerCfg.headerRetryTimeout).getOrElse(retryCfg.retryTimeout.toMillis)
54     val firstRetry : Long = env.header[Long](headerCfg.headerFirstRetry).getOrElse(System.currentTimeMillis())
55 
56     env
57       .withHeader(headerCfg.headerMaxRetries, maxRetries).get
58       .withHeader(headerCfg.headerRetryCount, retryCount).get
59       .withHeader(headerCfg.headerRetryTimeout, retryTimeout).get
60       .withHeader(headerCfg.headerFirstRetry, firstRetry).get
61       .withHeader(headerCfg.headerRetrying, "True").get
62   }
63 
64   val validate : LogLevel => FlowProcessor.IntegrationStep = level => env => Try {
65 
66     val mandatoryHeader : String => Long = h =>
67       env.header[Long](h) match {
68         case None    => throw new MissingRetryHeaderException(h)
69         case Some(l) => l
70       }
71 
72     val maxRetries : Long = mandatoryHeader(headerCfg.headerMaxRetries)
73     val retryCount : Long = mandatoryHeader(headerCfg.headerRetryCount)
74     val retryTimeout : Long = mandatoryHeader(headerCfg.headerRetryTimeout)
75     val firstRetry : Long = mandatoryHeader(headerCfg.headerFirstRetry)
76 
77     val remaining : FiniteDuration = (retryTimeout - (System.currentTimeMillis() - firstRetry)).millis
78     log.logEnv(env, level, s"Retrying envelope [${env.id}] : [$retryCount / $maxRetries] [${remaining}] remaining", false)
79 
80     if (maxRetries > 0 && retryCount > maxRetries) {
81       throw new RetryCountExceededException(maxRetries)
82     }
83 
84     if (System.currentTimeMillis() - firstRetry > retryTimeout) {
85       throw new RetryTimeoutException(firstRetry + retryTimeout)
86     }
87 
88     if (env.header[String](headerCfg.headerRetryDestination).isEmpty) {
89       throw new MissingRetryDestinationException(headerCfg.headerRetryDestination)
90     }
91 
92     env
93   }
94 
95   val flow : Graph[FlowShape[FlowEnvelope, FlowEnvelope], NotUsed] =
