20 /**
21  * Keep track of connection factories registered as services and create KeepAlive Streams as appropriate.
22  */
23 class JmsKeepAliveController(
24   ctCtxt : ContainerContext,
25   producer : KeepAliveProducerFactory
26 ) extends Actor {
27 
28   private val log : Logger = Logger[JmsKeepAliveController]
29 
30   override def preStart(): Unit =
31     context.become(running(Map.empty))
32 
33   override def supervisorStrategy: SupervisorStrategy = OneForOneStrategy(){
34     case _ => Stop
35   }
36 
37   private val cfKey : IdAwareConnectionFactory => String = cf => s"${cf.vendor}${cf.provider}"
38 
39   override def receive: Receive = Actor.emptyBehavior
40 
41   private def running(watched : Map[String, ActorRef]) : Receive = {
42     case AddedConnectionFactory(cf) =>
43       if (!watched.contains(cfKey(cf))) {
44         log.info(s"Starting keep Alive actor for JMS connection factory [${cf.vendor}:${cf.provider}]")
45         val actor : ActorRef = context.system.actorOf(JmsKeepAliveActor.props(ctCtxt, cf, producer))
46         context.become(running(watched + (cfKey(cf) -> actor)))
47       }
48 
49     case RemovedConnectionFactory(cf) => watched.get(cfKey(cf)).foreach { actor =>
50       log.info(s"Stopping keep Alive Actor for JMS connection factory [${cf.vendor}:${cf.provider}]")
51       context.system.stop(actor)
52       context.become(running(watched - cfKey(cf)))
53     }
54   }
55 }

67 class JmsKeepAliveActor(
68   ctCtxt : ContainerContext,
69   cf : IdAwareConnectionFactory,
70   prodFactory : KeepAliveProducerFactory
71 ) extends Actor
72   with JmsStreamSupport {
73 
74   private val log : Logger = Logger[JmsKeepAliveActor]
75   private implicit val eCtxt : ExecutionContext = context.system.dispatcher
76   private implicit val materializer : Materializer = ActorMaterializer()
77   private val headerConfig : FlowHeaderConfig = FlowHeaderConfig.create(ctCtxt)
78 
79   case object Tick
80 
81   override def preStart(): Unit = {
82     cf match {
83       case bcf : BlendedSingleConnectionFactory =>
84         if (bcf.config.keepAliveEnabled) {
85 
86           context.system.eventStream.subscribe(self, classOf[ProducerMaterialized])
87           context.system.eventStream.subscribe(self, classOf[MessageReceived])
88           context.system.eventStream.subscribe(self, classOf[ConnectionStateChanged])
89 
90           prodFactory.start(bcf)
91           // We start in idle state
92           context.become(idle(bcf.config))
93         } else {
94           log.info(s"KeepAlive for [${bcf.vendor}:${bcf.provider}] is disabled.")
95           context.stop(self)
96         }
97 
98       case _ =>
99         log.info(s"No keep alive configuration found for [${cf.vendor}:${cf.provider}]")
100         context.stop(self)
101     }
102   }
103 
104   override def postStop(): Unit = prodFactory.stop()
105 
106   override def receive: Receive = Actor.emptyBehavior
107 
108   private def setCounter(cfg : ConnectionConfig, actor : ActorRef, newCnt : Int, oldCnt : Option[Int] = None) : Unit = {
109     if (oldCnt.isEmpty || !oldCnt.contains(newCnt)) {
110       context.system.eventStream.publish(KeepAliveMissed(cf.vendor, cf.provider, newCnt))
111       log.debug(s"New Keep Alive missed counter for [${cf.vendor}:${cf.provider}] is [$newCnt]")
112     }
113 
114     val newTimer = context.system.scheduler.scheduleOnce(cfg.keepAliveInterval, self, Tick)
115     context.become(running(newTimer, cfg, actor, newCnt))
116   }
117 
118   private def idle(cfg : ConnectionConfig) : Receive = {
119     // we need a materialized producer to start tracking keep alives
120     case ProducerMaterialized(v,p,a) =>
121       if (v == cfg.vendor && p == cfg.provider) {
122         log.debug(s"Keep alive Stream for [$v:$p] materialized, keep alive interval is [${cfg.keepAliveInterval}]")
123         setCounter(cfg, a, 0)
124       }
125     case _ => // do nothing
126   }
127 
128   private def running(timer: Cancellable, cfg : ConnectionConfig, actor : ActorRef, cnt : Int) : Receive = {
129     case Tick if cnt == cfg.maxKeepAliveMissed =>
130       context.system.eventStream.publish(MaxKeepAliveExceeded(cf.vendor, cf.provider))
131 
132     case Tick =>
133       val env : FlowEnvelope = FlowEnvelope(FlowMessage.props(
134         "JMSCorrelationID" -> ctCtxt.uuid,
135         headerConfig.headerKeepAlivesMissed -> (cnt + 1)
136       ).get)
137 
138       log.debug(s"Scheduling keep alive message for [${cf.vendor}:${cf.provider}] : [$env]")
139       actor ! env
140       setCounter(cfg, actor, cnt + 1, Some(cnt))
141 
142     case MessageReceived(v, p, _) if v == cf.vendor && p == cf.provider =>
143       timer.cancel()
144       setCounter(cfg,actor,0,Some(cnt))
145 
146     case ConnectionStateChanged(state) if state.status != Connected =>
147       timer.cancel()
148       context.become(idle(cfg))
149 
150     case ProducerMaterialized(v, p, a) =>
151       if (v == cfg.vendor && p == cfg.provider) {
152         log.debug(s"Keep alive Stream for [$v:$p] materialized, keep alive interval is [${cfg.keepAliveInterval}]")
153         setCounter(cfg, a, 0)
154       }
155   }
