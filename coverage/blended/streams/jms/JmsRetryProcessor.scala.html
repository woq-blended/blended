64 class JmsRetryProcessor(
65   streamsCfg : BlendedStreamsConfig,
66   retryCfg : JmsRetryConfig
67 )(
68   implicit system : ActorSystem, materializer : Materializer
69 ) extends JmsStreamSupport {
70 
71   private[this] val id : String = retryCfg.headerCfg.prefix + ".retry." + retryCfg.retryDestName
72   private[this] val retryLog : FlowEnvelopeLogger = FlowEnvelopeLogger.create(retryCfg.headerCfg, Logger(id))
73   private[this] val log : Logger = Logger[JmsRetryProcessor]
74 
75   private[this] var actor : Option[ActorRef] = None
76 
77   private[this] val router = new JmsRetryRouter("route", retryCfg, retryLog)
78 
79   class RetryDestinationResolver(
80     override val headerConfig : FlowHeaderConfig,
81     override val settings : JmsProducerSettings,
82     val validator : FlowProcessor.IntegrationStep
83   ) extends FlowHeaderConfigAware with JmsEnvelopeHeader {
84 
85     override def sendParameter(session : Session, env : FlowEnvelope) : Try[JmsSendParameter] = Try {
86 
87       val dest : JmsDestination = env.exception match {
88         // If the envelope does not have an exception, we will send it to the original destination for reprocessing
89         // If the header "RetryDestination" is missing, we will send the message to the retry failed queue
90         case None =>
91           JmsDestination.create(env.headerWithDefault[String](headerConfig.headerRetryDestination, retryCfg.failedDestName)).get
92 
93         // If the envelope has an exception, we will try to resend it unless the retry router validation
94         // throws an exception (which always means we can't retry the message)
95         case Some(_) =>
96           validator(env) match {
97             case Success(_) => JmsDestination.create(retryCfg.retryDestName).get
98             case Failure(_) => JmsDestination.create(retryCfg.failedDestName).get          }
99       }
100 
101       JmsSendParameter(
102         message = createJmsMessage(session, env).get,
103         destination = dest,
104         deliveryMode = JmsDeliveryMode.Persistent,
105         priority = settings.priority,
106         ttl = settings.timeToLive
107       )
108     }
109   }
110 
111   protected def retrySource : Source[FlowEnvelope, NotUsed] = {
112     val settings = JmsConsumerSettings(
113       log = retryLog,
114       headerCfg = retryCfg.headerCfg,
115       connectionFactory = retryCfg.cf,
116       acknowledgeMode = AcknowledgeMode.ClientAcknowledge,
117       jmsDestination = Some(JmsDestination.create(retryCfg.retryDestName).get),
118       logLevel = _ => LogLevel.Debug
119     )
120 
121     jmsConsumer(
122       name = settings.jmsDestination.get.asString,
123       settings = settings,
124       minMessageDelay = Some(retryCfg.retryInterval)
125     )
126   }
127 
128   protected def resendMessage : Flow[FlowEnvelope, FlowEnvelope, NotUsed] = {
129     val producerSettings : JmsProducerSettings = JmsProducerSettings(
130       log = retryLog,
131       headerCfg = retryCfg.headerCfg,
132       connectionFactory = retryCfg.cf,
133       destinationResolver = s => new RetryDestinationResolver(retryCfg.headerCfg, s, router.validate(LogLevel.Trace)),
134       deliveryMode = JmsDeliveryMode.Persistent,
135       timeToLive = None,
136       clearPreviousException = true,
137       logLevel = env => if (router.validate(LogLevel.Trace)(env).isFailure) {
138         LogLevel.Info
139       } else {
140         LogLevel.Debug
141       }
142     )
143 
144     jmsProducer(
145       name = id + ".routeSend",
146       settings = producerSettings,
147       autoAck = false
148     )
149   }
150 
151   protected def sendTransaction : Flow[FlowEnvelope, FlowEnvelope, NotUsed] = {
152 
153     val wiretap = new TransactionWiretap(
154       cf = retryCfg.cf,
155       eventDest = JmsDestination.create(retryCfg.eventDestName).get,
156       headerCfg = retryCfg.headerCfg,
157       inbound = false,
158       trackSource = id,
159       log = retryLog
160     )
161 
162     Flow.fromGraph(FlowProcessor.fromFunction(id, retryLog)(router.validate(LogLevel.Trace)))
163       .via(FlowProcessor.log(LogLevel.Debug, retryLog, "Creating transaction failed event"))
164       .via(wiretap.flow(true))
165   }
166 
167   protected def sendToOriginal : Flow[FlowEnvelope, FlowEnvelope, NotUsed] = resendMessage
168 
169   protected def sendToRetry : Flow[FlowEnvelope, FlowEnvelope, NotUsed] = resendMessage
170 
171   protected def retryGraph : Graph[FlowShape[FlowEnvelope, FlowEnvelope], NotUsed] = {
172 
173     GraphDSL.create() { implicit b =>
174 
175       // determine the retry routing parameters from the message
176       val route = b.add(router.flow)
177 
178       val routeSend = b.add(sendToOriginal)
179 
180       // After determining the retry parameters we send the envelope to either the Retry Destination
181       // or the ReryFailed destination
182       route.out ~> routeSend.in
183 
184       // Partition the normal outcome and the error outcome of forwarding the message
185       val routeErrorSplit = b.add(FlowProcessor.partition[FlowEnvelope](_.exception.isEmpty))
186 
187       routeSend.out ~> routeErrorSplit.in
188 
189       // Merge the splitted branches
190       val merge = b.add(Merge[FlowEnvelope](2))
191 
192       // If no errors occurred so far, we will simply pass the envelope to acknowledgement
193       routeErrorSplit.out0 ~> merge.in(0)
194 
195       // In case of an error resending the message to the original JMS Destination we will
196       // resend it to the end of the retry destination right away
197       val retrySend = b.add(sendToRetry)
198       routeErrorSplit.out1 ~> retrySend ~> merge.in(1)
199 
200       // After sending processing the message we check whether we need to send a
201       // transaction failed event. We only need to send a transaction failed event in
202       // case the envelope is marked with an exception after trying to forward the
203       // message
204 
205       val transSplit = b.add(FlowProcessor.partition[FlowEnvelope] { env => env.exception.isEmpty && router.validate(LogLevel.Trace)(env).isFailure })
206       val transMerge = b.add(Merge[FlowEnvelope](2))
207 
208       merge.out ~> transSplit.in
209 
210       transSplit.out0 ~> sendTransaction ~> transMerge.in(0)
211       transSplit.out1 ~> transMerge.in(1)
212 
213       // Acknowledge / Deny the result of the overall retry flow
214       val ack = b.add(
215         Flow.fromGraph(FlowProcessor.log(LogLevel.Debug, retryLog, "Before Acknowledge"))
216           .via(new AckProcessor(id + ".ack").flow)
217       )
218       transMerge.out ~> ack.in
219 
220       // Finally we hook up the dangling endpoints of the flow
221       new FlowShape[FlowEnvelope, FlowEnvelope](route.in, ack.out)
222     }
223   }

225   def start() : Unit = {
226 
227     actor.synchronized {
228       if (actor.isEmpty) {
229         log.info(s"Starting Jms Retry processor [$id] with [$retryCfg]")
230 
231         actor = Some(system.actorOf(StreamController.props[FlowEnvelope, NotUsed](
232           streamName = id,
233           src = retrySource.via(retryGraph),
234           streamCfg = streamsCfg
235         )(onMaterialize = _ => () )))
236       }
237     }
238   }
239 
240   def stop() : Unit = {
241     actor.synchronized {
242       actor.foreach(system.stop)
243       actor = None
244     }
245   }
