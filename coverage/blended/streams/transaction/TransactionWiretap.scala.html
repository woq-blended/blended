49   private[transaction] val createTransaction : Flow[FlowEnvelope, FlowEnvelope, NotUsed] = {
50 
51     def startTransaction(env : FlowEnvelope) : FlowTransactionEvent = {
52       FlowTransactionStarted(env.id, env.flowMessage.header)
53     }
54 
55     def updateTransaction(env : FlowEnvelope) : FlowTransactionEvent = {
56 
57       env.exception match {
58         case None =>
59           val branch = env.header[String](headerCfg.headerBranch).getOrElse("default")
60           FlowTransactionUpdate(env.id, env.flowMessage.header, WorklistStateCompleted, branch)
61 
62         case Some(e) => FlowTransactionFailed(env.id, env.flowMessage.header, Some(e.getMessage()))
63       }
64     }
65 
66     val g = Flow.fromFunction[FlowEnvelope, FlowEnvelope] { env =>
67       val event : FlowTransactionEvent = if (inbound) {
68         startTransaction(env)
69       } else {
70         updateTransaction(env)
71       }
72 
73       log.logEnv(env, LogLevel.Debug, s"Generated bridge transaction event [$event]", false)
74       FlowTransactionEvent.event2envelope(headerCfg)(event)
75         .withHeader(headerCfg.headerTrackSource, trackSource).unwrap
76     }
77 
78     Flow.fromGraph(g)
79   }

81   private[transaction] def transactionSink() : Flow[FlowEnvelope, FlowEnvelope, _] = {
82 
83     val g = GraphDSL.create() { implicit b =>
84       import GraphDSL.Implicits._
85 
86       val settings = JmsProducerSettings(

88         headerCfg = headerCfg,

94       )
95 
96       val producer = b.add(jmsProducer(
97         name = "event",

101 
102       val switchOffTracking = b.add(Flow.fromFunction[FlowEnvelope, FlowEnvelope] { env =>
103         log.logEnv(env, LogLevel.Trace, s"About to send envelope [$env]")
104         env.withHeader(headerCfg.headerTrack, false).unwrap
105       })
106 
107       switchOffTracking ~> producer
108       FlowShape(switchOffTracking.in, producer.out)
109     }
110 
111     Flow.fromGraph(g)
112   }

114   def flow(clearException : Boolean = false) : Flow[FlowEnvelope, FlowEnvelope, NotUsed] = {
115     val g = GraphDSL.create() { implicit b =>
116       import GraphDSL.Implicits._
117

121       val zip = b.add(Zip[FlowEnvelope, FlowEnvelope]())
122 
123       val select = b.add(
124         Flow.fromFunction[(FlowEnvelope, FlowEnvelope), FlowEnvelope] { pair =>
125 
126           if (clearException) {
127             pair._2.clearException()
128           } else {
129             pair._2
130           }
131         }
132       )
133 
134       split.out(1) ~> zip.in1
135       split.out(0) ~> trans ~> sink ~> zip.in0
136 
137       zip.out ~> select
138 
139       FlowShape(split.in, select.out)
140     }
141 
142     Flow.fromGraph(g)
