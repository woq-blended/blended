15 /**
16   * Consume transaction events from JMS and produce appropiate logging entries.
17   *
18   * Other parties produce [[FlowTransactionEvent]]s and send them to JMS wrapped
19   * within a [[FlowEnvelope]]. The transformation between transaction events and
20   * envelopes is encapsulated in the companion object of [[FlowTransactionEvent]].
21   *
22   * When a transaction event is received in it's envelope, the following steps
23   * will be executed:
24   *
25   * 1) forward the envelope unmodified to a JMS topic. This is an optional step
26   *    that only applies if the [[internalCf]] parameter is not [[None]].
27   * 2) decode the inbound envelope back into a transaction event
28   * 3) use the underlying [[FlowTransactionManager]] to update the containers
29   *    persistent store of known transactions
30   * 4) log the transaction event
31   *
32   * @param internalCf if the transaction event shall be forwarded to JMS, this
33   *                   the [[IdAwareConnectionFactory]] to be used for sending the
34   *                   message
35   * @param headerCfg  The [[FlowHeaderConfig]] used to calculate the property names
36   * @param tMgr       The [[FlowTransactionManager]] used to manage the persistence of the
37   *                   known transactions of the container
38   * @param streamLogger a Logger to be used for any logging
39   * @param system     The underlying [[ActorSystem]]
40   */
41 class FlowTransactionStream(
42   internalCf : Option[IdAwareConnectionFactory],
43   headerCfg : FlowHeaderConfig,
44   tMgr : FlowTransactionManager,
45   streamLogger: FlowEnvelopeLogger
46 )(implicit system: ActorSystem) extends JmsStreamSupport {
47 
48   private implicit val materializer : Materializer = ActorMaterializer()
49 
50   // recreate the FlowTransactionEvent from the inbound envelope
51   private val updateEvent : FlowEnvelope => Try[FlowTransactionEvent] = { env =>
52     Try {
53       val event = FlowTransactionEvent.envelope2event(headerCfg)(env).get
54       streamLogger.logEnv(
55         env, LogLevel.Trace, s"Received transaction event [${event.transactionId}][${event.state}]"
56       )
57       event
58     }
59   }
60 
61   // Create a JMS producer to perform the logging to JMS
62   private val logEventToJms : IdAwareConnectionFactory => Flow[FlowEnvelope, FlowEnvelope, NotUsed] = { cf =>
63 
64     val settings : JmsProducerSettings = JmsProducerSettings(
65       log = streamLogger,
66       headerCfg = headerCfg,
67       connectionFactory = cf,
68       jmsDestination = Some(JmsTopic(s"${headerCfg.prefix}.topic.transactions")),
69       clearPreviousException = true,
70       logLevel = _ => LogLevel.Debug
71     )
72 
73     jmsProducer(name = "logToJms", settings, autoAck = false)
74   }
75 
76   // run the inbound transaction update through the transaction manager
77   private val recordTransaction : Graph[FlowShape[Try[FlowTransactionEvent], Try[FlowTransaction]], NotUsed] = {
78 
79     GraphDSL.create() { implicit b =>
80       import GraphDSL.Implicits._
81 
82       val callTMgr = b.add(Flow.fromFunction[FlowTransactionEvent, Try[FlowTransaction]]{e => tMgr.updateTransaction(e) })
83 
84       val errorFilter = b.add(FlowProcessor.partition[Try[FlowTransactionEvent]](_.isSuccess))
85 
86       // If the incoming trys are success
87       val succ = b.add(Flow.fromFunction[Try[FlowTransactionEvent], FlowTransactionEvent](_.get))
88       errorFilter.out0 ~> succ ~> callTMgr
89 
90       // if the incoming trys are failures
91       val failed = b.add(Flow.fromFunction[Try[FlowTransactionEvent], Try[FlowTransaction]](evt => Failure(evt.failed.get)))
92       errorFilter.out1 ~> failed
93 
94       val fanIn = b.add(Merge[Try[FlowTransaction]](2))
95       callTMgr.out ~> fanIn.in(0)
96       failed.out ~> fanIn.in(1)
97 
98       FlowShape(errorFilter.in, fanIn.out)
99     }
100   }
101 
102   private val logTransaction : Graph[FlowShape[Try[FlowTransaction], Try[FlowEnvelope]], NotUsed] = {
103 
104     val mdc : FlowTransaction => Map[String, String] = trans => FlowEnvelopeLogger.mdcMap(
105       FlowEnvelopeLogger.mdcPrefix(headerCfg), trans.creationProps
106     )
107 
108     GraphDSL.create() { implicit b =>
109 
110       val f = b.add(Flow.fromFunction[Try[FlowTransaction], Try[FlowEnvelope]]{
111         case Success(t) =>
112           t.state match {
113             case FlowTransactionStateStarted | FlowTransactionStateCompleted =>
114               streamLogger.underlying.infoMdc(mdc(t))(t.toString())
115             case FlowTransactionStateFailed =>
116               streamLogger.underlying.warnMdc(mdc(t))(t.toString())
117             case _ =>
118               streamLogger.underlying.debugMdc(mdc(t))(t.toString())
119           }
120 
121           Success(FlowTransaction.transaction2envelope(headerCfg)(t))
122         case Failure(t) =>
123           Failure(t)
124       })
125 
126       FlowShape(f.in, f.out)
127     }
128   }
129 
130   private val createResult : ((FlowEnvelope, Try[FlowEnvelope])) => FlowEnvelope = { case (orig, env) =>
131     env match {
132       case Success(e) =>
133         streamLogger.logEnv(e, LogLevel.Trace, s"Successfully processed transaction event [${e.id}]")
134         e.withAckHandler(orig.getAckHandler).withRequiresAcknowledge(orig.requiresAcknowledge).clearException()
135       case Failure(t) =>
136         streamLogger.logEnv(orig, LogLevel.Trace, s"Failed to process transaction event [${orig.id}]")
137         orig.withException(t)
138     }
139   }
140 
141   def build() : Flow[FlowEnvelope, FlowEnvelope, NotUsed] = {
142 
143     val g : Graph[FlowShape[FlowEnvelope, FlowEnvelope], NotUsed] = GraphDSL.create() { implicit b =>
144       import GraphDSL.Implicits._
145 
146       // decode the incoming FlowEvent
147       val update = b.add(Flow.fromFunction(updateEvent).named("updateEvent"))
148 
149       // update the persisted transaction
150       val record = b.add(Flow.fromGraph(recordTransaction).named("recordTransaction"))
151 
152       // log the transaction
153       val logTrans = b.add(Flow.fromGraph(logTransaction).named("logTransaction"))
154 
155       // We need to split, so that we keep the original message in one branch and process the
156       // event in the other branch
157       val split = b.add(Broadcast[FlowEnvelope](2))
158 
159 
160       val join = b.add(Zip[FlowEnvelope, Try[FlowEnvelope]])
161       split.out(0) ~> join.in0
162 
163       internalCf match {
164         case None =>
165           split.out(1) ~> update ~> record ~> logTrans
166         case Some(cf) =>
167           val logToJms = b.add(logEventToJms(cf).named("logToJms"))
168           split.out(1) ~> logToJms ~> update ~> record ~> logTrans
169       }
170 
171       logTrans.out ~> join.in1
172 
173       val result = b.add(Flow.fromFunction(createResult).named("result"))
174       join.out ~> result.in
175 
176       FlowShape(split.in, result.out)
177     }
178 
179     Flow.fromGraph(g)
180   }
