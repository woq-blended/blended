45   val transaction2envelope : FlowHeaderConfig => FlowTransaction => FlowEnvelope = { cfg => t =>

54   val envelope2Transaction : FlowHeaderConfig => FlowEnvelope => FlowTransaction = { cfg => env =>
55     val state : FlowTransactionState =
56       env.header[String](cfg.headerState).map(h => FlowTransactionState.apply(h).get).getOrElse(FlowTransactionStateStarted)
57 
58     val worklistState : Map[String, List[WorklistState]] = env.header[String](cfg.headerBranch).map { s =>
59       if (s.isEmpty) {
60         Map.empty[String, List[WorklistState]]
61       } else {
62 
63         val branches : Map[String, List[String]] =
64           s.split(branchSeparator)
65             .map(b => b.split("="))
66             .filter(_.length == 2)
67             .map { b =>
68               b(0) -> b(1).split(stateSeparator).toList
69             }.toMap
70 
71         branches.mapValues(s => s.map(i => WorklistState.apply(i).get))
72       }
73     }.getOrElse(Map.empty[String, List[WorklistState]])
74 
75     val created : Date = new Date(env.header[Long](cfg.headerTransCreated).getOrElse(System.currentTimeMillis()))
76     val updated : Date = new Date(env.header[Long](cfg.headerTransUpdated).getOrElse(created.getTime()))
77 
78     FlowTransaction(
79       id = env.id,
80       created = created,
81       lastUpdate = updated,
82       creationProps = env.flowMessage.header,
83       state = state, worklist = worklistState

87   private[transaction] def worklistState(currentState : FlowTransactionState, wl : Map[String, List[WorklistState]]) : List[FlowTransactionState] = {
88     wl.map { case (_,v) =>
89       if (v.contains(WorklistStateFailed) || v.contains(WorklistStateTimeout)) {
90         FlowTransactionStateFailed
91       } else if (v.contains(WorklistStateStarted) && v.contains(WorklistStateCompleted)) {
92         FlowTransactionStateCompleted
93       } else if (v.contains(WorklistStateStarted)) {
94         FlowTransactionStateStarted
95       } else if (v.contains(WorklistStateCompleted)) {
96         FlowTransactionStateUpdated
97       } else {
98         currentState
99       }
100     }.toList.distinct
101   }

103   private[transaction] def transactionState(currentState : FlowTransactionState, wl : Map[String, List[WorklistState]]) : FlowTransactionState = {
104 
105     val itemStates : List[FlowTransactionState] = FlowTransaction.worklistState(currentState, wl)
106 
107     if (itemStates.contains(FlowTransactionStateFailed)) {
108       FlowTransactionStateFailed
109     } else if (itemStates.size > 1) {
110       FlowTransactionStateUpdated
111     } else if (itemStates.equals(List(FlowTransactionStateUpdated))) {
112       FlowTransactionStateUpdated
113     } else if (itemStates.equals(List(FlowTransactionStateCompleted))) {
114       FlowTransactionStateCompleted
115     } else {
116       if (currentState == FlowTransactionStateStarted) {
117         FlowTransactionStateUpdated
118       } else {
119         currentState
120       }
121     }
122   }

163   private def applyUpdate(updated : FlowTransactionUpdate) : FlowTransaction = {
164     // We extract the id's for the transaction parts
165     val updatedItemIds : Map[String, List[WorklistState]] =
166       updated.branchIds
167         .map{ id => id -> List(updated.updatedState) }
168         .map{ case (k, s) =>
169           val oldState : List[WorklistState] = worklist.getOrElse(k, List.empty)
170           k -> (s ::: oldState).distinct
171         }
172         .toMap
173 
174     // We keep everything that is not in the update
175     val newWorklist : Map[String, List[WorklistState]] =
176       worklist.filterKeys { id => !updatedItemIds.contains(id) } ++ updatedItemIds
177 
178     copy(
179       worklist = newWorklist,
180       lastUpdate = new Date(),
181       state = FlowTransaction.transactionState(state, newWorklist)
182     )
183   }

185   private val applyEvent : FlowTransactionEvent => FlowTransaction = { event =>
186     if (event.transactionId == tid) {
187       if (state == FlowTransactionStateStarted || state == FlowTransactionStateUpdated) {
188         log.trace(s"Updating transaction [$tid] with [$event]")
189         event match {
190           case started: FlowTransactionStarted => applyStarted(started)
191           case _ : FlowTransactionCompleted => applyCompleted()
192           case _ : FlowTransactionFailed => applyFailed()
193           case updated : FlowTransactionUpdate => applyUpdate(updated)
194         }
195       } else {
196         log.trace(s"Ignoring event for already terminated transaction [$tid]")
197         this
198       }
199     } else {
200       this
201     }
202   }
203 
204   def updateTransaction(event : FlowTransactionEvent) : FlowTransaction = applyEvent(event)
