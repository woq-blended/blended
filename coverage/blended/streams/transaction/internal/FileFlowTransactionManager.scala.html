59   override def updateTransaction(e: FlowTransactionEvent): Try[FlowTransaction] = {
60     val fut : Future[Try[FlowTransaction]] = findTransaction(e.transactionId).map[(Option[FlowTransaction], FlowTransaction)] {
61       case None =>
62         val now: Date = new Date()
63         log.trace(s"Storing new transaction [${e.transactionId}]")
64 
65         val newT : FlowTransaction = FlowTransaction(
66           id = e.transactionId,
67           created = now,
68           lastUpdate = now,
69           creationProps = e.properties
70         )
71 
72         if (e.state == FlowTransactionStateStarted) {
73           (None, newT)
74         } else {
75           (None, newT.updateTransaction(e))
76         }
77       case Some(t) =>
78         log.trace(s"Updating transaction [${e.transactionId}]")
79         (Some(t), t.updateTransaction(e))
80     }.map {
81       case (old, updated) => store(old, updated)
82     }
83 
84     Await.result(fut, 3.seconds)
85   }

112     val n : Array[String] = p.toFile().getName().split("\\.")
113     if (n.length == 3) {

115         val state : FlowTransactionState = FlowTransactionState.apply(n(2)).get

139   private def store(old: Option[FlowTransaction], changed : FlowTransaction) : Try[FlowTransaction] = Try {
140 
141     val json : String = Pickle.intoString(changed)
142     val newFile : File = new File(dir, filename(changed))
143 
144     old.foreach(t => new File(dir, filename(t)).delete())
145 
146     var os : Option[OutputStream] = None
147     var writer : Option[BufferedWriter] = None
148 
149     try {
150       os = Some(new FileOutputStream(newFile))
151       writer = Some(new BufferedWriter(new PrintWriter(os.get)))
152       writer.foreach{ w => w.write(json) }
153       changed
154     } catch {
155       case NonFatal(e) =>
156         log.warn(s"Error writing transaction file [${newFile.getAbsolutePath()}][${e.getMessage()}]")
157         throw e
158     } finally {
159       writer.foreach { w =>
160         try {
161           w.close()
162         } catch {
163           case NonFatal(e) =>
164             log.warn(s"Error closing file [${newFile.getAbsolutePath()}][${e.getMessage()}]")
165         }
166       }
167 
168       os.foreach { s =>
169         try {
170           s.close()
171         } catch {
172           case NonFatal(e) =>
173             log.warn(s"Error closing file [${newFile.getAbsolutePath()}][${e.getMessage()}]")
174           }
175         }
176       }
177   }

179   private def loadExistingTransaction(f : File) : Try[FlowTransaction] = Try {
180     loadTransaction(f) match {

187   private def loadTransaction(f : File) : Try[Option[FlowTransaction]] = Try {
188     if (f.exists()) {
189       val json : String = loadFile(f).get
190       val t : FlowTransaction = Unpickle[FlowTransaction].fromString(json).get
191       Some(t)
192     } else {
193       None
194     }

197   private def loadFile(f : File) : Try[String] = {
198     val src : BufferedSource = scala.io.Source.fromFile(f, "UTF-8")
199 
200     try {
201       Success(src.getLines().mkString("\n"))
202     } catch {
203       case NonFatal(t) =>
204         log.warn(s"Exception encountered accessing transaction file [${f.getAbsolutePath()}]")
205         Failure(t)
206     } finally {
207       try {
208         src.close()
209       } catch {
210         case NonFatal(t) =>
211           log.warn(s"Error closing file [${f.getAbsolutePath()}]:[${t.getMessage()}]")
212       }
213     }
214   }

219   private def mapDirectoryStreamWithFilter(
220     dirStream : FilteredDirectoryStream
221   )(
222     select : FlowTransaction => Boolean
223   ) : Future[Seq[FlowTransaction]] = {
224 
225     val transactions : Future[Seq[FlowTransaction]] = dirStream.entries
226       .map{ p => loadExistingTransaction(p.toFile()) }
227       .filter(_.isSuccess)
228       .map(_.get)
229       .filter(select)
230       .runWith(Sink.seq)
231 
232     transactions.onComplete(_ => dirStream.close())

237   private def withDirectoryStream(
238     dirStream : FilteredDirectoryStream
239   )(
240     f : Path => Boolean
241   ) : Future[Int] = {
242 
243     val count : Future[Int] = dirStream.entries
244       .via(Flow.fromFunction{ p => if (f(p)) 1 else 0 })
245       .runFold(0)( (c,v) => c + v )
246 
247     count.onComplete(_ => dirStream.close())
