52   def envelope2event : FlowHeaderConfig => FlowEnvelope => Try[FlowTransactionEvent] = { cfg => envelope =>
53 
54     Try {
55       (envelope.header[String](cfg.headerTransId), envelope.header[String](cfg.headerState)) match {
56         case (Some(id), Some(state)) => FlowTransactionState.apply(state).get match {
57           case FlowTransactionStateStarted =>
58             val header = envelope.flowMessage.header.filter{ case (k, v) => !k.startsWith("JMS") }
59             FlowTransactionStarted(id, header)
60 
61           case FlowTransactionStateCompleted =>
62             FlowTransactionCompleted(id, envelope.flowMessage.header)
63 
64           case FlowTransactionStateFailed =>
65             val reason : Option[String] = envelope.flowMessage match {
66               case txtMsg : TextFlowMessage => Some(txtMsg.content)
67               case _                        => None
68             }
69             FlowTransactionFailed(id, envelope.flowMessage.header, reason)
70 
71           case FlowTransactionStateUpdated =>
72 
73             val branchIds : Seq[String] = envelope.header[String](cfg.headerBranch) match {
74               case Some(s) => if (s.isEmpty()) Seq() else s.split(",")
75               case None    => Seq()
76             }
77 
78             val updatedState : WorklistState = envelope.flowMessage match {
79               case txtMsg : TextFlowMessage => WorklistState.apply(txtMsg.content).get
80               case m => throw new InvalidTransactionEnvelopeException(s"Expected TextFlowMessage for an update envelope, actual [${m.getClass().getName()}]")
81             }
82 
83             FlowTransactionUpdate(id, envelope.flowMessage.header, updatedState, branchIds : _*)
84 
85           case s =>
86             throw new InvalidTransactionEnvelopeException(s"Invalid Transaction state in envelope [$s]")
87         }
88         case (_, _) => throw new InvalidTransactionEnvelopeException(s"Envelope must have headers [${cfg.headerTransId}] and [${cfg.headerState}]")
89       }
90     }
91   }
