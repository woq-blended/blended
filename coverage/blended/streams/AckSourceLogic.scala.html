48 /**
49   * Provide common logic for Source stages producing [[FlowEnvelope]]s requiring
50   * acknowledgement handling.
51   *
52   * The AckSource logic encapsulates the underlying logic for an arbitrary Source
53   * of FlowEnvelopes. The envelopes will be passed down stream and must be
54   * acknowledged or denied eventually. An acknowledgement that takes too long,
55   * will be treated as a denial as well.
56   *
57   * A concrete implementation must implement the actions to be executed upon acknowledgement
58   * and denial. For example, a JMS source would use a JMS acknowledge on the underlying
59   * JMS message and a session.recover() upon a denial. A file system based source may move or
60   * delete the original file upon acknowledge and restore the original file upon denial.
61 
62   * After picking up an envelope from an external system, the envelope will be considered to
63   * be inflight until either an acknowledge or deny has been called. The AckSource logic defines
64   * the maximum number of messages that can be inflight at any moment in time. No further messages
65   * will be pulled from the external system until a free inflight slot will become available.
66   *
67   * Concrete implementations must:
68   *
69   * - create and maintain any technical connections required to poll the external system
70   * - map the inbound data to a FlowEnvelope
71   * - implement the concrete actions for acknowledgement and denial
72   *
73   * We will use an AcknowledgeContext to hold an inflight envelope and the id of the inflight slot
74   * it is using along with any additional that might be required to perform an acknowledge or denial.
75   *
76   * As a result, each poll of the external system will produce an Option[AcknowledgeContext] which
77   * will then be inserted into a free inflight slot. As a consequence, polling of the external system
78   * will only be performed if and only if a free inflight slot is available.
79 
80   * As long as free inflight slots are available and no external messages are available, the
81   * poll will be executed \in regular intervals. Concrete implementations may overwrite the
82   * the nextPoll() method to modify the calculation of the next polling occurrence.
83 */
84 abstract class AckSourceLogic[T <: AcknowledgeContext](
85   val shape : Shape,
86   val out : Outlet[FlowEnvelope],
87   val ackTimeout : FiniteDuration = 1.second
88 ) extends TimerGraphStageLogic(shape) {

109   // The map of current inflight AcknowledgeContexts. An inflight slot is considered
110   // to be available if it's id does not occur in the keys of the inflight map.
111   private val inflightMap : mutable.Map[String, (T, AckState)] = mutable.Map.empty
112 
113   private var lastUsedSlot : Option[String] = None
114 
115   // TODO: Make this configurable ?
116   protected def nextPoll() : Option[FiniteDuration] = Some(1.second)

118   // A callback to fail the stage
119   protected val fail : AsyncCallback[Throwable]= getAsyncCallback[Throwable]{ t : Throwable =>
120     log.underlying.error(t)(s"Failing stage [$id]")
121     failStage(t)
122   }

124   private def addInflight(inflightId : String, ackCtxt : T, state : AckState) : Unit = {
125     inflightMap += ( inflightId -> (ackCtxt, state))
126     log.underlying.debug(s"Inflight message count for [$id] is [${inflightMap.size}]")
127   }
128 
129   private def removeInflight(inflightId : String) : Unit = {
130     inflightMap -= inflightId
131     log.underlying.debug(s"Inflight message count for [$id] is [${inflightMap.size}]")
132   }
133 
134   // A callback to immediately schedule the next poll
135   protected val pollImmediately : AsyncCallback[Unit] = getAsyncCallback[Unit](_ => poll())
136 
137   // A callback to update the ack state for an inflight message
138   protected val updateAckState : AsyncCallback[(String, AckState)] = getAsyncCallback[(String, AckState)]{ case (id, state) =>
139 
140     inflightMap.get(id) match {
141       case Some((ctxt, _ )) =>
142         log.logEnv(ctxt.envelope, LogLevel.Debug, s"Updating state of [${ctxt.envelope.id}] for [$id] to [$state]")
143         inflightMap.put(id, (ctxt, state))
144         state match {
145           case AckState.Acknowledged => acknowledged(ctxt)
146           case AckState.Denied => denied(ctxt)
147           case _ =>
148         }
149       case None =>
150         log.underlying.debug(s"AckContext [$id] no longer inflight - perhaps it has timed out ?")
151     }
152   }
153 
154   private def acknowledged(ackCtxt : T) : Unit = {
155     ackCtxt.acknowledge()
156     log.logEnv(ackCtxt.envelope, LogLevel.Debug, s"Flow envelope [${ackCtxt.envelope.id}] has been acknowledged in [$id]")
157     // Then we clear the message from the inflight map
158     removeInflight(ackCtxt.inflightId)
159     // If the poll timer is active we actually have executed a poll recently with no result
160     if (!isTimerActive(Poll)) {
161       pollImmediately.invoke()
162     }
163   }
164 
165   override def postStop() : Unit = {
166 
167     val p : Map[String, T] = pending()
168     if (p.nonEmpty) {
169       log.underlying.debug(s"[$id] has [${p.size}] envelopes still in inflight while stopping")
170     }
171 
172     // perform any implementation specific logic to roll back pending envelopes
173     p.values.foreach(_.deny())
174   }
175 
176   // this will be called whenever an inflight message has been denied
177   private def denied(ackCtxt: T) : Unit = {
178     ackCtxt.deny()
179     log.logEnv(ackCtxt.envelope, LogLevel.Debug, s"Flow Envelope [${ackCtxt.envelope.id}] has been denied in [$id]")
180     // we need to clean up the inflight map
181     removeInflight(ackCtxt.inflightId)
182     // If the poll timer is active we actually have executed a poll recently with no result
183     if (!isTimerActive(Poll)) {
184       pollImmediately.invoke()
185     }
186   }
187 
188   // this will be called whenever the acknowledgement for an inflight
189   // message has timed out. Per default this will be delegated to the
190   // denied() handler
191   protected def ackTimedOut(ackCtxt : T): Unit = denied(ackCtxt)
192 
193   protected def determineNextSlot(slotList : List[String]) : Option[String] = {
194     lastUsedSlot = slotList.filter { id => !inflightMap.keys.exists(_ == id) } match {
195       case Nil => None
196       case head :: Nil => Some(head)
197       case l =>
198         Some(lastUsedSlot.flatMap{ lu => l.find{i => i > lu} }.getOrElse(l.head))
199     }
200 
201     lastUsedSlot
202   }
203 
204   protected def freeInflightSlot() : Option[String] = {
205     determineNextSlot(inflightSlots)
206   }
207 
208   /* Concrete implementations must implement this method to realize the technical
209      poll from the external system */
210   protected def doPerformPoll(id : String, ackHandler : AcknowledgeHandler) : Try[Option[T]]
211 
212   /* Perform a poll of the external system within the context of a free inflight slot */
213   private def performPoll(id : String) : Unit = Try {
214     // make sure, the outlet has been pulled. If that is not the case,
215     // the out handler will be called eventually, triggering another
216     // poll()
217     if (isAvailable(out) && !isTimerActive(Poll)) {
218 
219       val ackHandler : AcknowledgeHandler = new AcknowledgeHandler {
220         override def acknowledge() : Try[Unit] = Try {
221           updateAckState.invoke((id, AckState.Acknowledged))
222         }
223 
224         override def deny() : Try[Unit] = Try {
225           updateAckState.invoke((id, AckState.Denied))
226         }
227       }
228 
229       log.underlying.trace(s"Performing poll for [$id]")
230       doPerformPoll(id, ackHandler) match {
231         case Success(None) =>
232 
233           nextPoll() match {
234             case None => pollImmediately.invoke()
235             case Some(d) => if (!isTimerActive(Poll)) {
236               log.underlying.trace(s"Scheduling next poll for [$id] in [$d]")
237               scheduleOnce(Poll, d)
238             }
239           }
240 
241         case Success(Some(ackCtxt)) =>
242           // add the context to the inflight messages
243           log.logEnv(ackCtxt.envelope, LogLevel.Debug, s"Received [${ackCtxt.envelope.flowMessage}] in [$id]")
244           addInflight(id, ackCtxt, AckState.Pending)
245           // push the envelope to the outlet
246           if (autoAcknowledge) {
247             log.logEnv(ackCtxt.envelope, LogLevel.Debug, s"Auto Acknowledging [${ackCtxt.envelope.id}] in [$id]")
248             acknowledged(ackCtxt)
249             push(out, ackCtxt.envelope.withRequiresAcknowledge(false).withAckHandler(None))
250           } else {
251             addInflight(id, ackCtxt, AckState.Pending)
252             push(out, ackCtxt.envelope)
253           }
254 
255         case Failure(t) =>
256           log.underlying.warn(t)(s"Failed to poll for new message in [$id]")
257           failStage(t)
258       }
259     }
260   }
261 
262   protected def pending() : Map[String, T] =
263     inflightMap.filter { case (_, (_, state)) => state == AckState.Pending }.toMap.mapValues(_._1)
264 
265   override protected def onTimer(timerKey : Any) : Unit = {
266     timerKey match {
267       case CheckAck =>
268 
269         val timedoutAcks : Map[String, T] =
270           pending().filter {
271             case (_, ctxt) =>
272               System.currentTimeMillis() - ctxt.created > ackTimeout.toMillis
273           }
274 
275         timedoutAcks.values.foreach { ctxt =>
276           log.logEnv(ctxt.envelope, LogLevel.Warn, s"Acknowledge for [${ctxt.envelope}] has timed out in [${ctxt.inflightId}]")
277           ackTimedOut(ctxt)
278         }
279 
280       case Poll =>
281         log.underlying.trace(s"Received scheduled poll event")
282         poll()
283     }
284   }
285 
286   /**
287    * Poll the external system whether a new message is available.
288    * If a message can be polled without any exceptions, the resulting envelope
289    * will be wrapped in an AcknowledgeContext and returned.
290    * If no message is available, the result will be Success(None)
291    * *
292    * Any exception while polling for a message will deny all remaining inflight
293    * messages and then fail the stage.
294    */
295 
296   protected def poll() : Unit = {
297     // select a free inflight slot and trigger to poll and process the next inbound
298     // message
299     freeInflightSlot().foreach(performPoll)
300   }
301 
302   /*
303      Finally set the handler for the outlet.
304   */
305 
306   setHandler(out, new OutHandler() {
307     override def onPull() : Unit = {
308       if (!isTimerActive(Poll)) {
309         poll()
310       }
311     }
312   })
