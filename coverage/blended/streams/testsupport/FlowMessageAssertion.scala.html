55 class ExpectedBodies(bodies : Option[Any]*) extends FlowMessageAssertion {
56 
57   private val unmatched : FlowMessage => Option[Any] => Boolean = msg => expected =>
58     msg match {
59       case txtMsg : TextFlowMessage =>
60         !expected.isInstanceOf[Option[String]] || expected.isEmpty ||
61           expected.forall(e => !e.toString.equals(txtMsg.content))
62 
63       case binMsg : BinaryFlowMessage => expected.isDefined && {
64         expected.forall { c =>
65           c match {
66             case byteString : ByteString => !byteString.equals(binMsg.content)
67             case byteArr : Array[Byte]   => !ByteString(byteArr).equals(binMsg.content)
68             case _                       => true
69           }
70         }
71       }
72 
73       case _ : BaseFlowMessage => expected.isDefined
74     }
75 
76   override def f : Seq[FlowEnvelope] => Try[String] = l => {
77 
78     def compareBodies(matchList : Map[Option[Any], FlowMessage]) : Try[String] = Try {
79       matchList.filter { case (expected, actual) => unmatched(actual)(expected) } match {
80         case s if s.isEmpty => "Collector has received the correct bodies"
81         case e =>
82           val msg = e.map { case (b, a) => s"[$b != ${a.body()}]" }.mkString(",")
83           throw new Exception(s"Unexpected Bodies: $msg")
84       }
85     }
86 
87     if (bodies.length == 1) {
88       val compMap : Map[Option[Any], FlowMessage] = l.map { m => (bodies(0), m.flowMessage) }.toMap
89       compareBodies(compMap)
90     } else {
91       l.size match {
92         case n if n == bodies.length =>
93           compareBodies(bodies.toList.zip(l.map { _.flowMessage }).toMap)
94         case _ => throw new Exception(s"The number of messages received [${l.size}] does not match the number of bodies [${bodies.length}]")
95       }
96     }
