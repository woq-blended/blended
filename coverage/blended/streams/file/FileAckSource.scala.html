44 class FileAckSource(
45   pollCfg : FilePollConfig,
46   logger : FlowEnvelopeLogger

48 
49   private val pollId : String =  s"${pollCfg.headerCfg.prefix}.FilePoller.${pollCfg.id}.source"
50   private val out : Outlet[FlowEnvelope] = Outlet(name = pollId)
51   private val sdf = new SimpleDateFormat("yyyyMMdd-HHmmssSSS")

55   private class FileAckContext(
56     inflightId : String,
57     env: FlowEnvelope,
58     val originalFile : File,
59     val fileToProcess : File
60   ) extends DefaultAcknowledgeContext(inflightId, env, System.currentTimeMillis()) {
61 
62     override def acknowledge() : Unit = {
63       logger.logEnv(env, LogLevel.Debug, s"Successfully processed envelope [${envelope.id}]")
64       pollCfg.backup match {
65         case None =>
66           if (fileToProcess.delete()) {
67             logger.logEnv(env, LogLevel.Debug, s"Deleted file for [${envelope.id}] : [${fileToProcess}]")
68           } else {
69             logger.logEnv(env, LogLevel.Warn, s"File for [${envelope.id}] could not be deleted : [${fileToProcess}]")
70           }
71         case Some(d) =>
72 
73           val backupDir = new File(d)
74           if (!backupDir.exists()) {
75             backupDir.mkdirs()
76           }
77 
78           val backupFileName =
79             if (pollCfg.backupTimestamp) {
80               originalFile.getName + "-" + sdf.format(new Date())
81             } else {
82               originalFile.getName
83             }
84 
85           val fFrom : File = fileToProcess
86           val fTo : File = new File(backupDir, backupFileName)
87 
88           if (FileHelper.renameFile(fFrom, fTo, true)) {
89             logger.logEnv(env, LogLevel.Debug, s"Moved file for [${envelope.id}] from [${fFrom.getAbsolutePath()}] to [${fTo.getAbsolutePath()}]")
90           } else {
91             logger.logEnv(env, LogLevel.Debug, s"File for [${envelope.id}] failed to be renamed from [${fFrom.getAbsolutePath()}] to [${fTo.getAbsolutePath()}]")
92           }
93       }
94     }
95 
96     override def deny() : Unit = {
97       logger.logEnv(env, LogLevel.Debug, s"Restoring file [${originalFile}] in [${inflightId}]")
98       FileHelper.renameFile(fileToProcess, originalFile, true)
99     }
100   }

102   private class FileSourceLogic() extends AckSourceLogic[FileAckContext](shape, out, pollCfg.ackTimeout) {
103     /** The id to identify the instance in the log files */
104     override val id : String = pollId
105 
106     /** A logger that must be defined by concrete implementations */
107     override protected def log: FlowEnvelopeLogger = logger
108 
109     log.underlying.debug(s"Initializing FileAckSource with config [$pollCfg], ackTimeout [$ackTimeout]")
110 
111     /** The id's of the available inflight slots */
112     override protected val inflightSlots : List[String] =
113       1.to(pollCfg.batchSize).map(i => s"FilePoller-${pollCfg.id}-$i").toList
114 
115     override protected def nextPoll(): Option[FiniteDuration] = Some(pollCfg.interval)
116 
117     override protected def doPerformPoll(id : String, ackHandler : AcknowledgeHandler) : Try[Option[FileAckContext]] = {
118 
119       def createEnvelope(f : File) : Try[Option[FileAckContext]] = Try {
120 
121         val sdf : SimpleDateFormat = new SimpleDateFormat("yyyyMMdd-HHmmssSSS")
122 
123         if (f.exists()) {
124           val uuid : String = UUID.randomUUID().toString() + "-" + f.getName()
125 
126           val fileToProcess : File = new File(f + pollCfg.tmpExt)
127 
128           // First we try to rename the file in order to check whether it can be accessed yet
129           if (FileHelper.renameFile(f, fileToProcess, false)) {
130             val bytes : Array[Byte] = FileHelper.readFile(fileToProcess.getAbsolutePath())
131 
132             val msg : FlowMessage = if (pollCfg.asText) {
133               val charSet : Charset = pollCfg.charSet match {
134                 case None => Charset.defaultCharset()
135                 case Some(s) => Charset.forName(s)
136               }
137 
138               FlowMessage(new String(bytes, charSet))(pollCfg.header)
139             } else {
140               FlowMessage(bytes)(pollCfg.header)
141             }
142 
143             val env : FlowEnvelope = FlowEnvelope(msg,uuid)
144               .withHeader(pollCfg.filenameProp, f.getName()).get
145               .withHeader(pollCfg.filepathProp, f.getAbsolutePath()).get
146               .withRequiresAcknowledge(true)
147               .withAckHandler(Some(ackHandler))
148 
149             log.logEnv(env, LogLevel.Info, s"Created Envelope [$env] in [$id]]")
150 
151             Some(new FileAckContext(
152               inflightId = id,
153               env = env,
154               originalFile = f,
155               fileToProcess = fileToProcess
156             ))
157           } else {
158             None
159           }
160         } else {
161           None
162         }
163       }
164 
165       // first we try to lock the directory if it can't be locked, we will assume that another file poller is currently
166       // working on the same directory
167       if (locked()) {
168         Success(None)
169       } else {
170         FileAckSource.nextFile(pollCfg) match {
171           case Some(f) => Success(createEnvelope(f).get)
172           case None => Success(None)
173         }
174       }
175     }
176 
177     /**
178       * The file poller can be locked by the existence of a lock file if specified.
179       */
180     private[this] def locked() : Boolean = pollCfg.lock match {
181       case None => false
182       case Some(l) =>
183         val f = if (l.startsWith("./")) {
184           new File(pollCfg.sourceDir, l.substring(2))
185         } else {
186           new File(l)
187         }
188 
189         if (f.exists()) {
190           log.underlying.debug(s"Directory for [${pollCfg.id}] is locked with file [${f.getAbsolutePath()}]")
191           true
192         } else {
193           false
194         }
195     }
196   }
