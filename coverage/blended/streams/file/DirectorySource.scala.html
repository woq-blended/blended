11 class DirectorySource(pollCfg : FilePollConfig) {
12 
13   private val log : Logger = Logger[DirectorySource]
14   private var pendingFiles : Map[File, Option[Long]] = Map.empty
15 
16   private val srcDir = new File(pollCfg.sourceDir)
17 
18   // First we make sure the poll directory exists
19   if (!srcDir.exists()) {
20     srcDir.mkdirs()
21   }
22 
23   def nextFile() : Option[File] = {
24     // If the directory does not yet exist or is not readable, we log a warning
25     if (!srcDir.exists() || !srcDir.isDirectory() || !srcDir.canRead()) {
26       log.warn(s"Directory [$srcDir] for [${pollCfg.id}] does not exist or is not readable.")
27       None
28     } else {
29 
30       // first we clear all entries from the pending files that have been delivered to a file poller longer
31       // than the poll interval ago
32       cleanup()
33 
34       // If we don't have a current file to deliver we will scan the directory again
35       if (file().isEmpty) {
36         scanDirectory()
37       }
38 
39       val result : Option[File] = file() match {
40         case None => None
41         case Some(f) =>
42           pendingFiles = pendingFiles.filterKeys(_ != f) + (f -> Some(System.currentTimeMillis()))
43           Some(f)
44       }
45 
46       result
47     }
48   }
49 
50   private def cleanup() : Unit = {
51     val now : Long = System.currentTimeMillis()
52     pendingFiles = pendingFiles.filter {
53       case (_, Some(v)) =>  now - v < pollCfg.interval.toMillis
54       case (_, None) => true
55     }
56   }
57 
58   private def scanDirectory() : Unit = {
59     log.info(s"Executing directory scan for directory [${pollCfg.sourceDir}] with pattern [${pollCfg.pattern}]")
60 
61     val filter : DirectoryStream.Filter[Path] = new DirectoryStream.Filter[Path] {
62       override def accept(entry: Path): Boolean = {
63         entry.getParent().toFile().equals(srcDir) &&
64           pollCfg.pattern.forall(p => entry.toFile().getName().matches(p)) &&
65           !pendingFiles.contains(entry.toFile())
66       }
67     }
68 
69     val dirStream : DirectoryStream[Path] = Files.newDirectoryStream(srcDir.toPath(), filter)
70 
71     try {
72       // scalastyle:off magic.number
73       dirStream.iterator().asScala.take(100).map(_.toFile()).foreach(f => pendingFiles += (f -> None))
74       // scalastyle:on magic.number
75     } catch {
76       case NonFatal(e) =>
77         log.warn(s"Error reading directory [${srcDir.getAbsolutePath()}] : [${e.getMessage()}]")
78         None
79     } finally {
80       dirStream.close()
81     }
82   }
83 
84   private def file() : Option[File] = pendingFiles.find{ case (_, v) => v.isEmpty }.map(_._1)
