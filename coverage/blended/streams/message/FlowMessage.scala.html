94   private def classMatch(v : Any, clazz : Class[_]) : Boolean = {
95 
96     val intClasses : Seq[Class[_]] = Seq(classOf[Integer], classOf[Int])
97     val longClasses : Seq[Class[_]] = Seq(classOf[java.lang.Long], classOf[Long])
98     val shortClasses : Seq[Class[_]] = Seq(classOf[java.lang.Short], classOf[Short])
99     val floatClasses : Seq[Class[_]] = Seq(classOf[java.lang.Float], classOf[Float])
100     val doubleClasses : Seq[Class[_]] = Seq(classOf[java.lang.Double], classOf[Double])
101     val boolClasses : Seq[Class[_]] = Seq(classOf[java.lang.Boolean], classOf[Boolean])
102     val byteClasses : Seq[Class[_]] = Seq(classOf[java.lang.Byte], classOf[Byte])
103 
104     val matches : Map[Class[_], Seq[Class[_]]] = Map(
105       classOf[java.lang.Integer] -> intClasses,
106       classOf[Int] -> intClasses,
107       classOf[java.lang.Long] -> longClasses,
108       classOf[Long] -> longClasses,
109       classOf[Short] -> shortClasses,
110       classOf[java.lang.Short] -> shortClasses,
111       classOf[java.lang.Float] -> floatClasses,
112       classOf[Float] -> floatClasses,
113       classOf[java.lang.Double] -> doubleClasses,
114       classOf[Double] -> doubleClasses,
115       classOf[java.lang.Boolean] -> boolClasses,
116       classOf[Boolean] -> boolClasses,
117       classOf[java.lang.Byte] -> byteClasses,
118       classOf[Byte] -> byteClasses
119     )
120 
121     v.getClass() == clazz || matches.get(v.getClass()).exists { clazzes =>
122       clazzes.exists(c => v.getClass().isAssignableFrom(c))
123     }
124   }

126   private def fromString[T](v : String)(implicit clazz : ClassTag[T]) : Option[T] = {
127 
128     val resMap : Map[Class[_], String => Any] = Map (
129       classOf[Short] -> (s => s.toShort),
130       classOf[Int] -> (s => s.toInt),
131       classOf[Long] -> (s => s.toLong),
132       classOf[Boolean] -> (s => s.toBoolean),
133       classOf[Byte] -> (s => s.toByte),
134       classOf[Float] -> (s => s.toFloat),
135       classOf[Double] -> (s => s.toDouble),
136       classOf[Unit] -> (_ => Unit)
137     )
138 
139     resMap.get(clazz.runtimeClass).map(_(v)).map(_.asInstanceOf[T])
140   }

142   def header[T](name : String)(implicit m : Manifest[T]) : Option[T] = {
143 
144     header.get(name) match {
145       case Some(v) if classMatch(v.value, m.runtimeClass) =>
146         Some(v.value.asInstanceOf[T])
147 
148       case Some(v) if v.isInstanceOf[StringMsgProperty] =>
149         fromString(v.value.toString)
150 
151       case Some(v) if v.isInstanceOf[UnitMsgProperty] && m.runtimeClass.getName().equals("void") =>
152         Some(v.value.asInstanceOf[T])
153 
154       case _ =>
155         None
156     }
157   }
158 
159   def headerWithDefault[T](name : String, default : T)(implicit m : Manifest[T]) : T = header[T](name) match {
