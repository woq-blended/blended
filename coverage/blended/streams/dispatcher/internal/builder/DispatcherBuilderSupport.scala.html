46   * Access a typed object in the given envelope and the given key. If an object
47   * for the key with the propert type is present, the given function will be applied
48   * and the result of the function will be returned.
49   * If the object is not present in the envelope or has the wrong type, an exception
50   * will be returned.
51   */
52   def withContextObject[T,R](key : String, env: FlowEnvelope, streamLogger: FlowEnvelopeLogger)(f : T => Try[R])(implicit classTag: ClassTag[T]) : Either[FlowEnvelope, R] = {
53 
54     env.getFromContext[T](key).get match {
55 
56       // The object can't be found for the key with the given type
57       case None => // Should not be possible
58         val e = new MissingContextObject(key, classTag.runtimeClass.getName)
59         streamLogger.underlying.error(e)(e.getMessage)
60         Left(env.withException(e))
61 
62       // We have found the object, now we try to apply the function
63       case Some(o) =>
64         f(o) match {
65           case Success(s) => Right(s)
66           case Failure(t) => Left(env.withException(t))
67         }
68     }
69   }
