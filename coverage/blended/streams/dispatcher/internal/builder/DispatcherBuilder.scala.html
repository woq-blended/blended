46   def core()(implicit system : ActorSystem) : Graph[FanOutShape3[FlowEnvelope, FlowEnvelope, WorklistEvent, FlowEnvelope], NotUsed] = {
47 
48     GraphDSL.create() { implicit builder =>
49 
50       // This is where we pick up the messages from the source, populate the headers
51       // and perform initial checks if the message can be processed
52       val processInbound = builder.add(DispatcherInbound(dispatcherCfg, ctCtxt, envLogger))
53 
54       // The fanout step will produce one envelope per outbound config of the resource type
55       // sent in the message. The envelope context will contain the config for the outbound
56       // branch and the overall resource type config.
57       // Each envelope will also contain the destination routing.
58       // The step will also emit a WorklistStarted event if the calculation of the
59       // fanout steps was successfull.
60       // The step will emit an error envelope if an exception was thrown.
61       val processFanout = builder.add(DispatcherFanout(dispatcherCfg, ctCtxt, envLogger).build())
62 
63       // The error splitter pushes all envelopes that have an exception defined to the error sink
64       // and all messages without an exception defined to the normal sink
65       val errorSplitter = builder.add(FlowProcessor.partition[FlowEnvelope](_.exception.isEmpty))
66 
67       // wire up the steps
68       processInbound ~> processFanout.in
69       processFanout.out0 ~> errorSplitter.in
70 
71       new FanOutShape3(
72         processInbound.in,
73         errorSplitter.out0, // Normal outcome
74         processFanout.out1, // WorklistStarted event
75         errorSplitter.out1  // Outcome with Exception
76       )
77     }

80   // We try to process the outbound message using the injected outbound flow
81   // and try to trans to resulting FlowEnvelope into a worklist.
82   // If the worklist creation yields an exception, a FlowEnvelope with
83   // exception will be passed downstream.
84   // If after the oubound flow the envelope is marked with an exception,
85   // We will generate a worklist failed event, otherwise we will generate
86   // a Worklist completed event with the in bound envelope.
87   def outbound() : Graph[FanOutShape2[FlowEnvelope, WorklistEvent, FlowEnvelope], NotUsed] = {
88 
89     GraphDSL.create() { implicit b =>
90 
91       val outbound = b.add(sendFlow)
92 
93       val toWorklist = b.add(Flow.fromFunction[FlowEnvelope, Either[FlowEnvelope, WorklistEvent]] { env =>
94         try {
95           val worklist = bs.worklist(env).unwrap
96           val event : WorklistEvent = env.exception match {
97             case None => WorklistStepCompleted(worklist = worklist, state = WorklistStateCompleted)
98             case Some(_) => WorklistStepCompleted(worklist = worklist, state = WorklistStateFailed)
99           }
100 
101           Right(event)
102         } catch {
103           case t : Throwable => Left(env.withException(t))
104         }
105 
106       })
107 
108       // This error happens if for some reason we cannot create the worklist for the worklist event
109       val transformError = b.add(FlowProcessor.splitEither[FlowEnvelope, WorklistEvent]())
110 
111       outbound ~> toWorklist ~> transformError.in
112 
113       new FanOutShape2(
114         outbound.in,
115         transformError.out1, // A WorklistEvent signalling either Completed or Failed
116         transformError.out0 // An exceptional outcome while creating the worklist
117       )
118     }
119   }

121   private[builder] def eventEnvelopes(worklistEvent : WorklistEvent) : Seq[FlowEnvelope] = {
122     val result = worklistEvent.worklist.items match {
123       // Should not happen
124       case Seq() =>
125         Seq(
126           FlowEnvelope(FlowMessage.noProps, worklistEvent.worklist.id)
127             .withException(new MismatchedEnvelopeException(worklistEvent.worklist.id))
128         )
129       case s =>
130         s.map {
131           case flowItem : FlowWorklistItem => flowItem.env
132           // Should not happen
133           case _ => FlowEnvelope(FlowMessage.noProps, worklistEvent.worklist.id)
134             .withException(new MismatchedEnvelopeException(worklistEvent.worklist.id))
135         }
136     }
137 
138     envLogger.underlying.debug(s"Found worklist envelopes : [$result]")
139     result
140   }

145   private[builder] def transactionUpdate(event : WorklistEvent) : (WorklistEvent, Option[FlowTransactionEvent]) = {
146 
147     val props : FlowMessageProps = event.worklist.items match {
148       case Seq() => FlowMessage.noProps
149       case h :: _ => h match {
150         case flowItem : FlowWorklistItem => flowItem.env.flowMessage.header
151         case _                           => FlowMessage.noProps
152       }
153     }
154 
155     val transEvent : Option[FlowTransactionEvent] = event match {
156       // The started event will just update the FlowTransaction with a new worklist
157       case started : WorklistStarted =>
158         Some(FlowTransactionUpdate(
159           transactionId = started.worklist.id,
160           properties = props,
161           updatedState = WorklistStateStarted,
162           branchIds = branchIds(eventEnvelopes(event)):_*
163         ))
164       // Worklist Termination does nothing for completed worklists,
165       // for failed worklists it produces a Transaction failed update
166       case term : WorklistTerminated =>
167         if (term.state == WorklistStateCompleted) {
168           val envelopes = eventEnvelopes(term)
169             // We only send transaction updates for a completed worklist, if auto completion is set to true ...
170             .filter { _.header[Boolean](bs.headerAutoComplete).getOrElse(true) }
171             // ... AND the bridge outbound destination lies within the internal JMS provider
172             // (if the message goes to external, the final bridge send will complete the transaction
173             .filter { env =>
174               (env.header[String](bs.headerBridgeVendor), env.header[String](bs.headerBridgeProvider)) match {
175                 case (Some(v), Some(p)) => dispatcherCfg.providerRegistry.jmsProvider(v, p).exists(_.internal)
176                 case (_, _)             => false
177               }
178             }
179 
180           if (envelopes.isEmpty) {
181             envLogger.underlying.debug(s"No item envelopes found for [${event.worklist.id}]")
182             None
183           } else {
184             Some(FlowTransactionUpdate(term.worklist.id, props, WorklistStateCompleted, branchIds(envelopes):_*))
185           }
186         } else {
187           Some(FlowTransactionFailed(event.worklist.id, props, term.reason.map(_.getMessage())))
188         }
189       // Completed worklist steps do nothing
190       case step : WorklistStepCompleted =>
191         envLogger.underlying.debug(s"No transaction event for completed worklist [${event.worklist.id}]")
192         None
193     }
194 
195     envLogger.underlying.debug(s"Transaction update for worklist event [${event.worklist.id}] is [$transEvent]")
196     (event, transEvent)
197   }

199   private[builder] def acknowledge(event : WorklistEvent) : FlowEnvelope = {
200 
201     eventEnvelopes(event) match {
202       case Seq() =>
203         FlowEnvelope(FlowMessage.noProps, event.worklist.id).withException(new MismatchedEnvelopeException(event.worklist.id))
204       case h :: _ =>
205         if (h.requiresAcknowledge) {
206           classLogger.debug(s"Acknowledging envelope [${h.id}]")
207           h.acknowledge()
208         }
209         h
210     }

213   def worklistEventHandler() : Graph[FanOutShape2[WorklistEvent, FlowTransactionEvent, FlowEnvelope], NotUsed] = {
214 
215     GraphDSL.create() { implicit b =>
216 
217       // The worklist Manager will track currently open worklist events and emit accumulated Worklist Events
218       // The worklist manager will produce Worklist started and Worklist Terminated events
219       val wlManager = b.add(WorklistManager.flow("worklistMgr", envLogger).named("worklistMgr"))
220 
221       // We process the outcome of the worklist manager and transform it into Transaction events
222       val processEvent = b.add(Flow.fromFunction[WorklistEvent, (WorklistEvent, Option[FlowTransactionEvent])](transactionUpdate))
223 
224       // if processEvent did come back with a TransactionEvent, this will be passed downstream
225       val branches = b.add(
226         Broadcast[(WorklistEvent, Option[FlowTransactionEvent])](2).named("wlBranch")
227       )
228 
229       // This will generate a FlowTransactionEvent if necessary
230       val processTrans = b.add(Flow[(WorklistEvent, Option[FlowTransactionEvent])]
231         .map(_._2).named("selectTrans")
232         .filter(_.isDefined).named("hasUpdate")
233         .map(_.get).named("getTrans"))
234 
235       // For completed worklist we will acknowledge the envelope and capture exceptions
236       val processWorklist = b.add(
237         Flow[(WorklistEvent, Option[FlowTransactionEvent])]
238         .map(_._1)
239         .filter(_.state == WorklistStateCompleted)
240         .map(acknowledge)
241         .filter(_.exception.isDefined)
242       )
243 
244       wlManager ~> processEvent ~> branches
245 
246       branches.out(0) ~> processTrans
247       branches.out(1) ~> processWorklist
248 
249       new FanOutShape2(wlManager.in, processTrans.out, processWorklist.out)
250     }
251   }

253   def errorHandler() : Flow[FlowEnvelope, FlowTransactionEvent, NotUsed] = {
254     val g : Graph[FlowShape[FlowEnvelope, FlowEnvelope], NotUsed] = GraphDSL.create() { implicit b =>
255       import GraphDSL.Implicits._
256 
257       val routeError = b.add(Flow.fromFunction[FlowEnvelope, FlowEnvelope] { env =>
258 
259         try {
260           val vendor = env.header[String](srcVendorHeader(bs.headerConfig.prefix)).get
261           val provider = env.header[String](srcProviderHeader(bs.headerConfig.prefix)).get
262 
263           val errProvider = dispatcherCfg.providerRegistry.jmsProvider(vendor, provider).get
264           val dest = errProvider.errors.asString
265 
266           envLogger.logEnv(env, LogLevel.Debug, s"Routing error envelope [${env.id}] to [$vendor:$provider:$dest]")
267 
268           env
269             .withHeader(deliveryModeHeader(bs.headerConfig.prefix), JmsDeliveryMode.Persistent.asString).get
270             .withHeader(bs.headerBridgeVendor, vendor).get
271             .withHeader(bs.headerBridgeProvider, provider).get
272             .withHeader(bs.headerBridgeDest, dest).get
273             .withHeader(bs.headerConfig.headerState, FlowTransactionStateFailed.toString).get
274         } catch {
275           case t : Throwable =>
276             envLogger.logEnv(env, LogLevel.Warn, s"Failed to resolve error routing for envelope [${env.id}] : [${t.getMessage()}]")
277             env
278         }
279       })
280 
281       val sendError = b.add(sendFlow)
282 
283       val ackError = b.add(Flow.fromFunction[FlowEnvelope, FlowEnvelope] { env =>
284         envLogger.logEnv(env, LogLevel.Debug, s"Acknowledging error envelope [${env.id}]")
285         env.acknowledge()
286         env
287       })
288 
289       routeError ~> sendError ~> ackError
290 
291       FlowShape(routeError.in, ackError.out)
292     }
293 
294     Flow.fromGraph(g)
295       .via(Flow.fromFunction[FlowEnvelope, FlowTransactionEvent] { env =>
296         val event = FlowTransactionFailed(env.id, env.flowMessage.header, env.exception.map(_.getMessage()))
297         envLogger.logEnv(env, LogLevel.Debug, s"Transaction event : [$event]")
298         event
299       })

302   // The dispatcher processes a stream of inbound FlowEnvelopes and generates TransactionEvents to update
303   // a monitored FlowTransaction.
304   //
305   // This happens in the core graph :
306   // 1. Generate a Transaction Started Event
307   // 2. Perform the default inbound processing and fanout
308   //    after this step we may end up with multiple FlowEnvelopes per inbound message
309   // 3. The total of all generated fanout messages is called the worklist
310   // 4. Update the open transaction with the worklist, so that it can track the individual branches.
311   // 5. Any exceptions are available on the error outlet
312 
313   // This happens in the out graph:
314   // 5. For each fanout message
315   //    a. process the message via the outbound flow
316   //    b. if processed successfully mark the worklist item as completed
317   //    c. if processed with exception, mark the worklist item as failed
318   //    d. for an internal error, generate mark the envelope as exceptional
319 
320   // This happens in the dispatcher :
321   // 6. The worklist manager collects all worklist events, updates the worklist state and pushes the resulting state downstream
322   // 7. The error handler collects all envelopes with exceptions and generates a transaction event from here
323   // 8. The worklist eventhandler generates a transaction failed event for each failed worklist
324 
325   // 9. The combined transaction events are passed down stream
326 
327   def dispatcher()(implicit system: ActorSystem) : Graph[FlowShape[FlowEnvelope, FlowTransactionEvent], NotUsed] = {
328 
329     GraphDSL.create() { implicit b =>
330       // of course we start with the core
331       val callCore = b.add(core())
332 
333       // we do need a send processor
334       val callSend = b.add(outbound())
335 
336       // We will collect the errors here
337       val error = b.add(Merge[FlowEnvelope](3))
338 
339       // We will collect Worklist Events here
340       val event = b.add(Merge[WorklistEvent](2))
341 
342       // we will collect transaction events here
343       val trans = b.add(Merge[FlowTransactionEvent](2))
344 
345       val completeStatsOk = b.add(
346         FlowProcessor.completeStats("completeStats", envLogger, bs.headerConfig)
347       )
348 
349       val completeStatsFailed = b.add(
350         FlowProcessor.completeStats("completeStats", envLogger, bs.headerConfig)
351       )
352 
353       // the normal outcome of core goes to send
354       callCore.out0 ~> completeStatsOk ~> callSend.in
355       // The worklist started event goes to the event channel
356       callCore.out1 ~> event
357       // errors go the error channel
358       callCore.out2 ~> completeStatsFailed ~> error
359 
360       // the normal output of the sendflow are worklist events, so they go to the event channel
361       val evtHandler = b.add(worklistEventHandler())
362       callSend.out0 ~> event // Worklist events
363 
364       // All worklist events will go to the worklist manager, so that we can track worklist completions or failures
365       event ~> evtHandler.in
366 
367       evtHandler.out0 ~> trans.in(0) // worklist events to FlowTransactions
368       evtHandler.out1 ~> error // Error channel - only occurs when envelope matching worklist branch can't be resolved
369 
370       // if envelopes marked with an exception again got to the error channel
371       callSend.out1 ~> error
372 
373       val errHandler = b.add(errorHandler())
374 
375       // generate and collect transaction failures for exceptions
376       error.out ~> errHandler ~> trans.in(1)
377 
378       FlowShape(callCore.in, trans.out)
379     }
380   }
