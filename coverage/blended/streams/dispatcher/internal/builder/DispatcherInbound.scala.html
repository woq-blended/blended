42       Try {
43         env.header[String](bs.headerResourceType) match {
44           case None =>

67 
68           if (rtCfg.withCBE) {
69             val newMsg = env.flowMessage
70               .withHeader(bs.headerEventVendor, dispatcherCfg.eventProvider.vendor).get
71               .withHeader(bs.headerEventProvider, dispatcherCfg.eventProvider.provider).get
72               .withHeader(bs.headerEventDest, dispatcherCfg.eventProvider.transactions.asString).get
73               .withHeader(bs.headerCbeEnabled, true).get
74 
75             env.copy(flowMessage = newMsg)
76           } else {
77             env.withHeader(bs.headerCbeEnabled, false).get
78           }
79         }
80       }
81     }
