25   private[builder] val funFanoutOutbound : FlowEnvelope => Try[Seq[(OutboundRouteConfig, FlowEnvelope)]] = { env =>
26 
27     bs.withContextObject[ResourceTypeConfig, Seq[(OutboundRouteConfig, FlowEnvelope)]](bs.rtConfigKey, env, streamLogger) { rtCfg: ResourceTypeConfig =>
28       Try {
29         rtCfg.outbound.map { ob =>
30           val obEnv =
31             env
32               .withContextObject(bs.outboundCfgKey, ob)
33               .withHeader(bs.headerConfig.headerBranch, ob.id).unwrap
34           (ob, outboundMsg(ob)(obEnv).unwrap)
35         }
36       }
37     } match {
38       case Right(s) => Success(s)
39       case Left(t) =>
40         t.exception match {
41           case None => streamLogger.logEnv(env, LogLevel.Error, s"Exception in fan out step [${env.id}]")
42           case Some(t) => streamLogger.logEnv(env.withException(t), LogLevel.Error, s"Exception in fan out step [${env.id}]")
43         }
44         Failure(t.exception.getOrElse(new Exception("Unexpected exception")))
45     }
46   }

52   private lazy val outboundMsg : OutboundRouteConfig => FlowEnvelope => Try[FlowEnvelope] = { outCfg => env =>
53 
54     val useHeaderBlock : OutboundHeaderConfig => Try[Boolean] = { oh =>
55       Try {
56         oh.condition match {
57           // if the block does not have a condition, the header block will be used
58           case None => true
59           case Some(c) =>
60             val resolve = ctCtxt.resolveString(c, env.flowMessage.header.mapValues(_.value))
61             streamLogger.logEnv(env, LogLevel.Debug, s"Resolved condition to [$resolve][${resolve.map(_.getClass().getName())}]")
62             val use = resolve.map(_.asInstanceOf[Boolean]).unwrap
63 
64             val s = s"using header for [${env.id}]:[outboundMsg] block with expression [$c]"
65             if (use) {
66               streamLogger.logEnv(env, LogLevel.Debug, s)
67             } else {
68               streamLogger.logEnv(env, LogLevel.Debug, "Not " + s )
69             }
70             use
71         }
72       }
73     }
74 
75     Try {
76 
77       outCfg.outboundHeader.filter(b => useHeaderBlock(b).unwrap).foldLeft(env) {
78         case (current, oh) =>
79           var newEnv : FlowEnvelope = current
80             .withHeader(bs.headerConfig.headerMaxRetries, oh.maxRetries).unwrap
81             .withHeader(bs.headerAutoComplete, oh.autoComplete).unwrap
82 
83           if (oh.timeToLive >= 0L) {
84             newEnv = newEnv.withHeader(bs.headerTimeToLive, oh.timeToLive).unwrap
85           } else {
86             newEnv = newEnv.removeHeader(bs.headerTimeToLive)
87           }
88 
89           newEnv = newEnv
90             .withHeader(deliveryModeHeader(bs.headerConfig.prefix), oh.deliveryMode).unwrap
91 
92         oh.header.foreach { case (header, value) =>
93           val resolved = ctCtxt.resolveString(value, env.flowMessage.header.mapValues(_.value)).get
94           streamLogger.logEnv(env, LogLevel.Trace,s"[${newEnv.id}]:[${outCfg.id}] - resolved property [$header] to [$resolved]")
95           newEnv = newEnv.withHeader(header, resolved).unwrap
96         }
97 
98           newEnv = if (oh.clearBody) {
99             newEnv.copy(flowMessage = newEnv.flowMessage.clearBody())
100           } else {
101             newEnv
102           }
103 
104           newEnv
105             .withContextObject(bs.appHeaderKey, oh.applicationLogHeader)
106             .withContextObject(bs.bridgeProviderKey, oh.bridgeProviderConfig)
107             .withContextObject(bs.bridgeDestinationKey, oh.bridgeDestination)
108       }
109     }

113

130   def build() : Graph[FanOutShape2[FlowEnvelope, FlowEnvelope, WorklistEvent], NotUsed] = {
131     GraphDSL.create() { implicit builder =>
132       import GraphDSL.Implicits._
133 
134       val fanout = builder.add(fanoutOutbound)
135 
136       val errorFilter = builder.add(Broadcast[Either[FlowEnvelope, Seq[(OutboundRouteConfig, FlowEnvelope)]]](2))
137       val withError = builder.add(Flow[Either[FlowEnvelope, Seq[(OutboundRouteConfig, FlowEnvelope)]]].filter(_.isLeft).map(_.left.get))
138       val noError = builder.add(Flow[Either[FlowEnvelope, Seq[(OutboundRouteConfig, FlowEnvelope)]]].filter(_.isRight).map(_.right.get))
139       val mapDestination = builder.add(decideRouting)
140 
141       val createWorklist = builder.add(Broadcast[Seq[(OutboundRouteConfig, FlowEnvelope)]](2))
142 
143       val envelopes = builder.add(Flow[Seq[(OutboundRouteConfig, FlowEnvelope)]].mapConcat(_.toList).map(_._2))
144       val worklist = builder.add(Flow[Seq[(OutboundRouteConfig, FlowEnvelope)]].map(toWorklist))
145 
146       val merge = builder.add(Merge[FlowEnvelope](2))
147 
148       fanout ~> errorFilter ~> withError ~> merge
149       errorFilter ~> noError ~> createWorklist.in
150 
151       createWorklist.out(0) ~> envelopes ~> mapDestination ~> merge
152       createWorklist.out(1) ~> worklist
153 
154       new FanOutShape2(
155         fanout.in,
156         merge.out,
157         worklist.out
158       )
159     }
160   }
