40   private[builder] def dispatcherSend(streamLogger : FlowEnvelopeLogger) : Flow[FlowEnvelope, FlowEnvelope, NotUsed] = {
41 
42     val g : Graph[FlowShape[FlowEnvelope, FlowEnvelope], NotUsed] = {
43       GraphDSL.create() { implicit b =>
44 
45         import GraphDSL.Implicits._
46 
47         val sendProducerSettings = JmsProducerSettings(
48           log = streamLogger,
49           connectionFactory = cf,
50           headerCfg = bs.headerConfig,
51           destinationResolver = s => new DispatcherDestinationResolver(s, registry, bs, streamLogger),
52           logLevel = env => if (
53             env.header[String](bs.headerConfig.headerBridgeVendor).contains(internal.vendor) &&
54               env.header[String](bs.headerConfig.headerBridgeProvider).contains(internal.provider)
55           ) {
56             // In case the final destination lies within the internal JMS provider, we log the message sent as info
57             LogLevel.Info
58           } else {
59             // Otherwise we log it as debug
60             LogLevel.Debug
61           }
62         )
63 
64         val clearRetrying = b.add(
65           FlowProcessor.fromFunction("clearRetrying", streamLogger){ env => Try {
66             env.removeHeader(bs.headerConfig.headerRetrying)
67           }}
68         )
69 
70         val performSend = b.add(jmsProducer(
71           name = "dispatcherSend",
72           autoAck = false,
73           settings = sendProducerSettings
74         ))
75 
76         clearRetrying ~> performSend
77 
78         FlowShape(clearRetrying.in, performSend.out)
79       }
80     }
81 
82     Flow.fromGraph(g)
83   }

87     Graph[FlowShape[FlowTransactionEvent, FlowEnvelope], NotUsed] = {
88 
89     GraphDSL.create() { implicit b =>
90       import GraphDSL.Implicits._
91 
92       val transform = b.add(Flow.fromFunction[FlowTransactionEvent, FlowEnvelope] { t =>
93         FlowTransactionEvent.event2envelope(bs.headerConfig)(t)
94           .withHeader(bs.headerConfig.headerTrackSource, streamLogger.underlying.name).get
95       })
96 
97       val transactionSendSettings = JmsProducerSettings(
98         log = streamLogger,
99         headerCfg = bs.headerConfig,
100         connectionFactory = cf,
101         destinationResolver = s => new TransactionDestinationResolver(s, JmsDestination.asString(internal.transactions)),
102         jmsDestination = None,
103         deliveryMode = JmsDeliveryMode.Persistent,
104         priority = JmsSendParameter.defaultPriority,
105         timeToLive = None,
106         logLevel = _ => LogLevel.Debug
107       )
108 
109       val producer = b.add(jmsProducer(
110         name = "transactionSend",
111         settings = transactionSendSettings,
112         autoAck = false
113       ))
114 
115       transform ~> producer
116       FlowShape(transform.in, producer.out)
117     }

186   private def sendStartupMessages(providerCfg : BridgeProviderConfig) : Unit = {
187 
188     val logger = FlowEnvelopeLogger.create(
189       bs.headerConfig,
190       Logger(bs.headerConfig.prefix + ".dispatcher." + providerCfg.vendor + "." + providerCfg.provider + ".startup")
191     )
192 
193     try {
194       implicit val eCtxt : ExecutionContext = system.dispatcher
195 
196       if (routerCfg.startupMap.nonEmpty) {
197 
198         logger.underlying.debug(s"Waiting for JMS connection [${providerCfg.vendor},${providerCfg.provider}] to connect ...")
199 
200         val stateActor : ActorRef = system.actorOf(Props(new Actor {
201 
202           val prodSettings: JmsProducerSettings = JmsProducerSettings(
203             log = logger,
204             headerCfg = bs.headerConfig,
205             connectionFactory = cf,
206             jmsDestination = Some(providerCfg.inbound),
207             deliveryMode = JmsDeliveryMode.NonPersistent,
208             timeToLive = None
209           )
210 
211           val msgs : Seq[FlowEnvelope] = routerCfg.startupMap.map { case (k, v) =>
212             FlowEnvelope(FlowMessage(v)(FlowMessage.props(bs.headerConfig.headerResourceType -> k).unwrap))
213           }.toSeq
214 
215           override def receive: Receive = {
216             case e : ConnectionStateChanged if e.state.status == Connected && providerCfg.vendor == e.state.vendor && providerCfg.provider == e.state.provider =>
217               logger.underlying.debug(s"Sending [${msgs.size}] messages to [${providerCfg.vendor}:${providerCfg.provider}:${providerCfg.inbound.asString}]")
218               sendMessages(prodSettings, logger, msgs:_*).unwrap
219               context.stop(self)
220           }
221         }))
222 
223         system.eventStream.subscribe(stateActor, classOf[ConnectionStateChanged])
224         system.eventStream.publish(QueryConnectionState(providerCfg.vendor, providerCfg.provider))
225 
226       } else {
227         logger.underlying.debug("No startup messages configured")
228       }
229     } catch {
230       case NonFatal(e) =>
231         logger.underlying.warn(s"Failed to send startup messages [${e.getMessage()}]")
232     }
233   }

235   def start() : Unit = {
236 
237     try {
238       // Get the internal provider
239       val internalProvider : BridgeProviderConfig = registry.internalProvider.get
240       sendStartupMessages(internalProvider)
241 
242       // The transaction stream will process the transaction events from the transactions destination
243       transStream = Some(transactionStream(tMgr).get)
244 
245       // Create one dispatcher for each configured provider
246       registry.allProvider.foreach { provider =>
247 
248         // Create a specific logger for each Dispatcher instance
249         val dispLogger = Logger(bs.headerConfig.prefix + ".dispatcher." + provider.vendor + "." + provider.provider)
250         val envLogger : FlowEnvelopeLogger = FlowEnvelopeLogger.create(bs.headerConfig, dispLogger)
251 
252         // The blueprint for the dispatcher flow
253         val dispatcher : Flow[FlowEnvelope, FlowTransactionEvent, NotUsed] =
254           Flow.fromGraph(
255             DispatcherBuilder(
256               ctCtxt = ctCtxt,
257               dispatcherCfg = routerCfg,
258               envLogger = envLogger,
259               sendFlow = dispatcherSend(envLogger),
260             )(bs).dispatcher()
261           )
262 
263         val startStats : Graph[FlowShape[FlowEnvelope, FlowEnvelope], NotUsed] = FlowProcessor.fromFunction("startStats", envLogger){ env => Try {
264           val resType : String = env.header[String](bs.headerConfig.headerResourceType).getOrElse("UNKNOWN")
265           val id : String = ServiceInvocationReporter.invoked(
266             component = "dispatcher",
267             subComponents = Map(
268               "jmsvendor" -> provider.vendor,
269               "provider" -> provider.provider,
270               "resourcetype" -> resType
271             )
272           )
273           env.withHeader(bs.headerConfig.headerStatsId, id).unwrap
274         }}
275 
276         // Connect the consumer to a dispatcher
277         val source : Source[FlowTransactionEvent, NotUsed] =
278           bridgeSource(internalProvider, provider, envLogger)
279             .via(startStats)
280             .via(dispatcher)
281 
282         // Wrap the dispatcher into a stream controller and make sure, the generated transaction events are sent to
283         // the proper JMS destination
284         val actor : ActorRef = system.actorOf(StreamController.props[FlowEnvelope, NotUsed](
285           streamName = dispLogger.name,
286           src = source.via(transactionSend(envLogger)),
287           streamCfg = streamsCfg
288         )(onMaterialize = _ => ()))
289 
290         dispLogger.debug(s"Started dispatcher flow for provider [${provider.id}]")
291         startedDispatchers.put(provider.id, actor)
292       }
293     } catch {
294       case t : Throwable => Logger[RunnableDispatcher].error(t)(s"Failed to start dispatcher [${t.getMessage()}]")
295     }
296   }

298   def stop() : Unit = {
299     transMgr.foreach(system.stop)

302     startedDispatchers.clear()
