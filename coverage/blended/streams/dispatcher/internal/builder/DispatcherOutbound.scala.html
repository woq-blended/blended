25   private[builder] def resolveProvider(
26     registry : BridgeProviderRegistry,
27     vendor : Option[String],
28     provider : Option[String]
29   ) : Try[BridgeProviderConfig] = Try {
30     (vendor, provider) match {
31       case (Some(v), Some(p)) => registry.jmsProvider(v, p) match {
32         case None    => throw new Exception(s"Could not resolve provider [$vendor:$provider]")
33         case Some(r) => r
34       }
35       case (_, _) => throw new Exception(s"Could not resolve provider [$vendor:$provider]")
36     }
37   }

39   val resolveDest : ContainerContext => DispatcherBuilderSupport => FlowEnvelope => Try[JmsDestination] = { ctCtxt => bs => env => Try {
40 
41     env.getFromContext[Option[String]](bs.bridgeDestinationKey).get match {
42       case None => throw new Exception(s"Failed to resolve context object [${bs.bridgeDestinationKey}]")
43       case Some(ctxtDest) => ctxtDest match {
44         case None => JmsDestination.create(JmsFlowSupport.replyToQueueName).get
45         case Some(d) =>

48             s = d,
49             additionalProps = env.flowMessage.header.mapValues(_.value)
50           ).map(_.toString()).get

52           JmsDestination.create(name).get

54       }
55     }

59   private[builder] def outboundRouting(
60     dispatcherCfg : ResourceTypeRouterConfig,
61     ctCtxt : ContainerContext,
62     bs : DispatcherBuilderSupport,
63     streamLogger : FlowEnvelopeLogger
64   )(env: FlowEnvelope) : Try[DispatcherTarget] = Try {
65 
66     val dest : JmsDestination = resolveDest(ctCtxt)(bs)(env).get
67 
68     val targetDest : JmsDestination = dest.name match {
69       case JmsFlowSupport.replyToQueueName =>
70         val replyToHeader : String = JmsFlowSupport.replyToHeader(bs.headerConfig.prefix)
71         env.header[String](replyToHeader) match {
72           case None    => throw new Exception(s"Header [$replyToHeader] must be set for replyTo dispatcher flow")
73           case Some(s) => JmsDestination.create(s).get
74         }
75 
76       case _ => dest
77     }
78 
79     val bridgeProvider : BridgeProviderConfig = dest.name match {
80       // For the replyto destination we have to respond to the src provider
81       case JmsFlowSupport.replyToQueueName =>
82 
83         val v : Option[String] = env.header[String](bs.srcVendorHeader(bs.headerConfig.prefix)).map { s =>
84           ctCtxt.resolveString(s).map(_.toString()).get
85         }
86 
87         val p : Option[String] = env.header[String](bs.srcProviderHeader(bs.headerConfig.prefix)).map { s =>
88           ctCtxt.resolveString(s).map(_.toString()).get
89         }
90 
91         resolveProvider(dispatcherCfg.providerRegistry, v, p).get
92 
93       case _ =>
94         env.getFromContext[BridgeProviderConfig](bs.bridgeProviderKey).get match {
95           case None    => throw new Exception(s"Failed to resolve context object [${bs.bridgeProviderKey}]")
96           case Some(p) => p
97         }
98     }
99 
100     val result = DispatcherTarget(
101       bridgeProvider.vendor, bridgeProvider.provider, targetDest
102     )
103     streamLogger.logEnv(env, LogLevel.Debug, s"Routing for [${env.id}] is [$result]")
104 
105     result
106   }
