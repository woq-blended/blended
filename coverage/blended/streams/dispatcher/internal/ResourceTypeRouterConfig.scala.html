25   def create(

28     cfg : Config

30 
31     val internalProvider = provider.internalProvider.get

33     val eventProvider = ProviderResolver.getProvider(
34       provider,

49             defaultEventProvider = eventProvider,
50             defaultLogHeader = logHeader,
51             cfg = value
52           ).get
53       }
54 
55     val defaultHeader : List[HeaderProcessorConfig] = cfg.getConfigList(defaultHeaderPath, List.empty).map { cfg =>
56       HeaderProcessorConfig.create(cfg)
57     }
58 
59     val startupMap : Map[String, String] = cfg.getStringMapOption(startupPath).getOrElse(Map.empty).mapValues(s => ctCtxt.resolveString(s).get.toString)
60 
61     ResourceTypeRouterConfig(
62       defaultProvider = internalProvider,
63       eventProvider = eventProvider,
64       applicationLogHeader = logHeader,
65       defaultHeader = defaultHeader,

188   private[this] val applicationLogHeaderPath = "applicationLogHeader"

210     val moduleLastOnComplete = cfg.getBoolean(moduleLastOnCompletePath, false)
211     val applicationLogHeader = cfg.getStringListOption(applicationLogHeaderPath).getOrElse(defaultLogHeader)
212 
213     val clearBody = cfg.getBoolean(clearBodyPath, false)
214     val autoComplete = cfg.getBoolean(autocompletePath, true)
215     val maxRetries = cfg.getLong(maxRetryPath, -1L)
216     val timeToLive = cfg.getDuration(timeToLivePath, 0.millis).toMillis
217     val delMode = cfg.getString(deliveryModePath, JmsDeliveryMode.Persistent.asString)
218 
219     new OutboundHeaderConfig(
220       bridgeProviderConfig = bridgeProvider,
221       bridgeDestination = bridgeDestination,
222       autoComplete = autoComplete,
223       condition = cfg.getStringOption(conditionPath),
224       applicationLogHeader = applicationLogHeader,
225       maxRetries = maxRetries,
226       timeToLive = timeToLive,
227       moduleLastOnComplete = moduleLastOnComplete,
